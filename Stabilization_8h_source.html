221 DualSolution getStabDualSolAdvanced(const DualSolution &nodeDuals) {
222 // If there's no stabilization center, return the input duals
223 if (cur_stab_center.empty()) { return nodeDuals; }
224
225 // Initialize subgradient if it hasn't been initialized yet
226 if (subgradient.empty()) { subgradient.assign(nodeDuals.size(), 0.0); }
227
228 // Initialize the separation duals and smooth dual solution if they are empty
229 if (duals_sep.empty()) {
230 duals_sep = nodeDuals; // Start with the input duals
231 smooth_dual_sol = duals_sep; // Initialize the smoothed dual solution
232 return nodeDuals; // Return the input duals for now
233 }
234
235 // Get the current stabilization center (duals_in) and the current node duals (duals_out)
236 const auto &duals_in = cur_stab_center;
237 const auto &duals_out = nodeDuals;
238
239 const size_t n = nodeDuals.size();
240 std::vector<double> duals_tilde(n, 0.0);
241 std::vector<double> duals_g(n, 0.0);
242 std::vector<double> rho(n, 0.0);
243
244 // Precompute norms that are used multiple times
245 double norm_in_out = norm(duals_in, duals_out);
246 double norm_subgradient = norm(subgradient);
247
248 // Compute π_tilde: a convex combination of duals_in and duals_out
249 // double one_minus_alpha = 1 - base_alpha;
250 for (size_t row_id = 0; row_id < n; ++row_id) {
251 duals_tilde[row_id] = cur_alpha * duals_in[row_id] + (1 - cur_alpha) * duals_out[row_id];
252 }
253
254 // Compute the coefficient for π_g based on the norm of duals_in and duals_out
255 double coef_g = (norm_subgradient != 0.0) ? norm_in_out / norm_subgradient : 0.0;
256
257 // Compute π_g: duals_in + coef_g * subgradient
258 for (size_t row_id = 0; row_id < n; ++row_id) {
259 duals_g[row_id] = duals_in[row_id] + coef_g * subgradient[row_id];
260 }
261
262 // Compute β: a weight factor that depends on the alignment of duals_g and duals_out
263 double dot_product = 0.0;
264 for (size_t row_id = 0; row_id < n; ++row_id) {
265 dot_product += (duals_out[row_id] - duals_in[row_id]) * (duals_g[row_id] - duals_in[row_id]);
266 }
267
268 double norm_in_g = norm(duals_in, duals_g);
269 beta = (norm_in_out * norm_in_g != 0.0) ? std::max(0.0, dot_product / (norm_in_out * norm_in_g)) : 0.0;
270
271 // Compute ρ: a combination of duals_g and duals_out based on β
272 for (size_t row_id = 0; row_id < n; ++row_id) {
273 rho[row_id] = beta * duals_g[row_id] + (1 - beta) * duals_out[row_id];
274 }
275
276 // Compute the coefficient for dual separation (coef_sep)
277 double norm_in_tilde = norm(duals_in, duals_tilde);
278 double norm_in_rho = norm(duals_in, rho);
279 double coef_sep = (norm_in_rho != 0.0) ? norm_in_tilde / norm_in_rho : 0.0;
280
281 // Update the duals_sep by adjusting duals_in towards ρ
282 for (size_t row_id = 0; row_id < n; ++row_id) {
283 duals_sep[row_id] = duals_in[row_id] + coef_sep * (rho[row_id] - duals_in[row_id]);
284 }
285
286 // Update the smooth dual solution and return it
287 smooth_dual_sol = duals_sep;
288 return duals_sep;
289 }

303 double dynamic_alpha_schedule(const ModelData &dados, const DualSolution &nodeDuals,
304 const std::vector<Label *> &best_pricing_cols) {
305
306 // Get the number of rows from the size of the nodeDuals
307 size_t number_of_rows = nodeDuals.size();
308
309 // Calculate the direction of separation between the current smoothed dual solution and stabilization center
310 std::vector<double> in_sep_direction(number_of_rows);
311 std::transform(smooth_dual_sol.begin(), smooth_dual_sol.end(), cur_stab_center.begin(),
312 in_sep_direction.begin(), std::minus<>());
313
314 // Calculate the norm of the smoothed dual solution (accumulating squared differences)
315 double norm_smooth_dual_sol = std::transform_reduce(in_sep_direction.begin(), in_sep_direction.end(), 0.0,
316 std::plus<>(), [](double a) { return a * a; });
317
318 // If the norm is zero, return the base alpha (no adjustment necessary)
319 if (norm_smooth_dual_sol == 0.0) { return base_alpha; }
320
321 // Initialize new_rows and apply contributions from best_pricing_cols, skipping positive-cost columns
322 new_rows.assign(number_of_rows, 0.0);
323 for (const auto *best_pricing_col : best_pricing_cols) {
324 if (best_pricing_col->cost > 0) { continue; }
325 for (const auto &row : best_pricing_col->nodes_covered) {
326 if (row > 0 && row != N_SIZE - 1) { // Ignore invalid rows and the last node
327 new_rows[row - 1] += 1;
328 }
329 }
330 }
331
332 // Define and set row bounds based on the problem constraints (dados.sense)
333 // Reuse vectors to avoid reallocations
334 static std::vector<double> new_row_lower_bounds, new_row_upper_bounds;
335 if (new_row_lower_bounds.size() != number_of_rows) {
336 new_row_lower_bounds.resize(number_of_rows);
337 new_row_upper_bounds.resize(number_of_rows);
338 }
339
340 // Set row bounds
341 for (size_t row_id = 0; row_id < number_of_rows; ++row_id) {
342 if (dados.sense[row_id] == '<') {
343 new_row_upper_bounds[row_id] = dados.b[row_id];
344 new_row_lower_bounds[row_id] = -std::numeric_limits<double>::infinity();
345 } else if (dados.sense[row_id] == '>') {
346 new_row_upper_bounds[row_id] = std::numeric_limits<double>::infinity();
347 new_row_lower_bounds[row_id] = dados.b[row_id];
348 } else {
349 new_row_upper_bounds[row_id] = dados.b[row_id];
350 new_row_lower_bounds[row_id] = dados.b[row_id];
351 }
352 }
353
354 // Update the subgradient based on new row values and bounds
355 subgradient.assign(number_of_rows, 0.0);
356 for (size_t row_id = 0; row_id < number_of_rows; ++row_id) {
357 subgradient[row_id] = std::min(0.0, new_row_upper_bounds[row_id] - new_rows[row_id]) +
358 std::max(0.0, new_row_lower_bounds[row_id] - new_rows[row_id]);
359 }
360
361 // Compute the norm of the subgradient
362 subgradient_norm = norm(subgradient);
363
364 // Compute the cosine of the angle between in_sep_direction and subgradient
365 double cos_angle =
366 std::inner_product(in_sep_direction.begin(), in_sep_direction.end(), subgradient.begin(), 0.0) /
367 (norm_smooth_dual_sol * subgradient_norm);
368
369 // Negate the cosine angle to invert the direction
370 cos_angle = -cos_angle;
371
372 // Adjust cur_alpha based on the cosine angle and a threshold
373 if (cos_angle > 1e-12) {
374 alpha = std::min(0.99, base_alpha + (1.0 - base_alpha) * 0.1); // Increase cur_alpha
375 } else {
376 alpha = std::max(0.0, base_alpha - 0.1); // Decrease cur_alpha
377 }
378
379 // Return the updated cur_alpha
380 return alpha;
381 }

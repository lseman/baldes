116 double _misprice_schedule(int nb_misprices, double base_alpha) {
117 // Table 1 (left): alpha_tilde = [1 - k * (1 - alpha)]_+
118 const int k = std::max(1, nb_misprices);
119 double alpha = std::max(0.0, 1.0 - k * (1.0 - base_alpha));

121 alpha = 0.0; // Deactivate stabilization

123 duals_in = duals_sep;

138 DualSolution getStabDualSol(const DualSolution &input_duals) {
139 if (input_duals.empty()) { return input_duals; }
140 DualSolution pi_out;
141 pi_out.assign(input_duals.begin(), input_duals.begin() + sizeDual);
142 if (cur_stab_center.empty()) { return pi_out; }
143 if (cur_alpha <= 0.0) {
144 smooth_dual_sol = pi_out;
145 duals_sep = pi_out;
146 return pi_out;
147 }
148
149 // Non-directional smoothing:
150 // pi_tilde = alpha * pi_in + (1-alpha) * pi_out
151 const size_t n = pi_out.size();
152 DualSolution pi_tilde(n);
153 for (size_t i = 0; i < n; ++i) { pi_tilde[i] = cur_alpha * cur_stab_center[i] + (1.0 - cur_alpha) * pi_out[i]; }
154
155 // If directional components are unavailable, return pi_tilde projected
156 // to the positive orthant.
157 if (subgradient.empty() || subgradient_norm <= EPSILON || beta <= 0.0) {
158 for (double &v : pi_tilde) v = std::max(0.0, v);
159 smooth_dual_sol = pi_tilde;
160 duals_sep = pi_tilde;
161 return pi_tilde;
162 }
163
164 // Directional smoothing (Table 2):
165 // pi_g = pi_in + (g_in / ||g_in||) * ||pi_out - pi_in||
166 // rho = beta*pi_g + (1-beta)*pi_out
167 // pi_sep = (pi_in + ||pi_tilde-pi_in||/||rho-pi_in|| * (rho-pi_in))_+
168 const double norm_in_out = std::sqrt(std::inner_product(pi_out.begin(), pi_out.end(), cur_stab_center.begin(),
169 0.0, std::plus<double>(),
170 [](double a, double b) {
171 const double d = a - b;
172 return d * d;
173 }) +
174 EPSILON);
175
176 DualSolution pi_g(n);
177 for (size_t i = 0; i < n; ++i) {
178 pi_g[i] = cur_stab_center[i] + (subgradient[i] / subgradient_norm) * norm_in_out;
179 }
180
181 DualSolution rho(n);
182 for (size_t i = 0; i < n; ++i) { rho[i] = beta * pi_g[i] + (1.0 - beta) * pi_out[i]; }
183
184 const double norm_tilde_in = std::sqrt(std::inner_product(pi_tilde.begin(), pi_tilde.end(),
185 cur_stab_center.begin(), 0.0, std::plus<double>(),
186 [](double a, double b) {
187 const double d = a - b;
188 return d * d;
189 }) +
190 EPSILON);
191 const double norm_rho_in =
192 std::sqrt(std::inner_product(rho.begin(), rho.end(), cur_stab_center.begin(), 0.0, std::plus<double>(),
193 [](double a, double b) {
194 const double d = a - b;
195 return d * d;
196 }) +
197 EPSILON);
198
199 DualSolution pi_sep(n);
200 for (size_t i = 0; i < n; ++i) {
201 pi_sep[i] = cur_stab_center[i] + (norm_tilde_in / norm_rho_in) * (rho[i] - cur_stab_center[i]);
202 pi_sep[i] = std::max(0.0, pi_sep[i]);
203 }
204 smooth_dual_sol = pi_sep;
205 duals_sep = pi_sep;
206 return pi_sep;
207 }

225 bool dynamic_alpha_schedule(const ModelData &dados) {
226 constexpr double DOT_TOLERANCE = 1e-3;
227 const size_t n = cur_stab_center.size();
228
229 // Compute relative distance: ||smooth_dual_sol - cur_stab_center|| /
230 // |lp_obj|
231 double rel_distance = norm(smooth_dual_sol, cur_stab_center) / (std::abs(lp_obj) + EPSILON);
232 if (rel_distance < NORM_TOLERANCE) {
233 alpha = 0.0;
234 return false;
235 }
236
237 // Compute the difference vector (direction = smooth_dual_sol -
238 // cur_stab_center)
239 std::vector<double> direction(n);
240 for (size_t i = 0; i < n; ++i) { direction[i] = smooth_dual_sol[i] - cur_stab_center[i]; }
241
242 // Compute norm of direction vector using inner_product
243 double dir_norm =
244 std::sqrt(std::inner_product(direction.begin(), direction.end(), direction.begin(), 0.0) + EPSILON);
245 if (dir_norm < EPSILON || subgradient_norm < EPSILON) { return false; }
246
247 // Normalize the direction and the subgradient vectors.
248 std::vector<double> normalized_direction(n);
249 std::vector<double> normalized_subgradient(n);
250 for (size_t i = 0; i < n; ++i) {
251 normalized_direction[i] = direction[i] / dir_norm;
252 normalized_subgradient[i] = subgradient[i] / subgradient_norm;
253 }
254
255 // Compute the cosine of the angle between normalized_direction and
256 // normalized_subgradient.
257 double cos_angle = std::inner_product(normalized_direction.begin(), normalized_direction.end(),
258 normalized_subgradient.begin(), 0.0);
259
260 // If the cosine of the angle is very close to zero, then the vectors
261 // are nearly orthogonal.
262 return cos_angle < DOT_TOLERANCE;
263 }

265 void update_subgradient(const ModelData &dados, const DualSolution &nodeDuals,
266 const std::vector<Label *> &best_pricing_cols, bool use_weighted_update = true) {
267 size_t number_of_rows = nodeDuals.size();
268 new_rows.assign(number_of_rows, 0.0);
269
270 // Determine how many columns to check.
271 auto cols_to_check = std::min(numK, static_cast<int>(best_pricing_cols.size()));
272 for (size_t i = 0; i < cols_to_check; ++i) {
273 const auto &col = best_pricing_cols[i];
274
275 // Compute a weight for this column.
276 double weight = 1.0;
277 if (use_weighted_update) {
278 // weight inversely proportional to cost
279 weight = 1.0 / (1.0 + std::abs(col->cost));
280 }
281
282 // Accumulate coverage for nodes that are not the artificial start
283 // or end.
284 for (const auto &node : col->getRoute()) {
285 if (node > 0 && node != N_SIZE - 1) { new_rows[node - 1] += weight; }
286 }
287 }
288
289 // Compute the raw subgradient based on the constraint senses.
290 subgradient.assign(number_of_rows, 0.0);
291 for (size_t i = 0; i < number_of_rows; ++i) {
292 if (dados.sense[i] == '<') {
293 subgradient[i] = new_rows[i] - dados.b[i];
294 } else if (dados.sense[i] == '>') {
295 subgradient[i] = dados.b[i] - new_rows[i];
296 } else {
297 subgradient[i] = std::abs(dados.b[i] - new_rows[i]);
298 }
299 }
300
301 // Optionally, blend the computed subgradient with the current duals.
302 // This can help smooth the updates over iterations.
303 // for (size_t i = 0; i < number_of_rows; ++i) {
304 // Here we use a simple average. You might use a different
305 // combination or a momentum term.
306 // subgradient[i] = 0.5 * subgradient[i] + 0.5 * nodeDuals[i];
307 // }
308
309 // Recompute the norm of the subgradient.
310 subgradient_norm = norm(subgradient);
311 }

322 void update_stabilization_after_pricing_optim(const ModelData &dados, const DualSolution &input_duals,
323 const double &lag_gap,
324 const std::vector<Label *> &best_pricing_cols) {
325 std::vector<double> nodeDuals(input_duals.begin(), input_duals.begin() + sizeDual);
326
327 if (nb_misprices == 0) {
328 update_subgradient(dados, nodeDuals, best_pricing_cols);
329
330 // Dynamic alpha schedule (Table 1 right):
331 // if g_sep · (pi_out - pi_in) > 0 -> fincr(alpha)
332 // else -> fdecr(alpha)
333 double g_dot_dir = 0.0;
334 if (!subgradient.empty() && subgradient.size() == nodeDuals.size() && !cur_stab_center.empty()) {
335 for (size_t i = 0; i < nodeDuals.size(); ++i) {
336 g_dot_dir += subgradient[i] * (nodeDuals[i] - cur_stab_center[i]);
337 }
338 }
339
340 auto fincr = [](double a) { return a + (1.0 - a) * 0.1; };
341 auto fdecr = [](double a) {
342 if (a >= 0.5) return a / 1.1;
343 return std::max(0.0, a - (1.0 - a) * 0.1);
344 };
345 alpha = (g_dot_dir > 0.0) ? fincr(alpha) : fdecr(alpha);
346 alpha = std::clamp(alpha, 0.0, kAlphaMax);
347 base_alpha = std::clamp(alpha, kAlphaMin, kAlphaMax);
348
349 // Adaptive beta schedule (Section directional smoothing):
350 // beta = cos(gamma) between (pi_out - pi_in) and (pi_g - pi_in).
351 if (!subgradient.empty() && subgradient_norm > EPSILON && !cur_stab_center.empty()) {
352 double dir_norm_sq = 0.0;
353 double dot = 0.0;
354 for (size_t i = 0; i < nodeDuals.size(); ++i) {
355 const double d = nodeDuals[i] - cur_stab_center[i];
356 dir_norm_sq += d * d;
357 dot += d * subgradient[i];
358 }
359 const double dir_norm = std::sqrt(dir_norm_sq + EPSILON);
360 beta = std::clamp(dot / (dir_norm * subgradient_norm), 0.0, 1.0);
361 } else {
362 beta = 0.0;
363 }
364
365 // In-point update between master iterations.
366 stab_center_for_next_iteration = smooth_dual_sol;
367 } else {
368 // During mispricing sequence keep in-point fixed.
369 stab_center_for_next_iteration = cur_stab_center;
370 }
371 cut_added = false;
372
373 lag_gap_prev = lag_gap;
374
375 if (std::isnan(alpha) || std::isinf(alpha)) {
376 stabilization_active = false;
377 cleanup();
378 }
379 }

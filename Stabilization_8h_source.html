180 DualSolution getStabDualSolAdvanced(const DualSolution &input_duals) {
181 constexpr double EPSILON = 1e-12;
182
183 // Use only the first sizeDual elements from input_duals.
184 DualSolution nodeDuals(input_duals.begin(),
185 input_duals.begin() + sizeDual);
186
187 // If stabilization center or subgradient are not available, return the
188 // input.
189 if (cur_stab_center.empty() || subgradient.empty()) {
190 return input_duals;
191 }
192
193 const size_t n = nodeDuals.size();
194
195 // Step 1: pi_tilde = cur_stab_center + (1 - cur_alpha)*(nodeDuals -
196 // cur_stab_center)
197 DualSolution pi_tilde(n);
198 for (size_t i = 0; i < n; ++i) {
199 pi_tilde[i] =
200 cur_stab_center[i] +
201 (1.0 - cur_alpha) * (nodeDuals[i] - cur_stab_center[i]);
202 }
203
204 // Step 2: Compute norm_in_out = ||nodeDuals - cur_stab_center||
205 double norm_in_out =
206 std::sqrt(std::inner_product(nodeDuals.begin(), nodeDuals.end(),
207 cur_stab_center.begin(), 0.0,
208 std::plus<double>(),
209 [](double a, double b) {
210 double d = a - b;
211 return d * d;
212 }) +
213 EPSILON);
214
215 // Compute pi_g = cur_stab_center +
216 // (subgradient/subgradient_norm)*norm_in_out
217 DualSolution pi_g(n);
218 for (size_t i = 0; i < n; ++i) {
219 pi_g[i] = cur_stab_center[i] +
220 (subgradient[i] / subgradient_norm) * norm_in_out;
221 }
222
223 // Step 3: rho = beta*pi_g + (1 - beta)*nodeDuals
224 DualSolution rho(n);
225 for (size_t i = 0; i < n; ++i) {
226 rho[i] = beta * pi_g[i] + (1.0 - beta) * nodeDuals[i];
227 }
228
229 // Compute norm_tilde_in = ||pi_tilde - cur_stab_center||
230 double norm_tilde_in =
231 std::sqrt(std::inner_product(pi_tilde.begin(), pi_tilde.end(),
232 cur_stab_center.begin(), 0.0,
233 std::plus<double>(),
234 [](double a, double b) {
235 double d = a - b;
236 return d * d;
237 }) +
238 EPSILON);
239
240 // Compute norm_rho_in = ||rho - cur_stab_center||
241 double norm_rho_in = std::sqrt(
242 std::inner_product(rho.begin(), rho.end(), cur_stab_center.begin(),
243 0.0, std::plus<double>(),
244 [](double a, double b) {
245 double d = a - b;
246 return d * d;
247 }) +
248 EPSILON);
249
250 // Step 4: new_duals = cur_stab_center + (norm_tilde_in / norm_rho_in) *
251 // (rho - cur_stab_center)
252 DualSolution new_duals(n);
253 for (size_t i = 0; i < n; ++i) {
254 new_duals[i] =
255 cur_stab_center[i] +
256 (norm_tilde_in / norm_rho_in) * (rho[i] - cur_stab_center[i]);
257 new_duals[i] = std::max(
258 0.0, new_duals[i]); // Project onto the positive orthant.
259 }
260
261 // Store the new dual solutions in smoothing variables.
262 smooth_dual_sol = new_duals;
263 duals_sep = new_duals;
264 return new_duals;
265 }

269 bool dynamic_alpha_schedule(const ModelData &dados) {
270 constexpr double DOT_TOLERANCE = 1e-3;
271 const size_t n = cur_stab_center.size();
272
273 // Compute relative distance: ||smooth_dual_sol - cur_stab_center|| /
274 // |lp_obj|
275 double rel_distance = norm(smooth_dual_sol, cur_stab_center) /
276 (std::abs(lp_obj) + EPSILON);
277 if (rel_distance < NORM_TOLERANCE) {
278 alpha = 0.0;
279 return false;
280 }
281
282 // Compute the difference vector (direction = smooth_dual_sol -
283 // cur_stab_center)
284 std::vector<double> direction(n);
285 for (size_t i = 0; i < n; ++i) {
286 direction[i] = smooth_dual_sol[i] - cur_stab_center[i];
287 }
288
289 // Compute norm of direction vector using inner_product
290 double dir_norm =
291 std::sqrt(std::inner_product(direction.begin(), direction.end(),
292 direction.begin(), 0.0) +
293 EPSILON);
294 if (dir_norm < EPSILON || subgradient_norm < EPSILON) {
295 return false;
296 }
297
298 // Normalize the direction and the subgradient vectors.
299 std::vector<double> normalized_direction(n);
300 std::vector<double> normalized_subgradient(n);
301 for (size_t i = 0; i < n; ++i) {
302 normalized_direction[i] = direction[i] / dir_norm;
303 normalized_subgradient[i] = subgradient[i] / subgradient_norm;
304 }
305
306 // Compute the cosine of the angle between normalized_direction and
307 // normalized_subgradient.
308 double cos_angle = std::inner_product(
309 normalized_direction.begin(), normalized_direction.end(),
310 normalized_subgradient.begin(), 0.0);
311
312 // If the cosine of the angle is very close to zero, then the vectors
313 // are nearly orthogonal.
314 return cos_angle < DOT_TOLERANCE;
315 }

317 void update_subgradient(const ModelData &dados,
318 const DualSolution &nodeDuals,
319 const std::vector<Label *> &best_pricing_cols,
320 bool use_weighted_update = true) {
321 size_t number_of_rows = nodeDuals.size();
322 new_rows.assign(number_of_rows, 0.0);
323
324 // Determine how many columns to check.
325 auto cols_to_check =
326 std::min(numK, static_cast<int>(best_pricing_cols.size()));
327 for (size_t i = 0; i < cols_to_check; ++i) {
328 const auto &col = best_pricing_cols[i];
329
330 // Compute a weight for this column.
331 double weight = 1.0;
332 if (use_weighted_update) {
333 // weight inversely proportional to cost
334 weight = 1.0 / (1.0 + std::abs(col->cost));
335 }
336
337 // Accumulate coverage for nodes that are not the artificial start
338 // or end.
339 for (const auto &node : col->nodes_covered) {
340 if (node > 0 && node != N_SIZE - 1) {
341 new_rows[node - 1] += weight;
342 }
343 }
344 }
345
346 // Compute the raw subgradient based on the constraint senses.
347 subgradient.assign(number_of_rows, 0.0);
348 for (size_t i = 0; i < number_of_rows; ++i) {
349 if (dados.sense[i] == '<') {
350 subgradient[i] = new_rows[i] - dados.b[i];
351 } else if (dados.sense[i] == '>') {
352 subgradient[i] = dados.b[i] - new_rows[i];
353 } else {
354 subgradient[i] = std::abs(dados.b[i] - new_rows[i]);
355 }
356 }
357
358 // Optionally, blend the computed subgradient with the current duals.
359 // This can help smooth the updates over iterations.
360 // for (size_t i = 0; i < number_of_rows; ++i) {
361 // Here we use a simple average. You might use a different
362 // combination or a momentum term.
363 // subgradient[i] = 0.5 * subgradient[i] + 0.5 * nodeDuals[i];
364 // }
365
366 // Recompute the norm of the subgradient.
367 subgradient_norm = norm(subgradient);
368 }

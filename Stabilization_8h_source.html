259 DualSolution getStabDualSolAdvanced(const DualSolution &input_duals) {
260 constexpr double EPSILON = 1e-12;
261
262 DualSolution nodeDuals(input_duals.begin(), input_duals.begin() + sizeDual);
263
264 if (cur_stab_center.empty() || subgradient.empty()) { return getStabDualSol(input_duals); }
265
266 const size_t n = nodeDuals.size();
267
268 // Step 1: π˜ = πin + (1−α)(πout − πin)
269 DualSolution pi_tilde(n);
270 for (size_t i = 0; i < n; ++i) {
271 pi_tilde[i] = cur_stab_center[i] + (1.0 - cur_alpha) * (nodeDuals[i] - cur_stab_center[i]);
272 }
273
274 // Step 2: πg = πin + (gin/||gin||) * ||πout − πin||
275 double norm_in_out = 0.0;
276 for (size_t i = 0; i < n; ++i) {
277 double diff = nodeDuals[i] - cur_stab_center[i];
278 norm_in_out += diff * diff;
279 }
280 norm_in_out = std::sqrt(norm_in_out + EPSILON);
281
282 DualSolution pi_g(n);
283 for (size_t i = 0; i < n; ++i) {
284 pi_g[i] = cur_stab_center[i] + (subgradient[i] / subgradient_norm) * norm_in_out;
285 }
286
287 // Step 3: ρ = βπg + (1−β)πout
288 DualSolution rho(n);
289 for (size_t i = 0; i < n; ++i) { rho[i] = beta * pi_g[i] + (1.0 - beta) * nodeDuals[i]; }
290
291 // Calculate ||π˜ − πin||
292 double norm_tilde_in = 0.0;
293 for (size_t i = 0; i < n; ++i) {
294 double diff = pi_tilde[i] - cur_stab_center[i];
295 norm_tilde_in += diff * diff;
296 }
297 norm_tilde_in = std::sqrt(norm_tilde_in + EPSILON);
298
299 // Calculate ||ρ − πin||
300 double norm_rho_in = 0.0;
301 for (size_t i = 0; i < n; ++i) {
302 double diff = rho[i] - cur_stab_center[i];
303 norm_rho_in += diff * diff;
304 }
305 norm_rho_in = std::sqrt(norm_rho_in + EPSILON);
306
307 // Step 4: πsep = πin + (||π˜ − πin||/||ρ − πin||)(ρ − πin)
308 DualSolution new_duals(n);
309 for (size_t i = 0; i < n; ++i) {
310 new_duals[i] = cur_stab_center[i] + (norm_tilde_in / norm_rho_in) * (rho[i] - cur_stab_center[i]);
311 new_duals[i] = std::max(0.0, new_duals[i]); // Project onto positive orthant
312 }
313
314 smooth_dual_sol = new_duals;
315 duals_sep = new_duals;
316 return new_duals;
317 }

362 bool dynamic_alpha_schedule(const ModelData &dados) {
363 const size_t n = cur_stab_center.size();
364
365 // 1. Check if stabilization is needed at all
366 double relative_distance = norm(smooth_dual_sol, cur_stab_center) / (std::abs(lp_obj) + EPSILON);
367 if (relative_distance < NORM_TOLERANCE) {
368 alpha = 0.0;
369 return false;
370 }
371
372 // 2. Calculate and normalize the in-out direction
373 std::vector<double> direction(n);
374 double direction_norm = 0.0;
375
376 for (size_t i = 0; i < n; i++) {
377 direction[i] = smooth_dual_sol[i] - cur_stab_center[i];
378 direction_norm += direction[i] * direction[i];
379 }
380 direction_norm = std::sqrt(direction_norm + EPSILON);
381
382 if (direction_norm < EPSILON || subgradient_norm < EPSILON) { return false; }
383
384 // 3. Normalize both vectors
385 std::vector<double> normalized_direction(n);
386 std::vector<double> normalized_subgradient(n);
387
388 for (size_t i = 0; i < n; i++) {
389 normalized_direction[i] = direction[i] / direction_norm;
390 normalized_subgradient[i] = subgradient[i] / subgradient_norm;
391 }
392
393 // 4. Calculate cosine of angle between normalized vectors
394 double cos_angle = 0.0;
395 for (size_t i = 0; i < n; i++) { cos_angle += normalized_direction[i] * normalized_subgradient[i]; }
396
397 // 5. Logic for alpha adjustment
398 // If cos_angle < 0: vectors point in opposite directions -> increase alpha
399 // If cos_angle > 0: vectors point in similar directions -> decrease alpha
400 return cos_angle < 0;
401 }

463 void update_stabilization_after_pricing_optim(const ModelData &dados, const DualSolution &input_duals,
464 const double &lag_gap, std::vector<Label *> best_pricing_cols) {
465 // Early exit checks and cycling detection
466 static int no_progress_count = 0;
467 static double last_gap = lag_gap;
468
469 // if (!stabilization_active) return;
470
471 std::vector<double> nodeDuals(input_duals.begin(), input_duals.begin() + sizeDual);
472
473 // Update stabilization parameters if no misprices
474 if (nb_misprices == 0) {
475 update_subgradient(dados, nodeDuals, best_pricing_cols);
476 bool should_increase = dynamic_alpha_schedule(dados);
477
478 constexpr double ALPHA_FACTOR = 0.15;
479 if (should_increase) {
480 alpha = std::min(0.99, alpha + (1.0 - alpha) * ALPHA_FACTOR);
481 } else {
482 alpha = std::max(0.1, alpha * (1.0 - ALPHA_FACTOR));
483 }
484
485 if (!std::isnan(alpha) && !std::isinf(alpha)) { base_alpha = alpha; }
486 }
487
488 // Get current stabilized solution
489 DualSolution stab_sol = smooth_dual_sol;
490
491 if (lag_gap < lag_gap_prev) {
492 // Update multi-point manager
493 mp_manager.updatePool(stab_sol, lag_gap); // Negative because we're maximizing
494
495 stab_center_for_next_iteration = smooth_dual_sol;
496 } else {
497 stab_center_for_next_iteration = cur_stab_center;
498 }
499
500 // Update metrics
501 lag_gap_prev = lag_gap;
502
503 // Safety checks
504 if (std::isnan(alpha) || std::isinf(alpha)) {
505 stabilization_active = false;
506 cleanup();
507 }
508 }

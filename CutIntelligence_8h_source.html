56 static double computeSimilarity(const std::vector<int> &set1,

58 std::vector<int> intersection;
59 std::set_intersection(set1.begin(), set1.end(), set2.begin(),

98 void applyAddNode(CandidateSet &current,
99 const std::vector<std::vector<int>> &node_scores) {
100 std::vector<int> nodes_vec(current.nodes.begin(), current.nodes.end());
101 std::vector<int> potential;
102
103 // if nodes_vec.size() = MAX_RANK, do not add any nodes.
104 if (nodes_vec.size() >= MAX_RANK) return;
105 // Candidates are those in neighbor set but not in nodes.
106 for (int candidate : current.neighbor) {
107 if (current.nodes.find(candidate) == current.nodes.end()) {
108 potential.push_back(candidate);

110 }
111 if (!potential.empty()) {
112 std::shuffle(potential.begin(), potential.end(), rng);
113 // Optionally, one could score potential nodes using node_scores.
114 int chosen = potential.front();
115 nodes_vec.push_back(chosen);
116 current.nodes = ankerl::unordered_dense::set<int>(nodes_vec.begin(),
117 nodes_vec.end());

119 for (const int &node : nodes_vec) {

180 bool acceptMove(double delta, const CandidateSet &current) {
181 if (delta > 0) return true;
182 std::uniform_real_distribution<double> dist(0.0, 1.0);
183 double acceptance_prob = std::exp(delta / temperature);
184 if (history.size() >= 2) {
185 double recent_improvement_rate =
186 static_cast<double>(history.back().improvements) /
187 LocalSearchConfig::SEGMENT_SIZE;
188 double acceptance_rate =
189 std::clamp(0.4 * (1.0 - recent_improvement_rate),
190 LocalSearchConfig::MIN_ACCEPTANCE_RATE,
191 LocalSearchConfig::MAX_ACCEPTANCE_RATE);
192 double avg_recent_violation =
193 std::accumulate(history.begin(), history.end(), 0.0,
194 [](double sum, const SegmentStats &seg) {
195 return sum + seg.avg_violation;
196 }) /
197 history.size();
198 if (current.violation < avg_recent_violation) {
199 return (dist(rng) < acceptance_rate *
200 LocalSearchConfig::IMPROVEMENT_BONUS) &&
201 (delta >
202 -std::abs(current.violation *
203 LocalSearchConfig::MAX_DETERIORATION));
204 }
205 return (dist(rng) < acceptance_rate) &&
206 (delta > -std::abs(current.violation *
207 LocalSearchConfig::MAX_DETERIORATION));
208 }
209 return dist(rng) < acceptance_prob;
210 }

242 void strategicRestart(CandidateSet &current, const CandidateSet &best,
243 std::vector<CandidateSet> &diverse_solutions) {
244 if (iterations_since_improvement > LocalSearchConfig::SEGMENT_SIZE &&
245 !diverse_solutions.empty()) {
246 std::vector<std::pair<double, size_t>> restart_candidates;
247 for (size_t i = 0; i < diverse_solutions.size(); ++i) {
248 const auto &sol = diverse_solutions[i];
249 double quality_score = sol.violation / best.violation;
250 double diversity_score = 0.0;
251 for (const auto &other : diverse_solutions) {
252 if (&other != &sol) {
253 double structural_div =
254 1.0 - computeSimilarity(
255 std::vector<int>(sol.nodes.begin(),
256 sol.nodes.end()),
257 std::vector<int>(other.nodes.begin(),
258 other.nodes.end()));
259 double violation_div =
260 std::abs(other.violation - sol.violation);
261 diversity_score +=
262 0.5 * (structural_div + violation_div);
263 }
264 }
265 restart_candidates.emplace_back(
266 quality_score * LocalSearchConfig::QUALITY_WEIGHT +
267 diversity_score * LocalSearchConfig::DIVERSITY_WEIGHT,
268 i);
269 }
270 pdqsort(restart_candidates.begin(), restart_candidates.end());
271 current = diverse_solutions[restart_candidates.back().second];
272 iterations_since_improvement = 0;
273 temperature = LocalSearchConfig::INITIAL_TEMPERATURE;
274 }

92 void setModel(const std::any &modelData) override {
93 GRBModel *oldModel = model;
94 bool old_owns = owns_model;
95 model = nullptr;
96 owns_model = false;
97
98 try {
99 if (modelData.type() == typeid(GRBModelWrapper)) {
100 const auto &wrapper = std::any_cast<const GRBModelWrapper &>(modelData);
101 if (!wrapper.model) { throw std::runtime_error("Null model in wrapper"); }
102 model = new GRBModel(*wrapper.model);
103 owns_model = true; // We own this model
104 } else if (modelData.type() == typeid(GRBModel *)) {
105 auto grbmodel = std::any_cast<GRBModel *>(modelData);
106 if (!grbmodel) { throw std::runtime_error("Null model pointer"); }
107 model = grbmodel;
108 owns_model = false; // We don't own this model
109 } else {
110 model = oldModel;
111 owns_model = old_owns;
112 throw std::invalid_argument("Invalid model type for Gurobi");
113 }
114
115 if (oldModel != nullptr && oldModel != model && old_owns) { delete oldModel; }
116
117 if (model) { numConstrs = model->get(GRB_IntAttr_NumConstrs); }
118
119 } catch (const std::exception &e) {
120 if (model == nullptr) {
121 model = oldModel;
122 owns_model = old_owns;
123 }

125 }

142 int numConstrs = model->get(GRB_IntAttr_NumConstrs);
143 std::vector<GRBConstr> constraints;

148 std::vector<double> duals(numConstrs);

167 int numVars = model->get(GRB_IntAttr_NumVars);
168
169 // Create vector to store basis status for each variable
170 for (int i = 0; i < numVars; i++) {
171 int basisStatus = vars[i].get(GRB_IntAttr_VBasis);
172 // Check if the variable is basic (status == 0)
173 if (basisStatus == 0) { basicVariableIndices.push_back(i); }
174 }
175
176 delete[] vars; // Clean up allocated memory
177 } catch (GRBException &e) {
178 std::cout << "Error code = " << e.getErrorCode() << std::endl;
179 std::cout << e.getMessage() << std::endl;
180 }

42std::size_t compute_cut_key(const std::array<uint64_t, num_words> &baseSet,
43 const std::vector<int> &perm_num,
44 const int perm_den) {
45 XXH3_state_t *state = XXH3_createState();
46 assert(state != nullptr);
47 XXH3_64bits_reset(state);
48
49 // Hash baseSet (array of uint64_t)
50 XXH3_64bits_update(state, baseSet.data(),
51 baseSet.size() * sizeof(uint64_t));
52 // Hash perm_num
53 XXH3_64bits_update(state, perm_num.data(), perm_num.size() * sizeof(int));
54 // Hash perm_den
55 XXH3_64bits_update(state, &perm_den, sizeof(int));
56
57 std::size_t cut_key = XXH3_64bits_digest(state);
58 XXH3_freeState(state);
59 return cut_key;
60}

357double LimitedMemoryRank1Cuts::computeLimitedMemoryCoefficient(
358 const std::array<uint64_t, num_words> &C,
359 const std::array<uint64_t, num_words> &AM, const SRCPermutation &p,
360 const std::vector<uint16_t> &P, std::vector<int> &order) noexcept {
361 double alpha = 0.0;
362 int S = 0;
363 auto den = p.den;
364
365 for (size_t j = 1; j < P.size() - 1; ++j) {
366 int vj = P[j];
367
368 // Precompute bitshift values for reuse
369 uint64_t am_mask = (1ULL << (vj & 63));
370 uint64_t am_index = vj >> 6;
371
372 // Check if vj is in AM using precomputed values
373 if (!(AM[am_index] & am_mask)) {
374 S = 0; // Reset S if vj is not in AM
375 } else if (C[am_index] & am_mask) {
376 // Get the position of vj in C by counting the set bits up to vj
377 int pos = order[vj];
378
379 S += p.num[pos];
380 if (S >= den) {
381 S -= den;
382 alpha += 1;
383 }
384 }
385 }
386
387 return alpha;
388}

42void CutStorage::addCut(Cut &cut) {
43 // Compute a unique key for the cut based on its base set and probability
44 // data.
45 cut.key = compute_cut_key(cut.baseSet, cut.p.num, cut.p.den);
46
47 // Check if the cut already exists in our map.
48 if (auto it = cutMaster_to_cut_map.find(cut.key);
49 it != cutMaster_to_cut_map.end()) {
50 // The cut already exists; update its id and merge neighbor information.
51 cut.id = it->second;
52 const auto &old_neighbors = cuts[cut.id].neighbors;
53 for (size_t i = 0; i < num_words; ++i) {
54 cut.neighbors[i] |= old_neighbors[i];
55 }
56 // Preserve 'added' flag if the existing cut has been marked as added.
57 if (cuts[cut.id].added) {
58 cut.added = true;
59 cut.updated = true;
60 }
61 cuts[cut.id] = cut;
62 } else {
63 // The cut is new; assign a new id and store it.
64 cut.id = cuts.size();
65 cuts.push_back(cut);
66 cutMaster_to_cut_map[cut.key] = cut.id;
67 }
68
69 // Safely update indexCuts with the cut's id.
70 indexCuts[cut.key].push_back(cut.id);
71}

86 const std::vector<double> &x) {
87 // Determine parallel parameters.
88 const int JOBS = std::thread::hardware_concurrency();
89 auto nC = N_SIZE;
90 const int chunk_size = (tasks.size() + JOBS - 1) / JOBS;
91 const size_t num_chunks = (tasks.size() + chunk_size - 1) / chunk_size;
92
93 // Reserve per-chunk storage to avoid locking.
94 std::vector<std::vector<std::pair<double, Cut>>> chunk_cuts(num_chunks);
95 for (auto &vec : chunk_cuts) {
96 vec.reserve(20); // Reserve an estimate; adjust as needed.
97 }
98
99 // Initialize SRCPermutation 'p' and compute a right-hand-side value.
100 SRCPermutation p;
101 p.num = {1, 1, 1};
102 p.den = 2;
103 double rhs = p.getRHS();
104
105 // === Parallel Processing of Tasks in Chunks ===
106 auto bulk_sender = stdexec::bulk(
107 stdexec::just(), num_chunks,
108 [this, &A, &x, nC, chunk_size, &p, rhs,
109 &chunk_cuts](std::size_t chunk_idx) {
110 // Each chunk gets its own temporary vector.
111 std::vector<std::pair<double, Cut>> local_cuts;
112 size_t start_idx = chunk_idx * chunk_size;
113 size_t end_idx = std::min(start_idx + chunk_size, tasks.size());
114
115 // Process each task in this chunk.
116 for (size_t task_idx = start_idx; task_idx < end_idx; ++task_idx) {
117 // Unpack task parameters (i, j, k).
118 const auto &[i, j, k] = tasks[task_idx];
119
120 // Initialize a vector to count occurrences for each column.
121 std::vector<int> expanded(A.num_cols, 0);
122 std::vector<int> buffer_int(A.num_cols);
123 int buffer_int_n = 0;
124 double lhs = 0.0;
125
126 // Accumulate counts from rows corresponding to i, j, and k.
127 // Compute lhs value based on columns with at least 2 counts.
128 for (int idx = 0; idx < A.num_cols; ++idx) {
129 expanded[idx] = std::min(1, vertex_route_map[i][idx]) +
130 std::min(1, vertex_route_map[j][idx]) +
131 std::min(1, vertex_route_map[k][idx]);
132 if (expanded[idx] >= 2) {
133 lhs += std::floor(expanded[idx] * 0.5) * x[idx];
134 buffer_int[buffer_int_n++] = idx;
135 }
136 }
137
138 // Only proceed if lhs exceeds threshold.
139 if (lhs > rhs + 1e-3) {
140 // Initialize bitmask arrays for cut representation.
141 std::array<uint64_t, num_words> C = {};
142 std::array<uint64_t, num_words> AM = {};
143 std::vector<int> order(N_SIZE, 0);
144 int ordering = 0;
145
146 // Build the base cut indices from the task parameters.
147 std::vector<int> C_index = {i, j, k};
148 for (int node : C_index) {
149 C[node / 64] |= (1ULL << (node % 64));
150 AM[node / 64] |= (1ULL << (node % 64));
151 order[node] = ordering++;
152 }
153 // Get the remaining nodes from buffer_int.
154 std::vector<int> remainingNodes(
155 buffer_int.begin(), buffer_int.begin() + buffer_int_n);
156
157 // Initialize cut coefficients for all paths.
158 std::vector<double> cut_coefficients(allPaths.size(), 0.0);
159
160 // Build a set from the base cut indices.
161 ankerl::unordered_dense::set<int> C_set(C_index.begin(),
162 C_index.end());
163
164 // Update AM based on the positions of nodes in each
165 // consumer path.
166 for (int node : remainingNodes) {
167 auto &consumers = allPaths[node]; // Access consumers.
168 int first = -1, second = -1;
169 // Identify first and second occurrences of elements
170 // from C_set.
171 for (size_t pos = 1; pos < consumers.size() - 1;
172 ++pos) {
173 if (C_set.contains(consumers[pos])) {
174 if (first == -1) {
175 first = pos;
176 } else {
177 second = pos;
178 for (int pos_inner = first + 1;
179 pos_inner < second; ++pos_inner) {
180 AM[consumers[pos_inner] / 64] |=
181 (1ULL
182 << (consumers[pos_inner] % 64));
183 }
184 break;
185 }
186 }
187 }
188 }
189 // set all AM vector to 1
190 // for (int node_idx = 0; node_idx < N_SIZE; ++node_idx) {
191 // AM[node_idx / 64] |= (1ULL << (node_idx % 64));
192 // }
193
194 // Compute the cut coefficients and accumulate lhs.
195 double computed_lhs = 0.0;
196 for (auto z : remainingNodes) {
197 auto &clients = allPaths[z].route;
198 auto ctr = 0;
199 for (auto client : C_index) {
200 if (std::find(clients.begin(), clients.end(),
201 client) != clients.end()) {
202 ctr++;
203 }
204 }
205
206 if (ctr >= 2) {
207 cut_coefficients[z] = 1.0;
208 }
209 // cut_coefficients[z] =
210 // computeLimitedMemoryCoefficient( C, AM, p, clients,
211 // order);
212 computed_lhs += cut_coefficients[z] * x[z];
213 }
214
215 // If violation is positive, record the cut.
216 if (numericutils::gte(computed_lhs, rhs)) {
217 Cut cut(C, AM, cut_coefficients, p);
218 cut.baseSetOrder = order;
219 local_cuts.emplace_back(computed_lhs, cut);
220 }
221 }
222 }
223 // Save the local cuts into the global vector for this chunk.
224 chunk_cuts[chunk_idx] = std::move(local_cuts);
225 });
226
227 // Submit the bulk work and wait for all tasks to complete.
228 auto work = stdexec::starts_on(sched, bulk_sender);
229 stdexec::sync_wait(std::move(work));
230
231 // === Merge results from all chunks ===
232 std::vector<std::pair<double, Cut>> tmp_cuts;
233 for (const auto &vec : chunk_cuts) {
234 tmp_cuts.insert(tmp_cuts.end(), vec.begin(), vec.end());
235 }
236
237 // === Post-Processing: Sort and Add Cuts ===
238 pdqsort(tmp_cuts.begin(), tmp_cuts.end(),
239 [](const auto &a, const auto &b) { return a.first > b.first; });
240
241 // Determine limits for adding cuts.
242 int max_cuts = std::min(static_cast<int>(tmp_cuts.size()), 3);
243 int max_trials = 15;
244 int cuts_added = 0;
245 int cuts_orig_size = cutStorage.size();
246
247 // Add cuts from tmp_cuts, up to limits.
248 for (int i = 0; i < static_cast<int>(tmp_cuts.size()); ++i) {
249 if (cuts_added >= max_cuts || max_trials <= 0) break;
250 auto &cut = tmp_cuts[i].second;
251 cutStorage.addCut(cut);
252 ++cuts_added;
253 }

357double LimitedMemoryRank1Cuts::computeLimitedMemoryCoefficient(
358 const std::array<uint64_t, num_words> &C,
359 const std::array<uint64_t, num_words> &AM, const SRCPermutation &p,
360 const std::vector<uint16_t> &P, std::vector<int> &order) noexcept {
361 double alpha = 0.0;
362 int S = 0;
363 auto den = p.den;
364
365 for (size_t j = 1; j < P.size() - 1; ++j) {
366 const int vj = P[j];
367
368 // Precompute bitshift values for reuse
369 const size_t word = vj >> 6;
370 const uint64_t bit_mask = 1ULL << (vj & 63);
371
372 // Check if vj is in AM using precomputed values
373 if (!(AM[word] & bit_mask)) {
374 S = 0; // Reset S if vj is not in AM
375 } else if (C[word] & bit_mask) {
376 // Get the position of vj in C by counting the set bits up to vj
377 const int pos = order[vj];
378 S += p.num[pos];
379 if (S >= den) {
380 S -= den;
381 alpha += 1.0;
382 }
383 }
384 }
385
386 return alpha;
387}

389std::vector<CandidateSet> LocalSearch::solve(
390 const CandidateSet &initial, const SparseMatrix &A,
391 const std::vector<double> &x,
392 const std::vector<std::vector<int>> &node_scores, int max_iterations) {
393 // Initialize best and current candidate sets.
394 CandidateSet best = initial;
395 CandidateSet current = initial;
396 std::vector<CandidateSet> diverse_solutions;
397 diverse_solutions.reserve(LocalSearchConfig::MAX_DIVERSE_SOLUTIONS + 1);
398
399 // Lambda: Check if a candidate solution is "diverse" enough.
400 auto isSolutionDiverse = [&](const CandidateSet &candidate) -> bool {
401 // Convert candidate nodes into a sorted vector.
402 std::vector<int> cand_nodes(candidate.nodes.begin(),
403 candidate.nodes.end());
404 std::sort(cand_nodes.begin(), cand_nodes.end());
405 for (const auto &sol : diverse_solutions) {
406 std::vector<int> sol_nodes(sol.nodes.begin(), sol.nodes.end());
407 std::sort(sol_nodes.begin(), sol_nodes.end());
408 double similarity = computeSimilarity(cand_nodes, sol_nodes);
409 if (similarity > 0.9 ||
410 std::abs(sol.violation - candidate.violation) <
411 LocalSearchConfig::DIVERSITY_THRESHOLD) {
412 return false;
413 }
414 }
415 return true;
416 };
417
418 // Lambda: Update the list of diverse solutions.
419 auto updateDiverseSolutions = [&](const CandidateSet &candidate) {
420 if (isSolutionDiverse(candidate)) {
421 diverse_solutions.push_back(candidate);
422 if (diverse_solutions.size() >
423 LocalSearchConfig::MAX_DIVERSE_SOLUTIONS) {
424 auto min_it = std::min_element(
425 diverse_solutions.begin(), diverse_solutions.end(),
426 [](const CandidateSet &a, const CandidateSet &b) {
427 return a.violation < b.violation;
428 });
429 diverse_solutions.erase(min_it);
430 }
431 }
432 };
433
434 // Lambda: Update segment statistics after each segment.
435 auto updateSegmentStatistics = [&]() {
436 segment_iterations++;
437 if (segment_iterations == LocalSearchConfig::SEGMENT_SIZE) {
438 history.push_back(current_segment);
439 if (history.size() > 3) {
440 history.pop_front();
441 }
442 current_segment = SegmentStats{};
443 segment_iterations = 0;
444 }
445 };
446
447 // Main iterative search loop.
448 for (int iter = 0; iter < max_iterations; ++iter) {
449 // Backup the current solution.
450 CandidateSet backup = current;
451 OperatorType selected_op = selectOperator();
452
453 // Apply the selected operator.
454 switch (selected_op) {
455 case OperatorType::REMOVE_NODE:
456 applyRemoveNode(current);
457 break;
458 case OperatorType::ADD_NODE:
459 applyAddNode(current, node_scores);
460 break;
461 case OperatorType::REMOVE_NEIGHBORS:
462 applyRemoveNeighbors(current,
463 LocalSearchConfig::MAX_REMOVE_COUNT);
464 break;
465 case OperatorType::ADD_NEIGHBORS:
466 applyAddNeighbors(current, node_scores,
467 LocalSearchConfig::MAX_ADD_COUNT);
468 break;
469 default:
470 // Optionally fallback or combine neighbor updates.
471 applyUpdateNeighbors(current, node_scores);
472 break;
473 }
474
475 // Evaluate the new candidate using parent's violation computation.
476 auto [new_violation, new_perm, rhs] =
477 parent->computeViolationWithBestPerm(current.nodes,
478 current.neighbor, A, x);
479 double delta = new_violation - backup.violation;
480
481 // Accept or reject the move.
482 if (acceptMove(delta, current)) {
483 // Accept move: update candidate solution.
484 current.violation = new_violation;
485 current.perm = std::move(new_perm);
486 current.rhs = std::move(rhs);
487 updateStatistics(selected_op, delta, current, new_violation);
488 if (new_violation > best.violation) {
489 best = current;
490 current_segment.improvements++;
491 iterations_since_improvement = 0;
492 current_segment.best_violation =
493 std::max(current_segment.best_violation, new_violation);
494 } else {
495 iterations_since_improvement++;
496 }
497 updateDiverseSolutions(current);
498 } else {
499 // Revert move and adjust operator score.
500 current = backup;
501 operators[static_cast<size_t>(selected_op)].score =
502 std::max(LocalSearchConfig::MIN_WEIGHT,
503 operators[static_cast<size_t>(selected_op)].score *
504 (1 - LocalSearchConfig::OPERATOR_LEARNING_RATE));
505 }
506
507 updateSegmentStatistics();
508 updateTemperature();
509 strategicRestart(current, best, diverse_solutions);
510 }
511
512 // Final processing: add the best solution and sort the diverse set.
513 diverse_solutions.push_back(std::move(best));
514 pdqsort(diverse_solutions.begin(), diverse_solutions.end(),
515 [](const CandidateSet &a, const CandidateSet &b) {
516 return a.violation > b.violation;
517 });
518
519 return diverse_solutions;
520}

99 static inline bool detectIntegerMasterSolution(const std::vector<double> &solution, const ModelData &matrix,
100 double &integer_obj, double bin_tol = kBinaryTol,
101 double row_tol = kRowFeasTol) {
102 const size_t n = std::min(solution.size(), matrix.c.size());
103 if (n == 0) return false;
104
105 std::vector<double> x_bin(n, 0.0);
106 for (size_t j = 0; j < n; ++j) {
107 const double x = solution[j];
108 if (!std::isfinite(x)) return false;
109 if (x < -bin_tol || x > 1.0 + bin_tol) return false;
110
111 if (std::abs(x) <= bin_tol) {
112 x_bin[j] = 0.0;
113 } else if (std::abs(x - 1.0) <= bin_tol) {
114 x_bin[j] = 1.0;
115 } else {
116 return false;
117 }
118 }
119
120 // Feasibility check on master rows.
121 auto lhs = matrix.A_sparse.multiply(x_bin);
122 if (lhs.size() != matrix.b.size() || lhs.size() != matrix.sense.size()) { return false; }
123 for (size_t r = 0; r < lhs.size(); ++r) {
124 const double ar = lhs[r];
125 const double br = matrix.b[r];
126 const char s = matrix.sense[r];
127 if (s == '<') {
128 if (ar > br + row_tol) return false;
129 } else if (s == '>') {
130 if (ar < br - row_tol) return false;
131 } else { // '='
132 if (std::abs(ar - br) > row_tol) return false;
133 }
134 }
135
136 integer_obj = 0.0;
137 for (size_t j = 0; j < n; ++j) { integer_obj += matrix.c[j] * x_bin[j]; }
138 return std::isfinite(integer_obj);
139 }

266 std::vector<std::pair<int, int>> collectHGSAvoidEdges(const HGS &hgs, int max_edges = 256) const {
267 std::vector<std::pair<int, int>> avoid_edges;
268 avoid_edges.reserve(static_cast<size_t>(max_edges));
269 const int num_nodes = static_cast<int>(nodes.size());
270 for (int i = 1; i < num_nodes && static_cast<int>(avoid_edges.size()) < max_edges; ++i) {
271 for (int j = 1; j < num_nodes && static_cast<int>(avoid_edges.size()) < max_edges; ++j) {
272 if (i == j) continue;
273 auto it = hgs.arc_stats.find(HGS::arcKey(i, j));
274 if (it == hgs.arc_stats.end() || it->second.total_instances <= 0) continue;
275 if (HGS::shouldFixArc(static_cast<int>(it->second.total_instances), it->second.avoidance_count, 0.75)) {
276 avoid_edges.emplace_back(i, j);
277 }
278 }
279 }
280 return avoid_edges;

283 static bool normalizeHGSRoute(const std::vector<int> &raw_route, std::vector<int> &route) {
284 route.clear();
285 if (raw_route.size() < 3 || raw_route.front() != 0) return false;
286
287 route.reserve(raw_route.size());
288 route.push_back(0);
289 std::vector<unsigned char> seen(static_cast<size_t>(N_SIZE), 0);
290 for (size_t idx = 1; idx + 1 < raw_route.size(); ++idx) {
291 const int node = raw_route[idx];
292 if (node <= 0 || node >= N_SIZE - 1) return false;
293 if (seen[static_cast<size_t>(node)]) return false;
294 seen[static_cast<size_t>(node)] = 1;
295 route.push_back(node);
296 }
297 if (route.size() < 2) return false;
298 route.push_back(N_SIZE - 1);
299 return true;

302 double routeTravelCost(const std::vector<int> &route) const {
303 double cost = 0.0;

305 const int from = route[i] == N_SIZE - 1 ? 0 : route[i];

312 double routeReducedCost(BNBNode *node, const std::vector<int> &route, double travel_cost,
313 const std::vector<double> &pricing_duals) {
314 double rc = travel_cost;
315
316 for (int node_id : route) {
317 if (node_id <= 0 || node_id >= N_SIZE - 1) continue;
318 const int row = node_id - 1;
319 if (row >= 0 && row < static_cast<int>(pricing_duals.size())) { rc -= pricing_duals[row]; }
320 }
321 if (N_SIZE - 2 >= 0 && N_SIZE - 2 < static_cast<int>(pricing_duals.size())) { rc -= pricing_duals[N_SIZE - 2]; }
322
323 Path path(route, travel_cost);
324 SRC_MODE_BLOCK({
325 auto &cuts = node->r1c->cutStorage;
326 auto &SRCconstraints = node->SRCconstraints;
327 auto coeffs = cuts.computeLimitedMemoryCoefficients(path.route);
328 const size_t n_terms = std::min(coeffs.size(), SRCconstraints.size());
329 for (size_t i = 0; i < n_terms; ++i) {
330 const int row = SRCconstraints[i]->index();
331 if (row >= 0 && row < static_cast<int>(pricing_duals.size())) { rc -= coeffs[i] * pricing_duals[row]; }
332 }
333 });
334
335 RCC_MODE_BLOCK({
336 auto &rccManager = node->rccManager;
337 auto coeffs = rccManager->computeRCCCoefficients(path.route);
338 const auto &cuts = rccManager->getCuts();
339 const size_t n_terms = std::min(coeffs.size(), cuts.size());
340 for (size_t i = 0; i < n_terms; ++i) {
341 const int row = cuts[i].ctr->index();
342 if (row >= 0 && row < static_cast<int>(pricing_duals.size())) { rc -= coeffs[i] * pricing_duals[row]; }
343 }
344 });
345
346 auto &branching = node->branchingDuals;
347 auto coeffs = branching->computeCoefficients(path.route);
348 const auto &ctrs = branching->getBranchingbaldesCtrs();
349 const size_t n_terms = std::min(coeffs.size(), ctrs.size());
350 for (size_t i = 0; i < n_terms; ++i) {
351 const int row = ctrs[i]->index();
352 if (row >= 0 && row < static_cast<int>(pricing_duals.size())) { rc -= coeffs[i] * pricing_duals[row]; }
353 }
354
355 return rc;
356 }

358 void recordHGSArcStats(HGS &hgs, const std::vector<std::vector<int>> &raw_routes) const {
359 const int num_nodes = static_cast<int>(nodes.size());
360 struct DistIdx {
361 double d;
362 int idx;
363 };
364
365 for (const auto &route : raw_routes) {
366 for (size_t k = 0; k + 1 < route.size(); ++k) { hgs.recordArcUsage(route[k], route[k + 1]); }
367 for (size_t ki = 0; ki < route.size(); ++ki) {
368 const int ci = route[ki];
369 if (ci <= 0 || ci >= num_nodes) continue;
370 const int taken_j = (ki + 1 < route.size()) ? route[ki + 1] : -1;
371
372 std::vector<DistIdx> neighbors;
373 neighbors.reserve(static_cast<size_t>(std::max(0, num_nodes - 2)));
374 for (int j = 1; j < num_nodes; ++j) {
375 if (j != ci) { neighbors.push_back({instance.getcij(ci, j), j}); }
376 }
377 std::sort(neighbors.begin(), neighbors.end(),
378 [](const DistIdx &a, const DistIdx &b) { return a.d < b.d; });
379
380 constexpr int nbGranular = 40;
381 for (int k = 0; k < std::min(nbGranular, static_cast<int>(neighbors.size())); ++k) {
382 const int cj = neighbors[k].idx;
383 if (cj != taken_j) { hgs.recordArcAvoidance(ci, cj); }
384 }
385 }
386 }

389 HGSIntegrationStats runIterativeHGSAndAddColumns(HGS &hgs, BNBNode *node, BucketGraph *bucket_graph,
390 const std::vector<double> &nodeDuals,
391 const std::vector<double> &originDuals, double &integer_solution,
392 int &colAdded) {
393 HGSIntegrationStats stats;
394 const auto hgs_duals = buildHGSDuals(nodeDuals);
395 const auto avoid_edges = collectHGSAvoidEdges(hgs);
396
397 std::vector<std::vector<int>> hgs_routes;
398 try {
399 hgs_routes = hgs.runIterative(instance, hgs_duals, avoid_edges, 500, 2.0);
400 } catch (const std::exception &e) {
401 print_heur("Iterative HGS skipped after exception: {}\n", e.what());
402 return stats;
403 } catch (const std::string &e) {
404 print_heur("Iterative HGS skipped after exception: {}\n", e);
405 return stats;
406 } catch (...) {
407 print_heur("Iterative HGS skipped after unknown exception\n");
408 return stats;
409 }
410
411 stats.raw_routes = static_cast<int>(hgs_routes.size());
412 if (hgs_routes.empty()) return stats;
413 recordHGSArcStats(hgs, hgs_routes);
414
415 const std::vector<double> pricing_duals = originDuals.empty() ? node->getDuals() : originDuals;
416 std::vector<std::unique_ptr<Label>> owned_labels;
417 std::vector<Label *> hgs_labels;
418 owned_labels.reserve(hgs_routes.size());
419 hgs_labels.reserve(hgs_routes.size());
420
421 std::vector<int> route;
422 for (const auto &raw_route : hgs_routes) {
423 if (!normalizeHGSRoute(raw_route, route)) {
424 ++stats.invalid_routes;
425 continue;
426 }
427
428 const double travel_cost = routeTravelCost(route);
429 const double reduced_cost = routeReducedCost(node, route, travel_cost, pricing_duals);
430 if (!hasNegativeReducedCost(reduced_cost)) {
431 ++stats.nonnegative_routes;
432 continue;
433 }
434
435 Path path(route, travel_cost);
436 if (node->pathSet.find(path) != node->pathSet.end()) {
437 ++stats.duplicate_routes;
438 continue;
439 }
440
441 auto label = std::make_unique<Label>();
442 label->real_cost = travel_cost;
443 label->cost = reduced_cost;
444 label->addRoute(route);
445 hgs_labels.push_back(label.get());
446 owned_labels.push_back(std::move(label));
447 ++stats.accepted_routes;
448 }
449
450 if (!hgs_labels.empty()) {
451 double hgs_inner = 0.0;
452 stats.added_columns = addColumn(node, hgs_labels, hgs_inner, false);
453 colAdded += stats.added_columns;
454 }
455
456 double hgs_incumbent = hgs.best_cost;
457 const auto best_hgs_routes = hgs.getBestRoutes();
458 if (!best_hgs_routes.empty()) {
459 double route_cost = 0.0;
460 bool valid_cost = true;
461 for (const auto &raw_route : best_hgs_routes) {
462 if (!normalizeHGSRoute(raw_route, route)) {
463 valid_cost = false;
464 break;
465 }
466 route_cost += routeTravelCost(route);
467 }
468 if (valid_cost) { hgs_incumbent = route_cost; }
469 }
470
471 if (std::isfinite(hgs_incumbent) && hgs_incumbent < integer_solution) {
472 print_heur("HGS improved incumbent: {:.2f} -> {:.2f}\n", integer_solution, hgs_incumbent);
473 integer_solution = hgs_incumbent;
474 node->integer_sol = hgs_incumbent;
475 bucket_graph->incumbent = hgs_incumbent;
476 }
477
478 return stats;
479 }

532

536 ss = bucket_graph->ss;

543 }

558#endif

561 inline int addPath(BNBNode *node, const std::vector<Path> &paths, bool enumerate = false) {
562 // Setup mode-specific objects.
563 SRC_MODE_BLOCK(auto &r1c = node->r1c; auto &cuts = r1c->cutStorage;)
564 RCC_MODE_BLOCK(auto &rccManager = node->rccManager;)
565
566 int numConstrsLocal = node->getIntAttr("NumConstrs");
567 auto &constrs = node->getConstrs();
568 auto &matrix = node->matrix;
569 auto &allPaths = node->paths;
570 auto &SRCconstraints = node->SRCconstraints;
571
572 // Pre-allocate a coefficient vector for each variable.
573 std::vector<double> coluna(numConstrsLocal, 0.0);
574
575 // Containers to collect variable data.
576 std::vector<double> lb, ub, obj;
577 std::vector<MIPColumn> cols;
578 std::vector<std::string> names;
579 std::vector<VarType> vtypes;
580 auto &pathSet = node->pathSet;
581 allPaths.reserve(allPaths.size() + std::min<size_t>(paths.size(), 10));
582 pathSet.reserve(pathSet.size() + std::min<size_t>(paths.size(), 10));
583
584 auto counter = 0;
585 std::vector<int> updatedIndices;
586 updatedIndices.reserve(N_SIZE);
587 // Process each candidate path.
588 for (auto &label : paths) {
589 if (counter >= 10) break;
590
591 // Skip if the route is empty or if (in non-enumeration mode) the
592 // cost is positive.
593 if (label.route.empty()) continue;
594
595 if (!pathSet.emplace(label).second) continue;
596
597 counter++;
598
599 double travel_cost = label.cost;

610 if (coluna[idx] == 0.0) { updatedIndices.push_back(idx); }
611 coluna[idx]++;

615 if (coluna[idx] != 0.0) { col.addTerm(idx, coluna[idx]); }
616 coluna[idx] = 0.0;
617 }
618 // Add the vehicle constraint term.
619 col.addTerm(N_SIZE - 2, 1.0);
620
621 // Step 2: Add SRC limited memory rank-1 cut terms.
622 SRC_MODE_BLOCK({
623 auto vec = cuts.computeLimitedMemoryCoefficients(label.route);
624 if (!vec.empty()) {
625 const size_t n_terms = std::min(vec.size(), SRCconstraints.size());
626 for (size_t i = 0; i < n_terms; ++i) {
627 if (std::abs(vec[i]) > 1e-3) { col.addTerm(SRCconstraints[i]->index(), vec[i]); }
628 }
629 }
630 });
631
632 // Step 3: Add RCC cut terms.
633 RCC_MODE_BLOCK({
634 auto RCCvec = rccManager->computeRCCCoefficients(label.route);
635 if (!RCCvec.empty()) {
636 const auto &rccCuts = rccManager->getCuts();
637 for (int i = 0; i < static_cast<int>(RCCvec.size()); i++) {
638 if (std::abs(RCCvec[i]) > 1e-3) { col.addTerm(rccCuts[i].ctr->index(), RCCvec[i]); }
639 }
640 }
641 });
642
643 // Step 4: Add branching dual terms.
644 auto &branching = node->branchingDuals;
645 auto branchingVec = branching->computeCoefficients(label.route);
646 if (!branchingVec.empty()) {
647 const auto &branchingCtrs = branching->getBranchingbaldesCtrs();
648 for (int i = 0; i < static_cast<int>(branchingVec.size()); i++) {
649 if (std::abs(branchingVec[i]) > 1e-3) { col.addTerm(branchingCtrs[i]->index(), branchingVec[i]); }
650 }
651 }
652
653 // Collect variable data.
654 lb.push_back(0.0);
655 ub.push_back(1.0);
656 obj.push_back(travel_cost);
657 cols.emplace_back(std::move(col));
658 names.emplace_back(std::move(var_name));
659 vtypes.push_back(VarType::Continuous);
660
661 // Save the path in the node's path set to avoid duplicates.
662 SRC_MODE_BLOCK(cuts.syncCoefficientCacheForPath(label, allPaths.size());)
663 node->addPath(label);
664 }
665
666 // Add variables to the MIP model if any were generated.
667 if (!lb.empty()) {
668 node->addVars(lb.data(), ub.data(), obj.data(), vtypes.data(), names.data(), cols.data(), lb.size());
669 node->update();
670 }
671
672 return counter;

820 inline void applyDeluxingReduction(BNBNode *node, const std::vector<double> &dual_solution, double upper_bound,
821 double lower_bound) {
822 if (upper_bound <= lower_bound) { return; }
823 auto &allPaths = node->paths;
824 if (allPaths.empty()) { return; }
825
826 double gap = upper_bound - lower_bound;
827 auto reduced_costs = node->mip.getAllReducedCosts(dual_solution);
828 if (reduced_costs.empty()) { return; }
829
830 std::vector<int> removeIndices;
831 removeIndices.reserve(std::min(allPaths.size(), reduced_costs.size()));
832 for (int i = 0; i < static_cast<int>(allPaths.size()) && i < static_cast<int>(reduced_costs.size()); ++i) {
833 if (reduced_costs[i] > gap) { removeIndices.push_back(i); }
834 }
835 if (removeIndices.empty()) { return; }
836
837 auto &pathSet = node->pathSet;
838 std::sort(removeIndices.begin(), removeIndices.end(), std::greater<int>());
839 SRC_MODE_BLOCK(node->r1c->cutStorage.eraseCoefficientIndices(removeIndices);)
840 for (int idx : removeIndices) {
841 if (idx >= 0 && idx < static_cast<int>(allPaths.size())) {
842 pathSet.erase(allPaths[idx]);
843 allPaths.erase(allPaths.begin() + idx);
844 node->mip.delete_variable(idx);
845 }
846 }
847 node->update();
848 }

854 bool cutHandler(std::shared_ptr<LimitedMemoryRank1Cuts> &r1c, BNBNode *node, std::vector<baldesCtrPtr> &constraints,
855 bool loaded = false) {
856 auto &cuts = r1c->cutStorage;
857 bool changed = false;
858
859 // If there are no cuts, nothing to update.
860 if (cuts.empty()) { return false; }
861 const auto &allPaths = node->paths;
862 // Process each cut.
863 auto ctr = 0;
864
865 std::vector<int> cutsToBeRemoved;
866 for (auto &cut : cuts) {
867 // If the cut was already added and hasn't been updated, skip it.
868 if ((cut.added && !cut.updated) && !loaded) { continue; }
869
870 // Mark that at least one cut has been changed.
871 changed = true;
872 int z;
873 if (!loaded) {
874 z = cut.id;
875 } else {
876 z = ctr++;
877 cut.id = z;
878 }
879
880 auto cutSize = cut.p.num.size();
881 if (loaded && (cutSize == 1)) { // || cutSize == 4 || cutSize == 5)) {
882 fmt::print("Cut size is {}, skipping...\n", cutSize);
883 // // delete the cut
884 cutsToBeRemoved.push_back(cut.id);
885 continue;
886 }
887
888 std::vector<double> coeffs;
889 if (!loaded) {
890 if (cut.coefficients.size() == allPaths.size()) {
891 coeffs = cut.coefficients;
892 } else if ((cut.added && cut.updated) || !cut.hasSparseCoefficients()) {
893 coeffs = r1c->cutStorage.loadCoefficients(allPaths, cut);
894 cut.coefficients = coeffs;
895 }
896 } else {
897 coeffs = r1c->cutStorage.loadCoefficients(allPaths, cut, true);
898 }
899
900 // If the cut was previously added and now updated, simply change
901 // coefficients.
902 if ((cut.added && cut.updated) && !loaded) {
903 cut.updated = false;
904 node->chgCoeff(constraints[z], coeffs);
905 }
906 // Otherwise, build and add the new constraint.
907 else {
908 LinearExpression lhs;
909 if (!loaded && cut.hasSparseCoefficients() && cut.coefficients.size() != allPaths.size()) {
910 for (size_t idx = 0; idx < cut.coefficient_indices.size(); ++idx) {
911 lhs += node->getVar(cut.coefficient_indices[idx]) * cut.coefficient_values[idx];
912 }
913 } else {
914 for (size_t i = 0; i < coeffs.size(); ++i) {
915 if (coeffs[i] != 0) { lhs += node->getVar(i) * coeffs[i]; }
916 }
917 }
918 std::string constraint_name = "SRC_" + std::to_string(z);
919 if (cutSize == 1) {
920 auto ctr = node->addConstr(lhs == 0, constraint_name);
921 constraints.push_back(ctr);
922 } else {
923 auto ctr = node->addConstr(lhs <= cut.p.getRHS(), constraint_name);
924 constraints.push_back(ctr);
925 }
926 }
927
928 // In either case, mark the cut as added and reset its updated flag.
929 if (!loaded && cut.hasSparseCoefficients() && cut.coefficients.size() != allPaths.size()) {
930 cuts.materializeSparseCoefficients(cut, allPaths.size());
931 }
932 cut.added = true;
933 cut.updated = false;
934 }
935
936 // iterate in reverse order to avoid invalidating indices
937 std::sort(cutsToBeRemoved.begin(), cutsToBeRemoved.end(), std::greater<int>());
938 for (int cut_id : cutsToBeRemoved) { r1c->cutStorage.removeCut(cut_id); }
939 return changed;
940 }

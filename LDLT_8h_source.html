26 Permutation(int size) : perm(size) {

31 template <typename T>

100 void analyzePattern_preordered(const CholMatrixType &ap, bool doLDLT) {
101 const StorageIndex size = StorageIndex(ap.rows());
102
103 // Pre-allocate all vectors at once with appropriate sizes
104 m_matrix.resize(size, size);
105 m_parent.resize(size);
106 m_nonZerosPerCol.resize(size);
107 std::vector<StorageIndex> tags(size);
108
109 // Initialize arrays - use memset for better performance on POD types
110 std::memset(m_parent.data(), -1, size * sizeof(StorageIndex));
111 std::memset(m_nonZerosPerCol.data(), 0, size * sizeof(StorageIndex));
112
113 // Compute elimination tree and count nonzeros per column
114 for (StorageIndex k = 0; k < size; ++k) {
115 tags[k] = k; // Mark node k as visited
116
117 // Traverse column k using iterator
118 for (typename CholMatrixType::InnerIterator it(ap, k); it; ++it) {
119 StorageIndex i = it.index();
120 if (i < k) {
121 // Use local variables to reduce memory access
122 StorageIndex current = i;
123 StorageIndex parent;
124
125 // Follow path from i to root of etree
126 while (tags[current] != k) {
127 parent = m_parent[current];
128 if (parent == -1) {
129 m_parent[current] = k;
130 parent = k;
131 }
132 m_nonZerosPerCol[current]++;
133 tags[current] = k;
134 current = parent;
135 }
136 }
137 }
138 }
139
140 // Build column pointers array
141 StorageIndex *Lp = m_matrix.outerIndexPtr();
142 StorageIndex running_total = 0;
143
144 // Use prefix sum for better cache utilization
145 Lp[0] = 0;
146 for (StorageIndex k = 0; k < size; ++k) {
147 running_total += m_nonZerosPerCol[k];
148 Lp[k + 1] = running_total;
149 }
150
151 // Allocate space for non-zeros
152 m_matrix.resizeNonZeros(running_total);
153
154 // Set status flags
155 m_isInitialized = true;
156 m_info = Success;
157 m_analysisIsOk = true;
158 m_factorizationIsOk = false;

182 template <bool DoLDLT, bool NonHermitian>
183 void factorize_preordered(const CholMatrixType &ap) {
184 // Matrix size and pointers into the factorized matrix storage.
185 const StorageIndex size = StorageIndex(ap.rows());
186 const StorageIndex *Lp = m_matrix.outerIndexPtr();
187 StorageIndex *Li = m_matrix.innerIndexPtr();
188 Scalar *Lx = m_matrix.valuePtr();
189
190 // Temporary aligned storage for working vectors.
191 alignas(64) std::vector<Scalar> y(size, Scalar(0));
192 alignas(64) std::vector<StorageIndex> pattern(size, 0);
193 alignas(64) std::vector<StorageIndex> tags(size, 0);
194 m_diag.resize(size);
195
196 // Preallocate thread-local storage for the main factorization loop.
197 const int hardware_threads = std::thread::hardware_concurrency();
198 std::vector<std::vector<Scalar>> y_local_storage(
199 hardware_threads, std::vector<Scalar>(size, Scalar(0)));
200
201 // Precompute the elimination patterns for each column.
202 // (Sequential due to dependencies in the tags vector.)
203 std::vector<std::vector<StorageIndex>> column_patterns(size);
204 for (StorageIndex k = 0; k < size; ++k) {
205 StorageIndex top = size;
206 tags[k] = k;
207 for (typename CholMatrixType::InnerIterator it(ap, k); it; ++it) {
208 StorageIndex i = it.index();
209 if (i <= k) {
210 StorageIndex len = 0;
211 for (; tags[i] != k; i = m_parent[i]) {
212 pattern[len++] = i;
213 tags[i] = k;
214 if (m_parent[i] == -1) break;
215 }
216 while (len > 0) {
217 pattern[--top] = pattern[--len];
218 }
219 }
220 }
221 column_patterns[k].assign(pattern.begin() + top, pattern.end());
222 }
223
224 // Parallel factorization of each column using a bulk sender.
225 auto bulk_sender = stdexec::bulk(
226 stdexec::just(),
227 size, // Each column is processed as an independent task.
228 [this, &y, &tags, Lp, Li, Lx, size, &ap, &y_local_storage,
229 hardware_threads, &column_patterns](std::size_t k) {
230 // Identify thread-local storage using a hash of the thread id.
231 size_t thread_id =
232 std::hash<std::thread::id>{}(std::this_thread::get_id()) %
233 hardware_threads;
234 auto &y_local = y_local_storage[thread_id];
235
236 // Reinitialize the tag for column k and count nonzeros.
237 tags[k] = k;
238 m_nonZerosPerCol[k] = 0;
239
240 // Accumulate contributions from the original matrix.
241 for (typename CholMatrixType::InnerIterator it(ap, k); it;
242 ++it) {
243 StorageIndex i = it.index();
244 if (i <= k) {
245 y_local[i] += getSymm(it.value());
246 }
247 }
248
249 // Compute the shifted diagonal element.
250 DiagonalScalar d =
251 getDiag(y_local[k]) * m_shiftScale + m_shiftOffset;
252 y_local[k] = Scalar(0);
253
254 // Use the precomputed pattern to update off-diagonals.
255 const auto &pattern = column_patterns[k];
256 for (StorageIndex i : pattern) {
257 const Scalar yi = y_local[i];
258 y_local[i] = Scalar(0);
259 if (yi != Scalar(0)) {
260 const Scalar diag_i =
261 getDiag(m_diag[i]); // Cache diagonal value.
262 const Scalar l_ki = yi / diag_i;
263 const StorageIndex p2 = Lp[i] + m_nonZerosPerCol[i];
264
265 // Update entries in the local buffer (SIMD-friendly
266 // inner loop hint).
267 for (StorageIndex p = Lp[i]; p < p2; ++p) {
268 y_local[Li[p]] -= getSymm(Lx[p]) * yi;
269 }
270
271 d -= getDiag(l_ki * getSymm(yi));
272 Li[p2] = k;
273 Lx[p2] = l_ki;
274 ++m_nonZerosPerCol[i];
275 }
276 }
277 m_diag[k] = d;
278 });
279
280 stdexec::sync_wait(stdexec::when_all(bulk_sender));
281
282 // Finalize factorization status.
283 m_info = Success;
284 m_factorizationIsOk = true;
285 }

287 template <int SrcMode_, int DstMode_, bool NonHermitian,
288 typename MatrixType, int DstOrder>
289 void permute_symm_to_symm(
290 const MatrixType &mat,
291 SparseMatrix<typename MatrixType::Scalar, DstOrder,
292 typename MatrixType::StorageIndex> &_dest,
293 const typename MatrixType::StorageIndex *perm) {
294 using StorageIndex = typename MatrixType::StorageIndex;
295 using Scalar = typename MatrixType::Scalar;
296 SparseMatrix<Scalar, DstOrder, StorageIndex> &dest(_dest.derived());
297 using VectorI = Matrix<StorageIndex, Dynamic, 1>;
298 using MatEval = internal::evaluator<MatrixType>;
299 using MatIterator = CustomMatIterator<MatrixType>;
300
301 enum {
302 SrcOrder = MatrixType::IsRowMajor ? RowMajor : ColMajor,
303 StorageOrderMatch = int(SrcOrder) == int(DstOrder),
304 DstMode = DstOrder == RowMajor ? (DstMode_ == Upper ? Lower : Upper)
305 : DstMode_,
306 SrcMode = SrcOrder == RowMajor ? (SrcMode_ == Upper ? Lower : Upper)
307 : SrcMode_
308 };
309
310 MatEval matEval(mat);
311 Index size = mat.rows();
312 VectorI count(size);
313 count.setZero();
314 dest.resize(size, size);
315
316 const bool isLower = int(SrcMode) == int(Lower);
317 const bool isUpper = int(SrcMode) == int(Upper);
318 const bool isDstLower = int(DstMode) == int(Lower);
319
320 // Precompute permutation
321 std::vector<StorageIndex> perm_cache(size);
322 for (StorageIndex j = 0; j < size; ++j) {
323 perm_cache[j] = perm ? perm[j] : j;
324 }
325
326 // First pass: Count the non-zero elements for each column/row
327 for (StorageIndex j = 0; j < size; ++j) {
328 StorageIndex jp = perm_cache[j];
329
330 for (MatIterator it(matEval, j); it; ++it) {
331 StorageIndex i = it.index();
332 if ((isLower && i < j) || (isUpper && i > j)) continue;
333
334 StorageIndex ip = perm_cache[i];
335
336 // Minimize conditional checks
337 StorageIndex min_ip_jp = std::min(ip, jp);
338 StorageIndex max_ip_jp = std::max(ip, jp);
339
340 count[isDstLower ? min_ip_jp : max_ip_jp]++;
341 }
342 }
343
344 // Allocate space based on the counted non-zero entries
345 dest.outerIndexPtr()[0] = 0;
346 for (Index j = 0; j < size; ++j) {
347 dest.outerIndexPtr()[j + 1] = dest.outerIndexPtr()[j] + count[j];
348 }
349 dest.resizeNonZeros(dest.outerIndexPtr()[size]);
350
351 // Reset counts for actual filling
352 for (Index j = 0; j < size; ++j) {
353 count[j] = dest.outerIndexPtr()[j];
354 }
355
356 // Main loop: Populate the destination sparse matrix
357 for (StorageIndex j = 0; j < size; ++j) {
358 StorageIndex jp = perm_cache[j];
359
360 for (MatIterator it(matEval, j); it; ++it) {
361 StorageIndex i = it.index();
362 if ((isLower && i < j) || (isUpper && i > j)) continue;
363
364 StorageIndex ip = perm_cache[i];
365
366 // Minimize conditional checks
367 StorageIndex min_ip_jp = std::min(ip, jp);
368 StorageIndex max_ip_jp = std::max(ip, jp);
369
370 Index k = count[isDstLower ? min_ip_jp : max_ip_jp]++;
371 dest.innerIndexPtr()[k] = isDstLower ? max_ip_jp : min_ip_jp;
372
373 if (!StorageOrderMatch) std::swap(ip, jp);
374 if ((isDstLower && ip < jp) || (!isDstLower && ip > jp)) {
375 dest.valuePtr()[k] =
376 NonHermitian ? it.value() : numext::conj(it.value());
377 } else {
378 dest.valuePtr()[k] = it.value();
379 }
380 }
381 }

384 template <int Mode, bool NonHermitian, typename MatrixType, int DestOrder>
385 void permute_symm_to_fullsymm(
386 const MatrixType &mat,
387 SparseMatrix<typename MatrixType::Scalar, DestOrder,
388 typename MatrixType::StorageIndex> &_dest,
389 const typename MatrixType::StorageIndex *perm) {
390 using StorageIndex = typename MatrixType::StorageIndex;
391 using Scalar = typename MatrixType::Scalar;
392 using Dest = SparseMatrix<Scalar, DestOrder, StorageIndex>;
393 using VectorI = Matrix<StorageIndex, Dynamic, 1>;
394 using MatEval = internal::evaluator<MatrixType>;
395 using MatIterator =
396 typename internal::evaluator<MatrixType>::InnerIterator;
397
398 MatEval matEval(mat);
399 Dest &dest(_dest.derived());
400
401 enum {
402 StorageOrderMatch =
403 int(Dest::IsRowMajor) == int(MatrixType::IsRowMajor)
404 };
405
406 Index size = mat.rows();
407 VectorI count(size);
408 count.setZero();
409 dest.resize(size, size);
410
411 // Precompute permutation
412 std::vector<StorageIndex> perm_cache(size);
413 for (Index j = 0; j < size; ++j) {
414 perm_cache[j] = perm ? perm[j] : j;
415 }
416
417 // First pass: Count non-zeros for each column
418 for (Index j = 0; j < size; ++j) {
419 Index jp = perm_cache[j];
420
421 for (MatIterator it(matEval, j); it; ++it) {
422 Index i = it.index();
423 Index r = it.row();
424 Index c = it.col();
425 Index ip = perm_cache[i];
426
427 if constexpr (Mode == int(Upper | Lower)) {
428 count[StorageOrderMatch ? jp : ip]++;
429 } else if (r == c) {
430 count[ip]++;
431 } else if ((Mode == Lower && r > c) ||
432 (Mode == Upper && r < c)) {
433 count[ip]++;
434 count[jp]++;
435 }
436 }
437 }
438
439 Index nnz = count.sum();
440
441 // Resize for non-zeros and fill outer index
442 dest.resizeNonZeros(nnz);
443 dest.outerIndexPtr()[0] = 0;
444
445 // Unrolling this loop for small loop performance gain
446 for (Index j = 0; j < size; ++j) {
447 dest.outerIndexPtr()[j + 1] = dest.outerIndexPtr()[j] + count[j];
448 }
449
450 // Reset count for actual insertion
451 for (Index j = 0; j < size; ++j) {
452 count[j] = dest.outerIndexPtr()[j];
453 }
454
455 // Second pass: Copy data into destination matrix
456 for (StorageIndex j = 0; j < size; ++j) {
457 for (MatIterator it(matEval, j); it; ++it) {
458 StorageIndex i =
459 internal::convert_index<StorageIndex>(it.index());
460 Index r = it.row();
461 Index c = it.col();
462
463 StorageIndex jp = perm_cache[j];
464 StorageIndex ip = perm_cache[i];
465
466 if constexpr (Mode == int(Upper | Lower)) {
467 Index k = count[StorageOrderMatch ? jp : ip]++;
468 dest.innerIndexPtr()[k] = StorageOrderMatch ? ip : jp;
469 dest.valuePtr()[k] = it.value();
470 } else if (r == c) {
471 Index k = count[ip]++;
472 dest.innerIndexPtr()[k] = ip;
473 dest.valuePtr()[k] = it.value();
474 } else if (((Mode & Lower) == Lower && r > c) ||
475 ((Mode & Upper) == Upper && r < c)) {
476 if (!StorageOrderMatch) std::swap(ip, jp);
477 Index k = count[jp]++;
478 dest.innerIndexPtr()[k] = ip;
479 dest.valuePtr()[k] = it.value();
480 k = count[ip]++;
481 dest.innerIndexPtr()[k] = jp;
482 dest.valuePtr()[k] =
483 (NonHermitian ? it.value() : numext::conj(it.value()));
484 }
485 }
486 }

531
532 Eigen::VectorXd dest;
533
534 // Apply forward permutation
535 if (m_P.size() > 0) {
536 dest.noalias() = m_P * b; // Use noalias to avoid temporary
537 } else {
538 dest = b;
539 }
540
541 // Precompute diagonal norm (if not already done)
542 static Scalar cachedDiagonalNorm = m_diag.norm();
543
544 // Compute the residual norm (only if necessary)
545 const Scalar residualNorm = (matrixL() * dest - b).norm();
546
547 // Compute an adaptive regularization factor
548 const Scalar minRegularization =
549 Scalar(1e-12); // Minimum regularization
550 const Scalar maxRegularization =
551 Scalar(1e-6); // Maximum regularization
552 const Scalar adaptiveRegularization = std::max(
553 minRegularization,
554 std::min(maxRegularization,
555 residualNorm / (cachedDiagonalNorm + residualNorm)));
556
557 // Regular solve path with adaptive regularization
558 solveTriangular<decltype(matrixL()), decltype(dest), Lower>(
559 matrixL(), dest, adaptiveRegularization);
560 dest.array() /= m_diag.array();
561 solveTriangular<decltype(matrixU()), decltype(dest), Upper>(
562 matrixU(), dest, adaptiveRegularization);
563
564 // Apply backward permutation
565 if (m_Pinv.size() > 0) {
566 dest.noalias() = m_Pinv * dest; // Use noalias to avoid temporary
567 }

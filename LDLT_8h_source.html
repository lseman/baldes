148 template <typename MatrixType>
149 Eigen::VectorXd solve(const MatrixType &A, const Eigen::VectorXd &b) {
150 Eigen::VectorXd x = Eigen::VectorXd::Zero(b.size()); // Initialize x = 0
151 Eigen::VectorXd r = b; // Initial residual r = b - A*x, but x=0 so r = b
152 Eigen::VectorXd p = preconditioner->solve(r); // Apply preconditioner to residual
153 Eigen::VectorXd z = p; // z = p for the first iteration
154 Eigen::VectorXd Ap(b.size());
155 double resNorm = r.norm();
156 double initialResNorm = resNorm;
157 double currentTolerance = tolerance; // Start with initial tolerance
158
159 m_info = Eigen::Success; // Initialize info to success at the beginning
160
161 for (int i = 0; i < maxIterations; ++i) {
162 if (resNorm < currentTolerance) {
163 m_info = Eigen::Success; // Successfully converged
164 break;
165 }
166
167 Ap = A * p; // Compute A*p
168 double alpha = r.dot(z) / p.dot(Ap); // Compute alpha
169 x += alpha * p; // Update solution
170 Eigen::VectorXd r_old = r; // Save old residual
171 r -= alpha * Ap; // Update residual
172 resNorm = r.norm();
173
174 // Dynamically adapt tolerance based on residual progress
175 if (resNorm / initialResNorm < 0.1) {
176 currentTolerance *= 0.5; // Tighten tolerance if significant progress is made
177 }
178
179 // Check for divergence or numerical issues
180 if (std::isnan(resNorm) || std::isinf(resNorm)) {
181 m_info = Eigen::NumericalIssue; // Numerical issues (NaN or Inf) detected
182 break;
183 }
184
185 z = preconditioner->solve(r); // Apply preconditioner to new residual
186 double beta = r.dot(z) / r_old.dot(preconditioner->solve(r_old)); // Compute beta
187 p = z + beta * p; // Update search direction
188 }
189
190 // If we reached the maximum number of iterations and haven't converged
191 if (resNorm >= currentTolerance) {
192 m_info = Eigen::NoConvergence; // Did not converge within maxIterations
193 }
194
195 return x; // Return the solution
196 }

201 preconditioner->compute(matrix);
202 }

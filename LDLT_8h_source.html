23class Permutation {
24public:
25 std::vector<int> perm;
26
27 Permutation(int size) : perm(size) {
28 for (int i = 0; i < size; ++i) perm[i] = i;
29 }
30
31 // Apply the permutation to a vector
32 template <typename T>
33 std::vector<T> apply(const std::vector<T> &vec) const {
34 std::vector<T> result(vec.size());
35 for (int i = 0; i < vec.size(); ++i) { result[i] = vec[perm[i]]; }
36 return result;
37 }
38
39 // Invert the permutation
40 Permutation inverse() const {
41 Permutation inv_perm(perm.size());
42 for (int i = 0; i < perm.size(); ++i) { inv_perm.perm[perm[i]] = i; }
43 return inv_perm;
44 }

85 void analyzePattern_preordered(const CholMatrixType &ap, bool doLDLT) {
86 const StorageIndex size = StorageIndex(ap.rows());
87
88 // Pre-allocate all vectors at once with appropriate sizes
89 m_matrix.resize(size, size);
90 m_parent.resize(size);
91 m_nonZerosPerCol.resize(size);
92 std::vector<StorageIndex> tags(size);
93
94 // Initialize arrays - use memset for better performance on POD types
95 std::memset(m_parent.data(), -1, size * sizeof(StorageIndex));
96 std::memset(m_nonZerosPerCol.data(), 0, size * sizeof(StorageIndex));
97
98 // Compute elimination tree and count nonzeros per column
99 for (StorageIndex k = 0; k < size; ++k) {
100 tags[k] = k; // Mark node k as visited
101
102 // Traverse column k using iterator
103 for (typename CholMatrixType::InnerIterator it(ap, k); it; ++it) {
104 StorageIndex i = it.index();
105 if (i < k) {
106 // Use local variables to reduce memory access
107 StorageIndex current = i;
108 StorageIndex parent;
109
110 // Follow path from i to root of etree
111 while (tags[current] != k) {
112 parent = m_parent[current];
113 if (parent == -1) {
114 m_parent[current] = k;
115 parent = k;
116 }
117 m_nonZerosPerCol[current]++;
118 tags[current] = k;
119 current = parent;
120 }
121 }
122 }
123 }
124
125 // Build column pointers array
126 StorageIndex *Lp = m_matrix.outerIndexPtr();
127 StorageIndex running_total = 0;
128
129 // Use prefix sum for better cache utilization
130 Lp[0] = 0;
131 for (StorageIndex k = 0; k < size; ++k) {
132 running_total += m_nonZerosPerCol[k];
133 Lp[k + 1] = running_total;
134 }
135
136 // Allocate space for non-zeros
137 m_matrix.resizeNonZeros(running_total);
138
139 // Set status flags
140 m_isInitialized = true;
141 m_info = Success;
142 m_analysisIsOk = true;
143 m_factorizationIsOk = false;

167 template <bool DoLDLT, bool NonHermitian>
168 void factorize_preordered(const CholMatrixType &ap) {
169 const StorageIndex size = StorageIndex(ap.rows());
170 const StorageIndex *Lp = m_matrix.outerIndexPtr();
171 StorageIndex *Li = m_matrix.innerIndexPtr();
172 Scalar *Lx = m_matrix.valuePtr();
173
174 alignas(64) std::vector<Scalar> y(size, Scalar(0));
175 alignas(64) std::vector<StorageIndex> pattern(size, 0);
176 alignas(64) std::vector<StorageIndex> tags(size, 0);
177
178 m_diag.resize(size);
179 std::atomic<bool> ok{true};
180
181 // Optimize thread count and chunk size
182 const int hardware_threads = std::thread::hardware_concurrency();
183 const int min_tasks_per_thread = 32;
184 const int chunk_size = std::max(size / (hardware_threads * min_tasks_per_thread), StorageIndex(1));
185
186 using simd_vec = std::experimental::native_simd<double>;
187 constexpr int simd_width = simd_vec::size();
188
189 auto bulk_sender = stdexec::bulk(
190 stdexec::just(), (size + chunk_size - 1) / chunk_size,
191 [this, &y, &pattern, &tags, Lp, Li, Lx, size, &ap, &ok, chunk_size, simd_width](std::size_t chunk_idx) {
192 const size_t start_k = chunk_idx * chunk_size;
193 const size_t end_k = std::min(start_k + chunk_size, size_t(size));
194
195 std::vector<Scalar> y_local(size, Scalar(0));
196
197 for (size_t k = start_k; k < end_k && ok; ++k) {
198 StorageIndex top = size;
199 tags[k] = k;
200 m_nonZerosPerCol[k] = 0;
201
202 // Process column k using local buffer
203 for (typename CholMatrixType::InnerIterator it(ap, k); it; ++it) {
204 StorageIndex i = it.index();
205 if (i <= k) {
206 y_local[i] += getSymm(it.value());
207
208 StorageIndex len = 0;
209 for (; tags[i] != k; i = m_parent[i]) {
210 pattern[len++] = i;
211 tags[i] = k;
212 }
213 while (len > 0) pattern[--top] = pattern[--len];
214 }
215 }
216
217 DiagonalScalar d = getDiag(y_local[k]) * m_shiftScale + m_shiftOffset;
218 y_local[k] = Scalar(0);
219
220 for (; top < size; ++top) {
221 const StorageIndex i = pattern[top];
222 const Scalar yi = y_local[i];
223 y_local[i] = Scalar(0);
224
225 const Scalar l_ki = yi / getDiag(m_diag[i]);
226 const StorageIndex p2 = Lp[i] + m_nonZerosPerCol[i];
227
228 // Vectorized sparse update with improved cache efficiency
229 for (StorageIndex p = Lp[i]; p < p2; p += simd_width) {
230 const int remaining = std::min(simd_width, static_cast<int>(p2 - p));
231 if (remaining < simd_width) {
232 // Handle remaining elements sequentially
233 for (int k = 0; k < remaining; ++k) { y_local[Li[p + k]] -= getSymm(Lx[p + k]) * yi; }
234 } else {
235 // Full SIMD processing
236 simd_vec lx_vec;
237 StorageIndex vec_indices[simd_width];
238
239 // Load data
240 for (int k = 0; k < simd_width; ++k) {
241 lx_vec[k] = getSymm(Lx[p + k]);
242 vec_indices[k] = Li[p + k];
243 }
244
245 // Compute and store
246 auto result = lx_vec * simd_vec(yi);
247 for (int k = 0; k < simd_width; ++k) { y_local[vec_indices[k]] -= result[k]; }
248 }
249 }
250
251 d -= getDiag(l_ki * getSymm(yi));
252 Li[p2] = k;
253 Lx[p2] = l_ki;
254 ++m_nonZerosPerCol[i];
255 }
256
257 m_diag[k] = d;
258 if (d == RealScalar(0)) {
259 ok.store(false, std::memory_order_relaxed);
260 break;
261 }
262 }
263 });
264
265 stdexec::sync_wait(stdexec::when_all(bulk_sender));
266 m_info = ok ? Success : NumericalIssue;
267 m_factorizationIsOk = true;
268 }

272 permute_symm_to_symm(const MatrixType &mat,
273 SparseMatrix<typename MatrixType::Scalar, DstOrder, typename MatrixType::StorageIndex> &_dest,
274 const typename MatrixType::StorageIndex *perm) {
275 using StorageIndex = typename MatrixType::StorageIndex;
276 using Scalar = typename MatrixType::Scalar;
277 SparseMatrix<Scalar, DstOrder, StorageIndex> &dest(_dest.derived());
278 using VectorI = Matrix<StorageIndex, Dynamic, 1>;
279 using MatEval = internal::evaluator<MatrixType>;
280 using MatIterator = CustomMatIterator<MatrixType>;
281
282 enum {
283 SrcOrder = MatrixType::IsRowMajor ? RowMajor : ColMajor,
284 StorageOrderMatch = int(SrcOrder) == int(DstOrder),
285 DstMode = DstOrder == RowMajor ? (DstMode_ == Upper ? Lower : Upper) : DstMode_,
286 SrcMode = SrcOrder == RowMajor ? (SrcMode_ == Upper ? Lower : Upper) : SrcMode_
287 };
288
289 MatEval matEval(mat);
290 Index size = mat.rows();
291 VectorI count(size);
292 count.setZero();
293 dest.resize(size, size);
294
295 const bool isLower = int(SrcMode) == int(Lower);
296 const bool isUpper = int(SrcMode) == int(Upper);
297 const bool isDstLower = int(DstMode) == int(Lower);
298
299 // Precompute permutation
300 std::vector<StorageIndex> perm_cache(size);
301 for (StorageIndex j = 0; j < size; ++j) { perm_cache[j] = perm ? perm[j] : j; }
302
303 // First pass: Count the non-zero elements for each column/row
304 for (StorageIndex j = 0; j < size; ++j) {
305 StorageIndex jp = perm_cache[j];
306
307 for (MatIterator it(matEval, j); it; ++it) {
308 StorageIndex i = it.index();
309 if ((isLower && i < j) || (isUpper && i > j)) continue;
310
311 StorageIndex ip = perm_cache[i];
312
313 // Minimize conditional checks
314 StorageIndex min_ip_jp = std::min(ip, jp);
315 StorageIndex max_ip_jp = std::max(ip, jp);
316
317 count[isDstLower ? min_ip_jp : max_ip_jp]++;
318 }
319 }
320
321 // Allocate space based on the counted non-zero entries
322 dest.outerIndexPtr()[0] = 0;
323 for (Index j = 0; j < size; ++j) { dest.outerIndexPtr()[j + 1] = dest.outerIndexPtr()[j] + count[j]; }
324 dest.resizeNonZeros(dest.outerIndexPtr()[size]);
325
326 // Reset counts for actual filling
327 for (Index j = 0; j < size; ++j) { count[j] = dest.outerIndexPtr()[j]; }
328
329 // Main loop: Populate the destination sparse matrix
330 for (StorageIndex j = 0; j < size; ++j) {
331 StorageIndex jp = perm_cache[j];
332 //__builtin_prefetch(&dest.innerIndexPtr()[count[j]], 1, 1); // Prefetch for insertion
333
334 for (MatIterator it(matEval, j); it; ++it) {
335 StorageIndex i = it.index();
336 if ((isLower && i < j) || (isUpper && i > j)) continue;
337
338 StorageIndex ip = perm_cache[i];
339
340 // Minimize conditional checks
341 StorageIndex min_ip_jp = std::min(ip, jp);
342 StorageIndex max_ip_jp = std::max(ip, jp);
343
344 Index k = count[isDstLower ? min_ip_jp : max_ip_jp]++;
345 dest.innerIndexPtr()[k] = isDstLower ? max_ip_jp : min_ip_jp;
346
347 // Prefetch values for efficient memory access
348 //__builtin_prefetch(&dest.valuePtr()[k], 1, 1);
349
350 if (!StorageOrderMatch) std::swap(ip, jp);
351 if ((isDstLower && ip < jp) || (!isDstLower && ip > jp)) {
352 dest.valuePtr()[k] = NonHermitian ? it.value() : numext::conj(it.value());
353 } else {
354 dest.valuePtr()[k] = it.value();
355 }
356 }
357 }

360 template <int Mode, bool NonHermitian, typename MatrixType, int DestOrder>
361 void permute_symm_to_fullsymm(
362 const MatrixType &mat,
363 SparseMatrix<typename MatrixType::Scalar, DestOrder, typename MatrixType::StorageIndex> &_dest,
364 const typename MatrixType::StorageIndex *perm) {
365
366 using StorageIndex = typename MatrixType::StorageIndex;
367 using Scalar = typename MatrixType::Scalar;
368 using Dest = SparseMatrix<Scalar, DestOrder, StorageIndex>;
369 using VectorI = Matrix<StorageIndex, Dynamic, 1>;
370 using MatEval = internal::evaluator<MatrixType>;
371 using MatIterator = typename internal::evaluator<MatrixType>::InnerIterator;
372
373 MatEval matEval(mat);
374 Dest &dest(_dest.derived());
375
376 enum { StorageOrderMatch = int(Dest::IsRowMajor) == int(MatrixType::IsRowMajor) };
377
378 Index size = mat.rows();
379 VectorI count(size);
380 count.setZero();
381 dest.resize(size, size);
382
383 // First pass: Count non-zeros for each column
384 for (Index j = 0; j < size; ++j) {
385 Index jp = perm ? perm[j] : j;
386
387 for (MatIterator it(matEval, j); it; ++it) {
388 Index i = it.index();
389 Index r = it.row();
390 Index c = it.col();
391 Index ip = perm ? perm[i] : i;
392
393 if constexpr (Mode == int(Upper | Lower)) {
394 count[StorageOrderMatch ? jp : ip]++;
395 } else if (r == c) {
396 count[ip]++;
397 } else if ((Mode == Lower && r > c) || (Mode == Upper && r < c)) {
398 count[ip]++;
399 count[jp]++;
400 }
401 }
402 }
403
404 Index nnz = count.sum();
405
406 // Resize for non-zeros and fill outer index
407 dest.resizeNonZeros(nnz);
408 dest.outerIndexPtr()[0] = 0;
409
410 // Unrolling this loop for small loop performance gain
411 for (Index j = 0; j < size; ++j) { dest.outerIndexPtr()[j + 1] = dest.outerIndexPtr()[j] + count[j]; }
412
413 // Reset count for actual insertion
414 for (Index j = 0; j < size; ++j) { count[j] = dest.outerIndexPtr()[j]; }
415
416 // Second pass: Copy data into destination matrix
417 for (StorageIndex j = 0; j < size; ++j) {
418 for (MatIterator it(matEval, j); it; ++it) {
419 StorageIndex i = internal::convert_index<StorageIndex>(it.index());
420 Index r = it.row();
421 Index c = it.col();
422
423 StorageIndex jp = perm ? perm[j] : j;
424 StorageIndex ip = perm ? perm[i] : i;
425
426 if constexpr (Mode == int(Upper | Lower)) {
427 Index k = count[StorageOrderMatch ? jp : ip]++;
428 dest.innerIndexPtr()[k] = StorageOrderMatch ? ip : jp;
429 dest.valuePtr()[k] = it.value();
430 } else if (r == c) {
431 Index k = count[ip]++;
432 dest.innerIndexPtr()[k] = ip;
433 dest.valuePtr()[k] = it.value();
434 } else if (((Mode & Lower) == Lower && r > c) || ((Mode & Upper) == Upper && r < c)) {
435 if (!StorageOrderMatch) std::swap(ip, jp);
436 Index k = count[jp]++;
437 dest.innerIndexPtr()[k] = ip;
438 dest.valuePtr()[k] = it.value();
439 k = count[ip]++;
440 dest.innerIndexPtr()[k] = jp;
441 dest.valuePtr()[k] = (NonHermitian ? it.value() : numext::conj(it.value()));
442 }
443 }
444 }

490 eigen_assert(m_isInitialized && "Decomposition not initialized.");
491
492 Eigen::VectorXd dest;
493
494 // Apply forward permutation (if needed)
495 if (m_P.size() > 0) {
496 dest = m_P * b;
497 } else {
498 dest = b;
499 }
500
501 // Residual-Based Regularization
502 const Scalar residualNorm = (matrixL() * dest - b).norm();
503 const Scalar regularizationFactor = std::min(Scalar(1e-10), residualNorm * Scalar(1e-12));
504
505 // Solve L * y = P * b
506 solveTriangular<decltype(matrixL()), decltype(dest), Lower>(matrixL(), dest, regularizationFactor);
507
508 // Solve D * z = y using Eigen's coefficient-wise operations
509 dest.array() /= m_diag.array();
510
511 // Solve U * x = z
512 solveTriangular<decltype(matrixU()), decltype(dest), Upper>(matrixU(), dest, regularizationFactor);
513
514 // Apply backward permutation (if needed)
515 if (m_Pinv.size() > 0) { dest = m_Pinv * dest; }
516
517 return dest;

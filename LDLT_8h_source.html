30class Permutation {
31public:
32 std::vector<int> perm;
33
34 Permutation(int size) : perm(size) {
35 for (int i = 0; i < size; ++i) perm[i] = i;
36 }
37
38 // Apply the permutation to a vector
39 template <typename T>
40 std::vector<T> apply(const std::vector<T> &vec) const {
41 std::vector<T> result(vec.size());
42 for (int i = 0; i < vec.size(); ++i) { result[i] = vec[perm[i]]; }
43 return result;
44 }
45
46 // Invert the permutation
47 Permutation inverse() const {
48 Permutation inv_perm(perm.size());
49 for (int i = 0; i < perm.size(); ++i) { inv_perm.perm[perm[i]] = i; }
50 return inv_perm;
51 }
52};

91 void analyzePattern_preordered(const CholMatrixType &ap, bool doLDLT) {
92 const StorageIndex size = StorageIndex(ap.rows());
93
94 // Pre-allocate all vectors at once with appropriate sizes
95 m_matrix.resize(size, size);
96 m_parent.resize(size);
97 m_nonZerosPerCol.resize(size);
98 std::vector<StorageIndex> tags(size);
99
100 // Initialize arrays - use memset for better performance on POD types
101 std::memset(m_parent.data(), -1, size * sizeof(StorageIndex));
102 std::memset(m_nonZerosPerCol.data(), 0, size * sizeof(StorageIndex));
103
104 // Compute elimination tree and count nonzeros per column
105 for (StorageIndex k = 0; k < size; ++k) {
106 tags[k] = k; // Mark node k as visited
107
108 // Traverse column k using iterator
109 for (typename CholMatrixType::InnerIterator it(ap, k); it; ++it) {
110 StorageIndex i = it.index();
111 if (i < k) {
112 // Use local variables to reduce memory access
113 StorageIndex current = i;
114 StorageIndex parent;
115
116 // Follow path from i to root of etree
117 while (tags[current] != k) {
118 parent = m_parent[current];
119 if (parent == -1) {
120 m_parent[current] = k;
121 parent = k;
122 }
123 m_nonZerosPerCol[current]++;
124 tags[current] = k;
125 current = parent;
126 }
127 }
128 }
129 }
130
131 // Build column pointers array
132 StorageIndex *Lp = m_matrix.outerIndexPtr();
133 StorageIndex running_total = 0;
134
135 // Use prefix sum for better cache utilization
136 Lp[0] = 0;
137 for (StorageIndex k = 0; k < size; ++k) {
138 running_total += m_nonZerosPerCol[k];
139 Lp[k + 1] = running_total;
140 }
141
142 // Allocate space for non-zeros
143 m_matrix.resizeNonZeros(running_total);
144
145 // Set status flags
146 m_isInitialized = true;
147 m_info = Success;
148 m_analysisIsOk = true;
149 m_factorizationIsOk = false;
150 }

173 template <bool DoLDLT, bool NonHermitian>
174 void factorize_preordered(const CholMatrixType &ap) {
175 // Matrix size and pointers into the factorized matrix storage.
176 const StorageIndex size = StorageIndex(ap.rows());
177 const StorageIndex *Lp = m_matrix.outerIndexPtr();
178 StorageIndex *Li = m_matrix.innerIndexPtr();
179 Scalar *Lx = m_matrix.valuePtr();
180
181 // Temporary aligned storage for working vectors.
182 alignas(64) std::vector<Scalar> y(size, Scalar(0));
183 alignas(64) std::vector<StorageIndex> pattern(size, 0);
184 alignas(64) std::vector<StorageIndex> tags(size, 0);
185 m_diag.resize(size);
186
187 // Preallocate thread-local storage for the main factorization loop.
188 const int hardware_threads = std::thread::hardware_concurrency();
189 std::vector<std::vector<Scalar>> y_local_storage(hardware_threads, std::vector<Scalar>(size, Scalar(0)));
190
191 // Precompute the elimination patterns for each column.
192 // (Sequential due to dependencies in the tags vector.)
193 std::vector<std::vector<StorageIndex>> column_patterns(size);
194 for (StorageIndex k = 0; k < size; ++k) {
195 StorageIndex top = size;
196 tags[k] = k;
197 for (typename CholMatrixType::InnerIterator it(ap, k); it; ++it) {
198 StorageIndex i = it.index();
199 if (i <= k) {
200 StorageIndex len = 0;
201 for (; tags[i] != k; i = m_parent[i]) {
202 pattern[len++] = i;
203 tags[i] = k;
204 if (m_parent[i] == -1) break;
205 }
206 while (len > 0) { pattern[--top] = pattern[--len]; }
207 }
208 }
209 column_patterns[k].assign(pattern.begin() + top, pattern.end());
210 }
211
212 // Parallel factorization of each column using a bulk sender.
213 auto bulk_sender =
214 stdexec::bulk(stdexec::just(),
215 size, // Each column is processed as an independent task.
216 [this, &y, &tags, Lp, Li, Lx, size, &ap, &y_local_storage, hardware_threads,
217 &column_patterns](std::size_t k) {
218 // Identify thread-local storage using a hash of the thread id.
219 size_t thread_id =
220 std::hash<std::thread::id>{}(std::this_thread::get_id()) % hardware_threads;
221 auto &y_local = y_local_storage[thread_id];
222
223 // Reinitialize the tag for column k and count nonzeros.
224 tags[k] = k;
225 m_nonZerosPerCol[k] = 0;
226
227 // Accumulate contributions from the original matrix.
228 for (typename CholMatrixType::InnerIterator it(ap, k); it; ++it) {
229 StorageIndex i = it.index();
230 if (i <= k) { y_local[i] += getSymm(it.value()); }
231 }
232
233 // Compute the shifted diagonal element.
234 DiagonalScalar d = getDiag(y_local[k]) * m_shiftScale + m_shiftOffset;
235 y_local[k] = Scalar(0);
236
237 // Use the precomputed pattern to update off-diagonals.
238 const auto &pattern = column_patterns[k];
239 for (StorageIndex i : pattern) {
240 const Scalar yi = y_local[i];
241 y_local[i] = Scalar(0);
242 if (yi != Scalar(0)) {
243 const Scalar diag_i = getDiag(m_diag[i]); // Cache diagonal value.
244 const Scalar l_ki = yi / diag_i;
245 const StorageIndex p2 = Lp[i] + m_nonZerosPerCol[i];
246
247 // Update entries in the local buffer (SIMD-friendly
248 // inner loop hint).
249 for (StorageIndex p = Lp[i]; p < p2; ++p) {
250 y_local[Li[p]] -= getSymm(Lx[p]) * yi;
251 }
252
253 d -= getDiag(l_ki * getSymm(yi));
254 Li[p2] = k;
255 Lx[p2] = l_ki;
256 ++m_nonZerosPerCol[i];
257 }
258 }
259 m_diag[k] = d;
260 });
261
262 stdexec::sync_wait(stdexec::when_all(bulk_sender));
263
264 // Finalize factorization status.
265 m_info = Success;
266 m_factorizationIsOk = true;
267 }

269 template <int SrcMode_, int DstMode_, bool NonHermitian, typename MatrixType, int DstOrder>
270 void
271 permute_symm_to_symm(const MatrixType &mat,
272 SparseMatrix<typename MatrixType::Scalar, DstOrder, typename MatrixType::StorageIndex> &_dest,
273 const typename MatrixType::StorageIndex *perm) {
274 using StorageIndex = typename MatrixType::StorageIndex;
275 using Scalar = typename MatrixType::Scalar;
276 SparseMatrix<Scalar, DstOrder, StorageIndex> &dest(_dest.derived());
277 using VectorI = Matrix<StorageIndex, Dynamic, 1>;
278 using MatEval = internal::evaluator<MatrixType>;
279 using MatIterator = CustomMatIterator<MatrixType>;
280
281 enum {
282 SrcOrder = MatrixType::IsRowMajor ? RowMajor : ColMajor,
283 StorageOrderMatch = int(SrcOrder) == int(DstOrder),
284 DstMode = DstOrder == RowMajor ? (DstMode_ == Upper ? Lower : Upper) : DstMode_,
285 SrcMode = SrcOrder == RowMajor ? (SrcMode_ == Upper ? Lower : Upper) : SrcMode_
286 };
287
288 MatEval matEval(mat);
289 Index size = mat.rows();
290 VectorI count(size);
291 count.setZero();
292 dest.resize(size, size);
293
294 const bool isLower = int(SrcMode) == int(Lower);
295 const bool isUpper = int(SrcMode) == int(Upper);
296 const bool isDstLower = int(DstMode) == int(Lower);
297
298 // Precompute permutation
299 std::vector<StorageIndex> perm_cache(size);
300 for (StorageIndex j = 0; j < size; ++j) { perm_cache[j] = perm ? perm[j] : j; }
301
302 // First pass: Count the non-zero elements for each column/row
303 for (StorageIndex j = 0; j < size; ++j) {
304 StorageIndex jp = perm_cache[j];
305
306 for (MatIterator it(matEval, j); it; ++it) {
307 StorageIndex i = it.index();
308 if ((isLower && i < j) || (isUpper && i > j)) continue;
309
310 StorageIndex ip = perm_cache[i];
311
312 // Minimize conditional checks
313 StorageIndex min_ip_jp = std::min(ip, jp);
314 StorageIndex max_ip_jp = std::max(ip, jp);
315
316 count[isDstLower ? min_ip_jp : max_ip_jp]++;
317 }
318 }
319
320 // Allocate space based on the counted non-zero entries
321 dest.outerIndexPtr()[0] = 0;
322 for (Index j = 0; j < size; ++j) { dest.outerIndexPtr()[j + 1] = dest.outerIndexPtr()[j] + count[j]; }
323 dest.resizeNonZeros(dest.outerIndexPtr()[size]);
324
325 // Reset counts for actual filling
326 for (Index j = 0; j < size; ++j) { count[j] = dest.outerIndexPtr()[j]; }
327
328 // Main loop: Populate the destination sparse matrix
329 for (StorageIndex j = 0; j < size; ++j) {
330 StorageIndex jp = perm_cache[j];
331
332 for (MatIterator it(matEval, j); it; ++it) {
333 StorageIndex i = it.index();
334 if ((isLower && i < j) || (isUpper && i > j)) continue;
335
336 StorageIndex ip = perm_cache[i];
337
338 // Minimize conditional checks
339 StorageIndex min_ip_jp = std::min(ip, jp);
340 StorageIndex max_ip_jp = std::max(ip, jp);
341
342 Index k = count[isDstLower ? min_ip_jp : max_ip_jp]++;
343 dest.innerIndexPtr()[k] = isDstLower ? max_ip_jp : min_ip_jp;
344
345 if (!StorageOrderMatch) std::swap(ip, jp);
346 if ((isDstLower && ip < jp) || (!isDstLower && ip > jp)) {
347 dest.valuePtr()[k] = NonHermitian ? it.value() : numext::conj(it.value());
348 } else {
349 dest.valuePtr()[k] = it.value();
350 }
351 }
352 }
353 }

355 template <int Mode, bool NonHermitian, typename MatrixType, int DestOrder>
356 void permute_symm_to_fullsymm(
357 const MatrixType &mat,
358 SparseMatrix<typename MatrixType::Scalar, DestOrder, typename MatrixType::StorageIndex> &_dest,
359 const typename MatrixType::StorageIndex *perm) {
360 using StorageIndex = typename MatrixType::StorageIndex;
361 using Scalar = typename MatrixType::Scalar;
362 using Dest = SparseMatrix<Scalar, DestOrder, StorageIndex>;
363 using VectorI = Matrix<StorageIndex, Dynamic, 1>;
364 using MatEval = internal::evaluator<MatrixType>;
365 using MatIterator = typename internal::evaluator<MatrixType>::InnerIterator;
366
367 MatEval matEval(mat);
368 Dest &dest(_dest.derived());
369
370 enum { StorageOrderMatch = int(Dest::IsRowMajor) == int(MatrixType::IsRowMajor) };
371
372 Index size = mat.rows();
373 VectorI count(size);
374 count.setZero();
375 dest.resize(size, size);
376
377 // Precompute permutation
378 std::vector<StorageIndex> perm_cache(size);
379 for (Index j = 0; j < size; ++j) { perm_cache[j] = perm ? perm[j] : j; }
380
381 // First pass: Count non-zeros for each column
382 for (Index j = 0; j < size; ++j) {
383 Index jp = perm_cache[j];
384
385 for (MatIterator it(matEval, j); it; ++it) {
386 Index i = it.index();
387 Index r = it.row();
388 Index c = it.col();
389 Index ip = perm_cache[i];
390
391 if constexpr (Mode == int(Upper | Lower)) {
392 count[StorageOrderMatch ? jp : ip]++;
393 } else if (r == c) {
394 count[ip]++;
395 } else if ((Mode == Lower && r > c) || (Mode == Upper && r < c)) {
396 count[ip]++;
397 count[jp]++;
398 }
399 }
400 }
401
402 Index nnz = count.sum();
403
404 // Resize for non-zeros and fill outer index
405 dest.resizeNonZeros(nnz);
406 dest.outerIndexPtr()[0] = 0;
407
408 // Unrolling this loop for small loop performance gain
409 for (Index j = 0; j < size; ++j) { dest.outerIndexPtr()[j + 1] = dest.outerIndexPtr()[j] + count[j]; }
410
411 // Reset count for actual insertion
412 for (Index j = 0; j < size; ++j) { count[j] = dest.outerIndexPtr()[j]; }
413
414 // Second pass: Copy data into destination matrix
415 for (StorageIndex j = 0; j < size; ++j) {
416 for (MatIterator it(matEval, j); it; ++it) {
417 StorageIndex i = internal::convert_index<StorageIndex>(it.index());
418 Index r = it.row();
419 Index c = it.col();
420
421 StorageIndex jp = perm_cache[j];
422 StorageIndex ip = perm_cache[i];
423
424 if constexpr (Mode == int(Upper | Lower)) {
425 Index k = count[StorageOrderMatch ? jp : ip]++;
426 dest.innerIndexPtr()[k] = StorageOrderMatch ? ip : jp;
427 dest.valuePtr()[k] = it.value();
428 } else if (r == c) {
429 Index k = count[ip]++;
430 dest.innerIndexPtr()[k] = ip;
431 dest.valuePtr()[k] = it.value();
432 } else if (((Mode & Lower) == Lower && r > c) || ((Mode & Upper) == Upper && r < c)) {
433 if (!StorageOrderMatch) std::swap(ip, jp);
434 Index k = count[jp]++;
435 dest.innerIndexPtr()[k] = ip;
436 dest.valuePtr()[k] = it.value();
437 k = count[ip]++;
438 dest.innerIndexPtr()[k] = jp;
439 dest.valuePtr()[k] = (NonHermitian ? it.value() : numext::conj(it.value()));
440 }
441 }
442 }

445 template <bool NonHermitian>
446 void ordering_local(const MatrixType &a, ConstCholMatrixPtr &pmat, CholMatrixType &ap) {
447 const Index size = a.rows();
448 pmat = &ap;
449
450 // Step 1: Permute the input matrix to full symmetric form
451 CholMatrixType C;
452 permute_symm_to_fullsymm<UpLo, NonHermitian>(a, C, nullptr);
453
454 // Step 2: Compute the symmetric pattern A + A^T
455 CholMatrixType symm;
456 internal::ordering_helper_at_plus_a(C, symm);
457
458 // Step 3: Compute the minimum degree ordering
459 internal::minimum_degree_ordering(symm, m_Pinv);
460
461 // Step 4: Compute the inverse permutation
462 m_P = m_Pinv.inverse();
463
464 // Step 5: Permute the original matrix using the computed ordering
465 ap.resize(size, size);
466 permute_symm_to_symm<UpLo, Upper, false>(a, ap, m_P.indices().data());
467 }

483 template <typename Rhs>
484 Eigen::VectorXd solve(const MatrixBase<Rhs> &b) const {
485 eigen_assert(m_isInitialized && "Decomposition not initialized.");
486
487 Eigen::VectorXd dest;
488
489 // Apply forward permutation
490 if (m_P.size() > 0) {
491 dest.noalias() = m_P * b; // Use noalias to avoid temporary
492 } else {
493 dest = b;
494 }
495
496 // Precompute diagonal norm (if not already done)
497 static Scalar cachedDiagonalNorm = m_diag.norm();
498
499 // Compute the residual norm (only if necessary)
500 const Scalar residualNorm = (matrixL() * dest - b).norm();
501
502 // Compute an adaptive regularization factor
503 const Scalar minRegularization = Scalar(1e-12); // Minimum regularization
504 const Scalar maxRegularization = Scalar(1e-6); // Maximum regularization
505 const Scalar adaptiveRegularization = std::max(
506 minRegularization, std::min(maxRegularization, residualNorm / (cachedDiagonalNorm + residualNorm)));
507
508 // Regular solve path with adaptive regularization
509 solveTriangular<decltype(matrixL()), decltype(dest), Lower>(matrixL(), dest, adaptiveRegularization);
510 dest.array() /= m_diag.array();
511 solveTriangular<decltype(matrixU()), decltype(dest), Upper>(matrixU(), dest, adaptiveRegularization);
512
513 // Apply backward permutation
514 if (m_Pinv.size() > 0) {
515 dest.noalias() = m_Pinv * dest; // Use noalias to avoid temporary
516 }
517
518 return dest;
519 }

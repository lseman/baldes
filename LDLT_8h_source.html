665 StorageIndex pattern_size = 0;
666
667 // First, collect all row indices for this supernode
668 for (StorageIndex j = 0; j < snode_size; ++j) {
669 StorageIndex col = supernode_cols[j];
670 if (col >= size) {
671 std::cout << "Invalid column index: " << col << std::endl;
672 continue;
673 }
674
675 for (typename CholMatrixType::InnerIterator it(m_matrix, col); it; ++it) {
676 StorageIndex row = it.row();
677 if (row >= first_col && flag[row] == -1) {
678 flag[row] = pattern_size;
679 pattern.push_back(row);
680 row_indices[pattern_size] = row;
681 pattern_size++;
682 }
683 }
684 }
685
686 // std::cout << "Supernode " << snode << " pattern size: " << pattern_size << std::endl;
687
688 // Allocate and initialize dense block
689 dense_block.resize(pattern_size * snode_size);
690 std::fill(dense_block.begin(), dense_block.end(), Scalar(0));
691
692 // Copy data to dense block with bounds checking
693 for (StorageIndex j = 0; j < snode_size; ++j) {
694 StorageIndex col = supernode_cols[j];
695 // std::cout << "Copying column " << j << "/" << snode_size << std::endl;
696
697 for (typename CholMatrixType::InnerIterator it(m_matrix, col); it; ++it) {
698 StorageIndex row = it.row();
699 StorageIndex row_pos = flag[row];
700
701 // std::cout << "Copying row " << row << "/" << row_pos << std::endl;
702
703 if (row_pos >= 0 && row_pos < pattern_size) {
704 dense_block[j * pattern_size + row_pos] = it.value();
705 }
706 }
707 }
708
709 // Perform LDLT factorization on dense block
710 for (StorageIndex k = 0; k < snode_size; ++k) {
711 StorageIndex global_k = supernode_cols[k];
712 if (k >= pattern_size) continue;
713
714 // Get and validate diagonal element
715 Scalar d = dense_block[k * pattern_size + k];
716 if (std::abs(d) < std::numeric_limits<RealScalar>::epsilon()) {
717 d = std::numeric_limits<RealScalar>::epsilon();
718 }
719
720 // Update diagonal vector
721 if (global_k < m_diag.size()) {
722 m_diag[global_k] = d;
723 } else {
724 // std::cout << "Invalid diagonal index: " << global_k << std::endl;
725 continue;
726 }
727
728 // Update L factors
729 for (StorageIndex i = k + 1; i < pattern_size; ++i) {
730 StorageIndex idx = k * pattern_size + i;
731 if (idx < dense_block.size()) { dense_block[idx] /= d; }
732 }
733
734 // Update remaining block
735 for (StorageIndex j = k + 1; j < snode_size && j < pattern_size; ++j) {
736 Scalar multiplier = dense_block[j * pattern_size + k];
737 for (StorageIndex i = j; i < pattern_size; ++i) {
738 StorageIndex update_idx = k * pattern_size + i;
739 StorageIndex target_idx = j * pattern_size + i;
740 if (update_idx < dense_block.size() && target_idx < dense_block.size()) {
741 dense_block[target_idx] -= multiplier * numext::conj(dense_block[update_idx]);
742 }
743 }
744 }
745 }
746
747 // Copy results back to sparse matrix
748 for (StorageIndex j = 0; j < snode_size; ++j) {
749 StorageIndex col = supernode_cols[j];
750 if (col >= size) continue;
751
752 StorageIndex col_start = m_matrix.outerIndexPtr()[col];
753 StorageIndex nnz = 0;
754
755 // Only copy elements below or on the diagonal
756 for (StorageIndex i = 0; i < pattern_size; ++i) {
757 StorageIndex row = row_indices[i];
758 if (row >= col) {
759 StorageIndex pos = j * pattern_size + i;
760 if (pos < dense_block.size() && col_start + nnz < m_matrix.nonZeros() &&
761 std::abs(dense_block[pos]) > std::numeric_limits<RealScalar>::epsilon()) {
762
763 m_matrix.innerIndexPtr()[col_start + nnz] = row;
764 m_matrix.valuePtr()[col_start + nnz] = dense_block[pos];
765 ++nnz;
766 }
767 }
768 }
769
770 // Update nonzeros count
771 if (col < m_nonZerosPerCol.size()) { m_nonZerosPerCol[col] = nnz; }
772 }
773
774 // Reset flags
775 for (StorageIndex i = 0; i < pattern_size; ++i) { flag[row_indices[i]] = -1; }
776 }
777 }

886 StorageIndex *Li = m_matrix.innerIndexPtr();

891 alignas(64) std::vector<StorageIndex> tags(size, 0);

893 if (m_num_supernodes > 0) {
894 fmt::print("Factorizing supernodal\n");
895 factorize_supernodal();

897 fmt::print("Factorized supernodal\n");
898 } else {
899 m_diag.resize(size);
900 std::atomic<bool> ok{true};
901
902 // Optimize thread count and chunk size
903 const int hardware_threads = std::thread::hardware_concurrency();
904 const int min_tasks_per_thread = 32;
905 const int chunk_size = std::max(size / (hardware_threads * min_tasks_per_thread), StorageIndex(1));
906
907 using simd_vec = std::experimental::native_simd<double>;
908 constexpr int simd_width = simd_vec::size();
909
910 auto bulk_sender = stdexec::bulk(
911 stdexec::just(), (size + chunk_size - 1) / chunk_size,
912 [this, &y, &pattern, &tags, Lp, Li, Lx, size, &ap, &ok, chunk_size, simd_width](std::size_t chunk_idx) {
913 const size_t start_k = chunk_idx * chunk_size;
914 const size_t end_k = std::min(start_k + chunk_size, size_t(size));
915
916 std::vector<Scalar> y_local(size, Scalar(0));
917
918 for (size_t k = start_k; k < end_k && ok; ++k) {
919 StorageIndex top = size;
920 tags[k] = k;
921 m_nonZerosPerCol[k] = 0;
922
923 // Process column k using local buffer
924 for (typename CholMatrixType::InnerIterator it(ap, k); it; ++it) {
925 StorageIndex i = it.index();
926 if (i <= k) {
927 y_local[i] += getSymm(it.value());
928
929 StorageIndex len = 0;
930 for (; tags[i] != k; i = m_parent[i]) {
931 pattern[len++] = i;
932 tags[i] = k;
933 }
934 while (len > 0) pattern[--top] = pattern[--len];
935 }
936 }
937
938 DiagonalScalar d = getDiag(y_local[k]) * m_shiftScale + m_shiftOffset;
939 y_local[k] = Scalar(0);
940
941 for (; top < size; ++top) {
942 const StorageIndex i = pattern[top];
943 const Scalar yi = y_local[i];
944 y_local[i] = Scalar(0);
945
946 const Scalar l_ki = yi / getDiag(m_diag[i]);
947 const StorageIndex p2 = Lp[i] + m_nonZerosPerCol[i];
948
949 // Vectorized sparse update with improved cache efficiency
950 for (StorageIndex p = Lp[i]; p < p2; p += simd_width) {
951 const int remaining = std::min(simd_width, static_cast<int>(p2 - p));
952 if (remaining < simd_width) {
953 // Handle remaining elements sequentially
954 for (int k = 0; k < remaining; ++k) {
955 y_local[Li[p + k]] -= getSymm(Lx[p + k]) * yi;
956 }
957 } else {
958 // Full SIMD processing
959 simd_vec lx_vec;
960 StorageIndex vec_indices[simd_width];
961
962 // Load data
963 for (int k = 0; k < simd_width; ++k) {
964 lx_vec[k] = getSymm(Lx[p + k]);
965 vec_indices[k] = Li[p + k];
966 }
967
968 // Compute and store
969 auto result = lx_vec * simd_vec(yi);
970 for (int k = 0; k < simd_width; ++k) { y_local[vec_indices[k]] -= result[k]; }
971 }
972 }
973
974 d -= getDiag(l_ki * getSymm(yi));
975 Li[p2] = k;
976 Lx[p2] = l_ki;
977 ++m_nonZerosPerCol[i];
978 }
979
980 m_diag[k] = d;
981 if (d == RealScalar(0)) {
982 ok.store(false, std::memory_order_relaxed);
983 break;
984 }
985 }
986 });
987
988 stdexec::sync_wait(stdexec::when_all(bulk_sender));
989 m_info = ok ? Success : NumericalIssue;
990 }
991 m_factorizationIsOk = true;

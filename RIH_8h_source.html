40 static std::vector<int> order_crossover(const std::vector<int> &parent1, const std::vector<int> &parent2, int i,
41 int j) {
42 int n = parent1.size();
43 if (i < 0 || j >= n || i >= j) { throw std::invalid_argument("Invalid crossover points."); }
44
45 // Initialize offspring with placeholders (-1 indicates empty)
46 std::vector<int> offspring(n, -1);
47
48 // Copy segment from parent1
49 std::copy(parent1.begin() + i, parent1.begin() + j + 1, offspring.begin() + i);
50
51 // Fill remaining positions with elements from parent2
52 auto it = parent2.begin();
53 for (int k = 0; k < n; ++k) {
54 if (offspring[k] == -1) { // Empty slot in offspring
55 // Skip elements already present in the offspring
56 while (std::find(offspring.begin(), offspring.end(), *it) != offspring.end()) { ++it; }
57 offspring[k] = *it;
58 ++it;
59 }
60 }
61
62 return offspring;
63 }

65 std::pair<std::vector<uint16_t>, std::vector<uint16_t>>
66 srex_crossover(const std::vector<uint16_t> &parent1, const std::vector<uint16_t> &parent2, int i, int j) {
67 int n = parent1.size();
68 if (i < 0 || j >= n || i >= j) { return {parent1, parent2}; }
69
70 // Copy subroutes
71 std::vector<uint16_t> segment1(parent1.begin() + i, parent1.begin() + j + 1);
72 std::vector<uint16_t> segment2(parent2.begin() + i, parent2.begin() + j + 1);
73
74 // Create offspring by swapping the segments
75 std::vector<uint16_t> offspring1, offspring2;
76
77 // Insert segment2 into parent1 and reconstruct
78 for (int node : parent1) {
79 if (std::find(segment2.begin(), segment2.end(), node) == segment2.end()) {
80 offspring1.push_back(node); // Add nodes not in the swapped segment
81 }
82 }
83 offspring1.insert(offspring1.begin() + i, segment2.begin(), segment2.end());
84
85 // Insert segment1 into parent2 and reconstruct
86 for (int node : parent2) {
87 if (std::find(segment1.begin(), segment1.end(), node) == segment1.end()) {
88 offspring2.push_back(node); // Add nodes not in the swapped segment
89 }
90 }
91 offspring2.insert(offspring2.begin() + i, segment1.begin(), segment1.end());
92
93 return {std::move(offspring1), std::move(offspring2)};
94 }

316 std::pair<std::vector<uint16_t>, std::vector<uint16_t>> extended_swap_star_fun(const std::vector<uint16_t> &route1,
317 const std::vector<uint16_t> &route2,
318 int i, int j, int chain_length = 2) {
319 // Ensure indices are within bounds and do not involve depot nodes
320 if (i <= 0 || i + chain_length >= static_cast<int>(route1.size()) || j <= 0 ||
321 j + chain_length >= static_cast<int>(route2.size())) {
322 return {route1, route2};
323 }
324
325 // Create mutable copies of the routes
326 std::vector<uint16_t> new_route1(route1.begin(), route1.end());
327 std::vector<uint16_t> new_route2(route2.begin(), route2.end());
328
329 // Extract and erase chains in-place
330 auto chain1_start = new_route1.begin() + i;
331 auto chain1_end = chain1_start + chain_length;
332 std::vector<uint16_t> chain1(std::make_move_iterator(chain1_start), std::make_move_iterator(chain1_end));
333 new_route1.erase(chain1_start, chain1_end);
334
335 auto chain2_start = new_route2.begin() + j;
336 auto chain2_end = chain2_start + chain_length;
337 std::vector<uint16_t> chain2(std::make_move_iterator(chain2_start), std::make_move_iterator(chain2_end));
338 new_route2.erase(chain2_start, chain2_end);
339
340 // Find the best insertion positions
341 int best_pos_route1 = find_best_insertion_position(new_route1, chain2);
342 int best_pos_route2 = find_best_insertion_position(new_route2, chain1);
343
344 // Insert the chains at the best positions
345 new_route1.insert(new_route1.begin() + best_pos_route1, std::make_move_iterator(chain2.begin()),
346 std::make_move_iterator(chain2.end()));
347 new_route2.insert(new_route2.begin() + best_pos_route2, std::make_move_iterator(chain1.begin()),
348 std::make_move_iterator(chain1.end()));
349
350 return {std::move(new_route1), std::move(new_route2)};
351 }

361 int find_best_insertion_position(const std::vector<uint16_t> &route, const std::vector<uint16_t> &chain) {
362 int best_pos = 1;
363 double best_cost = std::numeric_limits<double>::max();
364
365 // Precompute the cost of the original route
366 const double original_cost = this->compute_cost(route).second;
367
368 // Iterate over valid insertion positions
369 for (int pos = 1; pos < static_cast<int>(route.size()); ++pos) {
370 // Incrementally compute the new cost for inserting the chain at
371 // position `pos`
372 const double incremental_cost = compute_insertion_cost(route, chain, pos, original_cost);
373
374 // Track the best position based on the cost
375 if (incremental_cost < best_cost) {
376 best_cost = incremental_cost;
377 best_pos = pos;
378 }
379 }
380 return best_pos;
381 }

383 double compute_insertion_cost(const std::vector<uint16_t> &route, const std::vector<uint16_t> &chain, int pos,
384 double original_cost) {
385 // Simulate the insertion and calculate the cost incrementally
386 std::vector<uint16_t> new_route(route.size() + chain.size());
387 std::copy(route.begin(), route.begin() + pos, new_route.begin());
388 std::copy(chain.begin(), chain.end(), new_route.begin() + pos);
389 std::copy(route.begin() + pos, route.end(), new_route.begin() + pos + chain.size());
390
391 return this->compute_cost(new_route).second;
392 }

458 void update_operator_weights(double decay_factor = 0.9, double reward_factor = 0.1, double min_weight = 0.01) {
459 // First, compute total improvement and total successes.
460 double total_improvement = 0.0;
461 int total_successes = 0;
462 const size_t n = operator_weights.size();
463 for (size_t i = 0; i < n; ++i) {
464 total_improvement += operator_improvements[i];
465 total_successes += operator_success_count[i];
466 }
467
468 // Only update weights if we have any recorded improvement or success.
469 if (total_improvement > 0 || total_successes > 0) {
470 const double inv_total_improvement = (total_improvement > 0) ? (1.0 / total_improvement) : 0.0;
471 const double inv_total_successes = (total_successes > 0) ? (1.0 / total_successes) : 0.0;
472
473 // Update each operator weight and accumulate the total weight.
474 double total_weight = 0.0;
475 for (size_t i = 0; i < n; ++i) {
476 double normalized_improvement = operator_improvements[i] * inv_total_improvement;
477 double success_rate = operator_success_count[i] * inv_total_successes;
478 double new_weight =
479 decay_factor * operator_weights[i] + reward_factor * (normalized_improvement + success_rate);
480 // Ensure a minimum weight.
481 operator_weights[i] = (new_weight < min_weight) ? min_weight : new_weight;
482 total_weight += operator_weights[i];
483 }
484
485 // Normalize the weights so that they sum to 1.
486 if (total_weight > 0.0) {
487 for (size_t i = 0; i < n; ++i) { operator_weights[i] /= total_weight; }
488 }
489 }
490
491 // Reset improvements and success counts for the next iteration.
492 for (size_t i = 0; i < n; ++i) {
493 operator_improvements[i] = 0.0;
494 operator_success_count[i] = 0;
495 }
496 }

504 std::vector<Label *> perturbation(const std::vector<Label *> &paths) {
505 // Copy the input paths as the baseline best solution.
506 std::vector<Label *> best = paths;
507 std::vector<Label *> best_new;
508
509 bool is_stuck = false;
510 // Initialize best costs for each path to a large value.
511 std::vector<double> best_costs(paths.size(), std::numeric_limits<double>::max());
512
513 // Reserve an approximate number of tasks if possible.
514 std::vector<std::tuple<const Label *, const Label *, size_t, size_t>> tasks;
515 tasks.reserve(paths.size() * paths.size() * 4); // adjust estimate as needed
516
517 for (const Label *label_i : best) {
518 for (const Label *label_j : best) {
519 if (label_i == label_j) continue;
520 const auto &route_i = label_i->getRoute();
521 const auto &route_j = label_j->getRoute();
522 // Skip if routes are too short.
523 if (route_i.size() < 3 || route_j.size() < 3) continue;
524 // Generate tasks for valid internal positions.
525 for (size_t k = 1; k < route_i.size() - 1; ++k) {
526 for (size_t l = 1; l < route_j.size() - 1; ++l) { tasks.emplace_back(label_i, label_j, k, l); }
527 }
528 }
529 }
530
531 // We'll use per-chunk local storage to avoid locking per improvement.
532 const size_t num_tasks = tasks.size();
533 const size_t chunk_size = 1; // Adjust chunk size based on experiments.
534 const size_t num_chunks = (num_tasks + chunk_size - 1) / chunk_size;
535
536 // Prepare a vector of per-chunk improved labels.
537 std::vector<std::vector<Label *>> local_best_new(num_chunks);
538
539 // Optionally, you can also create per-chunk accumulators for operator
540 // metrics. For brevity, we'll leave operator metric updates using
541 // locks.
542
543 auto bulk_sender = stdexec::bulk(
544 stdexec::just(), num_chunks, [&tasks, chunk_size, &local_best_new, this](std::size_t chunk_idx) {
545 const size_t start_idx = chunk_idx * chunk_size;
546 const size_t end_idx = std::min(start_idx + chunk_size, tasks.size());
547
548 // Local vector to accumulate improved labels.
549 std::vector<Label *> local_improvements;
550 local_improvements.reserve(10);
551
552 for (size_t task_idx = start_idx; task_idx < end_idx; ++task_idx) {
553 const auto &[label_i, label_j, k, l] = tasks[task_idx];
554 const auto &route_i = label_i->getRoute();
555 const auto &route_j = label_j->getRoute();
556
557 if (route_i.size() < 3 || route_j.size() < 3) continue;
558
559 // Select operator based on its weight.
560 int op_idx = select_operator(rng);
561
562 // Apply the chosen operator.
563 auto [new_route_i, new_route_j] = apply_operator(op_idx, route_i, route_j, k, l);
564 if (new_route_i.empty() || new_route_j.empty()) continue;
565
566 // Compute new route costs.
567 auto cost_i = compute_cost(new_route_i);
568 auto cost_j = compute_cost(new_route_j);
569 Path new_path_i{new_route_i, cost_i.first};
570 Path new_path_j{new_route_j, cost_j.first};
571
572 // Lambda to update a label if an improvement is found.
573 auto add_improved_label = [&](const Path &path, const std::pair<double, double> &cost,
574 const Label *orig_label, int op_idx) {
575 double new_cost = cost.second;
576 double current_cost = orig_label->cost;
577 if (new_cost < current_cost - 1e-3) {
578 // Update operator performance metrics.
579 {
580 std::lock_guard<std::mutex> lock(operator_mutex);
581 operator_improvements[op_idx] += current_cost - new_cost;
582 operator_success_count[op_idx] += 1;
583 }
584 // Create a new improved label.
585 auto best_new_label = new Label();
586 best_new_label->addRoute(path.route);
587 best_new_label->real_cost = cost.first;
588 best_new_label->cost = new_cost;
589 local_improvements.push_back(best_new_label);
590 }
591 };
592
593 if (is_feasible(new_path_i)) add_improved_label(new_path_i, cost_i, label_i, op_idx);
594 if (is_feasible(new_path_j)) add_improved_label(new_path_j, cost_j, label_j, op_idx);
595 }
596 // Save the local improvements for this chunk.
597 local_best_new[chunk_idx] = std::move(local_improvements);
598 });
599
600 // Submit work and wait for completion.
601 auto work = stdexec::starts_on(sched, bulk_sender);
602 stdexec::sync_wait(std::move(work));
603
604 // Merge local improvements into the global best_new vector.
605 for (auto &vec : local_best_new) { best_new.insert(best_new.end(), vec.begin(), vec.end()); }
606
607 // Update the operator weights based on the collected performance.

632 // Feasibility check and cost computation functions
633 bool is_feasible(const Path &route) const {
634 double current_time = 0.0;
635 double current_capacity = 0.0;
636 const double total_capacity = instance.q;
637 const size_t route_size = route.route.size();
638
639 // Ensure the route starts at depot 0 and ends at depot N_SIZE - 1.
640 if (route_size < 2 || route.route.front() != 0 || route.route.back() != N_SIZE - 1) { return false; }
641
642 // Prepare a thread-local bitset for tracking visited nodes.
643 const size_t n_segments = (N_SIZE >> 6) + 1; // Same as N_SIZE/64 + 1.
644 static thread_local std::vector<uint64_t> visited(n_segments, 0);
645 std::fill(visited.begin(), visited.end(), 0);
646
647 // Cache pointer to the route data to avoid repeated bounds checking.
648 const uint16_t *const rdata = route.route.data();
649
650 // Iterate through the route.
651 for (size_t i = 0; i < route_size - 1; ++i) {
652 const int source_id = rdata[i];
653 const int target_id = rdata[i + 1];
654
655 // Determine the bitset segment and position for the source node.
656 const size_t segment = source_id >> 6;
657 const size_t bit_position = source_id & 63;
658
659 // Check if the source node has already been visited.
660 if (visited[segment] & (1ULL << bit_position)) { return false; }
661 // Mark the source node as visited.
662 visited[segment] |= (1ULL << bit_position);
663
664 // Cache node references.
665 const auto &source = nodes[source_id];
666 const auto &target = nodes[target_id];
667
668 // Compute travel time and arrival time.
669 const double travel_time = source.duration + instance.getcij(source_id, target_id);
670 const double arrival_time = std::max(static_cast<double>(target.lb[0]), current_time + travel_time);
671
672 // Check the time window for the target node.
673 if (arrival_time > target.ub[0]) { return false; }
674
675 // Update current time and capacity.
676 current_time = arrival_time;
677 current_capacity += source.demand;
678 if (current_capacity > total_capacity) { return false; }
679 }
680 return true;
681 }

683 std::pair<double, double> compute_cost(const std::vector<uint16_t> &route) {
684 const size_t n = route.size();
685 if (n < 2) return {0.0, 0.0};
686
687 double cost = 0.0;
688 double red_cost = 0.0;
689 const uint16_t *const rdata = route.data();
690
691 for (size_t i = 0; i < n - 1; ++i) {
692 // Cache current and next route entries.
693 const uint16_t cur = rdata[i];
694 const uint16_t nxt = rdata[i + 1];
695
696 // Compute travel cost between the current and next node.
697 const double travel = instance.getcij(cur, nxt);
698
699 // Add travel cost.
700 cost += travel;
701
702 // Subtract node-specific cost from travel cost to compute reduced
703 // cost.
704 red_cost += travel - nodes[cur].cost;
705 }
706
707 return {cost, red_cost};
708 }

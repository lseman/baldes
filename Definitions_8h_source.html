285inline ModelData extractModelDataSparse(GRBModel *model) {
286 ModelData data;
287 try {
288 // baldesVars
289 int numVars = model->get(GRB_IntAttr_NumVars);
290 GRBVar *vars = model->getVars();
291
292 // Reserve memory based on the actual number of variables
293 data.ub.reserve(numVars);
294 data.lb.reserve(numVars);
295 data.c.reserve(numVars);
296 data.vtype.reserve(numVars);
297 data.name.reserve(numVars);
298
299 std::vector<double> ub(numVars), lb(numVars), obj(numVars);
300 std::vector<char> vtype(numVars);
301
302 auto ubs =
303 model->get(GRB_DoubleAttr_UB, vars, numVars); // Fetch upper bounds
304 auto lbs =
305 model->get(GRB_DoubleAttr_LB, vars, numVars); // Fetch lower bounds
306 auto objs = model->get(GRB_DoubleAttr_Obj, vars,
307 numVars); // Fetch objective coefficients
308 auto vtypes = model->get(GRB_CharAttr_VType, vars,
309 numVars); // Fetch variable types
310
311 // Copy the values to the vectors
312 data.ub.assign(ubs, ubs + numVars);
313 data.lb.assign(lbs, lbs + numVars);
314 data.c.assign(objs, objs + numVars);
315 data.vtype.assign(vtypes, vtypes + numVars);
316
317 // baldesCtrs
318 int numConstrs = model->get(GRB_IntAttr_NumConstrs);
319 SparseMatrix A_sparse;
320
321 // Reserve memory for constraint matrices, estimating 10 non-zeros per
322 // row
323 A_sparse.values.reserve(numConstrs * 10); // Estimate non-zero elements
324 data.b.reserve(numConstrs);
325 data.cname.reserve(numConstrs);
326 data.sense.reserve(numConstrs);
327
328 for (int i = 0; i < numConstrs; ++i) {
329 GRBConstr constr = model->getConstr(i);
330
331 // Get the row corresponding to the constraint
332 GRBLinExpr expr = model->getRow(constr);
333 int exprSize = expr.size();
334
335 // Reserve space for constraint elements to minimize reallocation
336 A_sparse.values.reserve(A_sparse.values.size() + exprSize);
337
338 for (int j = 0; j < exprSize; ++j) {
339 GRBVar var = expr.getVar(j);
340 double coeff = expr.getCoeff(j);
341 int varIndex = var.index();
342
343 // Populate SparseElement for A_sparse
344 A_sparse.insert(i, varIndex, coeff);
345 }
346
347 // Batch fetch baldesCtr Name, RHS, and Sense (improve by reducing
348 // single fetches)
349 data.cname.push_back(constr.get(
350 GRB_StringAttr_ConstrName)); // Fetch constraint names
351 data.b.push_back(
352 constr.get(GRB_DoubleAttr_RHS)); // Fetch RHS values
353 data.sense.push_back(
354 constr.get(GRB_CharAttr_Sense)); // Fetch sense values
355 }
356
357 // Set matrix dimensions and build row_start
358 A_sparse.num_cols = numVars;
359 A_sparse.num_rows = numConstrs;
360 A_sparse.buildRowStart();
361
362 // Store the sparse matrix in data
363 data.A_sparse = std::move(A_sparse);
364
365 } catch (GRBException &e) {
366 std::cerr << "Error code = " << e.getErrorCode() << std::endl;
367 std::cerr << e.getMessage() << std::endl;
368 }
369
370 return data;
371}

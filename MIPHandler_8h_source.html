192 std::vector<int> reduceByRC(const std::vector<double> &dual_solution, const std::vector<int> &basic,
193 double keep_percentage = 0.7) {
194 if (keep_percentage <= 0.0 || keep_percentage > 1.0) {
195 throw std::invalid_argument("keep_percentage must be between 0 and 1");
196 }
197
198 // Get all RCs and their corresponding variable indices
199 std::vector<std::pair<double, int>> rc_pairs;
200 for (int i = 0; i < getVars().size(); i++) {
201 double rc = getRC(i, dual_solution);
202 rc_pairs.push_back({rc, i});
203 }
204
205 // Sort by absolute RC value in descending order
206 std::sort(rc_pairs.begin(), rc_pairs.end(), [](const auto &a, const auto &b) { return a.first < b.first; });
207
208 // Calculate how many variables to keep
209 size_t keep_count = static_cast<size_t>(std::ceil(rc_pairs.size() * keep_percentage));
210
211 // Create a set of indices to remove (the bottom 30%)
212 ankerl::unordered_dense::set<int> indices_to_remove;
213 for (size_t i = keep_count; i < rc_pairs.size(); i++) {
214 int var_index = rc_pairs[i].second;
215 // Skip variables that are in the basic set
216 if (std::find(basic.begin(), basic.end(), var_index) != basic.end()) { continue; }
217 indices_to_remove.insert(var_index);
218 }
219
220 // Remove variables from highest index to lowest to maintain validity of remaining indices
221 std::vector<int> sorted_indices(indices_to_remove.begin(), indices_to_remove.end());
222 std::sort(sorted_indices.begin(), sorted_indices.end(), std::greater<int>());
223
224 // Delete variables
225 for (int idx : sorted_indices) { delete_variable(idx); }
226
227 // Rebuild var_name_to_index
228 for (int i = 0; i < variables.size(); ++i) { var_name_to_index[variables[i]->get_name()] = i; }
229
230 return sorted_indices;
231 }

294 baldesCtrPtr add_constraint(const LinearExpression &expression, double rhs, char relation) {
295 int constraint_index = constraints.size(); // Get the current index
296 // Add the constraint to the list of constraints and set its index
297 auto new_constraint = std::make_shared<baldesCtr>(expression, rhs, relation);
298 b_vec.push_back(rhs);
299 new_constraint->set_index(constraint_index); // Set the constraint's index
300
301 constraints.push_back(new_constraint); // Add to constraints list
302
303 // Convert the LinearExpression to sparse row format and add to the sparse matrix
304 int row_index = constraint_index;
305 // sparse_matrix.num_rows++;
306
307 // Assuming you have many variables, use a more efficient lookup like a map
308 for (const auto &[var_name, coeff] : expression.get_terms()) {
309 auto it = var_name_to_index.find(var_name);
310 if (it != var_name_to_index.end()) {
311 int col_index = it->second;
312 sparse_matrix.insert(row_index, col_index, coeff);
313 } else {
314 fmt::print("baldesVar {} not found in the problem's variables list!\n", var_name);
315 }
316 }
317
318 // Update row start for CRS
319 // sparse_matrix.buildRowStart();
320 return new_constraint;
321 }

324 int constraint_index = constraints.size(); // Get the current index
325 constraint->set_index(constraint_index);

332 const LinearExpression &expression = constraint->get_expression();
333

340 sparse_matrix.insert(row_index, col_index, coeff);
341 }
342
343 // Return reference to the added constraint
344 return constraints.back();

547 // Function to populate a Gurobi model from this MIPProblem instance
548 GRBModel toGurobiModel(GRBEnv &env) {
549 // Create a new Gurobi model
550 GRBModel gurobiModel(env);
551
552 // Map to store Gurobi variables
553 ankerl::unordered_dense::map<std::string, GRBVar> gurobiVars;
554
555 // check if we have repeted variable name
556 std::unordered_map<std::string, int> var_count;
557 // Step 1: Add variables to the Gurobi model
558 for (const auto &var : variables) {
559 // Add each variable to the Gurobi model, according to its type and bounds
560 GRBVar gurobiVar = gurobiModel.addVar(var->get_lb(), var->get_ub(), var->get_objective_coefficient(),
561 toGRBVarType(var->get_type()), var->get_name());
562 gurobiVars[var->get_name()] = gurobiVar;
563 }
564
565 // Step 2: Add constraints to the Gurobi model
566 for (const auto &constraint : constraints) {
567 GRBLinExpr gurobiExpr = convertToGurobiExpr(constraint, gurobiVars);
568 if (constraint->get_relation() == '<') {
569 gurobiModel.addConstr(gurobiExpr <= constraint->get_rhs(), constraint->get_name());
570 } else if (constraint->get_relation() == '>') {
571 gurobiModel.addConstr(gurobiExpr >= constraint->get_rhs(), constraint->get_name());
572 } else if (constraint->get_relation() == '=') {
573 gurobiModel.addConstr(gurobiExpr == constraint->get_rhs(), constraint->get_name());
574 }
575 }
576
577 // Step 3: Set objective if needed (assuming a linear objective function)
578 GRBLinExpr objective;
579 for (const auto &var : variables) {
580 objective += gurobiVars[var->get_name()] * var->get_objective_coefficient();
581 }
582 gurobiModel.setObjective(objective, GRB_MINIMIZE); // Assume minimization problem
583 gurobiModel.update();
584 return gurobiModel;
585 }

598 GRBLinExpr convertToGurobiExpr(const baldesCtrPtr constraint,

624 // Set up the number of variables and constraints
625 highsModel.lp_.num_col_ = numVars;
626 highsModel.lp_.num_row_ = numConstrs;
627
628 // Resize vectors for variable bounds and objective coefficients
629 highsModel.lp_.col_lower_.resize(numVars);
630 highsModel.lp_.col_upper_.resize(numVars);
631 highsModel.lp_.col_cost_.resize(numVars);
632
633 // Set variable bounds and cost
634 int var_index = 0;
635 for (const auto &var : variables) {
636 highsModel.lp_.col_lower_[var_index] = var->get_lb();
637 highsModel.lp_.col_upper_[var_index] = var->get_ub();
638 highsModel.lp_.col_cost_[var_index] = var->get_objective_coefficient();
639 var_index++;
640 }
641
642 // Resize vectors for constraint bounds
643 highsModel.lp_.row_lower_.resize(numConstrs);
644 highsModel.lp_.row_upper_.resize(numConstrs);
645
646 // Set constraint bounds
647 int row_index = 0;
648 for (const auto &constraint : constraints) {
649 double lower_bound, upper_bound;
650 char relation = constraint->get_relation();
651 double rhs = constraint->get_rhs();
652
653 // Set lower and upper bounds for each constraint based on its type
654 if (relation == '<') {
655 lower_bound = -kHighsInf;
656 upper_bound = rhs;
657 } else if (relation == '>') {
658 lower_bound = rhs;
659 upper_bound = kHighsInf;
660 } else if (relation == '=') {
661 lower_bound = upper_bound = rhs;
662 }
663
664 highsModel.lp_.row_lower_[row_index] = lower_bound;
665 highsModel.lp_.row_upper_[row_index] = upper_bound;
666 row_index++;
667 }
668
669 // Assign sparse matrix data directly from `sparse_matrix`
670 highsModel.lp_.a_matrix_.start_ = sparse_matrix.getRowStart();
671 highsModel.lp_.a_matrix_.index_ = sparse_matrix.getIndices();
672 highsModel.lp_.a_matrix_.value_ = sparse_matrix.getValues();
673 highsModel.lp_.a_matrix_.format_ = MatrixFormat::kRowwise;
674
675 // Set the objective direction (assuming minimization)
676 highsModel.lp_.sense_ = ObjSense::kMinimize;
677
678 // Add the model to HiGHS
679 highs.passModel(highsModel);
680
681 return highsModel;
682 }

688 double getSlack(int row, const std::vector<double> &solution) {
689 // Ensure the sparse matrix row structure is built (we stay in COO mode)
690 auto rhs = constraints[row]->get_rhs(); // Get the right-hand side value for this row
691
692 // Compute the dot product of the solution vector and the specified row
693 double row_value = 0.0;
694
695 // Iterate through the non-zero elements in COO format
696 for (size_t i = 0; i < sparse_matrix.rows.size(); ++i) {
697 if (sparse_matrix.rows[i] == row) {
698 int col_index = sparse_matrix.cols[i]; // Get the column index for the current element
699 double value = sparse_matrix.values[i]; // Get the value of the current element
700
701 // Perform the dot product with the corresponding element in the solution vector
702 row_value += value * solution[col_index];
703 }
704 }
705
706 // Return the slack: rhs - (dot product of the row and solution)
707 return rhs - row_value;
708 }

710 double getRC(int col, const std::vector<double> &dual_solution) {
711 // Get the objective coefficient for this variable
712 double obj_coeff = variables[col]->get_objective_coefficient();
713
714 // Calculate the sum of (dual_value * coefficient) for all constraints
715 double dual_sum = 0.0;
716
717 // Iterate through the non-zero elements in COO format
718 for (size_t i = 0; i < sparse_matrix.rows.size(); ++i) {
719 if (sparse_matrix.cols[i] == col) {
720 int row_index = sparse_matrix.rows[i]; // Get the row index
721 double value = sparse_matrix.values[i]; // Get the matrix coefficient
722
723 // Multiply the dual variable value by the coefficient and add to sum
724 dual_sum += dual_solution[row_index] * value;
725 }
726 }
727
728 // For minimization problems:
729 // Reduced Cost = objective_coefficient - sum(dual_values * coefficients)
730 return obj_coeff - dual_sum;
731 }

733 std::vector<double> getAllReducedCosts(const std::vector<double> &dual_solution) {
734 // Initialize reduced costs with objective coefficients
735 std::vector<double> reduced_costs(variables.size());
736 for (size_t i = 0; i < variables.size(); ++i) { reduced_costs[i] = variables[i]->get_objective_coefficient(); }
737
738 // Subtract the dual contributions in one pass through the sparse matrix
739 for (size_t i = 0; i < sparse_matrix.rows.size(); ++i) {
740 int col = sparse_matrix.cols[i];
741 int row = sparse_matrix.rows[i];
742 double value = sparse_matrix.values[i];
743
744 // Subtract dual_value * coefficient from the corresponding reduced cost
745 reduced_costs[col] -= dual_solution[row] * value;
746 }
747
748 return reduced_costs;
749 }

751 ReducedCostResult getMostViolatingReducedCost(const std::vector<double> &dual_solution,
752 bool is_minimization = true) {
753 std::vector<double> reduced_costs = getAllReducedCosts(dual_solution);
754
755 ReducedCostResult result{0.0, -1};
756
757 // For initialization
758 if (!reduced_costs.empty()) {
759 result.value = reduced_costs[0];
760 result.column_index = 0;
761 }
762
763 // Find the most violating reduced cost based on optimization direction
764 for (size_t i = 1; i < reduced_costs.size(); ++i) {
765 if (is_minimization) {
766 // For minimization, we want the most negative reduced cost
767 if (reduced_costs[i] < result.value) {
768 result.value = reduced_costs[i];
769 result.column_index = i;
770 }
771 } else {
772 // For maximization, we want the most positive reduced cost
773 if (reduced_costs[i] > result.value) {
774 result.value = reduced_costs[i];
775 result.column_index = i;
776 }
777 }
778 }
779
780 std::vector<int> col;
781 col.assign(b_vec.size(), 0);
782 // iterate over sparse matrix at column_index col and populate the col vector
783 for (size_t i = 0; i < sparse_matrix.rows.size(); ++i) {
784 if (sparse_matrix.cols[i] == result.column_index) { col[sparse_matrix.rows[i]] = sparse_matrix.values[i]; }
785 }
786 result.col = col;
787
788 return result;
789 }

269 MIPProblem *clone() const {
270 // Create a new MIPProblem instance with the same name and matrix
271 // dimensions
272 auto *clone = new MIPProblem(name, sparse_matrix.num_rows,
273 sparse_matrix.num_cols);
274
275 // Clone variables
276 for (const auto var : variables) {
277 baldesVarPtr newVar = std::make_shared<baldesVar>(
278 var->get_name(), var->get_type(), var->get_lb(), var->get_ub(),
279 var->get_objective_coefficient());
280 newVar->set_index(var->index());
281 clone->variables.push_back(newVar);
282 clone->var_name_to_index[var->get_name()] = newVar->index();
283 }
284
285 for (const auto &constraint : constraints) {
286 auto clonedCtr = constraint->clone();
287 clone->constraints.push_back(clonedCtr);
288 clone->b_vec.push_back(clonedCtr->get_rhs());
289 }
290
291 // Clone the objective
292 clone->objective = objective;
293 clone->objective_type = objective_type;
294
295 // Deep copy the sparse matrix
296 clone->sparse_matrix = sparse_matrix;
297
298 return clone;
299 }

301 std::vector<int> reduceByRC(const std::vector<double> &dual_solution,
302 const std::vector<int> &basic,
303 double keep_percentage = 0.7) {
304 if (keep_percentage <= 0.0 || keep_percentage > 1.0) {
305 throw std::invalid_argument(
306 "keep_percentage must be between 0 and 1");
307 }
308
309 // Get all RCs and their corresponding variable indices
310 std::vector<std::pair<double, int>> rc_pairs;
311 for (int i = 0; i < getVars().size(); i++) {
312 double rc = getRC(i, dual_solution);
313 rc_pairs.push_back({rc, i});
314 }
315
316 // Sort by absolute RC value in descending order
317 std::sort(
318 rc_pairs.begin(), rc_pairs.end(),
319 [](const auto &a, const auto &b) { return a.first < b.first; });
320
321 // Calculate how many variables to keep
322 size_t keep_count =
323 static_cast<size_t>(std::ceil(rc_pairs.size() * keep_percentage));
324
325 // Create a set of indices to remove (the bottom 30%)
326 ankerl::unordered_dense::set<int> indices_to_remove;
327 for (size_t i = keep_count; i < rc_pairs.size(); i++) {
328 int var_index = rc_pairs[i].second;
329 // Skip variables that are in the basic set
330 if (std::find(basic.begin(), basic.end(), var_index) !=
331 basic.end()) {
332 continue;
333 }
334 indices_to_remove.insert(var_index);
335 }
336
337 // Remove variables from highest index to lowest to maintain
338 // validity of remaining indices
339 std::vector<int> sorted_indices(indices_to_remove.begin(),
340 indices_to_remove.end());
341 std::sort(sorted_indices.begin(), sorted_indices.end(),
342 std::greater<int>());
343
344 // Delete variables
345 for (int idx : sorted_indices) {
346 delete_variable(idx);
347 }
348
349 // Rebuild var_name_to_index
350 for (int i = 0; i < variables.size(); ++i) {
351 var_name_to_index[variables[i]->get_name()] = i;
352 }
353
354 return sorted_indices;
355 }

437 baldesCtrPtr add_constraint(const LinearExpression &expression, double rhs,
438 char relation) {
439 // Determine the new constraint's index.
440 int constraint_index = constraints.size();
441
442 // Create the new constraint and add it to the Gurobi cache (if
443 // applicable).
444 auto new_constraint =
445 std::make_shared<baldesCtr>(expression, rhs, relation);
446#ifdef GUROBI
447 gurobiCache->addConstraint(new_constraint);
448#endif
449
450 // Record the RHS value and set the constraint index.
451 b_vec.push_back(rhs);
452 new_constraint->set_index(constraint_index);
453 constraints.push_back(new_constraint);
454
455 // Prepare for batch insertion into the sparse matrix.
456 int row_index = constraint_index;
457 const auto &terms = expression.get_terms();
458
459 std::vector<int> batch_rows;
460 std::vector<int> batch_cols;
461 std::vector<double> batch_values;
462 batch_rows.reserve(terms.size());
463 batch_cols.reserve(terms.size());
464 batch_values.reserve(terms.size());
465
466 // Collect terms for batch insertion.
467 for (const auto &[var_name, coeff] : terms) {
468 auto it = var_name_to_index.find(var_name);
469 if (it != var_name_to_index.end()) {
470 batch_rows.push_back(row_index);
471 batch_cols.push_back(it->second);
472 batch_values.push_back(coeff);
473 } else {
474 fmt::print(
475 "baldesVar {} not found in the problem's variables list!\n",
476 var_name);
477 }
478 }
479
480 // Batch insert the collected terms into the sparse matrix.
481 sparse_matrix.insert_batch(batch_rows, batch_cols, batch_values);
482
483 return new_constraint;
484 }

486 baldesCtrPtr add_constraint(baldesCtrPtr constraint,
487 const std::string &name) {
488 // Set the new constraint's index and name.
489 int constraint_index = constraints.size();
490 constraint->set_index(constraint_index);
491 constraint->set_name(name);
492
493#ifdef GUROBI
494 gurobiCache->addConstraint(constraint);
495#endif
496
497 // Add the constraint to our list and record its RHS.
498 constraints.push_back(constraint);
499 b_vec.push_back(constraint->get_rhs());
500
501 // Retrieve the linear expression for later use.
502 const LinearExpression &expression = constraint->get_expression();
503 int row_index = constraint_index;
504
505 // Prepare vectors for batch insertion into the sparse matrix.
506 std::vector<int> batch_rows;
507 std::vector<int> batch_cols;
508 std::vector<double> batch_values;
509 auto numTerms = expression.get_terms().size();
510 batch_rows.reserve(numTerms);
511 batch_cols.reserve(numTerms);
512 batch_values.reserve(numTerms);
513
514 // Collect all terms (variable name and coefficient) for this
515 // constraint.
516 for (const auto &[var_name, coeff] : expression.get_terms()) {
517 // Use the lookup to get the column index for this variable.
518 int col_index = var_name_to_index[var_name];
519 batch_rows.push_back(row_index);
520 batch_cols.push_back(col_index);
521 batch_values.push_back(coeff);
522 }
523
524 // Insert the batch of coefficients into the sparse matrix.
525 sparse_matrix.insert_batch(batch_rows, batch_cols, batch_values);
526
527 // Return the newly added constraint.
528 return constraints.back();
529 }

584 // Main implementation that does the actual deletion work
585 void delete_constraint(int constraint_index) {
586 // Delete the row from the sparse matrix.
587 sparse_matrix.delete_row(constraint_index);
588
589#ifdef GUROBI
590 gurobiCache->deleteConstraint(constraint_index);
591#endif
592
593 // Erase the constraint and its right-hand side value.
594 constraints.erase(constraints.begin() + constraint_index);
595 b_vec.erase(b_vec.begin() + constraint_index);
596
597 // Update indices for the remaining constraints.
598 // If constraints.size() is large, this loop might be parallelized.
599 for (int i = constraint_index; i < static_cast<int>(constraints.size());
600 ++i) {
601 if (constraints[i]) {
602 constraints[i]->set_index(i);
603 } else {
604 std::cerr << "Null constraint found at position " << i
605 << std::endl;
606 }
607 }
608 }

676 void chgCoeff(int constraintIndex, const std::vector<double> &values) {
677 // Check for a valid constraint index.
678 if (constraintIndex < 0 ||
679 constraintIndex >= static_cast<int>(constraints.size())) {
680 fmt::print("Invalid constraint index: {}\n", constraintIndex);
681 throw std::out_of_range("Invalid constraint index");
682 }
683
684#ifdef GUROBI
685 // Modify constraint in the GUROBI cache.
686 gurobiCache->modifyConstraint(constraintIndex, values);
687#endif
688
689 // Retrieve the constraint and its linear expression.
690 baldesCtrPtr constraint = constraints[constraintIndex];
691 LinearExpression &expression = constraint->get_expression();
692
693 // Iterate over each term in the coefficient vector.
694 for (size_t i = 0; i < values.size(); ++i) {
695 double new_value = values[i];
696
697 // Update the sparse matrix with the new value.
698 sparse_matrix.modify_or_delete(constraintIndex, static_cast<int>(i),
699 new_value);
700
701 // Retrieve the variable's name based on its index.
702 const std::string &var_name = variables[i]->get_name();
703
704 // Update the linear expression:
705 // If the coefficient is nonzero, add or update the term.
706 // Otherwise, remove the term from the expression.
707 if (new_value != 0.0) {
708 expression.add_or_update_term(var_name, new_value);
709 } else {
710 expression.remove_term(var_name);
711 }
712 }

718 void chgCoeff(int constraintIndex, int variableIndex, double value) {
719 // Validate indices.

733 // entry if zero.
734 sparse_matrix.modify_or_delete(constraintIndex, variableIndex, value);

736 // Retrieve the constraint and its linear expression.
737 baldesCtrPtr constraint = constraints[constraintIndex];
738 LinearExpression &expression = constraint->get_expression();
739
740 // Get the variable's name for lookup.
741 const std::string &var_name = variables[variableIndex]->get_name();
742
743 // Update the linear expression: add or update the term if nonzero;
744 // remove otherwise.
745 if (value != 0.0) {
746 expression.add_or_update_term(var_name, value);
747 } else {
748 expression.remove_term(var_name);

995 double getSlack(int row, const std::vector<double> &solution) {
996 // Get the right-hand side value for this row.
997 const double rhs = constraints[row]->get_rhs();
998 double dot_product = 0.0;
999
1000 // Cache local references to the COO arrays.
1001 const auto &rows = sparse_matrix.rows;
1002 const auto &cols = sparse_matrix.cols;
1003 const auto &values = sparse_matrix.values;
1004 const size_t nnz = rows.size();
1005
1006 // Iterate through the non-zero elements in COO format.
1007 for (size_t i = 0; i < nnz; ++i) {
1008 if (rows[i] == row) {
1009 // Use the cached column and value to update the dot product.
1010 dot_product += values[i] * solution[cols[i]];
1011 }
1012 }
1013
1014 // Return the slack: rhs - (dot product of the row and solution)
1015 return rhs - dot_product;
1016 }

1018 double getRC(int col, const std::vector<double> &dual_solution) {
1019 // Get the objective coefficient for this variable
1020 double obj_coeff = variables[col]->get_objective_coefficient();
1021
1022 // Calculate the sum of (dual_value * coefficient) for all
1023 // constraints
1024 double dual_sum = 0.0;
1025
1026 // Iterate through the non-zero elements in COO format
1027 for (size_t i = 0; i < sparse_matrix.rows.size(); ++i) {
1028 if (sparse_matrix.cols[i] == col) {
1029 int row_index = sparse_matrix.rows[i]; // Get the row index
1030 double value =
1031 sparse_matrix.values[i]; // Get the matrix coefficient
1032
1033 // Multiply the dual variable value by the coefficient and
1034 // add to sum
1035 dual_sum += dual_solution[row_index] * value;
1036 }
1037 }
1038
1039 // For minimization problems:
1040 // Reduced Cost = objective_coefficient - sum(dual_values *
1041 // coefficients)
1042 return obj_coeff - dual_sum;
1043 }

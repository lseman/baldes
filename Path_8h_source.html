107 // Add an arc (i -> j) to the arcMap.
108 void addArc(uint16_t i, uint16_t j) {
109 ArcKey arc{i, j};
110 auto it = arcMap.find(arc);
111 if (it != arcMap.end()) {
112 it->second++;
113 } else {
114 arcMap.emplace(arc, 1);
115 }
116 }

118 // Precompute arcs (iterates over the route and counts arcs)
119 void precomputeArcs() {
120 // Pre-allocate expected size for better performance.
121 arcMap.reserve(route.size() > 0 ? route.size() - 1 : 0);
122 for (size_t n = 0; n + 1 < route.size(); ++n) {
123 addArc(route[n], route[n + 1]);
124 }
125 }

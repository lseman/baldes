71 size_t seed = v.size();

153struct CandidateSetCompare {
154 bool operator()(const CandidateSet &a, const CandidateSet &b) const {
155 // First check if they represent the same core elements
156 if (a.nodes == b.nodes && a.perm.num == b.perm.num &&
157 a.perm.den == b.perm.den) {
158 // If they're the same, keep the one with higher violation
159 // by making it "less than" so it wins
160 return a.violation > b.violation;
161 }
162
163 // For different elements, establish consistent ordering
164 if (a.nodes != b.nodes) return a.nodes < b.nodes;
165 if (a.perm.num != b.perm.num) return a.perm.num < b.perm.num;
166 return a.perm.den < b.perm.den;
167 }
168};

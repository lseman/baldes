34// Inline helper function: Given a base vector and a denominator, generate all
35// unique runtime permutations (using std::next_permutation) and return them as
36// a vector of Permutations.
37inline std::vector<SRCPermutation> generateRuntimePermutations(const std::vector<int> &base, int den) {
38 std::vector<SRCPermutation> perms;
39 std::vector<int> temp = base;
40 std::sort(temp.begin(), temp.end());
41 do { perms.emplace_back(temp, den); } while (std::next_permutation(temp.begin(), temp.end()));
42 return perms;
43}

45// Inline genetic generator: For a given candidate size, apply several heuristic
46// plans, generate permutations for each plan, and return all as a vector of
47// Permutations. This function mimics the structure of the reference generator
48// with plans 0 through 6.
49inline std::vector<SRCPermutation> generateGeneticPermutations(int candidateSize) {
50 std::vector<SRCPermutation> allPerms;
51 std::vector<std::pair<std::vector<int>, int>> plans;
52
53 // Basic rank-3 and rank-5 patterns rely on small denominators.
54 plans.emplace_back(std::vector<int>(candidateSize, 1), 2);
55
56 if (candidateSize >= 4) {
57 plans.emplace_back(std::vector<int>(candidateSize, 1), 3);
58 plans.emplace_back(std::vector<int>(candidateSize, 1), 4);
59 }
60
61 if (candidateSize >= 5) { plans.emplace_back(std::vector<int>(candidateSize, 1), 5); }
62
63 if (candidateSize >= 4) {
64 auto base = std::vector<int>(candidateSize, 1);
65 base[0] = candidateSize - 2;
66 plans.emplace_back(base, candidateSize - 1);
67 plans.emplace_back(base, candidateSize);
68 plans.emplace_back(base, candidateSize + 1);
69 }
70
71 if (candidateSize >= 5) {
72 auto base = std::vector<int>(candidateSize, 1);
73 base[0] = candidateSize - 3;
74 base[1] = 2;
75 plans.emplace_back(base, candidateSize - 1);
76 plans.emplace_back(base, candidateSize);
77 }
78
79 if (candidateSize >= 5) {
80 auto base = std::vector<int>(candidateSize, 1);
81 base[0] = candidateSize - 2;
82 base[1] = candidateSize - 2;
83 if (candidateSize >= 3) base[2] = 2;
84 plans.emplace_back(base, candidateSize - 2);
85 plans.emplace_back(base, candidateSize - 1);
86 }
87
88 if (candidateSize >= 5) {
89 auto base = std::vector<int>(candidateSize, 1);
90 base[0] = candidateSize - 2;
91 base[1] = 2;
92 base[2] = 2;
93 plans.emplace_back(base, candidateSize);
94 }
95
96 if (candidateSize >= 4) {
97 auto base = std::vector<int>(candidateSize, 1);
98 base[0] = candidateSize - 2;
99 base[1] = candidateSize - 2;
100 base[2] = 2;
101 if (candidateSize >= 4) base[3] = 2;
102 plans.emplace_back(base, candidateSize - 1);
103 plans.emplace_back(base, candidateSize);
104 }
105
106 // Deduplicate plans by base/den pair.
107 std::vector<std::pair<std::vector<int>, int>> unique_plans;
108 for (auto &plan : plans) {
109 bool found = false;
110 for (const auto &existing : unique_plans) {
111 if (existing.second == plan.second && existing.first == plan.first) {
112 found = true;
113 break;
114 }
115 }
116 if (found) continue;
117 unique_plans.emplace_back(plan);
118 auto perms = generateRuntimePermutations(plan.first, plan.second);
119 allPerms.insert(allPerms.end(), perms.begin(), perms.end());
120 }
121
122 return allPerms;
123}

125inline std::vector<SRCPermutation> generateExactPermutations(int candidateSize) {
126 std::vector<std::pair<std::vector<int>, int>> plans;
127 auto add_plan = [&](std::vector<int> base, int den) {
128 if (den < 2) return;
129 plans.emplace_back(std::move(base), den);
130 };
131
132 if (candidateSize % 2 == 1) { add_plan(std::vector<int>(candidateSize, 1), 2); }
133
134 if (candidateSize >= 5 && (candidateSize - 2) % 3 == 0) { add_plan(std::vector<int>(candidateSize, 1), 3); }
135
136 if (candidateSize >= 4) {
137 auto plan = std::vector<int>(candidateSize, 1);
138 plan[0] = candidateSize - 2;
139 add_plan(plan, candidateSize - 1);
140 }
141
142 if (candidateSize >= 5) {
143 auto plan2 = std::vector<int>(candidateSize, 1);
144 plan2[0] = candidateSize - 3;
145 plan2[1] = candidateSize - 3;
146 plan2[2] = 2;
147 add_plan(plan2, candidateSize - 2);
148
149 auto plan3 = std::vector<int>(candidateSize, 1);
150 plan3[0] = candidateSize - 2;
151 plan3[1] = candidateSize - 2;
152 plan3[2] = 2;
153 plan3[3] = 2;
154 add_plan(plan3, candidateSize - 1);
155
156 auto plan4 = std::vector<int>(candidateSize, 1);
157 plan4[0] = candidateSize - 3;
158 plan4[1] = 2;
159 add_plan(plan4, candidateSize - 1);
160
161 auto plan6 = std::vector<int>(candidateSize, 1);
162 plan6[0] = candidateSize - 2;
163 plan6[1] = 2;
164 plan6[2] = 2;
165 add_plan(plan6, candidateSize);
166 }
167
168 if (candidateSize >= 4) {
169 auto plan5 = std::vector<int>(candidateSize, 1);
170 plan5[0] = candidateSize - 2;
171 add_plan(plan5, candidateSize - 1);
172 }
173
174 std::sort(plans.begin(), plans.end(), [](const auto &a, const auto &b) {
175 if (a.second != b.second) return a.second < b.second;
176 return a.first < b.first;
177 });
178 plans.erase(std::unique(plans.begin(), plans.end()), plans.end());
179
180 std::vector<SRCPermutation> allPerms;
181 for (auto &plan : plans) {
182 auto perms = generateRuntimePermutations(plan.first, plan.second);
183 allPerms.insert(allPerms.end(), perms.begin(), perms.end());
184 }
185
186 return allPerms;
187}

193 // Handle empty vectors.
194 if (v.empty()) { return 0; }
195 // Compute the hash using XXH64.

238 // Equality operator for comparison
239 bool operator==(const CandidateSet &other) const {
240 return nodes == other.nodes && neighbor == other.neighbor && perm.den == other.perm.den &&
241 perm.num == other.perm.num;
242 }
243
244 // Less than operator for std::set
245 bool operator<(const CandidateSet &other) const {
246 auto a = *this;
247 auto b = other;
248 if (a.nodes == b.nodes && a.perm.num == b.perm.num && a.perm.den == b.perm.den) {
249 // If they're the same, keep the one with higher violation
250 // by making it "less than" so it wins
251 return a.violation > b.violation;
252 }
253
254 // For different elements, establish consistent ordering
255 if (a.nodes != b.nodes) return a.nodes.size() < b.nodes.size();
256 if (a.perm.num != b.perm.num) return a.perm.num < b.perm.num;
257 return a.perm.den < b.perm.den;
258 }

261struct CandidateSetCompare {
262 bool operator()(const CandidateSet &a, const CandidateSet &b) const {
263 // First check if they represent the same core elements
264 if (a.nodes == b.nodes && a.perm.num == b.perm.num && a.perm.den == b.perm.den && a.neighbor == b.neighbor) {
265 // If they're the same, keep the one with higher violation
266 // by making it "less than" so it wins
267 return a.violation > b.violation;
268 }
269
270 // For different elements, establish consistent ordering
271 if (a.nodes != b.nodes) return a.nodes.size() < b.nodes.size();
272 if (a.perm.num != b.perm.num) return a.perm.num < b.perm.num;
273 if (a.perm.den != b.perm.den) return a.perm.den < b.perm.den;
274 if (a.neighbor.size() != b.neighbor.size()) return a.neighbor.size() < b.neighbor.size();
275 std::vector<int> a_neigh(a.neighbor.begin(), a.neighbor.end());
276 std::vector<int> b_neigh(b.neighbor.begin(), b.neighbor.end());
277 std::sort(a_neigh.begin(), a_neigh.end());
278 std::sort(b_neigh.begin(), b_neigh.end());
279 return a_neigh < b_neigh;
280 }

283struct CandidateSetHasher {
284 using is_transparent = void;
285 uint64_t operator()(const CandidateSet &cs) const {
286 XXH3_state_t *state = XXH3_createState();
287 assert(state != nullptr);
288 XXH3_64bits_reset(state);
289 // Convert unordered nodes and neighbors to sorted vectors.
290 std::vector<int> sorted_nodes(cs.nodes.begin(), cs.nodes.end());
291 std::sort(sorted_nodes.begin(), sorted_nodes.end());
292 XXH3_64bits_update(state, sorted_nodes.data(), sorted_nodes.size() * sizeof(int));
293 std::vector<int> sorted_neighbor(cs.neighbor.begin(), cs.neighbor.end());
294 std::sort(sorted_neighbor.begin(), sorted_neighbor.end());
295 XXH3_64bits_update(state, sorted_neighbor.data(), sorted_neighbor.size() * sizeof(int));
296 // Hash the permutation numerator (assumed to be a vector<int>).
297 XXH3_64bits_update(state, cs.perm.num.data(), cs.perm.num.size() * sizeof(int));
298 // Hash the permutation denominator.
299 XXH3_64bits_update(state, &cs.perm.den, sizeof(int));
300 uint64_t hash = XXH3_64bits_digest(state);
301 XXH3_freeState(state);
302 return hash;
303 }

305 uint64_t mixed_hash(const CandidateSet &cs) const { return operator()(cs); }

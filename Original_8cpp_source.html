15GRBModel solve_vrp(const InstanceData &instance) {
16 GRBEnv env = GRBEnv(true);
17 // env.set(GRB_IntParam_OutputFlag, 1);
18 env.start();
19
20 GRBModel model = GRBModel(env);
21
22 int num_customers = instance.nC;
23 int num_vehicles = 1; // instance.nV;
24 int depot = 0; // Assume depot is node 0
25
26 // Decision variables
27 // x[i][j][v]: Binary variable that represents if vehicle v travels from node i to node j
28 std::vector<std::vector<std::vector<GRBVar>>> x(
29 num_customers + 1, std::vector<std::vector<GRBVar>>(num_customers + 1, std::vector<GRBVar>(num_vehicles)));
30
31 // t[i]: Continuous variable that represents the arrival time at node i
32 std::vector<GRBVar> t(num_customers + 1);
33
34 // Create the decision variables
35 for (int i = 0; i <= num_customers; ++i) {
36 t[i] = model.addVar(instance.window_open[i], instance.window_close[i], 0.0, GRB_CONTINUOUS,
37 "t_" + std::to_string(i));
38 for (int j = 0; j <= num_customers; ++j) {
39 if (i != j) {
40 for (int v = 0; v < num_vehicles; ++v) {
41 x[i][j][v] =
42 model.addVar(0.0, 1.0, instance.travel_cost[i][j], GRB_BINARY,
43 "x_" + std::to_string(i) + "_" + std::to_string(j) + "_v" + std::to_string(v));
44 }
45 }
46 }
47 }
48
49 model.update(); // Integrate the variables into the model
50
51 // Objective: Minimize total travel cost
52 GRBLinExpr obj = 0;
53 for (int i = 0; i <= num_customers; ++i) {
54 for (int j = 0; j <= num_customers; ++j) {
55 if (i != j) {
56 for (int v = 0; v < num_vehicles; ++v) { obj += instance.travel_cost[i][j] * x[i][j][v]; }
57 }
58 }
59 }
60 model.setObjective(obj, GRB_MINIMIZE);
61
62 // baldesCtr 1: Flow conservation at each customer
63 for (int i = 1; i <= num_customers; ++i) {
64 GRBLinExpr flow_in = 0;
65 GRBLinExpr flow_out = 0;
66 for (int v = 0; v < num_vehicles; ++v) {
67 for (int j = 0; j <= num_customers; ++j) {
68 if (i != j) {
69 flow_in += x[j][i][v];
70 flow_out += x[i][j][v];
71 }
72 }
73 }
74 model.addConstr(flow_in == 1, "flow_in_" + std::to_string(i)); // Exactly one vehicle enters each customer
75 model.addConstr(flow_out == 1, "flow_out_" + std::to_string(i)); // Exactly one vehicle leaves each customer
76 }
77
78 // baldesCtr 2: Each vehicle leaves and returns to the depot
79 for (int v = 0; v < num_vehicles; ++v) {
80 GRBLinExpr vehicle_start = 0;
81 GRBLinExpr vehicle_end = 0;
82 for (int i = 1; i <= num_customers; ++i) {
83 vehicle_start += x[depot][i][v];
84 vehicle_end += x[i][depot][v];
85 }
86 model.addConstr(vehicle_start == 1, "vehicle_start_" + std::to_string(v)); // Each vehicle leaves depot once
87 model.addConstr(vehicle_end == 1, "vehicle_end_" + std::to_string(v)); // Each vehicle returns to depot once
88 }
89
90 // baldesCtr 3: Time window constraints
91 for (int i = 1; i <= num_customers; ++i) {
92 for (int j = 1; j <= num_customers; ++j) {
93 if (i != j) {
94 for (int v = 0; v < num_vehicles; ++v) {
95 model.addConstr(t[i] + instance.service_time[i] + instance.travel_cost[i][j] - t[j] <=
96 (1 - x[i][j][v]) * instance.T_max,
97 "time_" + std::to_string(i) + "_" + std::to_string(j) + "_v" + std::to_string(v));
98 }
99 }
100 }
101 }
102
103 // baldesCtr 4: Vehicle capacity constraint
104 for (int v = 0; v < num_vehicles; ++v) {
105 GRBLinExpr load = 0;
106 for (int i = 1; i <= num_customers; ++i) {
107 for (int j = 1; j <= num_customers; ++j) {
108 if (i != j) { load += instance.getDemand(i) * x[i][j][v]; }
109 }
110 }
111 model.addConstr(load <= instance.getCapacity(), "capacity_v" + std::to_string(v));
112 }
113
114 // Optimize the model
115 model.update();
116 model.optimize();
117 return model;
118}

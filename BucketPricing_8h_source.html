95inline void BucketGraph::transition_after_bucket_pricing(Stage active_stage, const BucketPricingResult &result) {
96 if (result.skip_stage_transition) return;
97
98 switch (active_stage) {
99 case Stage::One:
100 if (result.best_reduced_cost >= -1 || iter >= 10) { activate_bucket_pricing_stage(Stage::Two); }
101 break;
102 case Stage::Two:
103 if (result.best_reduced_cost >= -10 || iter > 500) { activate_bucket_pricing_stage(Stage::Three); }
104 break;
105 case Stage::Three:
106 if (result.best_reduced_cost >= -0.5) {
107 activate_bucket_pricing_stage(Stage::Four);
108 transition = true;
109 }
110 break;
111 case Stage::Four: {
112#ifdef FIX_BUCKETS
113 if (transition) { break; }
114#endif
115 if (status != Status::Rollback && considerRegenerate()) {
116 status = Status::Rollback;
117 break;
118 }
119 threshold = stats.computeThreshold(iter, result.best_reduced_cost);
120 constexpr double kExactPricingTol = 1e-6;
121 if (result.best_reduced_cost >= -kExactPricingTol) {
122 ss = true;
123#if !defined(SRC) && !defined(SRC3)
124 status = Status::Optimal;
125#else
126 status = Status::Separation;
127#endif
128 } else {
129 ss = false;
130 status = Status::NotOptimal;
131 }
132 break;
133 }
134 case Stage::Enumerate:
135 if (result.enumeration_failed) {
136 activate_bucket_pricing_stage(Stage::Four);
137 status = Status::NotOptimal;
138 } else {
139 status = Status::Optimal;
140 }
141 break;
142 default:
143 break;
144 }

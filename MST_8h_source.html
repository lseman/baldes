34 // Function to compute the MST using Prim's algorithm
35 std::vector<Arc> compute_mst() {
36 std::priority_queue<Arc, std::vector<Arc>, std::greater<Arc>> min_heap;
37 std::vector<bool> in_mst(nodes_.size(), false);
38 std::vector<Arc> mst;
39 mst.reserve(nodes_.size() - 1);
40
41 int start_node = 0;
42 in_mst[start_node] = true;
43 add_adjacent_arcs(start_node, min_heap);
44
45 while (!min_heap.empty() && mst.size() < nodes_.size() - 1) {
46 auto [weight, from, to] = min_heap.top();
47 min_heap.pop();
48
49 if (in_mst[to]) continue;
50
51 mst.emplace_back(weight, from, to);
52 in_mst[to] = true;
53 add_adjacent_arcs(to, min_heap);
54 }
55
56 return mst;
57 }

59 // Function to perform clustering by removing edges with costs exceeding
60 // threshold Θ
61 std::vector<std::vector<int>> cluster(double theta) {
62 auto mst = compute_mst();
63
64 // Calculate average and standard deviation of edge weights in the MST
65 double sum = 0.0;
66 for (const auto &[weight, from, to] : mst) {
67 sum += weight;
68 }
69 double avg = sum / mst.size();
70
71 double sum_sq = 0.0;
72 for (const auto &[weight, from, to] : mst) {
73 sum_sq += std::pow(weight - avg, 2);
74 }
75 double std_dev = std::sqrt(sum_sq / mst.size());
76
77 // Threshold Θ = avg(T) + θ * std(T)
78 double threshold = avg + theta * std_dev;
79
80 // Remove edges with weights greater than the threshold to form clusters
81 std::vector<Arc> filtered_edges;
82 filtered_edges.reserve(mst.size());
83 for (const auto &[weight, from, to] : mst) {
84 if (weight <= threshold) {
85 filtered_edges.emplace_back(weight, from, to);
86 }
87 }
88
89 // Find the connected components (clusters) in the filtered MST
90 return find_clusters(filtered_edges);
91 }

99 void add_adjacent_arcs(int node_id,

138 // Union-Find class for finding connected components
139 class UnionFind {
140 public:
141 UnionFind(int n) : parent(n), rank(n, 0) {
142 for (int i = 0; i < n; ++i) parent[i] = i;
143 }
144
145 int find(int x) {
146 if (parent[x] != x) parent[x] = find(parent[x]);
147 return parent[x];
148 }
149
150 void union_sets(int x, int y) {
151 int root_x = find(x);
152 int root_y = find(y);
153 if (root_x != root_y) {
154 if (rank[root_x] > rank[root_y])
155 parent[root_y] = root_x;
156 else if (rank[root_x] < rank[root_y])
157 parent[root_x] = root_y;
158 else {
159 parent[root_y] = root_x;
160 rank[root_x]++;
161 }
162 }
163 }
164
165 private:
166 std::vector<int> parent;
167 std::vector<int> rank;

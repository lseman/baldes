68 static std::vector<BranchingQueueItem> calculateAggregatedbaldesVars(BNBNode *node, const InstanceData *instance,
69 std::vector<VRPNode> &nodes) {
70 std::vector<BranchingQueueItem> queueItems;
71 auto LPSolution = node->extractSolution();
72 auto routes = node->getPaths();
73
74 // Aggregated variables for all vehicle types and routes
75 ankerl::unordered_dense::map<int, double> g_m; // Aggregated g_m (number of vehicles of type m used)
76 ankerl::unordered_dense::map<int, double>
77 g_m_v; // Aggregated g_m_v (whether customer v is served by a vehicle of type m)
78 ankerl::unordered_dense::map<std::pair<int, int>, double>
79 g_v_vp; // Aggregated g_{v,v'} (whether edge (v, v') is used)
80
81 auto clusters = calculateMSTClusters(instance, nodes, 1.0);
82 std::vector<double> clusterUsages(clusters.size(), 0.0); // Vector to store usage of each cluster
83
84 // Iterate over all routes in the LPSolution
85 for (int i = 0; i < LPSolution.size(); ++i) {
86 if (LPSolution[i] == 0) continue; // Skip if no solution for this route
87
88 if (LPSolution[i] > TOLERANCE_ZERO && isFractional(LPSolution[i])) {
89
90 double totalWindowDifference = 0.0;
91
92 // Assuming vehicleType is stored in the instance for each route
93 int vehicleType = 0;
94
95 // Calculate g_m: Sum the solution value for each vehicle type m
96 g_m[vehicleType] += LPSolution[i]; // Accumulate the usage of vehicle type m
97
98 // Iterate over the route to calculate g_m_v and g_{v,v'}
99 for (size_t c = 0; c < routes[i].route.size() - 1; ++c) {
100 auto source = routes[i].route[c];
101 auto target = routes[i].route[c + 1];
102
103 // Calculate window differences (for scoring, not directly affecting g_m_v or g_{v,v'})
104 double windowDifference =
105 std::fabs(instance->window_open[source] - instance->window_open[target]) +
106 std::fabs(instance->window_close[source] - instance->window_close[target]);
107 totalWindowDifference += windowDifference;
108
109 // Accumulate g_{v,v'}: Edge usage for both directions (v, v') and (v', v)
110 g_v_vp[{source, target}] += LPSolution[i]; // Accumulate for (v, v')
111 }
112
113 // Calculate g_m_v: Accumulate if customer v is served by a vehicle of type m
114 for (size_t c = 1; c < routes[i].route.size() - 1; ++c) { // Skip depot (assumed to be first and last)
115 auto customer = routes[i].route[c];
116 g_m_v[customer] +=
117 LPSolution[i]; // Accumulate the value for this customer being served by vehicle type m
118 }
119
120 // Aggregate fractional values into clusters
121 for (size_t clusterIdx = 0; clusterIdx < clusters.size(); ++clusterIdx) {
122 for (int nodeId : clusters[clusterIdx]) {
123 if (std::find(routes[i].route.begin(), routes[i].route.end(), nodeId) !=
124 routes[i].route.end()) {
125 clusterUsages[clusterIdx] += LPSolution[i];
126 }
127 }
128 }
129 }
130 }
131
132 // Create separate queue items for g_m, g_m_v, and g_{v,v'} variables
133
134 // 1. Queue items for g_m (vehicle types)
135 for (auto &[vehicleType, value] : g_m) {
136 queueItems.push_back(
137 BranchingQueueItem{0, 0, value, 0.0, {{vehicleType, value}}, {}, {}, CandidateType::Vehicle});
138 }
139
140 // 2. Queue items for g_m_v (customers served by vehicles)
141 for (auto &[customer, value] : g_m_v) {
142 queueItems.push_back(
143 BranchingQueueItem{0, customer, value, 0.0, {}, {{customer, value}}, {}, CandidateType::Node});
144 }
145
146 // 3. Queue items for g_{v,v'} (edge usage)
147 for (auto &[edge, value] : g_v_vp) {
148 queueItems.push_back(
149 BranchingQueueItem{edge.first, edge.second, value, 0.0, {}, {}, {{edge, value}}, CandidateType::Edge});
150 }
151
152 // Add queue items for clusters with significant fractional usage
153 for (size_t clusterIdx = 0; clusterIdx < clusters.size(); ++clusterIdx) {
154 if (clusterUsages[clusterIdx] > TOLERANCE_ZERO) {
155 BranchingQueueItem item =
156 BranchingQueueItem{0, 0, clusterUsages[clusterIdx], 0.0, {}, {}, {}, CandidateType::Cluster};
157 item.g_c[clusterIdx] = clusterUsages[clusterIdx];
158 item.clusters = clusters[clusterIdx];
159 queueItems.push_back(item);
160 }
161 }
162
163 return queueItems;
164 }
165 static void addbaldesCtrForCandidate(BNBNode *childNode, const BranchingQueueItem &item, double bound,
166 BranchingDirection direction) {
167 if (item.candidateType == CandidateType::Vehicle) {
168 childNode->addBranchingbaldesCtr(bound, direction, CandidateType::Vehicle);
169 } else if (item.candidateType == CandidateType::Node) {
170 childNode->addBranchingbaldesCtr(bound, direction, CandidateType::Node, item.targetNode);
171 } else if (item.candidateType == CandidateType::Edge) {
172 std::pair<int, int> edge = {item.sourceNode, item.targetNode};
173 childNode->addBranchingbaldesCtr(bound, direction, CandidateType::Edge, edge);
174 } else if (item.candidateType == CandidateType::Cluster) {
175 childNode->addBranchingbaldesCtr(bound, direction, CandidateType::Cluster, item.g_c.begin()->first);
176 }
177 }

180 static std::pair<BNBNode *, BNBNode *> applyBranchingbaldesCtrs(BNBNode *parentNode,
181 const BranchingQueueItem &item,
182 double fractionalValue, CandidateType type) {
183 BNBNode *childNode1 = parentNode->newChild();
184 BNBNode *childNode2 = parentNode->newChild();
185
186 double lowerBound = std::floor(fractionalValue);
187 double upperBound = std::ceil(fractionalValue);
188 if (type == CandidateType::Node) {
189 lowerBound = 0;
190 upperBound = 1;
191 }
192
193 addbaldesCtrForCandidate(childNode1, item, lowerBound, BranchingDirection::Less);
194 addbaldesCtrForCandidate(childNode2, item, upperBound, BranchingDirection::Greater);
195
196 return {childNode1, childNode2};

203 static std::vector<BranchingQueueItem>
204 evaluateWithBranching(BNBNode *node, const std::vector<BranchingQueueItem> &phase0Candidates) {
205 std::vector<BranchingQueueItem> results;
206 std::mutex results_mutex; // Mutex to protect shared results
207
208 const int JOBS = std::thread::hardware_concurrency();
209 exec::static_thread_pool pool(JOBS); // Pool with concurrent threads
210 auto sched = pool.get_scheduler();
211
212 // Define chunk size based on performance tuning (adjust as needed)
213 const int chunk_size = 10;
214 auto total_chunks = (phase0Candidates.size() + chunk_size - 1) / chunk_size;
215
216 // Parallel bulk execution
217 auto bulk_sender = stdexec::bulk(
218 stdexec::just(), total_chunks,
219 [&results, &results_mutex, node, &phase0Candidates, chunk_size](std::size_t chunk_idx) {
220 size_t start_idx = chunk_idx * chunk_size;
221 size_t end_idx = std::min(start_idx + chunk_size, phase0Candidates.size());
222
223 std::vector<BranchingQueueItem> local_results; // Local storage to avoid locking too often
224
225 // Process the chunk of candidates
226 for (size_t idx = start_idx; idx < end_idx; ++idx) {
227 const auto &candidate = phase0Candidates[idx];
228
229 // Add branching constraints and create two child nodes
230 auto [childNode1, childNode2] =
231 applyBranchingbaldesCtrs(node, candidate, candidate.fractionalValue, candidate.candidateType);
232
233 // Solve the Restricted Master LP for each child node
234 double deltaLB1, deltaLB2;
235 double feasLB1, feasLB2;
236 std::tie(feasLB1, deltaLB1) = childNode1->solveRestrictedMasterLP();
237 std::tie(feasLB2, deltaLB2) = childNode2->solveRestrictedMasterLP();
238 // print the deltaLB1 and deltaLB2
239
240 // Calculate product value for branching
241 double productValue = deltaLB1 * deltaLB2;
242
243 // Store the result locally
244 local_results.push_back({candidate.sourceNode, candidate.targetNode, candidate.fractionalValue,
245 productValue, candidate.g_m, candidate.g_m_v, candidate.g_v_vp,
246 candidate.candidateType, std::make_pair(feasLB1, feasLB2)});
247 }
248
249 // Add local results to shared results with mutex protection
250 {
251 std::lock_guard<std::mutex> lock(results_mutex);
252 results.insert(results.end(), local_results.begin(), local_results.end());
253 }
254 });
255
256 // Execute bulk task using stdexec
257 stdexec::sync_wait(stdexec::when_all(bulk_sender));
258
259 // Sort results by product value in descending order (highest product value first)
260 pdqsort(results.begin(), results.end(), [](const BranchingQueueItem &a, const BranchingQueueItem &b) {
261 return a.productValue > b.productValue;
262 });
263
264 return results;
265 }

272 const std::vector<BranchingQueueItem> &phase1Candidates) {

275 // Lambda to generate candidates
276 auto generateCandidates =
277 [&](CandidateType type, int sourceNode, int targetNode, double fractionalValue, std::pair<bool, bool> flags,
278 std::optional<std::variant<int, std::pair<int, int>, std::vector<int>>> payload = std::nullopt) {
279 // check if flag first is true
280 double lower_bound = std::floor(fractionalValue);
281 double upper_bound = std::ceil(fractionalValue);

284 lower_bound = 0;
285 upper_bound = 1;
286 }
287 if (flags.first) {

290 if (type == CandidateType::Cluster) {

293 candidates.push_back(candidate);
294 }
295 if (flags.second) {
296 auto candidate = new VRPCandidate(sourceNode, targetNode, BranchingDirection::Greater, upper_bound,
297 type, payload);
298 if (type == CandidateType::Cluster) {
299 candidate->cluster = std::get<std::vector<int>>(payload.value());
300 }
301 candidates.push_back(candidate);
302 }
303 };
304
305 for (const auto &item : phase1Candidates) {
306 // print candidate product value
307 if (!item.g_m.empty()) {
308 generateCandidates(CandidateType::Vehicle, item.sourceNode, item.targetNode, item.fractionalValue,
309 item.flags,
310 std::nullopt); // No payload for vehicles
311 }
312 if (!item.g_m_v.empty()) {
313 generateCandidates(CandidateType::Node, item.sourceNode, item.targetNode, item.fractionalValue,
314 item.flags,
315 item.targetNode); // Node as payload
316 }
317 if (!item.g_v_vp.empty()) {
318 std::pair<int, int> edge = {item.sourceNode, item.targetNode};
319 generateCandidates(CandidateType::Edge, item.sourceNode, item.targetNode, item.fractionalValue,
320 item.flags,
321 edge); // Edge as payload
322 }
323 if (!item.g_c.empty()) {
324 generateCandidates(CandidateType::Cluster, item.sourceNode, item.targetNode, item.fractionalValue,
325 item.flags,
326 item.clusters); // Clusters as payload
327 }
328 }
329
330 return candidates;
331 }

375 static std::vector<BranchingQueueItem>
376 selectCandidatesPhase0(const std::vector<BranchingQueueItem> &allCandidates,
377 const std::vector<BranchingQueueItem> &historyCandidates, size_t maxCandidatesPhase0,
378 InstanceData *instance) {
379
380 std::vector<BranchingQueueItem> selectedCandidates;
381
382 // Step 1: Select half of the candidates from history using pseudo-costs if available
383 size_t historySize = historyCandidates.size();
384 size_t numFromHistory = std::min(maxCandidatesPhase0 / 2, historySize);
385
386 // Sort history candidates by pseudo-costs (assumed to be stored in score field)
387 std::vector<BranchingQueueItem> sortedHistoryCandidates = historyCandidates;
388 pdqsort(sortedHistoryCandidates.begin(), sortedHistoryCandidates.end(),
389 [](const BranchingQueueItem &a, const BranchingQueueItem &b) {
390 return a.productValue > b.productValue; // Higher pseudo-cost first
391 });
392
393 // Select top candidates from history
394 for (size_t i = 0; i < numFromHistory; ++i) { selectedCandidates.push_back(sortedHistoryCandidates[i]); }
395
396 // Step 2: Select remaining candidates using the three branching strategies
397 size_t remainingCandidates = maxCandidatesPhase0 - selectedCandidates.size();
398 std::vector<BranchingQueueItem> strategyCandidates;
399
400 // Strategy 1: Fractional value distance to the closest integer (for branching on variables)
401 for (const auto &candidate : allCandidates) {
402 if (remainingCandidates == 0) break;
403
404 // Check if candidate is suitable based on its fractional value
405 if (candidate.fractionalValue > TOLERANCE_INTEGER) {
406 strategyCandidates.push_back(candidate);
407 remainingCandidates--;
408 }
409 }
410
411 // Strategy 2: Branching on edges with proximity to the depot
412 for (const auto &candidate : allCandidates) {
413 if (remainingCandidates == 0) break;
414
415 if (candidate.candidateType == CandidateType::Vehicle) continue; // Skip vehicle candidates
416
417 // Assuming `getDepotDistance()` computes distance of the edge to the closest depot
418 double depotDistance = instance->getcij(0, candidate.targetNode);
419 if (depotDistance < DEPOT_DISTANCE_THRESHOLD) { // Threshold for proximity to depot
420 strategyCandidates.push_back(candidate);
421 remainingCandidates--;
422 }
423 }
424
425 // Step 3: Sort remaining candidates based on distance to the closest integer or depot distance
426 pdqsort(strategyCandidates.begin(), strategyCandidates.end(),
427 [](const BranchingQueueItem &a, const BranchingQueueItem &b) {
428 return a.fractionalValue < b.fractionalValue; // Closest to integer first for fractional strategy
429 });
430
431 // Add remaining candidates to the selected candidates list
432 for (size_t i = 0; i < strategyCandidates.size() && selectedCandidates.size() < maxCandidatesPhase0; ++i) {
433 selectedCandidates.push_back(strategyCandidates[i]);
434 }
435
436 return selectedCandidates;
437 }

79template <Direction D, Stage S, Full F>
80std::vector<double> BucketGraph::labeling_algorithm() {
81 // Cache bucket and auxiliary structure references to avoid repeated
82 // lookups.
83 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
84 const auto &topological_order = assign_buckets<D>(fw_topological_order, bw_topological_order);
85 const auto &sccs = assign_buckets<D>(fw_sccs, bw_sccs);
86 const auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
87 auto &c_bar = assign_buckets<D>(fw_c_bar, bw_c_bar);
88 auto &n_labels = assign_buckets<D>(n_fw_labels, n_bw_labels);
89 const auto &sorted_sccs = assign_buckets<D>(fw_sccs_sorted, bw_sccs_sorted);
90 const auto &n_buckets = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
91 auto &stat_n_labels = assign_buckets<D>(stat_n_labels_fw, stat_n_labels_bw);
92 auto &stat_n_dom = assign_buckets<D>(stat_n_dom_fw, stat_n_dom_bw);
93 auto &arc_scores = assign_buckets<D>(fw_arc_scores, bw_arc_scores);
94 auto &best_label = assign_buckets<D>(fw_best_label, bw_best_label);
95
96 const bool profile_labeling = options.profile_labeling;
97 const double src_compensation_bound =
98 (S == Stage::Four || S == Stage::Enumerate) ? src_cost_compensation_bound() : 0.0;
99
100 // Reset the total label count.
101 n_labels = 0;
102
103 // Pre-calculate bitmap segments for dominance-checking.
104 const size_t n_segments = (n_buckets + 63) >> 6;
105 std::vector<uint64_t> Bvisited(n_segments, 0);
106 std::vector<uint32_t> touched_segments;
107 touched_segments.reserve(std::min<size_t>(n_segments, 64));
108 std::vector<int> bucket_pos_in_scc(n_buckets, -1);
109
110 // Process only newly queued labels while preserving the bucket order
111 // inside each SCC.
112 std::vector<std::vector<Label *>> pending_labels_by_bucket;
113 std::vector<size_t> pending_label_cursor;
114 std::vector<uint8_t> bucket_is_active;
115 std::vector<int> active_bucket_heap;
116
117 // Precompute main resource index and q_star value for partial solutions.
118 constexpr bool is_partial = (F == Full::Partial);
119 int main_res_index = 0;
120 double q_star_value = 0.0;
121 if constexpr (is_partial) {
122 main_res_index = options.main_resources[0];
123 q_star_value = std::floor(q_star[main_res_index]);
124 }
125
126 // Reuse a thread-local vector for collecting labels to dominate.
127 static thread_local std::vector<Label *> tl_labels_to_dominate;
128 tl_labels_to_dominate.clear();
129 tl_labels_to_dominate.reserve(64); // Pre-reserve space to avoid reallocations
130
131 // Process each strongly connected component (SCC) in topological order.
132 for (const auto &scc_index : topological_order) {
133 const auto &scc_buckets = sorted_sccs[scc_index];
134 if (scc_buckets.empty()) continue;
135
136 pending_labels_by_bucket.clear();
137 pending_labels_by_bucket.resize(scc_buckets.size());
138 pending_label_cursor.assign(scc_buckets.size(), 0);
139 bucket_is_active.assign(scc_buckets.size(), 0);
140 active_bucket_heap.clear();
141
142 auto activate_bucket = [&](int bucket_pos) {
143 if (bucket_pos < 0 || bucket_is_active[bucket_pos]) return;
144 bucket_is_active[bucket_pos] = 1;
145 active_bucket_heap.push_back(bucket_pos);
146 std::push_heap(active_bucket_heap.begin(), active_bucket_heap.end(), std::greater<int>());
147 };
148
149 for (size_t bucket_idx = 0; bucket_idx < scc_buckets.size(); ++bucket_idx) {
150 const int bucket = scc_buckets[bucket_idx];
151 bucket_pos_in_scc[bucket] = static_cast<int>(bucket_idx);
152
153 const auto &bucket_labels = buckets[bucket].get_sorted_labels();
154 const auto &extra_labels = buckets[bucket].get_extra_labels();
155 auto &pending_labels = pending_labels_by_bucket[bucket_idx];
156 pending_labels.reserve(bucket_labels.size() + extra_labels.size());
157
158 for (Label *label : bucket_labels) {
159 if (!label->is_extended && !label->is_dominated) pending_labels.push_back(label);
160 }
161 for (Label *label : extra_labels) {
162 if (!label->is_extended && !label->is_dominated) pending_labels.push_back(label);
163 }
164
165 if (!pending_labels.empty()) activate_bucket(static_cast<int>(bucket_idx));
166 }
167
168 while (!active_bucket_heap.empty()) {
169 std::pop_heap(active_bucket_heap.begin(), active_bucket_heap.end(), std::greater<int>());
170 const int bucket_idx = active_bucket_heap.back();
171 active_bucket_heap.pop_back();
172
173 const int bucket = scc_buckets[bucket_idx];
174 auto &pending_labels = pending_labels_by_bucket[bucket_idx];
175 size_t &pending_cursor = pending_label_cursor[bucket_idx];
176
177 while (pending_cursor < pending_labels.size()) {
178 Label *label = pending_labels[pending_cursor++];
179
180 // Skip labels already processed.
181 if (__builtin_expect(label->is_extended || label->is_dominated, 0)) continue;
182
183 // For partial solutions, check resource feasibility.
184 if constexpr (F != Full::PSTEP && F != Full::TSP) {
185 if constexpr (is_partial) {
186 const auto main_resource = label->resources[main_res_index];
187 if ((D == Direction::Forward && main_resource > q_star_value) ||
188 (D == Direction::Backward && main_resource <= q_star_value)) {
189 label->set_extended(true);
190 continue;
191 }
192 }
193 }
194
195 for (uint32_t segment_idx : touched_segments) { Bvisited[segment_idx] = 0; }
196 touched_segments.clear();
197
198 // Check if the label is dominated by labels in smaller
199 // buckets.
200 if constexpr (F != Full::TSP) {
201 if (profile_labeling) profile_record_dominance_check(D, S);
202 if (DominatedInCompWiseSmallerBuckets<D, S>(label, bucket, c_bar, Bvisited, touched_segments,
203 stat_n_dom))
204 continue;
205 }
206
207 // Retrieve outgoing arcs for the current label.
208 const auto &node_arcs = buckets[bucket].template get_bucket_arcs<D>();
209 if (__builtin_expect(node_arcs.empty(), 0)) {
210 label->set_extended(true);
211 continue;
212 }
213
214 // Prefetch first few arcs
215 if (!node_arcs.empty()) {
216 for (size_t i = 0; i < std::min(size_t(4), node_arcs.size()); ++i) {
217 __builtin_prefetch(&node_arcs[i], 0, 3);
218 }
219 }
220
221 // Extend first, then process labels grouped by destination
222 // bucket. This keeps the destination bucket and its SoA cache
223 // hot across the whole group and amortizes staged-tier merges.
224 static thread_local std::vector<Label *> destination_batch;
225 destination_batch.clear();
226 destination_batch.reserve(node_arcs.size());
227 for (size_t arc_idx = 0; arc_idx < node_arcs.size(); ++arc_idx) {
228 const auto &arc = node_arcs[arc_idx];
229
230 // Prefetch next arc if available
231 if (arc_idx + 4 < node_arcs.size()) { __builtin_prefetch(&node_arcs[arc_idx + 4], 0, 3); }
232
233 auto new_label = Extend<D, S, ArcType::Bucket, Mutability::Mut, F>(label, arc);
234 if (new_label != nullptr) destination_batch.push_back(new_label);
235 }
236
237 std::stable_sort(destination_batch.begin(), destination_batch.end(), [](const Label *a, const Label *b) {
238 return a->vertex < b->vertex;
239 });
240
241 // Remove dominance among sibling extensions before scanning
242 // the committed destination bucket. Labels are grouped by
243 // destination, so each pair here is directly comparable.
244 for (size_t group_begin = 0; group_begin < destination_batch.size();) {
245 size_t group_end = group_begin + 1;
246 while (group_end < destination_batch.size() &&
247 destination_batch[group_end]->vertex == destination_batch[group_begin]->vertex) {
248 ++group_end;
249 }
250
251 for (size_t i = group_begin; i < group_end; ++i) {
252 Label *lhs = destination_batch[i];
253 if (lhs->is_dominated) continue;
254 for (size_t j = i + 1; j < group_end; ++j) {
255 Label *rhs = destination_batch[j];
256 if (rhs->is_dominated) continue;
257 if constexpr (S == Stage::One) {
258 if (lhs->cost <= rhs->cost) {
259 rhs->set_dominated(true);
260 } else {
261 lhs->set_dominated(true);
262 break;
263 }
264 } else {
265 if (is_dominated<D, S>(rhs, lhs)) rhs->set_dominated(true);
266 if (!rhs->is_dominated && is_dominated<D, S>(lhs, rhs)) {
267 lhs->set_dominated(true);
268 break;
269 }
270 }
271 }
272 }
273 group_begin = group_end;
274 }
275
276 for (Label *new_label : destination_batch) {
277 // Process each new label produced by the extension.
278 if (new_label != nullptr) {
279 if (new_label->is_dominated) continue;
280 if constexpr (S == Stage::Enumerate) {
281 if (n_labels >= enumeration_policy.max_labels_per_direction) {
282 enumeration_failed.store(true, std::memory_order_relaxed);
283 label->set_extended(true);
284 continue;
285 }
286 }
287 ++stat_n_labels;
288 const int to_bucket = new_label->vertex;
289 auto &mother_bucket = buckets[to_bucket];
290 uint64_t bucket_scan_count = 0;
291
292 if (profile_labeling) { profile_record_new_label(D, S); }
293
294 // Prefetch the mother bucket and its labels
295 __builtin_prefetch(&mother_bucket, 0, 3);
296
297 mother_bucket.flush_extra_labels_if_large();
298 const auto &to_bucket_labels = mother_bucket.get_sorted_labels();
299 const auto &to_bucket_extra = mother_bucket.get_extra_labels();
300 constexpr bool uses_visited_dominance =
301 S == Stage::Three || S == Stage::Four || S == Stage::Enumerate;
302 const uint64_t new_visited_signature = new_label->visited_signature();
303#if defined(BALDES_HAS_SIMD)
304 if constexpr (uses_visited_dominance) {
305 if (!to_bucket_labels.empty()) mother_bucket.ensure_label_cache();
306 }
307#endif
308
309 // Prefetch first few destination bucket labels
310 if (!to_bucket_labels.empty()) {
311 for (size_t i = 0; i < std::min(size_t(4), to_bucket_labels.size()); ++i) {
312 __builtin_prefetch(to_bucket_labels[i], 0, 3);
313 }
314 }
315
316 bool dominated = false;
317 if constexpr (S == Stage::One) {
318 // Stage One: mark higher-cost labels as
319 // dominated.
320 for (auto *existing_label : to_bucket_labels) {
321 ++bucket_scan_count;
322 if (existing_label->is_dominated) continue;
323 if (new_label->cost < existing_label->cost)
324 existing_label->set_dominated(true);
325 else {
326 dominated = true;
327 break;
328 }
329 }
330 if (!dominated) {
331 for (auto *existing_label : to_bucket_extra) {
332 ++bucket_scan_count;
333 if (existing_label->is_dominated) continue;
334 if (new_label->cost < existing_label->cost)
335 existing_label->set_dominated(true);
336 else {
337 dominated = true;
338 break;
339 }
340 }
341 }
342 if (profile_labeling) {
343 profile_record_dominance_check(D, S);
344 profile_record_inner_bin_scan(D, S, bucket_scan_count);
345 if (D == Direction::Forward) {
346 dominance_checks_per_bucket[to_bucket] += static_cast<int>(bucket_scan_count);
347 }
348 }
349 } else {
350 uint64_t inner_scan_count = 0;
351
352 // RC-bracketed single-pass dominance (RouteOpt
353 // dominance.hpp doDominance style). The committed
354 // tier is RC-sorted; for non-SRC stages, raw cost
355 // decides which dominance direction is possible:
356 // cur.cost < new.cost - eps: only cur can
357 // dominate new, so test resource/path dominance
358 // of new by cur; stop if true.
359 // cur.cost > new.cost + eps: only new can
360 // dominate cur, so test resource/path dominance
361 // of cur by new; mark cur dominated if true.
362 // near-equal costs: test both full predicates.
363 // SRC compensation can overturn raw cost order, so
364 // widen the bracket by a conservative bound: the
365 // sum of absolute active SRC duals. Outside this
366 // interval only one dominance direction is
367 // possible; inside it the full SRC predicate is
368 // evaluated in both directions.
369 const double cost_lo =
370 new_label->cost - src_compensation_bound - numericutils::eps;
371 const double cost_hi =
372 new_label->cost + src_compensation_bound + numericutils::eps;

606 if (is_dominated<D, S>(cur, new_label)) { cur->set_dominated(true); }
607 }
608 };
609
610 // Forward path: walk sorted committed labels once.
611 const size_t to_bucket_size = to_bucket_labels.size();
612 {
613 static constexpr size_t kLinearCostBracketScanLimit = 32;
614 if (to_bucket_size <= kLinearCostBracketScanLimit) {
615 for (size_t i = 0; i < to_bucket_size; ++i) {
616 if (i + 8 < to_bucket_size) {
617 __builtin_prefetch(to_bucket_labels[i + 8], 0, 3);
618 }
619 Label *cur = to_bucket_labels[i];
620 ++inner_scan_count;
621 if (__builtin_expect(cur->is_dominated, 0)) continue;
622 const double cur_cost = cur->cost;
623 if (cur_cost < cost_lo) {
624 if (dominates_resource_path<D, S>(new_label, cur)) {
625 ++stat_n_dom;
626 dominated = true;
627 break;
628 }
629 } else if (cur_cost > cost_hi) {
630 if (dominates_resource_path<D, S>(cur, new_label)) {
631 cur->set_dominated(true);
632 }
633 } else {
634 if (is_dominated<D, S>(new_label, cur)) {
635 ++stat_n_dom;
636 dominated = true;
637 break;
638 }
639 if (is_dominated<D, S>(cur, new_label)) { cur->set_dominated(true); }
640 }
641 }
642 } else {
643#if defined(BALDES_HAS_SIMD)
644 mother_bucket.ensure_label_cache();
645#endif
646 const auto bracket_begin_it = std::lower_bound(
647 to_bucket_labels.begin(), to_bucket_labels.end(), cost_lo, label_cost_less);
648 const auto expensive_begin_it = std::upper_bound(
649 to_bucket_labels.begin(), to_bucket_labels.end(), cost_hi,
650 [](double value, const Label *label) { return value < label->cost; });
651 const size_t bracket_begin =
652 static_cast<size_t>(std::distance(to_bucket_labels.begin(), bracket_begin_it));
653 const size_t expensive_begin = static_cast<size_t>(
654 std::distance(to_bucket_labels.begin(), expensive_begin_it));
655
656 scan_new_dominated_by(0, bracket_begin);
657 if (!dominated) { scan_bracket(bracket_begin, expensive_begin); }
658 if (!dominated) { scan_existing_dominated_by_new(expensive_begin, to_bucket_size); }
659 }
660 }
661
662 // Extra (unsorted) tier: same trichotomy without
663 // sort assumptions. Only walked if not dominated
664 // by the sorted tier.
665 if (!dominated) {
666 for (Label *cur : to_bucket_extra) {
667 ++inner_scan_count;
668 if (__builtin_expect(cur->is_dominated, 0)) continue;
669 const double cur_cost = cur->cost;
670 if (cur_cost < cost_lo) {
671 if (dominates_resource_path<D, S>(new_label, cur)) {
672 ++stat_n_dom;
673 dominated = true;
674 break;
675 }
676 } else if (cur_cost > cost_hi) {
677 if (dominates_resource_path<D, S>(cur, new_label)) { cur->set_dominated(true); }
678 } else {
679 if (is_dominated<D, S>(new_label, cur)) {
680 ++stat_n_dom;
681 dominated = true;
682 break;
683 }
684 if (is_dominated<D, S>(cur, new_label)) { cur->set_dominated(true); }
685 }
686 }
687 }
688 if (profile_labeling) {
689 profile_record_dominance_check(D, S);
690 profile_record_inner_bin_scan(D, S, inner_scan_count);
691 if (D == Direction::Forward) {
692 dominance_checks_per_bucket[to_bucket] += static_cast<int>(inner_scan_count);
693 }
694 }
695 }
696
697 if (!dominated) {
698 if (profile_labeling) profile_record_non_dominated_label(D, S);
699 if constexpr (D == Direction::Forward) {
700 ++non_dominated_labels_per_bucket;
701 } else {
702 ++non_dominated_labels_per_bucket_bw;
703 }
704 ++n_labels;
705#ifdef SORTED_LABELS
706 mother_bucket.add_sorted_label(new_label);
707#elif defined(LIMITED_BUCKETS)
708 mother_bucket.sorted_label(new_label, BUCKET_CAPACITY);
709#else
710 mother_bucket.add_label(new_label);
711#endif
712 const int to_bucket_pos = bucket_pos_in_scc[to_bucket];
713 if (to_bucket_pos != -1) {
714 pending_labels_by_bucket[to_bucket_pos].push_back(new_label);
715 activate_bucket(to_bucket_pos);
716 }
717 }
718 }
719 }
720 label->set_extended(true);
721 }
722
723 bucket_is_active[bucket_idx] = 0;
724 }
725
726 for (const int bucket : scc_buckets) {
727 bucket_pos_in_scc[bucket] = -1;
728 buckets[bucket].compact_dominated_labels();
729 }
730
731 // Update c_bar values for all buckets in the current SCC. The topology
732 // builder already stores these buckets in direction-specific resource
733 // order, so avoid copying and sorting them on every labeling pass.
734 const auto &sorted_buckets = sorted_sccs[scc_index];
735
736 // Prefetch data for c_bar updates
737 if (!sorted_buckets.empty()) {
738 for (size_t i = 0; i < std::min(size_t(4), sorted_buckets.size()); ++i) {
739 __builtin_prefetch(&buckets[sorted_buckets[i]], 0, 3);
740 __builtin_prefetch(&Phi[sorted_buckets[i]], 0, 3);
741 }
742 }
743
744 // Batch processing of c_bar updates for improved cache locality.
745 for (size_t i = 0; i < sorted_buckets.size(); ++i) {
746 const int bucket = sorted_buckets[i];
747
748 // Prefetch data for next iterations
749 if (i + 4 < sorted_buckets.size()) {
750 __builtin_prefetch(&buckets[sorted_buckets[i + 4]], 0, 3);
751 __builtin_prefetch(&Phi[sorted_buckets[i + 4]], 0, 3);
752 }
753
754 double min_c_bar = buckets[bucket].get_cb();
755 const auto &phi_buckets = Phi[bucket];
756
757 // Prefetch first few phi buckets
758 if (!phi_buckets.empty()) {
759 for (size_t j = 0; j < std::min(size_t(4), phi_buckets.size()); ++j) {
760 __builtin_prefetch(&c_bar[phi_buckets[j]], 0, 3);
761 }
762 }
763
764 // Process phi buckets in blocks of 4 for better cache efficiency
765 size_t j = 0;
766 for (; j + 3 < phi_buckets.size(); j += 4) {
767 // Prefetch next block
768 if (j + 8 < phi_buckets.size()) {
769 __builtin_prefetch(&c_bar[phi_buckets[j + 4]], 0, 3);
770 __builtin_prefetch(&c_bar[phi_buckets[j + 5]], 0, 3);
771 __builtin_prefetch(&c_bar[phi_buckets[j + 6]], 0, 3);
772 __builtin_prefetch(&c_bar[phi_buckets[j + 7]], 0, 3);
773 }
774
775 // Compute min across 4 values at once
776 double val0 = c_bar[phi_buckets[j]];
777 double val1 = c_bar[phi_buckets[j + 1]];
778 double val2 = c_bar[phi_buckets[j + 2]];
779 double val3 = c_bar[phi_buckets[j + 3]];
780
781 min_c_bar = std::min(min_c_bar, std::min(std::min(val0, val1), std::min(val2, val3)));
782 }
783
784 // Process remaining phi buckets
785 for (; j < phi_buckets.size(); ++j) { min_c_bar = std::min(min_c_bar, c_bar[phi_buckets[j]]); }
786
787 c_bar[bucket] = min_c_bar;
788 }
789 }
790
791 // Store the best label from the labeling process.
792 if (Label *best_label_run = get_best_label<D>(topological_order, c_bar, sccs)) best_label = best_label_run;
793
794 return c_bar;
795}

802inline auto BucketGraph::Extend(const std::conditional_t<M == Mutability::Mut, Label *, const Label *> L_prime,
803 const std::conditional_t<A == ArcType::Bucket, BucketArc,
804 std::conditional_t<A == ArcType::Jump, JumpArc, Arc>> &gamma,
805 int depth) noexcept {
806 // Pre-allocated result vector to avoid allocations for the common case of
807 // 0-1 results
808 static thread_local std::vector<Label *> result_vector;
809 Label *result = nullptr;
810
811 // Prepare temporary resource storage (resize to current resources size)
812 static thread_local std::vector<double> new_resources;
813 const size_t n_resources = options.resources.size();
814 new_resources.resize(n_resources);

816 // Cache initial values from L_prime
817 const int initial_node_id = L_prime->node_id;
818 const auto &initial_resources = L_prime->resources; // Use reference instead of copy

822 // as a marked BucketArc whose to_bucket is the jump bucket and whose
823 // jump_to_node is the head of the original arc.
824 int node_id;
825 int jump_bucket_id = -1;
826 bool is_jump_arc = false;
827 if constexpr (A == ArcType::Bucket) {
828 if (gamma.jump) {
829 if (gamma.jump_to_node < 0) {
830 if constexpr (F == Full::Reverse)
831 return -1;
832 else
833 return result;
834 }
835 node_id = gamma.jump_to_node;
836 jump_bucket_id = gamma.to_bucket;
837 is_jump_arc = true;
838 } else {
839 node_id = assign_buckets<D>(fw_buckets, bw_buckets)[gamma.to_bucket].node_id;
840 }
841 } else if constexpr (A == ArcType::Node) {
842 node_id = gamma.to;
843 } else { // Jump arc
844 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
845 node_id = gamma.to_job;
846 jump_bucket_id = gamma.jump_bucket;
847 is_jump_arc = true;
848 if (node_id < 0 || jump_bucket_id < 0 || jump_bucket_id >= static_cast<int>(buckets.size())) {
849 if constexpr (F == Full::Reverse)
850 return -1;
851 else
852 return result;
853 }
854 }
855
856 // Early rejection: if we loop back or the node was already visited
857 if (node_id == initial_node_id || is_node_visited(L_prime->visited_bitmap, node_id)) {
858 if constexpr (F == Full::Reverse)
859 return -1;
860 else
861 return result;
862 }
863
864 // For stages where fixed arcs matter
865 if constexpr (S >= Stage::Three || S == Stage::Eliminate) {
866 if ((D == Direction::Forward && is_arc_fixed(initial_node_id, node_id)) ||
867 (D == Direction::Backward && is_arc_fixed(node_id, initial_node_id))) {
868 if constexpr (F == Full::Reverse)
869 return -1;
870 else
871 return result;
872 }
873 }
874
875 // Process resources (feasibility check)
876 if constexpr (F != Full::TSP) {
877 auto process_jump_resources = [&]() -> bool {
878 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
879 if (jump_bucket_id < 0 || jump_bucket_id >= static_cast<int>(buckets.size())) return false;
880 const auto &jump_bucket = buckets[jump_bucket_id];
881
882 std::array<double, R_SIZE> jumped_resources = initial_resources;
883 for (size_t i = 0; i < n_resources; ++i) {
884 if constexpr (D == Direction::Forward) {
885 jumped_resources[i] = std::max(jumped_resources[i], static_cast<double>(jump_bucket.lb[i]));
886 } else {
887 jumped_resources[i] = std::min(jumped_resources[i], static_cast<double>(jump_bucket.ub[i]));
888 }
889 }
890 return process_all_resources<D>(new_resources, jumped_resources, gamma, nodes[node_id], n_resources);
891 };
892
893 if constexpr (A != ArcType::Jump) {
894 // Only copy resources for non-jump arcs (jump arcs already copied)
895 std::memcpy(new_resources.data(), initial_resources.data(), n_resources * sizeof(double));
896
897 const bool feasible = is_jump_arc ? process_jump_resources()
898 : process_all_resources<D>(new_resources, initial_resources, gamma,
899 nodes[node_id], n_resources);
900 if (!feasible) {
901 if constexpr (F == Full::Reverse)
902 return -1;
903 else
904 return result;
905 }
906 } else {
907 if (!process_jump_resources()) {
908 if constexpr (F == Full::Reverse)
909 return -1;
910 else
911 return result;
912 }
913 }
914 }
915
916 // For PSTEP/TSP: Check path length constraints
917 if constexpr (F == Full::PSTEP || F == Full::TSP) {
918 const int next_path_len = L_prime->path_len + 1;
919 if (next_path_len > options.max_path_size ||
920 (next_path_len == options.max_path_size && node_id != options.end_depot)) {
921 if constexpr (F == Full::Reverse)
922 return -1;
923 else
924 return result;
925 }
926 }
927
928 // Get bucket number for the new label
929 int to_bucket = -1;
930 if constexpr (A == ArcType::Bucket) {
931 to_bucket = get_bucket_number<D>(node_id, new_resources);
932 } else if constexpr (A == ArcType::Jump) {
933 to_bucket = get_bucket_number<D>(node_id, new_resources);
934 } else if constexpr (A == ArcType::Node) {
935 to_bucket = get_bucket_number<D>(node_id, new_resources);
936 }
937
938 const auto &direction_buckets = assign_buckets<D>(fw_buckets, bw_buckets);
939 if (to_bucket < 0 || to_bucket >= static_cast<int>(direction_buckets.size())) {
940 if constexpr (F == Full::Reverse)
941 return -1;
942 else
943 return result;
944 }
945
946 // Handle Reverse mode early
947 if constexpr (F == Full::Reverse) {
948 if (new_resources[options.main_resources[0]] <= std::floor(q_star[options.main_resources[0]])) return -1;
949
950 return to_bucket;
951 }
952
953 // Calculate new cost
954#ifdef CUSTOM_COST
955 const auto distance = getcij(initial_node_id, node_id);
956 double new_cost = cost_calculator.calculate_cost(initial_cost, distance, node_id, new_resources);
957#else
958 // Cache the distance value to avoid repeated calls
959 const auto distance = getcij(initial_node_id, node_id);
960 double new_cost = initial_cost + distance;
961#endif
962
963 const auto &VRPNode = nodes[node_id];
964
965 if constexpr (F != Full::PSTEP) {
966 new_cost -= VRPNode.cost;
967 } else {
968 if (L_prime->path_len > 1 && initial_node_id != options.depot) {
969 const auto three_two_dual = pstep_duals.getThreeTwoDualValue(initial_node_id);
970 new_cost += three_two_dual + pstep_duals.getThreeThreeDualValue(initial_node_id);
971 }
972 new_cost += -pstep_duals.getThreeTwoDualValue(node_id) + pstep_duals.getArcDualValue(initial_node_id, node_id);
973 }
974
975 // Apply branching dual adjustments if present - use local variables to
976 // avoid repeated pointer dereferencing
977 if (!branching_duals->empty()) {
978 if (D == Direction::Forward) {
979 new_cost += branching_duals->getDual(initial_node_id, node_id);
980 } else {
981 new_cost += branching_duals->getDual(node_id, initial_node_id);
982 }
983 new_cost += branching_duals->getDual(node_id);
984 }
985
986 RCC_MODE_BLOCK(if constexpr (S == Stage::Four) {
987 if (D == Direction::Forward) {
988 new_cost -= arc_duals.getDual(initial_node_id, node_id);
989 } else {
990 new_cost -= arc_duals.getDual(node_id, initial_node_id);
991 }
992 })
993
994 // Exact pricing can use the already-computed backward labels as a
995 // current-cost completion lower bound. Account conservatively for every
996 // possible future SRC discount before rejecting the extension. This runs
997 // before allocation and before bitmap/route/SRC state copies.
998 if constexpr (D == Direction::Forward && F == Full::Partial &&
999 (S == Stage::Four || S == Stage::Enumerate)) {
1000 if (forward_completion_bounds_ready) {
1001 const int opposite_bucket = get_opposite_bucket_number<D>(to_bucket, new_resources);
1002 if (opposite_bucket >= 0 && opposite_bucket < static_cast<int>(bw_c_bar.size())) {
1003 const double completion = bw_c_bar[opposite_bucket];
1004 if (std::isfinite(completion)) {
1005 completion_bound_attempts.fetch_add(1, std::memory_order_relaxed);
1006 const double cutoff = S == Stage::Enumerate
1007 ? std::min(gap, enumeration_route_cutoff.load(std::memory_order_relaxed))
1008 : 0.0;
1009 if (new_cost + completion - src_cost_compensation_bound() >= cutoff - numericutils::eps) {
1010 completion_bound_rejections.fetch_add(1, std::memory_order_relaxed);
1011 return result;
1012 }
1013 }
1014 }
1015 }
1016 }
1017
1018 // Create and initialize a new label
1019 auto &label_pool = assign_buckets<D>(label_pool_fw, label_pool_bw);
1020 auto new_label = label_pool->acquire(to_bucket);
1021 new_label->initialize(to_bucket, new_cost, new_resources, node_id);
1022 new_label->vertex = to_bucket;
1023 new_label->real_cost = L_prime->real_cost + distance; // Use cached distance
1024 new_label->path_len = L_prime->path_len + 1;
1025
1026 // Update visited bitmap: intersect parent's visited bits with neighborhood
1027 // mask
1028 if constexpr (F != Full::PSTEP) {
1029 for (size_t i = 0; i < new_label->visited_bitmap.size(); ++i) {
1030 const uint64_t current_visited = L_prime->visited_bitmap[i];
1031 if (current_visited) { new_label->visited_bitmap[i] = current_visited & neighborhoods_bitmap[node_id][i]; }
1032 }
1033 } else {
1034 // Use memcpy for direct bitmap copy (likely faster than assignment for
1035 // large bitmaps)
1036 std::memcpy(new_label->visited_bitmap.data(), L_prime->visited_bitmap.data(),
1037 new_label->visited_bitmap.size() * sizeof(uint64_t));
1038 }
1039
1040 set_node_visited(new_label->visited_bitmap, node_id);
1041
1042 // Reuse pooled vector capacity when extending the route.
1043 auto &extended_route = new_label->nodes_covered;
1044 extended_route.clear();
1045 extended_route.reserve(L_prime->nodes_covered.size() + 1);
1046 extended_route.insert(extended_route.end(), L_prime->nodes_covered.begin(), L_prime->nodes_covered.end());
1047 extended_route.push_back(node_id);
1048
1049#if defined(SRC)
1050 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
1051 auto &cutter = cut_storage;
1052 const auto n_cuts = cutter->activeSize();
1053 if (__builtin_expect(n_cuts == 0, 0)) { // Branch prediction: unlikely
1054 if constexpr (F == Full::Reverse) {
1055 return -1;
1056 } else {
1057 return new_label;
1058 }
1059 }
1060
1061 new_label->SRCmap = L_prime->SRCmap;
1062 const auto active_cuts = cutter->getActiveCuts();
1063 double total_cost_update = 0.0;
1064
1065 // Prefetch critical data structures
1066 __builtin_prefetch(active_cuts.data(), 0, 3);
1067 __builtin_prefetch(new_label->SRCmap.data(), 1, 3); // 1 = read-write
1068
1069#if !defined(SRC_MEMORY_MODE_ARC)
1070 auto &masks = cutter->getSegmentMasks();
1071 __builtin_prefetch(&masks, 0, 3);
1072 for (const auto &update : cutter->getSRCNodeUpdates(node_id)) {
1073 auto &src_map_value = new_label->SRCmap[update.active_idx];
1074 src_map_value += update.add;
1075 const bool overflow = src_map_value >= update.den;
1076 src_map_value -= overflow ? update.den : 0;
1077 total_cost_update -= overflow ? update.dual : 0;
1078 }
1079
1080 for (const auto active_idx : cutter->getSRCNodeClears(node_id)) { new_label->SRCmap[active_idx] = 0; }
1081#else
1082 for (const auto &active_cut : active_cuts) {
1083 const auto &cut = *active_cut.cut_ptr;
1084 auto &src_map_value = new_label->SRCmap[active_cut.index];
1085 if (!cut.isSRCMemoryArc(L_prime->node_id, node_id)) { src_map_value = 0; }
1086 if (cut.isSRCset(node_id)) {
1087 src_map_value += cut.srcMultiplier(node_id);
1088 if (src_map_value >= cut.p.den) {
1089 src_map_value -= cut.p.den;
1090 total_cost_update -= active_cut.dual_value;
1091 }
1092 }
1093 }
1094#endif
1095
1096 new_label->cost += total_cost_update;
1097 }
1098#endif
1099 if constexpr (F != Full::Reverse) {
1100 return new_label;
1101 } else {
1102 return -1;
1103 }

1110template <Direction D, Stage S>
1111inline bool BucketGraph::dominates_resource_path(const Label *__restrict new_label,
1112 const Label *__restrict label) noexcept {
1113 // Compare resource vectors.
1114 const auto *__restrict new_res = new_label->resources.data();
1115 const auto *__restrict lbl_res = label->resources.data();
1116
1117 // Prefetch resource data
1118 __builtin_prefetch(new_res, 0,
1119 3); // 0 = read only, 3 = high temporal locality
1120 __builtin_prefetch(lbl_res, 0, 3);
1121
1122 const size_t n_res = options.resources.size();
1123 if constexpr (D == Direction::Forward) {
1124 // In the Forward direction, each resource value of 'label' must
1125 // not exceed that of 'new_label'. Check resource 0 last because bucket
1126 // traversal already gives it partial ordering.
1127 for (size_t i = 1; i < n_res; ++i) {
1128 if (numericutils::gt(lbl_res[i], new_res[i])) { return false; }
1129 }
1130 if (n_res > 0 && numericutils::gt(lbl_res[0], new_res[0])) { return false; }
1131 } else if constexpr (D == Direction::Backward) {
1132 // In the Backward direction, each resource value of 'label'
1133 // must not be less than that of 'new_label'. Check resource 0 last.
1134 for (size_t i = 1; i < n_res; ++i) {
1135 if (numericutils::lt(lbl_res[i], new_res[i])) { return false; }
1136 }
1137 if (n_res > 0 && numericutils::lt(lbl_res[0], new_res[0])) { return false; }
1138 }
1139
1140 // visits of label.
1141 if constexpr (S == Stage::Three || S == Stage::Four || S == Stage::Enumerate) {
1142 const size_t n_bitmap = label->visited_bitmap.size();
1143
1144 // Prefetch bitmap data
1145 __builtin_prefetch(label->visited_bitmap.data(), 0, 3);
1146 __builtin_prefetch(new_label->visited_bitmap.data(), 0, 3);
1147
1148 for (size_t i = 0; i < n_bitmap; ++i) {
1149 // Every visited node in 'label' must also be visited in
1150 // 'new_label'.
1151 if ((label->visited_bitmap[i] & new_label->visited_bitmap[i]) != label->visited_bitmap[i]) { return false; }
1152 }
1153 }
1154 return true;
1155}

1157template <Direction D, Stage S>
1158inline bool BucketGraph::is_dominated(const Label *__restrict new_label, const Label *__restrict label) noexcept {
1159 // A label cannot dominate if its cost is higher.
1160 // const double cost_diff = label->cost - new_label->cost;
1161#if defined(SRC)
1162 if constexpr (!(S == Stage::Four || S == Stage::Enumerate)) {
1163 if (numericutils::gt(label->cost, new_label->cost)) { return false; }
1164 }
1165#else
1166 if (numericutils::gt(label->cost, new_label->cost)) { return false; }
1167#endif
1168
1169 if (!dominates_resource_path<D, S>(new_label, label)) { return false; }
1170
1171#ifdef SRC
1172 // For Stage Four or Enumerate, apply additional SRC-based cost
1173 // adjustments.
1174 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
1175 const double local_cost_diff = label->cost - new_label->cost;
1176 const auto *__restrict lbl_srcs = label->SRCmap.data();
1177 const auto *__restrict new_lbl_srcs = new_label->SRCmap.data();
1178
1179 // Prefetch SRCmap data
1180 __builtin_prefetch(lbl_srcs, 0, 3);
1181 __builtin_prefetch(new_lbl_srcs, 0, 3);
1182
1183 const auto &active_cuts = cut_storage->getActiveCuts();
1184 double dual_sum = 0.0;
1185 const size_t n = active_cuts.size();
1186 for (size_t i = 0; i < n; ++i) {
1187 const auto &cut = active_cuts[i];
1188 // Branch prediction hint: likely false.
1189 if (__builtin_expect(lbl_srcs[cut.index] > new_lbl_srcs[cut.index], 0)) { dual_sum += cut.dual_value; }
1190 }
1191 return !numericutils::gt(local_cost_diff, dual_sum);
1192 }

1265 // dominate L?". Per Sadykov & Vanderbeck (2021), this routine MUST NOT
1266 // mutate labels in compwise-smaller buckets — they belong to already-
1267 // processed (or concurrently processed) buckets, and marking them
1268 // dominated mid-walk silently drops queued pending labels. Cost is a
1269 // necessary condition for cur-dominates-L; SRC-aware stages widen the
1270 // sorted-bin stopping threshold by the safe compensation bound.

1277 const size_t size = view.labels.size();
1278 if (profile_labeling) profile_record_inner_bin_scan(D, S, size);
1279#if defined(BALDES_HAS_SIMD)
1280 {

1283 using mask_d = typename simd_d::mask_type;
1284
1285 constexpr size_t simd_width = simd_d::size();

1287 const simd_d eps(numericutils::eps);
1288
1289 size_t i = 0;
1290 for (; i + simd_width <= size; i += simd_width) {

1292 mask_d candidate = costs <= cost_limit;

1300 const simd_d res(view.resources[r].data() + i, stdx::element_aligned);
1301 candidate &= res <= (simd_d(label_res[r]) + eps);
1302 }

1305 candidate &= res0 <= (simd_d(label_res[0]) + eps);
1306 }
1307 } else {

1309 const simd_d res(view.resources[r].data() + i, stdx::element_aligned);
1310 candidate &= res >= (simd_d(label_res[r]) - eps);
1311 }

1314 candidate &= res0 >= (simd_d(label_res[0]) - eps);

1323 if (profile_labeling) profile_record_signature_rejection(D, S);

1333 }
1334 }

1343 resource_candidate = false;
1344 break;
1345 }
1346 }
1347 if (resource_candidate && n_res > 0 && numericutils::gt(view.resources[0][i], label_res[0])) {
1348 resource_candidate = false;
1349 }
1350 } else {
1351 for (size_t r = 1; r < n_res; ++r) {
1352 if (numericutils::lt(view.resources[r][i], label_res[r])) {
1353 resource_candidate = false;
1354 break;
1355 }
1356 }
1357 if (resource_candidate && n_res > 0 && numericutils::lt(view.resources[0][i], label_res[0])) {
1358 resource_candidate = false;
1359 }
1360 }
1361 if (!resource_candidate) continue;
1362
1363 if constexpr (uses_visited_dominance) {
1364 if ((view.visited_signatures[i] & ~label_visited_signature) != 0) {
1365 if (profile_labeling) profile_record_signature_rejection(D, S);

1367 }

1375 }

1377 return false;
1378 }
1379#endif

1381 // Bin is RC-sorted ascending; once cur is more expensive than L,
1382 // no later label can dominate L (cost ordering).

1386 if constexpr (D == Direction::Forward) {

1389 resource_candidate = false;
1390 break;
1391 }
1392 }
1393 if (resource_candidate && n_res > 0 && numericutils::gt(view.resources[0][i], label_res[0])) {
1394 resource_candidate = false;

1399 resource_candidate = false;

1404 resource_candidate = false;

1407 if (!resource_candidate) continue;
1408
1409 if constexpr (uses_visited_dominance) {

1426 auto inline_check_extra_dominance = [&](std::span<Label *const> labels, uint &n_dom) -> bool {
1427 if (profile_labeling) profile_record_inner_bin_scan(D, S, labels.size());
1428 for (Label *label : labels) {
1429 if (label->is_dominated) continue;
1430 // extra is unsorted -> no break, just skip.
1431 if (label->cost > bounded_cost_hi) continue;
1432 if (is_dominated<D, S>(L, label)) {
1433 ++n_dom;
1434 return true;
1435 }
1436 }
1437 return false;
1438 };
1439
1440 // Main DFS traversal loop
1441 while (!stack_buffer.empty()) {
1442 const int current_bucket = stack_buffer.back();
1443 stack_buffer.pop_back();
1444
1445 const size_t segment = static_cast<size_t>(current_bucket) >> 6;
1446 const uint64_t bit = bit_mask_lookup_ptr[current_bucket & 63];
1447
1448 if (unlikely(Bvisited[segment] & bit)) continue;
1449
1450 // Mark as visited
1451 if (Bvisited[segment] == 0) { touched_segments.push_back(static_cast<uint32_t>(segment)); }
1452 Bvisited[segment] |= bit;
1453
1454 // Fast path: if label cost is lower and bucket precedes L in the SCC
1455 // order, return early
1456 if (likely(label_cost < c_bar[current_bucket] && bucket_scc_rank[current_bucket] < label_rank)) return false;
1457
1458 // Whole-bucket prune: if no label in this bucket or any predecessor bin
1459 // at this node can dominate L in the second resource, skip it entirely.
1460 if (options.resources.size() > 1) {
1461 const auto &bucket = buckets[current_bucket];
1462 const int node_id = bucket.node_id;
1463 const int local_bucket = current_bucket - num_buckets_index[node_id];
1464 if (local_bucket >= 0 && local_bucket < num_buckets[node_id]) {
1465 const double label_rc2 = L->resources[1];
1466 if constexpr (D == Direction::Forward) {
1467 if (numericutils::gt(rc2_till_this_bin[node_id][local_bucket], label_rc2)) { continue; }
1468 } else {
1469 if (numericutils::lt(rc2_till_this_bin[node_id][local_bucket], label_rc2)) { continue; }
1470 }
1471 }
1472 }
1473
1474 // Skip dominance check for L's own bucket
1475 if (b_L != current_bucket) {
1476 const auto &mother_bucket = buckets[current_bucket];
1477 if (mother_bucket.check_dominance_soa(L, inline_check_dominance, inline_check_extra_dominance, stat_n_dom))
1478 return true;
1479 }
1480
1481 // Process neighbor buckets using pointer arithmetic and prefetching
1482 const auto &neighbors = Phi[current_bucket];
1483 const size_t n_neighbors = neighbors.size();
1484 const int *neighbor_ptr = neighbors.data();
1485 const int *neighbor_end = neighbor_ptr + n_neighbors;
1486 for (; neighbor_ptr != neighbor_end; ++neighbor_ptr) {
1487 const int b_prime = *neighbor_ptr;
1488 const auto &bucket = buckets[b_prime];
1489 const int node_id = bucket.node_id;
1490 const int local_b = b_prime - num_buckets_index[node_id];
1491 if (options.resources.size() > 1 && local_b >= 0 && local_b < num_buckets[node_id]) {
1492 const double label_rc2 = L->resources[1];
1493 if constexpr (D == Direction::Forward) {
1494 if (numericutils::gt(rc2_till_this_bin[node_id][local_b], label_rc2)) { continue; }
1495 } else {
1496 if (numericutils::lt(rc2_till_this_bin[node_id][local_b], label_rc2)) { continue; }
1497 }
1498 }
1499
1500 const size_t word_idx = static_cast<size_t>(b_prime) >> 6;
1501 const uint64_t mask = bit_mask_lookup_ptr[b_prime & 63];
1502 __builtin_prefetch(&Bvisited[word_idx], 0,
1503 1); // Prefetch Bvisited for this segment
1504 if (!(Bvisited[word_idx] & mask)) stack_buffer.push_back(b_prime);
1505 }
1506 }
1507
1508 return false;
1509}

1521template <Stage state, Full fullness>
1522void BucketGraph::run_labeling_algorithms(std::vector<double> &forward_cbar, std::vector<double> &backward_cbar) {
1523 if constexpr (state == Stage::Four || state == Stage::Enumerate) {
1524 ++completion_exact_calls;
1525 bool use_backward_first = false;
1526#if defined(BALDES_COMPLETION_SCHEDULE_BACKWARD_FIRST)
1527 use_backward_first = true;
1528#elif defined(BALDES_COMPLETION_SCHEDULE_PARALLEL)
1529 use_backward_first = false;
1530#else
1531 // AUTO starts parallel, probes backward-first every eight exact calls,
1532 // and periodically rechecks parallel even after choosing sequential.
1533 const uint64_t phase = completion_exact_calls % 8;
1534 use_backward_first = auto_prefer_backward_first ? phase != 0 : phase == 4;
1535#endif
1536
1537 if (use_backward_first) {
1538 completion_bound_attempts.store(0, std::memory_order_relaxed);
1539 completion_bound_rejections.store(0, std::memory_order_relaxed);
1540 const auto backward_start = std::chrono::steady_clock::now();
1541 forward_completion_bounds_ready = false;
1542 backward_cbar = labeling_algorithm<Direction::Backward, state, fullness>();
1543 const auto backward_end = std::chrono::steady_clock::now();
1544 bw_c_bar = backward_cbar;
1545 forward_completion_bounds_ready = true;
1546 forward_cbar = labeling_algorithm<Direction::Forward, state, fullness>();
1547 const auto forward_end = std::chrono::steady_clock::now();
1548 forward_completion_bounds_ready = false;
1549
1550 const double backward_ms =
1551 std::chrono::duration<double, std::milli>(backward_end - backward_start).count();
1552 const double forward_ms =
1553 std::chrono::duration<double, std::milli>(forward_end - backward_end).count();
1554 last_sequential_completion_ms = backward_ms + forward_ms;
1555
1556#if defined(BALDES_COMPLETION_SCHEDULE_AUTO)
1557 const double attempts =
1558 static_cast<double>(completion_bound_attempts.load(std::memory_order_relaxed));
1559 const double rejected =
1560 static_cast<double>(completion_bound_rejections.load(std::memory_order_relaxed));
1561 const double rejection_ratio = attempts > 0.0 ? rejected / attempts : 0.0;
1562 const double estimated_unpruned_forward_ms = forward_ms / std::max(0.10, 1.0 - rejection_ratio);
1563 const double estimated_parallel_ms = std::max(backward_ms, estimated_unpruned_forward_ms);
1564 auto_prefer_backward_first = last_sequential_completion_ms < 0.95 * estimated_parallel_ms;
1565#endif
1566 return;
1567 }
1568 }
1569
1570 forward_completion_bounds_ready = false;
1571 double forward_ms = 0.0;
1572 double backward_ms = 0.0;
1573 // Schedule the forward labeling algorithm task on the scheduler.
1574 auto forward_task = stdexec::schedule(bi_sched) | stdexec::then([&]() {
1575 const auto start = std::chrono::steady_clock::now();
1576 auto result = labeling_algorithm<Direction::Forward, state, fullness>();
1577 forward_ms = std::chrono::duration<double, std::milli>(std::chrono::steady_clock::now() - start).count();
1578 return result;
1579 });
1580
1581 // Schedule the backward labeling algorithm task on the scheduler.
1582 auto backward_task = stdexec::schedule(bi_sched) | stdexec::then([&]() {
1583 const auto start = std::chrono::steady_clock::now();
1584 auto result = labeling_algorithm<Direction::Backward, state, fullness>();
1585 backward_ms = std::chrono::duration<double, std::milli>(std::chrono::steady_clock::now() - start).count();
1586 return result;
1587 });
1588
1589 // Combine the forward and backward tasks.
1590 auto combined_work = stdexec::when_all(std::move(forward_task), std::move(backward_task)) |
1591 stdexec::then([&](auto forward_result, auto backward_result) {
1592 // Store the results into the provided vectors.
1593 forward_cbar = std::move(forward_result);
1594 backward_cbar = std::move(backward_result);
1595 });
1596
1597 // Wait for all tasks to complete.
1598 stdexec::sync_wait(std::move(combined_work));
1599 last_parallel_completion_ms = std::max(forward_ms, backward_ms);
1600#if defined(BALDES_COMPLETION_SCHEDULE_AUTO)
1601 if constexpr (state == Stage::Four || state == Stage::Enumerate) {
1602 if (completion_exact_calls % 8 == 0 && std::isfinite(last_sequential_completion_ms)) {
1603 auto_prefer_backward_first = last_sequential_completion_ms < 0.95 * last_parallel_completion_ms;
1604 }
1605 }
1606#endif
1607}

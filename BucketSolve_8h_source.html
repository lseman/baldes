145inline std::vector<Label *> BucketGraph::solveHeuristic() {
146 // Initialize the status as not optimal at the start
147 status = Status::NotOptimal;
148
149 updateSplit(); // Update the split values for the bucket graph
150
151 // Placeholder for the final paths (labels) and inner objective value
152 std::vector<Label *> paths;
153 double inner_obj;
154
156 // ADAPTIVE STAGE HANDLING
158 // Stage 1: Apply a light heuristic (Stage::One)
159 if (s1) {
160 stage = 1;
161 paths = bi_labeling_algorithm<Stage::One>(); // Solve the problem with Stage 1 heuristic
162 inner_obj = paths[0]->cost;
163
164 // Transition from Stage 1 to Stage 2 if the objective improves or after 10 iterations
165 if (inner_obj >= -5 || iter >= 10) {
166 s1 = false;
167 s2 = true; // Move to Stage 2
168 }
169 }
170 // Stage 2: Apply a more expensive pricing heuristic (Stage::Two)
171 else if (s2) {
172 s2 = true;
173 stage = 2;
174 paths = bi_labeling_algorithm<Stage::Two>(); // Solve the problem with Stage 2 heuristic
175 inner_obj = paths[0]->cost;
176
177 // Transition from Stage 2 to Stage 3 if the objective improves or after 800 iterations
178 if (inner_obj >= -100 || iter > 800) { status = Status::Optimal; }
179 }
180
181 iter++; // Increment the iteration counter
182
183 return paths; // Return the final paths after processing
184}

190template <Direction D, Stage S, Full F>
191std::vector<double> BucketGraph::labeling_algorithm() {
192 // Assign the correct direction buckets, ordered SCCs, and other related structures depending on the direction D
193 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
194 auto &ordered_sccs = assign_buckets<D>(fw_ordered_sccs, bw_ordered_sccs);
195 auto &topological_order = assign_buckets<D>(fw_topological_order, bw_topological_order);
196 auto &sccs = assign_buckets<D>(fw_sccs, bw_sccs);
197 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
198 auto &c_bar = assign_buckets<D>(fw_c_bar, bw_c_bar);
199 auto &fixed_buckets = assign_buckets<D>(fw_fixed_buckets, bw_fixed_buckets);
200 auto &n_labels = assign_buckets<D>(n_fw_labels, n_bw_labels);
201 auto &sorted_sccs = assign_buckets<D>(fw_sccs_sorted, bw_sccs_sorted);
202 auto &n_buckets = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
203 auto &stat_n_labels = assign_buckets<D>(stat_n_labels_fw, stat_n_labels_bw);
204 auto &stat_n_dom = assign_buckets<D>(stat_n_dom_fw, stat_n_dom_bw);
205
206 n_labels = 0;
207
208 // Pre-calculate n_segments and create Bvisited once
209 const size_t n_segments = (n_buckets + 63) / 64;
210 std::vector<uint64_t> Bvisited(n_segments);
211
212 // Process SCCs in topological order
213 for (const auto &scc_index : topological_order) {
214 bool all_ext;
215 do {
216 all_ext = true;
217
218 // Process buckets in the sorted order within each SCC
219 for (const auto bucket : sorted_sccs[scc_index]) {
220 const auto &bucket_labels = buckets[bucket].get_labels();
221
222 for (Label *label : bucket_labels) {
223 if (label->is_extended) continue;
224
225 // Early resource checks
226 if constexpr (F != Full::PSTEP && F != Full::TSP) {
227 if constexpr (F == Full::Partial) {
228 if constexpr (D == Direction::Forward) {
229 if (label->resources[options.main_resources[0]] > q_star[options.main_resources[0]]) {
230 label->set_extended(true);
231 continue;
232 }
233 } else if constexpr (D == Direction::Backward) {
234 if (label->resources[options.main_resources[0]] <= q_star[options.main_resources[0]]) {
235 label->set_extended(true);
236 continue;
237 }
238 }
239 }
240 }
241
242 // Clear visited buckets efficiently based on size
243 if (n_segments <= 8) {
244 for (size_t i = 0; i < n_segments; ++i) { Bvisited[i] = 0; }
245 } else {
246 std::memset(Bvisited.data(), 0, n_segments * sizeof(uint64_t));
247 }
248
249 // Check dominance in smaller buckets
250 bool domin_smaller = false;
251 if constexpr (F != Full::TSP) {
252 domin_smaller =
253 DominatedInCompWiseSmallerBuckets<D, S>(label, bucket, c_bar, Bvisited, ordered_sccs);
254 }
255
256 if (!domin_smaller) {
257 // Lambda for processing new labels - optimized for each stage
258 auto process_new_label = [&](Label *new_label) {
259 stat_n_labels++;
260
261 const int to_bucket = new_label->vertex;
262 auto &mother_bucket = buckets[to_bucket];
263 bool dominated = false;
264
265 const auto &to_bucket_labels = mother_bucket.get_labels();
266
267 if constexpr (F != Full::PSTEP && F != Full::TSP) {
268 if constexpr (S == Stage::Four) {
269 if constexpr (D == Direction::Forward) {
270 dominance_checks_per_bucket[to_bucket] += to_bucket_labels.size();
271 }
272 }
273 }
274
275 if constexpr (S == Stage::One) {
276 for (auto *existing_label : to_bucket_labels) {
277 if (label->cost < existing_label->cost) {
278 mother_bucket.remove_label(existing_label);
279 } else {
280 dominated = true;
281 break;
282 }
283 }
284 } else {
285 auto check_dominance_in_bucket = [&](const std::vector<Label *> &labels) {
286#ifndef __AVX2__
287 for (auto *existing_label : labels) {
288 if (is_dominated<D, S>(new_label, existing_label)) {
289 stat_n_dom++;
290 return true;
291 }
292 }
293 return false;
294#else
295 return check_dominance_against_vector<D, S>(new_label, labels, cut_storage,
296 options.resources.size());
297#endif
298 };
299
300 dominated =
301 mother_bucket.check_dominance(new_label, check_dominance_in_bucket, stat_n_dom);
302 }
303
304 if (!dominated) {
305 if constexpr (S != Stage::Enumerate) {
306 for (auto *existing_label : to_bucket_labels) {
307 if (is_dominated<D, S>(existing_label, new_label)) {
308 mother_bucket.remove_label(existing_label);
309 }
310 }
311 }
312
313 n_labels++;
314
315#ifdef SORTED_LABELS
316 mother_bucket.add_sorted_label(new_label);
317#elif LIMITED_BUCKETS
318 mother_bucket.sorted_label(new_label, BUCKET_CAPACITY);
319#else
320 mother_bucket.add_label(new_label);
321#endif
322 all_ext = false;
323 }
324 };
325
326 // Process arcs for the current label
327 const auto &arcs = nodes[label->node_id].get_arcs<D>(scc_index);
328 for (const auto &arc : arcs) {
329 auto new_labels = Extend<D, S, ArcType::Node, Mutability::Mut, F>(label, arc);
330 if (new_labels.empty()) {
331#ifdef UNREACHABLE_DOMINANCE
332 set_node_unreachable(label->unreachable_bitmap, arc.to);
333#endif
334 } else {
335 for (auto *new_label : new_labels) { process_new_label(new_label); }
336 }
337 }
338 }
339
340 label->set_extended(true);
341 }
342 }
343
344 // Update c_bar values for the current SCC
345 for (const int bucket : sorted_sccs[scc_index]) {
346 c_bar[bucket] = std::min(c_bar[bucket], buckets[bucket].get_cb());
347 for (const auto phi_bucket : Phi[bucket]) {
348 c_bar[bucket] = std::min(c_bar[bucket], c_bar[phi_bucket]);
349 }
350 }
351 } while (!all_ext);
352 }
353
354 // Get and store best label
355 Label *best_label = get_best_label<D>(topological_order, c_bar, sccs);
356 if constexpr (D == Direction::Forward) {
357 fw_best_label = best_label;
358 } else {
359 bw_best_label = best_label;
360 }
361
362 return c_bar;
363}

369template <Stage S, Symmetry SYM>
370std::vector<Label *> BucketGraph::bi_labeling_algorithm() {
371 // Stage-specific initializations
372 if constexpr (S == Stage::Three) {
373 heuristic_fixing<S>();
374 } else if constexpr (S == Stage::Four) {
375 if (first_reset) {
376 reset_fixed();
377 first_reset = false;
378 }
379#ifdef FIX_BUCKETS
380 if (options.bucket_fixing) { bucket_fixing<S>(); }
381#endif
382 }
383
384 reset_pool();
385 common_initialization();
386
387 std::vector<double> forward_cbar(fw_buckets.size());
388 std::vector<double> backward_cbar(bw_buckets.size());
389
390 if constexpr (SYM == Symmetry::Asymmetric) {
391 run_labeling_algorithms<S, Full::Partial>(forward_cbar, backward_cbar);
392 } else {
393 forward_cbar = labeling_algorithm<Direction::Forward, S, Full::Partial>();
394 }
395
396 auto best_label = label_pool_fw->acquire();
397 if constexpr (SYM == Symmetry::Asymmetric) {
398 if (check_feasibility(fw_best_label, bw_best_label)) {
399 best_label = compute_label<S>(fw_best_label, bw_best_label);
400 } else {
401 best_label->cost = 0.0;
402 best_label->real_cost = std::numeric_limits<double>::infinity();
403 best_label->nodes_covered.clear();
404 }
405 } else {
406 best_label->cost = best_label->real_cost;
407 best_label->real_cost = std::numeric_limits<double>::infinity();
408 best_label->nodes_covered.clear();
409 }
410 merged_labels.push_back(best_label);
411
412 if constexpr (S == Stage::Enumerate) { fmt::print("Labels generated, concatenating...\n"); }
413
414 const size_t n_segments = (fw_buckets_size + 63) / 64;
415 std::vector<uint64_t> Bvisited(n_segments);
416
417 for (int bucket = 0; bucket < fw_buckets_size; ++bucket) {
418 const auto &bucket_labels = fw_buckets[bucket].get_labels();
419
420 if constexpr (S == Stage::Four) { non_dominated_labels_per_bucket += bucket_labels.size(); }
421
422 for (const Label *L : bucket_labels) {
423 const auto &to_arcs = nodes[L->node_id].get_arcs<Direction::Forward>();
424
425 for (const auto &arc : to_arcs) {
426 const int to_node = arc.to;
427
428 if constexpr (S == Stage::Three || S == Stage::Eliminate) {
429 if (fixed_arcs[L->node_id][to_node] == 1) { continue; }
430 }
431
432 auto extended_labels =
433 Extend<Direction::Forward, S, ArcType::Node, Mutability::Const, Full::Reverse>(L, arc);
434 for (Label *L_prime : extended_labels) {
435 int bucket_to_process = L_prime->vertex; // Create mutable copy
436 std::memset(Bvisited.data(), 0, n_segments * sizeof(uint64_t));
437 ConcatenateLabel<S, SYM>(L, bucket_to_process, best_label, Bvisited);
438 }
439 }
440 }
441 }
442
443 pdqsort(merged_labels.begin(), merged_labels.end(),
444 [](const Label *a, const Label *b) { return a->cost < b->cost; });
445
446#ifdef SCHRODINGER
447 if (merged_labels.size() > N_ADD) {
448 std::vector<Path> paths;
449 const int labels_size = merged_labels.size();
450 const int end_idx = std::min(N_ADD + N_ADD, labels_size);
451
452 paths.reserve(end_idx - N_ADD);
453 for (int i = N_ADD; i < end_idx; ++i) {
454 if (merged_labels[i]->nodes_covered.size() <= 3) { continue; }
455 paths.emplace_back(merged_labels[i]->nodes_covered, merged_labels[i]->real_cost);
456 }
457
458 sPool.add_paths(paths);
459 sPool.iterate();
460 }
461#endif
462
463#ifdef RIH
464 std::vector<Label *> top_labels;
465 top_labels.reserve(N_ADD);
466
467 const int n_candidates = std::min(N_ADD, static_cast<int>(merged_labels.size()));
468 for (int i = 0; i < n_candidates; ++i) {
469 if (merged_labels[i]->nodes_covered.size() <= 3) { continue; }
470 top_labels.push_back(merged_labels[i]);
471 }
472 ils->submit_task(top_labels, nodes);
473#endif
474
475 inner_obj = merged_labels[0]->cost;
476 return merged_labels;
477}

486 const std::conditional_t<A == ArcType::Bucket, BucketArc,

492 auto &fixed_buckets = assign_buckets<D>(fw_fixed_buckets, bw_fixed_buckets);
493
494 // Precompute some values from the current label (L_prime) to avoid recalculating inside the loop
495 const int initial_node_id = L_prime->node_id;
496 auto initial_resources = L_prime->resources; // Copy the current label's resources
497 const double initial_cost = L_prime->cost; // Store the initial cost of the label
498
499 int node_id;
500 // Determine the target node based on the arc type (Bucket, Node, or Jump)
501 if constexpr (A == ArcType::Bucket) {
502 node_id = buckets[gamma.to_bucket].node_id; // Use the node ID from the bucket
503 } else if constexpr (A == ArcType::Node) {
504 node_id = gamma.to; // Use the node ID from the arc
505 } else if constexpr (A == ArcType::Jump) {
506 node_id = buckets[gamma.jump_bucket].node_id; // Use the node ID from the jump bucket
507 // Update resources based on jump bucket bounds
508 for (size_t i = 0; i < options.resources.size(); ++i) {
509 if constexpr (D == Direction::Forward) {
510 initial_resources[i] =

549 // counter the number of bits set in L_prime->visited_bitmap

580 // Compute travel cost between the initial and current nodes
581 const double travel_cost = getcij(initial_node_id, node_id);

603 RCC_MODE_BLOCK(if constexpr (S == Stage::Four) {

609 if (new_resources[options.main_resources[0]] <= q_star[options.main_resources[0]]) { return {}; }
610 auto new_label = label_pool->acquire();
611 new_label->vertex = to_bucket;

615 auto new_label = label_pool->acquire();
616 new_label->initialize(to_bucket, new_cost, new_resources, node_id);
617 new_label->vertex = to_bucket;
618
619 if constexpr (F == Full::PSTEP) {
620 new_label->visited_bitmap = L_prime->visited_bitmap;
621 set_node_visited(new_label->visited_bitmap, node_id);
622 }

626 new_label->unreachable_bitmap = L_prime->unreachable_bitmap;
627#endif
628

630 new_label->real_cost = L_prime->real_cost + travel_cost;
631
632 // Set the parent label, depending on mutability

634 new_label->parent = L_prime;
635 // L_prime->children.push_back(new_label);

639 if constexpr (S != Stage::Enumerate) {
640 for (size_t i = 0; i < new_label->visited_bitmap.size(); ++i) {
641 const uint64_t current_visited = L_prime->visited_bitmap[i];
642 if (current_visited) {
643 new_label->visited_bitmap[i] = current_visited & neighborhoods_bitmap[node_id][i];
644 }
645 }

647 } else {
648 new_label->visited_bitmap = L_prime->visited_bitmap;
649 }
650 set_node_visited(new_label->visited_bitmap, node_id); // Mark the new node as visited
651 // new_label->nodes_covered = L_prime->nodes_covered; // Copy the list of covered nodes
652 // new_label->nodes_covered.push_back(node_id); // Add the new node to the list of covered nodes
653
654#if defined(SRC)
655 new_label->SRCmap = L_prime->SRCmap;
656 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
657 auto &cutter = cut_storage;
658 std::span<const double> SRCDuals{cutter->SRCDuals};

667 static thread_local std::vector<size_t> indices;
668 indices.resize(cut_size);
669 std::iota(indices.begin(), indices.end(), 0);
670#endif
671
672 auto process_cut = [&](size_t idx) -> double {
673 if (SRCDuals[idx] > -1e-3) { return 0.0; }
674
675 const auto &cut = cutter->getCut(idx);
676 if (!(cut.neighbors[segment] & bit_mask)) {
677 new_label->SRCmap[idx] = 0.0;
678 return 0.0;
679 }
680
681 if (cut.baseSet[segment] & bit_mask) {
682 const auto &multipliers = cut.p;
683 const auto &den = multipliers.den;
684 auto &src_map_value = new_label->SRCmap[idx];
685 src_map_value += multipliers.num[cut.baseSetOrder[node_id]];
686 if (src_map_value >= den) {
687 src_map_value -= den;
688 return -SRCDuals[idx];
689 }
690 }
691 return 0.0;
692 };
693
694 double total_cost_update;
695 if (cut_size <= SMALL_SIZE) {
696 total_cost_update = 0.0;
697 for (size_t i = 0; i < cut_size; ++i) { total_cost_update += process_cut(i); }
698 } else {
699#if defined(__cpp_lib_parallel_algorithm)
700 total_cost_update = std::transform_reduce(std::execution::par_unseq, std::begin(indices), std::end(indices),
701 0.0, std::plus<>(), process_cut);
702#else
703 total_cost_update =
704 std::transform_reduce(std::begin(indices), std::end(indices), 0.0, std::plus<>(), process_cut);
705#endif
706 }
707
708 new_label->cost += total_cost_update;
709 }
710#endif
711
712 // Return the array (as Label**)
713 return std::vector<Label *>{new_label};
714}

720template <Direction D, Stage S>
721inline bool BucketGraph::is_dominated(const Label *new_label, const Label *label) noexcept {
722
723 // Extract resources for the new label and the comparison label
724 const auto &new_resources = new_label->resources;
725 const auto &label_resources = label->resources;
726
727 {
728 // Simple cost check: if the comparison label has a higher cost, it is not dominated
729 if (label->cost > new_label->cost) { return false; }
730 }
731
732 // Check resource conditions based on the direction (Forward or Backward)
733 for (size_t i = 0; i < options.resources.size(); ++i) {
734 if constexpr (D == Direction::Forward) {
735 // In Forward direction: the comparison label must not have more resources than the new label
736 if (label_resources[i] > new_resources[i]) { return false; }
737 } else if constexpr (D == Direction::Backward) {
738 // In Backward direction: the comparison label must not have fewer resources than the new label
739 if (label_resources[i] < new_resources[i]) { return false; }
740 }
741 }
742
743 // TODO:: check again this unreachable dominance
744#ifndef UNREACHABLE_DOMINANCE
745 // Check visited nodes (bitmap comparison) for Stages 3, 4, and Enumerate
746 if constexpr (S == Stage::Three || S == Stage::Four || S == Stage::Enumerate) {
747 // Iterate through the visited bitmap and ensure that the new label visits all nodes that the comparison
748 // label visits
749 for (size_t i = 0; i < label->visited_bitmap.size(); ++i) {
750 // If the comparison label visits a node that the new label does not, it is not dominated
751 if (((label->visited_bitmap[i] & new_label->visited_bitmap[i]) ^ label->visited_bitmap[i]) != 0) {
752 return false;
753 }
754 }
755 }
756#else
757 // Unreachable dominance logic: check visited and unreachable nodes in Stages 3, 4, and Enumerate
758 if constexpr (S == Stage::Three || S == Stage::Four || S == Stage::Enumerate) {
759 for (size_t i = 0; i < label->visited_bitmap.size(); ++i) {
760 // Combine visited and unreachable nodes in the comparison label's bitmap
761 auto combined_label_bitmap = label->visited_bitmap[i] | label->unreachable_bitmap[i];
762 // Ensure the new label visits all nodes that the comparison label (or its unreachable nodes) visits
763 if ((combined_label_bitmap & ~new_label->visited_bitmap[i]) != 0) { return false; }
764 }
765 }
766#endif
767
768#ifdef SRC
769 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
770 double sumSRC = 0;
771
772 const auto &SRCDuals = cut_storage->SRCDuals;
773 if (!SRCDuals.empty()) {
774 const auto &labelSRCMap = label->SRCmap;
775 const auto &newLabelSRCMap = new_label->SRCmap;
776 for (size_t i = 0; i < SRCDuals.size(); ++i) {
777 const auto &den = cut_storage->getCut(i).p.den;
778 const auto labelMod = labelSRCMap[i]; // % den;
779 const auto newLabelMod = newLabelSRCMap[i]; // % den;
780 if (labelMod > newLabelMod) { sumSRC += SRCDuals[i]; }
781 }
782 }
783 if (label->cost - sumSRC > new_label->cost) { return false; }
784 }
785#endif
786
787 // If all conditions are met, return true, indicating that the new label is dominated by the comparison label
788 return true;
789}

814template <Direction D, Stage S>
815inline bool BucketGraph::DominatedInCompWiseSmallerBuckets(
816 const Label *__restrict__ L, int bucket, const std::vector<double> &__restrict__ c_bar,
817 std::vector<uint64_t> &__restrict__ Bvisited,
818 const std::vector<std::vector<int>> &__restrict__ bucket_order) noexcept {
819
820 // Use references to avoid indirection
821 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
822 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
823 auto &uf = assign_buckets<D>(fw_union_find, bw_union_find);
824
825 // Cache frequently used values
826 const int b_L = L->vertex;
827 const double label_cost = L->cost;
828 const size_t res_size = options.resources.size();
829
830 // Pre-compute bit operation constants
831 constexpr uint64_t one = 1ULL;
832 constexpr uint64_t bit_mask_lookup[64] = {
833 1ULL << 0, 1ULL << 1, /* ... pre-compute all 64 values ... */
834 };
835
836 // Stack-based implementation with fixed size for small bucket counts
837 constexpr size_t MAX_STACK_SIZE = 128;
838 int stack_buffer[MAX_STACK_SIZE];
839 int stack_size = 1;
840 stack_buffer[0] = bucket;
841
842 // SIMD-optimized dominance check function
843 auto check_dominance = [&](const std::vector<Label *> &labels) {
844#ifdef __AVX2__
845 return check_dominance_against_vector<D, S>(L, labels, cut_storage, res_size);
846#else
847 const size_t n = labels.size();
848 for (size_t i = 0; i < n; ++i) {
849 if (is_dominated<D, S>(L, labels[i])) return true;
850 }
851 return false;
852#endif
853 };
854
855 do {
856 const int current_bucket = stack_buffer[--stack_size];
857
858 // Optimize bit operations
859 const size_t segment = current_bucket >> 6;
860 const uint64_t bit = bit_mask_lookup[current_bucket & 63];
861 Bvisited[segment] |= bit;
862
863 // Early exit check
864 if (__builtin_expect(
865 (label_cost < c_bar[current_bucket] && ::precedes<int>(bucket_order, current_bucket, b_L, uf)), 0)) {
866 return false;
867 }
868
869 // Check dominance only if necessary
870 if (b_L != current_bucket) {
871 auto &mother_bucket = buckets[current_bucket];
872 int stat_n_dom = 0;
873 if (mother_bucket.check_dominance(L, check_dominance, stat_n_dom)) { return true; }
874 }
875
876 // Process neighbors using SIMD if available
877 const auto &bucket_neighbors = Phi[current_bucket];
878 const size_t n_neighbors = bucket_neighbors.size();
879
880 // Regular processing
881 for (size_t i = 0; i < n_neighbors; ++i) {
882 const int b_prime = bucket_neighbors[i];
883 const size_t seg_prime = b_prime >> 6;
884 if (!(Bvisited[seg_prime] & bit_mask_lookup[b_prime & 63])) {
885 if (stack_size < MAX_STACK_SIZE) {
886 stack_buffer[stack_size++] = b_prime;
887 } else {
888 fmt::print("Stack overflow in DominatedInCompWiseSmallerBuckets\n");
889 return false;
890 }
891 }
892 }
893
894 } while (stack_size > 0);
895
896 return false;
897}

907template <Stage state, Full fullness>
908void BucketGraph::run_labeling_algorithms(std::vector<double> &forward_cbar, std::vector<double> &backward_cbar) {
909 // Create tasks for forward and backward labeling algorithms
910
911 auto forward_task = stdexec::schedule(bi_sched) |
912 stdexec::then([&]() { return labeling_algorithm<Direction::Forward, state, fullness>(); });
913
914 auto backward_task = stdexec::schedule(bi_sched) |
915 stdexec::then([&]() { return labeling_algorithm<Direction::Backward, state, fullness>(); });
916
917 // Execute the tasks in parallel and synchronize
918 auto work = stdexec::when_all(std::move(forward_task), std::move(backward_task)) |
919 stdexec::then([&](auto forward_result, auto backward_result) {
920 forward_cbar = std::move(forward_result);
921 backward_cbar = std::move(backward_result);
922 });
923
924 stdexec::sync_wait(std::move(work));
925}

931template <Stage S>
932Label *BucketGraph::compute_label(const Label *L, const Label *L_prime) {
933 double cij_cost = getcij(L->node_id, L_prime->node_id);
934 double new_cost = L->cost + L_prime->cost + cij_cost;
935 double real_cost = L->real_cost + L_prime->real_cost + cij_cost;
936
937 if constexpr (S == Stage::Four) {
938#if defined(RCC) || defined(EXACT_RCC)
939 auto arc_dual = arc_duals.getDual(L->node_id, L_prime->node_id);
940 new_cost -= arc_dual;
941#endif
942 }
943
944 // Branching duals
945 if (branching_duals->size() > 0) { new_cost -= branching_duals->getDual(L->node_id, L_prime->node_id); }
946
947 // Directly acquire new_label and set the cost
948 auto new_label = label_pool_fw->acquire();
949 new_label->cost = new_cost;
950 new_label->real_cost = real_cost;
951
952 if constexpr (S == Stage::Four) {
953 SRC_MODE_BLOCK(
954 // Check SRCDuals condition for specific stages
955 auto sumSRC = 0.0; const auto &SRCDuals = cut_storage->SRCDuals; if (!SRCDuals.empty()) {
956 size_t idx = 0;
957 auto sumSRC =
958 std::transform_reduce(SRCDuals.begin(), SRCDuals.end(), 0.0, std::plus<>(), [&](const auto &dual) {
959 size_t curr_idx = idx++;
960 auto den = cut_storage->getCut(curr_idx).p.den;
961 auto sum = (L->SRCmap[curr_idx] + L_prime->SRCmap[curr_idx]);
962 return (sum >= den) ? dual : 0.0;
963 });
964
965 new_label->cost -= sumSRC;
966 })
967 }
968
969 new_label->nodes_covered.clear();
970
971 // Start by inserting backward list elements
972 size_t forward_size = 0;
973 auto L_bw = L_prime;
974 for (; L_bw != nullptr; L_bw = L_bw->parent) {
975 new_label->nodes_covered.push_back(L_bw->node_id); // Insert backward elements directly
976 if (L_bw->parent == nullptr && L_bw->fresh == false) {
977 for (size_t i = 0; i < L_bw->nodes_covered.size(); ++i) {
978 new_label->nodes_covered.push_back(L_bw->nodes_covered[i]);
979 }
980 }
981 }
982
983 // Now insert forward list elements in reverse order without using std::reverse
984 auto L_fw = L;
985 for (; L_fw != nullptr; L_fw = L_fw->parent) {
986 new_label->nodes_covered.insert(new_label->nodes_covered.begin(),
987 L_fw->node_id); // Insert forward elements at the front
988 if (L_fw->parent == nullptr && L_fw->fresh == false) {
989 for (size_t i = 0; i < L_fw->nodes_covered.size(); ++i) {
990 new_label->nodes_covered.insert(new_label->nodes_covered.begin(), L_fw->nodes_covered[i]);
991 }
992 }
993 }
994
995 // new_label->nodes_covered = L_prime->nodes_covered;
996 // reverse the nodes_covered
997 // std::reverse(new_label->nodes_covered.begin(), new_label->nodes_covered.end());
998 // new_label->nodes_covered.insert(new_label->nodes_covered.begin(), L->nodes_covered.begin(),
999 // L->nodes_covered.end());
1000
1001 return new_label;
1002}

55inline std::vector<Label *> BucketGraph::solve(bool trigger) {
56 // Start with a nonoptimal status.
57 status = Status::NotOptimal;
58 if (trigger) {
59 transition = true;
60 fixed = false;
61 }
62
63 updateSplit(); // Update split values for the bucket graph.
64 std::vector<Label *> paths;
65
67 // Adaptive Stage Handling
69
70 // ----- Stage 1: Light Heuristic -----
71 if (s1 && depth == 0) {
72 stage = 1;
73 paths = bi_labeling_algorithm<Stage::One>();
74 // inner_obj = paths[0]->cost; // (if available)
75 if (inner_obj >= -1 || iter >= 10) {
76 s1 = false;
77 s2 = true; // Transition to Stage 2.
78 }
79 }
80 // ----- Stage 2: Expensive Pricing Heuristic -----
81 else if (s2 && depth == 0) {
82 stage = 2;
83 paths = bi_labeling_algorithm<Stage::Two>();
84 // inner_obj = paths[0]->cost; // (if available)
85 if (inner_obj >= -10 || iter > 500) {
86 s2 = false;
87 s3 = true; // Transition directly to Stage 4.
88 }
89 }
90 // ----- Stage 3: Heuristic Fixing Approach -----
91 else if (s3 && depth == 0) {
92 stage = 3;
93 paths = bi_labeling_algorithm<Stage::Three>();
94 // inner_obj = paths[0]->cost; // (if available)
95 if (inner_obj >= -0.5) {
96 s3 = false;
97 s4 = true;
98 transition = true; // Prepare for Stage 4 transition.
99 }
100 }
101 // ----- Stage 4: Exact Labeling with Fixing Enabled -----
102 else if (s4 && depth == 0) {
103 stage = 4;
104#ifdef FIX_BUCKETS
105 if (transition) {
106 // During a transition to Stage 4, fix buckets temporarily.
107 bool original_fixed = fixed;
108 fixed = true;
109 paths = bi_labeling_algorithm<Stage::Four>();
110 transition = false;
111 fixed = original_fixed;
112 min_red_cost = paths[0]->cost; // Update minimum reduced cost.
113 iter++;
114 return paths;
115 }
116#endif
117 // Standard Stage 4 processing.
118 paths = bi_labeling_algorithm<Stage::Four>();
119 // inner_obj = paths[0]->cost; // (if available)
120 bool rollback = false;
121 if (status != Status::Rollback) {
122 rollback = shallUpdateStep();
123 }
124 if (rollback) {
125 status = Status::Rollback;
126 return paths;
127 }
128 // Compute a new threshold for separation based on stats.
129 threshold = stats.computeThreshold(iter, inner_obj);
130 // If objective is sufficiently high, enter separation mode.
131 if (inner_obj > threshold) {
132 ss = true;
133#if !defined(SRC) && !defined(SRC3)
134 status = Status::Optimal;
135 return paths;
136#endif
137 status = Status::Separation;
138 }
139 }
140 // ----- Stage 5: Enumeration -----
141 else if (s5 && depth == 0) {
142 stage = 5;
143 print_info("Starting enumeration with gap {}\n", gap);
144 paths = bi_labeling_algorithm<Stage::Enumerate>();
145 print_info("Finished enumeration with {} paths\n", paths.size());
146 status = Status::Optimal;
147 }
148
149 // Increment the iteration counter.
150 iter++;
151 return paths;
152}

206std::vector<double> BucketGraph::labeling_algorithm() {
207 // Cache bucket and auxiliary structure references to avoid repeated
208 // lookups.
209 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
210 const auto &ordered_sccs =
211 assign_buckets<D>(fw_ordered_sccs, bw_ordered_sccs);
212 const auto &topological_order =
213 assign_buckets<D>(fw_topological_order, bw_topological_order);
214 const auto &sccs = assign_buckets<D>(fw_sccs, bw_sccs);
215 const auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
216 auto &c_bar = assign_buckets<D>(fw_c_bar, bw_c_bar);
217 auto &n_labels = assign_buckets<D>(n_fw_labels, n_bw_labels);
218 const auto &sorted_sccs = assign_buckets<D>(fw_sccs_sorted, bw_sccs_sorted);
219 const auto &n_buckets = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
220 auto &stat_n_labels = assign_buckets<D>(stat_n_labels_fw, stat_n_labels_bw);
221 auto &stat_n_dom = assign_buckets<D>(stat_n_dom_fw, stat_n_dom_bw);
222 auto &arc_scores = assign_buckets<D>(fw_arc_scores, bw_arc_scores);
223 auto &best_label = assign_buckets<D>(fw_best_label, bw_best_label);
224
225 // Reset the total label count.
226 n_labels = 0;
227
228 // Pre-calculate bitmap segments for dominance-checking.
229 const size_t n_segments = (n_buckets + 63) >> 6;
230 std::vector<uint64_t> Bvisited(n_segments, 0);
231
232 // Preallocate vector for extended labels to avoid repeated allocations.
233 std::vector<Label *> extended_labels;
234 extended_labels.reserve(64);
235
236 // Precompute main resource index and q_star value for partial solutions.
237 constexpr bool is_partial = (F == Full::Partial);
238 int main_res_index = 0;
239 double q_star_value = 0.0;
240 if constexpr (is_partial) {
241 main_res_index = options.main_resources[0];
242 q_star_value = std::floor(q_star[main_res_index]);
243 }
244
245 // Reuse a thread-local vector for collecting labels to dominate.
246 static thread_local std::vector<Label *> tl_labels_to_dominate;
247 tl_labels_to_dominate.clear();
248 tl_labels_to_dominate.reserve(
249 64); // Pre-reserve space to avoid reallocations
250
251 // Process each strongly connected component (SCC) in topological order.
252 for (const auto &scc_index : topological_order) {
253 // Prefetch the SCC data
254 __builtin_prefetch(&sorted_sccs[scc_index], 0, 3);
255
256 bool all_ext = false;
257 do {
258 all_ext = true; // Assume no new extensions until one occurs.
259 const auto &scc_buckets = sorted_sccs[scc_index];
260
261 // Prefetch the first few buckets
262 if (!scc_buckets.empty()) {
263 for (size_t i = 0; i < std::min(size_t(4), scc_buckets.size());
264 ++i) {
265 __builtin_prefetch(&buckets[scc_buckets[i]], 0, 3);
266 }
267 }
268
269 for (size_t bucket_idx = 0; bucket_idx < scc_buckets.size();
270 ++bucket_idx) {
271 const auto bucket = scc_buckets[bucket_idx];
272
273 // Prefetch next bucket if available
274 if (bucket_idx + 4 < scc_buckets.size()) {
275 __builtin_prefetch(&buckets[scc_buckets[bucket_idx + 4]], 0,
276 3);
277 }
278
279 const auto &bucket_labels = buckets[bucket].get_labels();
280
281 // Prefetch the first few labels
282 if (!bucket_labels.empty()) {
283 for (size_t i = 0;
284 i < std::min(size_t(4), bucket_labels.size()); ++i) {
285 __builtin_prefetch(bucket_labels[i], 0, 3);
286 }
287 }
288
289 for (size_t label_idx = 0; label_idx < bucket_labels.size();
290 ++label_idx) {
291 Label *label = bucket_labels[label_idx];
292
293 // Prefetch next label if available
294 if (label_idx + 4 < bucket_labels.size()) {
295 __builtin_prefetch(bucket_labels[label_idx + 4], 0, 3);
296 }
297
298 // Skip labels already processed.
299 if (__builtin_expect(
300 label->is_extended || label->is_dominated, 0))
301 continue;
302
303 // For partial solutions, check resource feasibility.
304 if constexpr (F != Full::PSTEP && F != Full::TSP) {
305 if constexpr (is_partial) {
306 const auto main_resource =
307 label->resources[main_res_index];
308 if ((D == Direction::Forward &&
309 main_resource > q_star_value) ||
310 (D == Direction::Backward &&
311 main_resource <= q_star_value)) {
312 label->set_extended(true);
313 continue;
314 }
315 }
316 }
317
318 // Clear the bitmap using memset (faster than std::fill for
319 // large arrays).
320 std::memset(Bvisited.data(), 0,
321 n_segments * sizeof(uint64_t));
322
323 // Check if the label is dominated by labels in smaller
324 // buckets.
325 if constexpr (F != Full::TSP) {
326 if (DominatedInCompWiseSmallerBuckets<D, S>(
327 label, bucket, c_bar, Bvisited, stat_n_dom))
328 continue;
329 }
330
331 // Retrieve outgoing arcs for the current label.
332 const auto &node_arcs =
333 buckets[bucket].template get_bucket_arcs<D>();
334 if (__builtin_expect(node_arcs.empty(), 0)) {
335 label->set_extended(true);
336 continue;
337 }
338
339 // Prefetch first few arcs
340 if (!node_arcs.empty()) {
341 for (size_t i = 0;
342 i < std::min(size_t(4), node_arcs.size()); ++i) {
343 __builtin_prefetch(&node_arcs[i], 0, 3);
344 }
345 }
346
347 for (size_t arc_idx = 0; arc_idx < node_arcs.size();
348 ++arc_idx) {
349 const auto &arc = node_arcs[arc_idx];
350
351 // Prefetch next arc if available
352 if (arc_idx + 4 < node_arcs.size()) {
353 __builtin_prefetch(&node_arcs[arc_idx + 4], 0, 3);
354 }
355
356 auto new_label =
357 Extend<D, S, ArcType::Bucket, Mutability::Mut, F>(
358 label, arc);
359
360 // Process each new label produced by the extension.
361 if (new_label != nullptr) {
362 ++stat_n_labels;
363 const int to_bucket = new_label->vertex;
364 auto &mother_bucket = buckets[to_bucket];
365
366 // Prefetch the mother bucket and its labels
367 __builtin_prefetch(&mother_bucket, 0, 3);
368
369 const auto &to_bucket_labels =
370 mother_bucket.get_labels();
371
372 // Prefetch first few destination bucket labels
373 if (!to_bucket_labels.empty()) {
374 for (size_t i = 0;
375 i < std::min(size_t(4),
376 to_bucket_labels.size());
377 ++i) {
378 __builtin_prefetch(to_bucket_labels[i], 0,
379 3);
380 }
381 }
382
383 bool dominated = false;
384 if constexpr (S == Stage::One) {
385 // Stage One: mark higher-cost labels as
386 // dominated.
387 for (auto *existing_label : to_bucket_labels) {
388 if (existing_label->is_dominated) continue;
389 if (label->cost < existing_label->cost)
390 existing_label->set_dominated(true);
391 else {
392 dominated = true;
393 break;
394 }
395 }
396 } else {
397 // Later stages: use a SIMD-friendly, unrolled
398 // loop.
399 auto check_dominance_in_bucket =
400 [&](const std::pmr::vector<Label *> &labels,
401 uint &stat_n_dom) -> bool {
402#ifdef __AVX2__
403 if (labels.size() >= AVX_LIM)
404 return check_dominance_against_vector<
405 D, S>(new_label, labels,
406 cut_storage,
407 options.resources.size(),
408 stat_n_dom);
409 else
410#endif
411 {
412 const size_t size = labels.size();
413 size_t i = 0;
414
415 // Process one label at a time but with
416 // prefetching ahead
417 for (; i < size; ++i) {
418 // Prefetch several labels ahead for
419 // better memory access patterns
420 if (i + 8 < size) {
421 __builtin_prefetch(
422 labels[i + 8], 0, 3);
423 }
424
425 // Process one label with branch
426 // prediction hint
427 Label *current_label = labels[i];
428 if (__builtin_expect(
429 !current_label
430 ->is_dominated,
431 1)) {
432 // Cache access to current_label
433 // to avoid multiple
434 // dereferences
435 if (is_dominated<D, S>(
436 new_label,
437 current_label)) {
438 ++stat_n_dom;
439 return true;
440 }
441 }
442 }
443 return false;
444 }
445 };
446 dominated = mother_bucket.check_dominance(
447 new_label, check_dominance_in_bucket,
448 stat_n_dom);
449 }
450
451 if (!dominated) {
452 // Reuse thread-local vector to avoid
453 // re-allocation.
454 tl_labels_to_dominate.clear();
455
456 const size_t to_bucket_size =
457 to_bucket_labels.size();
458 size_t i = 0;
459
460 // Process dominance checks in blocks for better
461 // cache usage
462 for (; i + 3 < to_bucket_size; i += 4) {
463 // Prefetch ahead
464 if (i + 8 < to_bucket_size) {
465 __builtin_prefetch(
466 to_bucket_labels[i + 4], 0, 3);
467 __builtin_prefetch(
468 to_bucket_labels[i + 5], 0, 3);
469 __builtin_prefetch(
470 to_bucket_labels[i + 6], 0, 3);
471 __builtin_prefetch(
472 to_bucket_labels[i + 7], 0, 3);
473 }
474
475 auto *l0 = to_bucket_labels[i];
476 auto *l1 = to_bucket_labels[i + 1];
477 auto *l2 = to_bucket_labels[i + 2];
478 auto *l3 = to_bucket_labels[i + 3];
479
480 if (!l0->is_dominated &&
481 is_dominated<D, S>(l0, new_label))
482 tl_labels_to_dominate.push_back(l0);
483 if (!l1->is_dominated &&
484 is_dominated<D, S>(l1, new_label))
485 tl_labels_to_dominate.push_back(l1);
486 if (!l2->is_dominated &&
487 is_dominated<D, S>(l2, new_label))
488 tl_labels_to_dominate.push_back(l2);
489 if (!l3->is_dominated &&
490 is_dominated<D, S>(l3, new_label))
491 tl_labels_to_dominate.push_back(l3);
492 }
493
494 // Process remaining labels
495 for (; i < to_bucket_size; ++i) {
496 auto *existing_label = to_bucket_labels[i];
497 if (!existing_label->is_dominated &&
498 is_dominated<D, S>(existing_label,
499 new_label))
500 tl_labels_to_dominate.push_back(
501 existing_label);
502 }
503
504 // Mark dominated labels in batch
505 for (auto *existing_label :
506 tl_labels_to_dominate)
507 existing_label->set_dominated(true);
508
509 ++n_labels;
510#ifdef SORTED_LABELS
511 mother_bucket.add_sorted_label(new_label);
512#elif defined(LIMITED_BUCKETS)
513 mother_bucket.sorted_label(new_label,
514 BUCKET_CAPACITY);
515#else
516 mother_bucket.add_label(new_label);
517#endif
518 // Mark that a new extension was added.
519 all_ext = false;
520 }
521 }
522 }
523 label->set_extended(true);
524 }
525 }
526 } while (!all_ext);
527
528 // Update c_bar values for all buckets in the current SCC.
529 auto sorted_buckets = sorted_sccs[scc_index];
530 if constexpr (D == Direction::Forward) {
531 pdqsort(sorted_buckets.begin(), sorted_buckets.end(),
532 [&](int a, int b) {
533 return buckets[a].get_lb() < buckets[b].get_lb();
534 });
535 } else {
536 pdqsort(sorted_buckets.begin(), sorted_buckets.end(),
537 [&](int a, int b) {
538 return buckets[a].get_ub() > buckets[b].get_ub();
539 });
540 }
541
542 // Prefetch data for c_bar updates
543 if (!sorted_buckets.empty()) {
544 for (size_t i = 0; i < std::min(size_t(4), sorted_buckets.size());
545 ++i) {
546 __builtin_prefetch(&buckets[sorted_buckets[i]], 0, 3);
547 __builtin_prefetch(&Phi[sorted_buckets[i]], 0, 3);
548 }
549 }
550
551 // Batch processing of c_bar updates for improved cache locality.
552 for (size_t i = 0; i < sorted_buckets.size(); ++i) {
553 const int bucket = sorted_buckets[i];
554
555 // Prefetch data for next iterations
556 if (i + 4 < sorted_buckets.size()) {
557 __builtin_prefetch(&buckets[sorted_buckets[i + 4]], 0, 3);
558 __builtin_prefetch(&Phi[sorted_buckets[i + 4]], 0, 3);
559 }
560
561 double min_c_bar = buckets[bucket].get_cb();
562 const auto &phi_buckets = Phi[bucket];
563
564 // Prefetch first few phi buckets
565 if (!phi_buckets.empty()) {
566 for (size_t j = 0; j < std::min(size_t(4), phi_buckets.size());
567 ++j) {
568 __builtin_prefetch(&c_bar[phi_buckets[j]], 0, 3);
569 }
570 }
571
572 // Process phi buckets in blocks of 4 for better cache efficiency
573 size_t j = 0;
574 for (; j + 3 < phi_buckets.size(); j += 4) {
575 // Prefetch next block
576 if (j + 8 < phi_buckets.size()) {
577 __builtin_prefetch(&c_bar[phi_buckets[j + 4]], 0, 3);
578 __builtin_prefetch(&c_bar[phi_buckets[j + 5]], 0, 3);
579 __builtin_prefetch(&c_bar[phi_buckets[j + 6]], 0, 3);
580 __builtin_prefetch(&c_bar[phi_buckets[j + 7]], 0, 3);
581 }
582
583 // Compute min across 4 values at once
584 double val0 = c_bar[phi_buckets[j]];
585 double val1 = c_bar[phi_buckets[j + 1]];
586 double val2 = c_bar[phi_buckets[j + 2]];
587 double val3 = c_bar[phi_buckets[j + 3]];
588
589 min_c_bar = std::min(min_c_bar, std::min(std::min(val0, val1),
590 std::min(val2, val3)));
591 }
592
593 // Process remaining phi buckets
594 for (; j < phi_buckets.size(); ++j) {
595 min_c_bar = std::min(min_c_bar, c_bar[phi_buckets[j]]);
596 }
597
598 c_bar[bucket] = min_c_bar;
599 }
600 }
601
602 // Store the best label from the labeling process.
603 if (Label *best_label_run =
604 get_best_label<D>(topological_order, c_bar, sccs))
605 best_label = best_label_run;
606
607 return c_bar;
608}

614template <Stage S, Symmetry SYM>
615std::vector<Label *> BucketGraph::bi_labeling_algorithm() {
616 // === Stage-specific Initializations ===
617 if constexpr (S == Stage::Three) {
618 heuristic_fixing<S>();
619 } else if constexpr (S == Stage::Four) {
620 if (first_reset) {
621 reset_fixed();
622 first_reset = false;
623 }
624#ifdef FIX_BUCKETS
625 if (options.bucket_fixing) {
626 // heuristic_fixing<S>();
627
628 bucket_fixing<S>();
629 }
630#endif
631 // heuristic_fixing<S>();
632 }
633
634 // Reset pools and perform common initialization.
635 reset_pool();
636 common_initialization();
637
638 // Pre-allocate c-bar vectors for forward and backward buckets.
639 std::vector<double> forward_cbar(fw_buckets.size());
640 std::vector<double> backward_cbar(bw_buckets.size());
641
642 // Run the labeling algorithms.
643 if constexpr (SYM == Symmetry::Asymmetric) {
644 run_labeling_algorithms<S, Full::Partial>(forward_cbar, backward_cbar);
645 } else {
646 forward_cbar =
647 labeling_algorithm<Direction::Forward, S, Full::Partial>();
648 }
649
650 // Acquire an initial best label from the forward label pool.
651 auto best_label = label_pool_fw->acquire();
652 if constexpr (SYM == Symmetry::Asymmetric) {
653 // Compute the best label if the forward and backward labels are
654 // feasible.
655 if (check_feasibility(fw_best_label, bw_best_label)) {
656 best_label = compute_label<S>(fw_best_label, bw_best_label);
657 } else {
658 best_label->cost = 0.0;
659 best_label->real_cost = std::numeric_limits<double>::infinity();
660 best_label->nodes_covered.clear();
661 }
662 } else {
663 best_label->cost = best_label->real_cost;
664 best_label->real_cost = std::numeric_limits<double>::infinity();
665 best_label->nodes_covered.clear();
666 }
667 merged_labels.push_back(best_label);
668
669 // Declare atomics to track the best cost and non-dominated label
670 // count.
671 std::atomic<double> best_cost{0.0};
672 std::atomic<size_t> non_dominated_labels_per_bucket{0};
673
674 // === Parallel Bucket Processing ===
675 // Choose a chunk size based on hardware concurrency.
676 const size_t chunk_size =
677 fw_buckets_size / MERGE_SCHED_CONCURRENCY; // 1/4 of buckets per core
678 const size_t n_chunks = (fw_buckets_size + chunk_size - 1) / chunk_size;
679
680 auto bulk_sender = stdexec::bulk(
681 stdexec::just(), n_chunks,
682 [this, chunk_size, &best_cost,
683 &non_dominated_labels_per_bucket](std::size_t chunk_idx) {
684 const size_t start_bucket = chunk_idx * chunk_size;
685 const size_t end_bucket =
686 std::min(start_bucket + chunk_size,
687 static_cast<size_t>(fw_buckets_size));
688
689 // Process each bucket within the current chunk.
690 for (size_t bucket = start_bucket; bucket < end_bucket; ++bucket) {
691 // Process non-dominated labels in the current bucket.
692 const auto &bucket_labels = fw_buckets[bucket].get_labels();
693 for (const Label *L : bucket_labels) {
694 if (L->is_dominated) continue;
695
696 non_dominated_labels_per_bucket.fetch_add(
697 1, std::memory_order_relaxed);
698
699 // Retrieve the outgoing arcs for the current label.
700 // const auto &to_arcs =
701 // nodes[L->node_id].get_arcs<Direction::Forward>();
702 const auto &to_arcs =
703 fw_buckets[bucket]
704 .template get_bucket_arcs<Direction::Forward>();
705 for (const auto &arc : to_arcs) {
706 const int to_node = fw_buckets[arc.to_bucket].node_id;
707 if constexpr (S >= Stage::Three ||
708 S == Stage::Eliminate) {
709 if (is_arc_fixed(L->node_id, to_node)) continue;
710 }
711
712 // Extend the label along the current arc.
713 auto L_prime =
714 Extend<Direction::Forward, S, ArcType::Bucket,
715 Mutability::Const, Full::Reverse>(L, arc);
716
717 // Process each extended label.
718 if (L_prime != -1) {
719 int bucket_to_process = L_prime;
720
721 // Concatenate the label with bucket
722 // information, updating best_cost.
723 ConcatenateLabel<S, SYM>(L, bucket_to_process,
724 best_cost);
725 }
726 }
727 }
728 }
729 });
730
731 // Submit the bulk work to the merge scheduler and wait for
732 // completion.
733 auto work = stdexec::starts_on(merge_sched, bulk_sender);
734 stdexec::sync_wait(std::move(work));
735
736 // === Post-processing and Sorting ===
737 // Sort merged labels based on cost (ascending order).
738 pdqsort(merged_labels.begin(), merged_labels.end(),
739 [](const Label *a, const Label *b) { return a->cost < b->cost; });
740
741#ifdef SCHRODINGER
742 // If using the Schrdinger approach, process additional paths.
743 if (merged_labels.size() > N_ADD) {
744 std::vector<Path> paths;
745 const int labels_size = merged_labels.size();
746 const int end_idx = std::min(N_ADD + N_ADD, labels_size);
747 paths.reserve(end_idx - N_ADD);
748 for (int i = N_ADD; i < end_idx; ++i) {
749 if (merged_labels[i]->nodes_covered.size() <= 3) continue;
750 paths.emplace_back(merged_labels[i]->nodes_covered,
751 merged_labels[i]->real_cost);
752 }
753 sPool.add_paths(paths);
754 sPool.iterate();
755 }
756#endif
757
758#ifdef RIH
759 // If using RIH stage modifications, submit the top labels to ILS.
760 if constexpr (S == Stage::Four) {
761 std::vector<Label *> top_labels;
762 top_labels.reserve(N_ADD);
763 const int n_candidates =
764 std::min(N_ADD, static_cast<int>(merged_labels.size()));
765 for (int i = 0; i < n_candidates; ++i) {
766 if (merged_labels[i]->nodes_covered.size() <= 3) continue;
767 top_labels.push_back(merged_labels[i]);
768 }
769 ils->submit_task(top_labels, nodes);
770 }
771#endif
772
773 // Store the cost of the best (first) merged label as the inner
774 // objective.
775 inner_obj = merged_labels[0]->cost;
776 return merged_labels;
777}

783template <Direction D, Stage S, ArcType A, Mutability M, Full F>
784inline auto BucketGraph::Extend(
785 const std::conditional_t<M == Mutability::Mut, Label *, const Label *>
786 L_prime,
787 const std::conditional_t<
788 A == ArcType::Bucket, BucketArc,
789 std::conditional_t<A == ArcType::Jump, JumpArc, Arc>> &gamma,
790 int depth) noexcept {
791 // Pre-allocated result vector to avoid allocations for the common case of
792 // 0-1 results
793 static thread_local std::vector<Label *> result_vector;
794 Label *result = nullptr;
795
796 // Prepare temporary resource storage (resize to current resources size)
797 static thread_local std::vector<double> new_resources;
798 const size_t n_resources = options.resources.size();
799 new_resources.resize(n_resources);
800
801 // Cache initial values from L_prime
802 const int initial_node_id = L_prime->node_id;
803 const auto &initial_resources =
804 L_prime->resources; // Use reference instead of copy
805 const double initial_cost = L_prime->cost;
806
807 // Determine target node ID based on arc type
808 int node_id;
809 if constexpr (A == ArcType::Bucket) {
810 node_id =
811 assign_buckets<D>(fw_buckets, bw_buckets)[gamma.to_bucket].node_id;
812 } else if constexpr (A == ArcType::Node) {
813 node_id = gamma.to;
814 } else { // Jump arc
815 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
816 node_id = buckets[gamma.jump_bucket].node_id;
817 const auto &bucket = buckets[gamma.jump_bucket];
818
819 // Fast copy the resources
820 std::memcpy(new_resources.data(), initial_resources.data(),
821 n_resources * sizeof(double));
822
823 // Update resource bounds based on jump bucket properties
824 for (size_t i = 0; i < n_resources; ++i) {
825 if constexpr (D == Direction::Forward) {
826 new_resources[i] = std::max(new_resources[i],
827 static_cast<double>(bucket.lb[i]));
828 } else {
829 new_resources[i] = std::min(new_resources[i],
830 static_cast<double>(bucket.ub[i]));
831 }
832 }
833 }
834
835 // Early rejection: if we loop back or the node was already visited
836 if (node_id == initial_node_id ||
837 is_node_visited(L_prime->visited_bitmap, node_id)) {
838 if constexpr (F == Full::Reverse)
839 return -1;
840 else
841 return result;
842 }
843
844 // For stages where fixed arcs matter
845 if constexpr (S >= Stage::Three || S == Stage::Eliminate) {
846 if ((D == Direction::Forward &&
847 is_arc_fixed(initial_node_id, node_id)) ||
848 (D == Direction::Backward &&
849 is_arc_fixed(node_id, initial_node_id))) {
850 if constexpr (F == Full::Reverse)
851 return -1;
852 else
853 return result;
854 }
855 }
856
857 // Process resources (feasibility check)
858 if constexpr (F != Full::TSP) {
859 if constexpr (A != ArcType::Jump) {
860 // Only copy resources for non-jump arcs (jump arcs already copied)
861 std::memcpy(new_resources.data(), initial_resources.data(),
862 n_resources * sizeof(double));
863
864 if (!process_all_resources<D>(new_resources, initial_resources,
865 gamma, nodes[node_id], n_resources)) {
866 if constexpr (F == Full::Reverse)
867 return -1;
868 else
869 return result;
870 }
871 } else {
872 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
873 const auto &bucket = buckets[gamma.jump_bucket];
874 for (size_t i = 0; i < n_resources; ++i) {
875 if constexpr (D == Direction::Forward) {
876 new_resources[i] =
877 std::max(new_resources[i] + gamma.resource_increment[i],
878 static_cast<double>(bucket.lb[i]));
879 if (numericutils::gt(new_resources[i], bucket.ub[i])) {
880 if constexpr (F == Full::Reverse)
881 return -1;
882 else
883 return nullptr;
884 }
885
886 } else {
887 new_resources[i] =
888 std::min(new_resources[i] - gamma.resource_increment[i],
889 static_cast<double>(bucket.ub[i]));
890 if (numericutils::lt(new_resources[i], bucket.lb[i])) {
891 if constexpr (F == Full::Reverse)
892 return -1;
893 else
894 return nullptr;
895 }
896 }
897 }
898 }
899 }
900
901 // For PSTEP/TSP: Check path length constraints
902 if constexpr (F == Full::PSTEP || F == Full::TSP) {
903 int n_visited = 0;
904 for (const auto &bitmap : L_prime->visited_bitmap)
905 n_visited += __builtin_popcountll(bitmap);
906 if (n_visited > options.max_path_size ||
907 (n_visited == options.max_path_size &&
908 node_id != options.end_depot)) {
909 if constexpr (F == Full::Reverse)
910 return -1;
911 else
912 return result;
913 }
914 }
915
916 // Get bucket number for the new label
917 int to_bucket = -1;
918 if constexpr (A == ArcType::Bucket) {
919 to_bucket = get_bucket_number<D>(node_id, new_resources);
920 } else if constexpr (A == ArcType::Jump) {
921 to_bucket = gamma.jump_bucket;
922 } else if constexpr (A == ArcType::Node) {
923 to_bucket = get_bucket_number<D>(node_id, new_resources);
924 }
925
926 // Handle Reverse mode early
927 if constexpr (F == Full::Reverse) {
928 if (new_resources[options.main_resources[0]] <=
929 std::floor(q_star[options.main_resources[0]]))
930 return -1;
931
932 return to_bucket;
933 }
934
935 // Calculate new cost
936#ifdef CUSTOM_COST
937 const auto distance = getcij(initial_node_id, node_id);
938 double new_cost = cost_calculator.calculate_cost(initial_cost, distance,
939 node_id, new_resources);
940#else
941 // Cache the distance value to avoid repeated calls
942 const auto distance = getcij(initial_node_id, node_id);
943 double new_cost = initial_cost + distance;
944#endif
945
946 const auto &VRPNode = nodes[node_id];
947
948 if constexpr (F != Full::PSTEP) {
949 new_cost -= VRPNode.cost;
950 } else {
951 int n_visited = 0;
952 for (const auto &bitmap : L_prime->visited_bitmap)
953 n_visited += __builtin_popcountll(bitmap);
954 if (n_visited > 1 && initial_node_id != options.depot) {
955 const auto three_two_dual =
956 pstep_duals.getThreeTwoDualValue(initial_node_id);
957 new_cost += three_two_dual +
958 pstep_duals.getThreeThreeDualValue(initial_node_id);
959 }
960 new_cost += -pstep_duals.getThreeTwoDualValue(node_id) +
961 pstep_duals.getArcDualValue(initial_node_id, node_id);
962 }
963
964 // Apply branching dual adjustments if present - use local variables to
965 // avoid repeated pointer dereferencing
966 if (!branching_duals->empty()) {
967 if (D == Direction::Forward) {
968 new_cost += branching_duals->getDual(initial_node_id, node_id);
969 } else {
970 new_cost += branching_duals->getDual(node_id, initial_node_id);
971 }
972 new_cost += branching_duals->getDual(node_id);
973 }
974
975 RCC_MODE_BLOCK(if constexpr (S == Stage::Four) {
976 if (D == Direction::Forward) {
977 new_cost -= arc_duals.getDual(initial_node_id, node_id);
978 } else {
979 new_cost -= arc_duals.getDual(node_id, initial_node_id);
980 }
981 })
982
983 // Create and initialize a new label
984 auto &label_pool = assign_buckets<D>(label_pool_fw, label_pool_bw);
985 auto new_label = label_pool->acquire();
986 new_label->initialize(to_bucket, new_cost, new_resources, node_id);
987 new_label->vertex = to_bucket;
988 new_label->real_cost =
989 L_prime->real_cost + distance; // Use cached distance
990
991 if constexpr (M == Mutability::Mut) {
992 L_prime->children.push_back(new_label);
993 }
994
995 // Update visited bitmap: intersect parent's visited bits with neighborhood
996 // mask
997 if constexpr (F != Full::PSTEP) {
998 for (size_t i = 0; i < new_label->visited_bitmap.size(); ++i) {
999 const uint64_t current_visited = L_prime->visited_bitmap[i];
1000 if (current_visited) {
1001 new_label->visited_bitmap[i] =
1002 current_visited & neighborhoods_bitmap[node_id][i];
1003 }
1004 }
1005 } else {
1006 // Use memcpy for direct bitmap copy (likely faster than assignment for
1007 // large bitmaps)
1008 std::memcpy(new_label->visited_bitmap.data(),
1009 L_prime->visited_bitmap.data(),
1010 new_label->visited_bitmap.size() * sizeof(uint64_t));
1011 }
1012
1013 set_node_visited(new_label->visited_bitmap, node_id);
1014
1015 // Update the path (nodes covered) by copying parent's path and adding
1016 // current node
1017 new_label->nodes_covered = L_prime->nodes_covered;
1018 new_label->nodes_covered.push_back(node_id);
1019
1020#if defined(SRC)
1021 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
1022 new_label->SRCmap = L_prime->SRCmap;
1023 auto &cutter = cut_storage;
1024 const auto active_cuts = cutter->getActiveCuts();
1025 const auto n_cuts = cutter->activeSize();
1026 if (__builtin_expect(n_cuts == 0, 0)) { // Branch prediction: unlikely
1027 if constexpr (F == Full::Reverse) {
1028 return -1;
1029 } else {
1030 return new_label;
1031 }
1032 }
1033
1034 double total_cost_update = 0.0;
1035 const size_t segment = node_id >> 6;
1036 const size_t bit_position = node_id & 63;
1037
1038 // Prefetch critical data structures
1039 auto &masks = cutter->getSegmentMasks();
1040 __builtin_prefetch(&masks, 0, 3);
1041 __builtin_prefetch(active_cuts.data(), 0, 3);
1042 __builtin_prefetch(new_label->SRCmap.data(), 1, 3); // 1 = read-write
1043
1044 const auto cut_limit_mask = masks.get_cut_limit_mask();
1045 const uint64_t valid_cuts =
1046 masks.get_valid_cut_mask(segment, bit_position) & cut_limit_mask;
1047
1048 // Skip processing if no valid cuts
1049 if (__builtin_expect(valid_cuts != 0,
1050 1)) { // Branch prediction: likely
1051 // Process valid cuts (4 at a time if possible)
1052 uint64_t remaining_cuts = valid_cuts;
1053
1054 // Process blocks of 4 cuts if available (manual loop unrolling)
1055 while (__builtin_popcountll(remaining_cuts) >= 4) {
1056 for (int i = 0; i < 4; i++) {
1057 const int cut_idx = __builtin_ctzll(remaining_cuts);
1058 const auto &active_cut = active_cuts[cut_idx];
1059 const auto &cut = *active_cut.cut_ptr;
1060
1061 // Prefetch the next cut data we'll need
1062 if (i < 3) {
1063 uint64_t next_bits =
1064 remaining_cuts & (remaining_cuts - 1);
1065 const int next_cut_idx = __builtin_ctzll(next_bits);
1066 __builtin_prefetch(&active_cuts[next_cut_idx], 0, 3);
1067 __builtin_prefetch(active_cuts[next_cut_idx].cut_ptr, 0,
1068 3);
1069 }
1070
1071 auto &src_map_value = new_label->SRCmap[active_cut.index];
1072 const double num_value =
1073 cut.p.num[cut.baseSetOrder[node_id]];
1074 src_map_value += num_value;
1075
1076 // Use branchless version when possible
1077 const bool overflow = src_map_value >= cut.p.den;
1078 src_map_value -= overflow ? cut.p.den : 0;
1079 total_cost_update -= overflow ? active_cut.dual_value : 0;
1080
1081 remaining_cuts &=
1082 (remaining_cuts - 1); // Clear processed bit
1083 }
1084 }
1085
1086 // Process remaining cuts (less than 4)
1087 while (remaining_cuts) {
1088 const int cut_idx = __builtin_ctzll(remaining_cuts);
1089 const auto &active_cut = active_cuts[cut_idx];
1090 const auto &cut = *active_cut.cut_ptr;
1091
1092 auto &src_map_value = new_label->SRCmap[active_cut.index];
1093 src_map_value += cut.p.num[cut.baseSetOrder[node_id]];
1094 if (src_map_value >= cut.p.den) {
1095 src_map_value -= cut.p.den;
1096 total_cost_update -= active_cut.dual_value;
1097 }
1098 remaining_cuts &= (remaining_cuts - 1);
1099 }
1100 }
1101
1102 // Process cuts to clear
1103 uint64_t to_clear =
1104 (~masks.get_neighbor_mask(segment, bit_position)) & cut_limit_mask;
1105
1106 if (__builtin_expect(to_clear != 0, 1)) { // Branch prediction: likely
1107 // Use vectorization for clearing multiple cuts when possible
1108 constexpr int vector_size =
1109 8; // Process 8 cuts at a time if possible
1110
1111 while (__builtin_popcountll(to_clear) >= vector_size) {
1112 for (int i = 0; i < vector_size; i++) {
1113 const int cut_idx = __builtin_ctzll(to_clear);
1114 new_label->SRCmap[active_cuts[cut_idx].index] = 0.0;
1115 to_clear &= (to_clear - 1);
1116 }
1117 }
1118
1119 // Process remaining cuts to clear
1120 while (to_clear) {
1121 const int cut_idx = __builtin_ctzll(to_clear);
1122 new_label->SRCmap[active_cuts[cut_idx].index] = 0.0;
1123 to_clear &= (to_clear - 1);
1124 }
1125 }
1126
1127 new_label->cost += total_cost_update;
1128 }
1129#endif
1130 if constexpr (F != Full::Reverse) {
1131 return new_label;
1132 } else {
1133 return -1;
1134 }

1141template <Direction D, Stage S>
1142inline bool BucketGraph::is_dominated(const Label *__restrict new_label,
1143 const Label *__restrict label) noexcept {
1144 // A label cannot dominate if its cost is higher.
1145 // const double cost_diff = label->cost - new_label->cost;
1146 if (numericutils::gt(label->cost, new_label->cost)) {
1147 return false;
1148 }
1149
1150 // Compare resource vectors.
1151 const auto *__restrict new_res = new_label->resources.data();
1152 const auto *__restrict lbl_res = label->resources.data();
1153
1154 // Prefetch resource data
1155 __builtin_prefetch(new_res, 0,
1156 3); // 0 = read only, 3 = high temporal locality
1157 __builtin_prefetch(lbl_res, 0, 3);
1158
1159 const size_t n_res = options.resources.size();
1160 if constexpr (D == Direction::Forward) {
1161 // In the Forward direction, each resource value of 'label' must
1162 // not exceed that of 'new_label'
1163 for (size_t i = 0; i < n_res; ++i) {
1164 if (numericutils::gt(lbl_res[i], new_res[i])) {
1165 return false;
1166 }
1167 }
1168 } else if constexpr (D == Direction::Backward) {
1169 // In the Backward direction, each resource value of 'label'
1170 // must not be less than that of 'new_label'
1171 for (size_t i = 0; i < n_res; ++i) {
1172 if (numericutils::lt(lbl_res[i], new_res[i])) {
1173 return false;
1174 }
1175 }
1176 }
1177
1178 // visits of label.
1179 if constexpr (S == Stage::Three || S == Stage::Four ||
1180 S == Stage::Enumerate) {
1181 const size_t n_bitmap = label->visited_bitmap.size();
1182
1183 // Prefetch bitmap data
1184 __builtin_prefetch(label->visited_bitmap.data(), 0, 3);
1185 __builtin_prefetch(new_label->visited_bitmap.data(), 0, 3);
1186
1187 for (size_t i = 0; i < n_bitmap; ++i) {
1188 // Every visited node in 'label' must also be visited in
1189 // 'new_label'.
1190 if ((label->visited_bitmap[i] & new_label->visited_bitmap[i]) !=
1191 label->visited_bitmap[i]) {
1192 return false;
1193 }
1194 }
1195 }
1196#ifdef SRC
1197 // For Stage Four or Enumerate, apply additional SRC-based cost
1198 // adjustments.
1199 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
1200 const double local_cost_diff = label->cost - new_label->cost;
1201 const auto *__restrict lbl_srcs = label->SRCmap.data();
1202 const auto *__restrict new_lbl_srcs = new_label->SRCmap.data();
1203
1204 // Prefetch SRCmap data
1205 __builtin_prefetch(lbl_srcs, 0, 3);
1206 __builtin_prefetch(new_lbl_srcs, 0, 3);
1207
1208 const auto &active_cuts = cut_storage->getActiveCuts();
1209 double dual_sum = 0.0;
1210 const size_t n = active_cuts.size();
1211 for (size_t i = 0; i < n; ++i) {
1212 const auto &cut = active_cuts[i];
1213 // Branch prediction hint: likely false.
1214 if (__builtin_expect(lbl_srcs[cut.index] > new_lbl_srcs[cut.index],
1215 0)) {
1216 dual_sum += cut.dual_value;
1217 if (__builtin_expect(
1218 (numericutils::gt(local_cost_diff, dual_sum)), 0)) {
1219 return false;
1220 }
1221 }
1222 }
1223 }
1224#endif
1225 return true;
1226}

1239inline bool precedes(
1240 const T a, const T b, const UnionFind &uf,
1241 ankerl::unordered_dense::map<std::pair<T, T>, bool> &cache) {
1242 // Create a cache key
1243 const std::pair<T, T> key{a, b};

1249 // correct ordering
1250 bool result = uf.compareSubsets(a, b);
1251 cache.emplace(key, result);
1252 return result;
1253}

1263template <Direction D, Stage S>
1264inline bool BucketGraph::DominatedInCompWiseSmallerBuckets(
1265 const Label *__restrict__ L, int bucket,
1266 const std::vector<double> &__restrict__ c_bar,
1267 std::vector<uint64_t> &__restrict__ Bvisited, uint &stat_n_dom) noexcept {
1268 // Cache direction-specific data
1269 const auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
1270 const auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
1271 const auto &uf = assign_buckets<D>(fw_union_find, bw_union_find);
1272 auto &union_cache = assign_buckets<D>(fw_union_cache, bw_union_cache);
1273
1274 // Cache frequently used label properties
1275 const int b_L = L->vertex;
1276 const double label_cost = L->cost;
1277 const size_t res_size = options.resources.size();
1278
1279 // Use static thread-local stack to avoid repeated allocations
1280 static thread_local std::vector<int> stack_buffer;
1281 if (stack_buffer.capacity() < R_SIZE) {
1282 stack_buffer.reserve(R_SIZE);
1283 }
1284 stack_buffer.clear();
1285 stack_buffer.push_back(bucket);
1286
1287 // Direct pointer to bit mask lookup for fast bit operations
1288 const uint64_t *const bit_mask_lookup_ptr = bit_mask_lookup.data();
1289
1290 // Inline dominance check function to avoid lambda overhead
1291 auto inline_check_dominance = [&](const std::pmr::vector<Label *> &labels,
1292 uint &n_dom) -> bool {
1293#ifdef __AVX2__
1294 if (labels.size() >= AVX_LIM)
1295 return check_dominance_against_vector<D, S>(L, labels, cut_storage,
1296 res_size, n_dom);
1297#endif
1298 const size_t size = labels.size();
1299 size_t i = 0;
1300 // Loop unrolling: process 4 labels at a time
1301 for (; i + 3 < size; i += 4) {
1302 if (is_dominated<D, S>(L, labels[i]) ||
1303 is_dominated<D, S>(L, labels[i + 1]) ||
1304 is_dominated<D, S>(L, labels[i + 2]) ||
1305 is_dominated<D, S>(L, labels[i + 3])) {
1306 ++n_dom;
1307 return true;
1308 }
1309 }
1310 // Process remaining labels
1311 for (; i < size; ++i) {
1312 if (is_dominated<D, S>(L, labels[i])) {
1313 ++n_dom;
1314 return true;
1315 }
1316 }
1317 return false;
1318 };
1319
1320 // Main DFS traversal loop
1321 while (!stack_buffer.empty()) {
1322 const int current_bucket = stack_buffer.back();
1323 stack_buffer.pop_back();
1324
1325 const size_t segment = static_cast<size_t>(current_bucket) >> 6;
1326 const uint64_t bit = bit_mask_lookup_ptr[current_bucket & 63];
1327
1328 if (unlikely(Bvisited[segment] & bit)) continue;
1329
1330 // Mark as visited
1331 Bvisited[segment] |= bit;
1332
1333 // Fast path: if label cost is lower and bucket precedes L in the SCC
1334 // order, return early
1335 if (likely(label_cost < c_bar[current_bucket] &&
1336 ::precedes(current_bucket, b_L, uf, union_cache)))
1337 return false;
1338
1339 // Skip dominance check for L's own bucket
1340 if (b_L != current_bucket) {
1341 const auto &mother_bucket = buckets[current_bucket];
1342 if (mother_bucket.check_dominance(L, inline_check_dominance,
1343 stat_n_dom))
1344 return true;
1345 }
1346
1347 // Process neighbor buckets using pointer arithmetic and prefetching
1348 const auto &neighbors = Phi[current_bucket];
1349 const size_t n_neighbors = neighbors.size();
1350 const int *neighbor_ptr = neighbors.data();
1351 const int *neighbor_end = neighbor_ptr + n_neighbors;
1352 for (; neighbor_ptr != neighbor_end; ++neighbor_ptr) {
1353 const int b_prime = *neighbor_ptr;
1354 const size_t word_idx = static_cast<size_t>(b_prime) >> 6;
1355 const uint64_t mask = bit_mask_lookup_ptr[b_prime & 63];
1356 __builtin_prefetch(&Bvisited[word_idx], 0,
1357 1); // Prefetch Bvisited for this segment
1358 if (!(Bvisited[word_idx] & mask)) stack_buffer.push_back(b_prime);
1359 }
1360 }
1361
1362 return false;
1363}

1375template <Stage state, Full fullness>
1376void BucketGraph::run_labeling_algorithms(std::vector<double> &forward_cbar,
1377 std::vector<double> &backward_cbar) {
1378 // Schedule the forward labeling algorithm task on the scheduler.
1379 auto forward_task =
1380 stdexec::schedule(bi_sched) | stdexec::then([&]() {
1381 return labeling_algorithm<Direction::Forward, state, fullness>();
1382 });
1383
1384 // Schedule the backward labeling algorithm task on the scheduler.
1385 auto backward_task =
1386 stdexec::schedule(bi_sched) | stdexec::then([&]() {
1387 return labeling_algorithm<Direction::Backward, state, fullness>();
1388 });
1389
1390 // Combine the forward and backward tasks.
1391 auto combined_work =
1392 stdexec::when_all(std::move(forward_task), std::move(backward_task)) |
1393 stdexec::then([&](auto forward_result, auto backward_result) {
1394 // Store the results into the provided vectors.
1395 forward_cbar = std::move(forward_result);
1396 backward_cbar = std::move(backward_result);
1397 });
1398

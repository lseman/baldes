54template <Symmetry SYM>
55inline std::vector<Label *> BucketGraph::solve(bool trigger) {
56 // Start with a nonoptimal status.
57 status = Status::NotOptimal;
58 if (trigger) {
59 transition = true;
60 fixed = false;
61 }
62
63 updateSplit(); // Update split values for the bucket graph.
64 std::vector<Label *> paths;
65
67 // Adaptive Stage Handling
69
70 // ----- Stage 1: Light Heuristic -----
71 if (s1 && depth == 0) {
72 stage = 1;
73 paths = bi_labeling_algorithm<Stage::One>();
74 // inner_obj = paths[0]->cost; // (if available)
75 if (inner_obj >= -1 || iter >= 10) {
76 s1 = false;
77 s2 = true; // Transition to Stage 2.
78 }
79 }
80 // ----- Stage 2: Expensive Pricing Heuristic -----
81 else if (s2 && depth == 0) {
82 stage = 2;
83 paths = bi_labeling_algorithm<Stage::Two>();
84 // inner_obj = paths[0]->cost; // (if available)
85 if (inner_obj >= -10 || iter > 500) {
86 s2 = false;
87 s3 = true; // Transition directly to Stage 4.
88 }
89 }
90 // ----- Stage 3: Heuristic Fixing Approach -----
91 else if (s3 && depth == 0) {
92 stage = 3;
93 paths = bi_labeling_algorithm<Stage::Three>();
94 // inner_obj = paths[0]->cost; // (if available)
95 if (inner_obj >= -0.5) {
96 s3 = false;
97 s4 = true;
98 transition = true; // Prepare for Stage 4 transition.
99 }
100 }
101 // ----- Stage 4: Exact Labeling with Fixing Enabled -----
102 else if (s4 && depth == 0) {
103 stage = 4;
104#ifdef FIX_BUCKETS
105 if (transition) {
106 // During a transition to Stage 4, fix buckets temporarily.
107 bool original_fixed = fixed;
108 fixed = true;
109 paths = bi_labeling_algorithm<Stage::Four>();
110 transition = false;
111 fixed = original_fixed;
112 min_red_cost = paths[0]->cost; // Update minimum reduced cost.
113 iter++;
114 return paths;
115 }
116#endif
117 // Standard Stage 4 processing.
118 paths = bi_labeling_algorithm<Stage::Four>();
119 // inner_obj = paths[0]->cost; // (if available)
120 bool rollback = false;
121 if (status != Status::Rollback) {
122 rollback = shallUpdateStep();
123 }
124 if (rollback) {
125 status = Status::Rollback;
126 return paths;
127 }
128 // Compute a new threshold for separation based on stats.
129 threshold = stats.computeThreshold(iter, inner_obj);
130 // If objective is sufficiently high, enter separation mode.
131 if (inner_obj > threshold) {
132 ss = true;
133#if !defined(SRC) && !defined(SRC3)
134 status = Status::Optimal;
135 return paths;
136#endif
137 status = Status::Separation;
138 }
139 }
140 // ----- Stage 5: Enumeration -----
141 else if (s5 && depth == 0) {
142 stage = 5;
143 print_info("Starting enumeration with gap {}\n", gap);
144 paths = bi_labeling_algorithm<Stage::Enumerate>();
145 print_info("Finished enumeration with {} paths\n", paths.size());
146 status = Status::Optimal;
147 }
148
149 // Increment the iteration counter.
150 iter++;
151 return paths;
152}

154inline std::vector<Label *> BucketGraph::solveHeuristic() {
155 // Initialize the status as not optimal at the start
156 status = Status::NotOptimal;
157
158 updateSplit(); // Update the split values for the bucket graph
159
160 // Placeholder for the final paths (labels) and inner objective value
161 std::vector<Label *> paths;
162 double inner_obj;
163
165 // ADAPTIVE STAGE HANDLING
167 // Stage 1: Apply a light heuristic (Stage::One)
168 if (s1) {
169 stage = 1;
170 paths = bi_labeling_algorithm<Stage::One>(); // Solve the problem with
171 // Stage 1 heuristic
172 inner_obj = paths[0]->cost;
173
174 // Transition from Stage 1 to Stage 2 if the objective improves or after
175 // 10 iterations
176 if (inner_obj >= -5 || iter >= 10) {
177 s1 = false;
178 s2 = true; // Move to Stage 2
179 }
180 }
181 // Stage 2: Apply a more expensive pricing heuristic (Stage::Two)
182 else if (s2) {
183 s2 = true;
184 stage = 2;
185 paths = bi_labeling_algorithm<Stage::Two>(); // Solve the problem with
186 // Stage 2 heuristic
187 inner_obj = paths[0]->cost;
188
189 // Transition from Stage 2 to Stage 3 if the objective improves or after
190 // 800 iterations
191 if (inner_obj >= -100 || iter > 800) {
192 status = Status::Optimal;
193 }
194 }
195
196 iter++; // Increment the iteration counter
197
198 return paths; // Return the final paths after processing
199}

205template <Direction D, Stage S, Full F>
206std::vector<double> BucketGraph::labeling_algorithm() {
207 // Cache bucket and auxiliary structure references to avoid repeated
208 // lookups.
209 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
210 const auto &ordered_sccs =
211 assign_buckets<D>(fw_ordered_sccs, bw_ordered_sccs);
212 const auto &topological_order =
213 assign_buckets<D>(fw_topological_order, bw_topological_order);
214 const auto &sccs = assign_buckets<D>(fw_sccs, bw_sccs);
215 const auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
216 auto &c_bar = assign_buckets<D>(fw_c_bar, bw_c_bar);
217 auto &n_labels = assign_buckets<D>(n_fw_labels, n_bw_labels);
218 const auto &sorted_sccs = assign_buckets<D>(fw_sccs_sorted, bw_sccs_sorted);
219 const auto &n_buckets = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
220 auto &stat_n_labels = assign_buckets<D>(stat_n_labels_fw, stat_n_labels_bw);
221 auto &stat_n_dom = assign_buckets<D>(stat_n_dom_fw, stat_n_dom_bw);
222 auto &arc_scores = assign_buckets<D>(fw_arc_scores, bw_arc_scores);
223 auto &best_label = assign_buckets<D>(fw_best_label, bw_best_label);
224
225 // Reset the total label count.
226 n_labels = 0;
227
228 // Pre-calculate bitmap segments for dominance-checking.
229 const size_t n_segments = (n_buckets + 63) >> 6;
230 std::vector<uint64_t> Bvisited(n_segments, 0);
231
232 // Preallocate vector for extended labels to avoid repeated allocations.
233 std::vector<Label *> extended_labels;
234 extended_labels.reserve(64);
235
236 // Precompute main resource index and q_star value for partial solutions.
237 constexpr bool is_partial = (F == Full::Partial);
238 int main_res_index = 0;
239 double q_star_value = 0.0;
240 if constexpr (is_partial) {
241 main_res_index = options.main_resources[0];
242 q_star_value = std::floor(q_star[main_res_index]);
243 }
244
245 // Reuse a thread-local vector for collecting labels to dominate.
246 static thread_local std::vector<Label *> tl_labels_to_dominate;
247 tl_labels_to_dominate.clear();
248
249 // Process each strongly connected component (SCC) in topological order.
250 for (const auto &scc_index : topological_order) {
251 bool all_ext = false;
252 do {
253 all_ext = true; // Assume no new extensions until one occurs.
254 const auto &scc_buckets = sorted_sccs[scc_index];
255 for (const auto bucket : scc_buckets) {
256 const auto &bucket_labels = buckets[bucket].get_labels();
257 for (Label *label : bucket_labels) {
258 // Skip labels already processed.
259 if (label->is_extended || label->is_dominated) continue;
260
261 // For partial solutions, check resource feasibility.
262 if constexpr (F != Full::PSTEP && F != Full::TSP) {
263 if constexpr (is_partial) {
264 const auto main_resource =
265 label->resources[main_res_index];
266 if ((D == Direction::Forward &&
267 main_resource > q_star_value) ||
268 (D == Direction::Backward &&
269 main_resource <= q_star_value)) {
270 label->set_extended(true);
271 continue;
272 }
273 }
274 }
275
276 // Clear the bitmap using memset (faster than std::fill for
277 // large arrays).
278 std::memset(Bvisited.data(), 0,
279 n_segments * sizeof(uint64_t));
280
281 // Check if the label is dominated by labels in smaller
282 // buckets.
283 if constexpr (F != Full::TSP) {
284 if (DominatedInCompWiseSmallerBuckets<D, S>(
285 label, bucket, c_bar, Bvisited, stat_n_dom))
286 continue;
287 }
288
289 // Retrieve outgoing arcs for the current label.
290 const auto &node_arcs =
291 buckets[bucket].template get_bucket_arcs<D>();
292 if (node_arcs.empty()) {
293 label->set_extended(true);
294 continue;
295 }
296
297 for (const auto &arc : node_arcs) {
298 extended_labels.clear();
299 extended_labels =
300 Extend<D, S, ArcType::Bucket, Mutability::Mut, F>(
301 label, arc, &extended_labels);
302
303 // Process each new label produced by the extension.
304 for (Label *new_label : extended_labels) {
305 ++stat_n_labels;
306 const int to_bucket = new_label->vertex;
307 auto &mother_bucket = buckets[to_bucket];
308 const auto &to_bucket_labels =
309 mother_bucket.get_labels();
310
311 bool dominated = false;
312 if constexpr (S == Stage::One) {
313 // Stage One: mark higher-cost labels as
314 // dominated.
315 for (auto *existing_label : to_bucket_labels) {
316 if (existing_label->is_dominated) continue;
317 if (label->cost < existing_label->cost)
318 existing_label->set_dominated(true);
319 else {
320 dominated = true;
321 break;
322 }
323 }
324 } else {
325 // Later stages: use a SIMD-friendly, unrolled
326 // loop.
327 auto check_dominance_in_bucket =
328 [&](const std::vector<Label *> &labels,
329 int &stat_n_dom) -> bool {
330#ifdef __AVX2__
331 if (labels.size() >= AVX_LIM)
332 return check_dominance_against_vector<
333 D, S>(new_label, labels,
334 cut_storage,
335 options.resources.size(),
336 stat_n_dom);
337 else
338#endif
339 {
340 const size_t size = labels.size();
341 size_t i = 0;
342 for (; i + 3 < size; i += 4) {
343 if ((!labels[i]->is_dominated &&
344 is_dominated<D, S>(
345 new_label, labels[i])) ||
346 (!labels[i + 1]->is_dominated &&
347 is_dominated<D, S>(
348 new_label,
349 labels[i + 1])) ||
350 (!labels[i + 2]->is_dominated &&
351 is_dominated<D, S>(
352 new_label,
353 labels[i + 2])) ||
354 (!labels[i + 3]->is_dominated &&
355 is_dominated<D, S>(
356 new_label,
357 labels[i + 3]))) {
358 ++stat_n_dom;
359 return true;
360 }
361 }
362 for (; i < size; ++i) {
363 if (!labels[i]->is_dominated &&
364 is_dominated<D, S>(new_label,
365 labels[i])) {
366 ++stat_n_dom;
367 return true;
368 }
369 }
370 return false;
371 }
372 };
373 dominated = mother_bucket.check_dominance(
374 new_label, check_dominance_in_bucket,
375 stat_n_dom);
376 }
377
378 if (!dominated) {
379 // Reuse thread-local vector to avoid
380 // re-allocation.
381 tl_labels_to_dominate.clear();
382 for (auto *existing_label : to_bucket_labels) {
383 if (existing_label->is_dominated) continue;
384 if (is_dominated<D, S>(existing_label,
385 new_label))
386 tl_labels_to_dominate.push_back(
387 existing_label);
388 }
389 for (auto *existing_label :
390 tl_labels_to_dominate)
391 existing_label->set_dominated(true);
392
393 ++n_labels;
394#ifdef SORTED_LABELS
395 mother_bucket.add_sorted_label(new_label);
396#elif defined(LIMITED_BUCKETS)
397 mother_bucket.sorted_label(new_label,
398 BUCKET_CAPACITY);
399#else
400 mother_bucket.add_label(new_label);
401#endif
402 // Mark that a new extension was added.
403 all_ext = false;
404 }
405 }
406 }
407 label->set_extended(true);
408 }
409 }
410 } while (!all_ext);
411
412 // Update c_bar values for all buckets in the current SCC.
413 auto sorted_buckets = sorted_sccs[scc_index];
414 if constexpr (D == Direction::Forward) {
415 pdqsort(sorted_buckets.begin(), sorted_buckets.end(),
416 [&](int a, int b) {
417 return buckets[a].get_lb() < buckets[b].get_lb();
418 });
419 } else {
420 pdqsort(sorted_buckets.begin(), sorted_buckets.end(),
421 [&](int a, int b) {
422 return buckets[a].get_ub() > buckets[b].get_ub();
423 });
424 }
425 // Batch processing of c_bar updates for improved cache locality.
426 for (const int bucket : sorted_buckets) {
427 double min_c_bar = buckets[bucket].get_cb();
428 const auto &phi_buckets = Phi[bucket];
429 for (const auto phi_bucket : phi_buckets)
430 min_c_bar = std::min(min_c_bar, c_bar[phi_bucket]);
431 c_bar[bucket] = min_c_bar;
432 }
433 }
434
435 // Store the best label from the labeling process.
436 if (Label *best_label_run =
437 get_best_label<D>(topological_order, c_bar, sccs))
438 best_label = best_label_run;
439
440 return c_bar;
441}

447template <Stage S, Symmetry SYM>
448std::vector<Label *> BucketGraph::bi_labeling_algorithm() {
449 // === Stage-specific Initializations ===
450 if constexpr (S == Stage::Three) {
451 heuristic_fixing<S>();
452 } else if constexpr (S == Stage::Four) {
453 if (first_reset) {
454 reset_fixed();
455 first_reset = false;
456 }
457#ifdef FIX_BUCKETS
458 if (options.bucket_fixing) {
459 // heuristic_fixing<S>();
460
461 bucket_fixing<S>();
462 }
463#endif
464 // heuristic_fixing<S>();
465 }
466
467 // Reset pools and perform common initialization.
468 reset_pool();
469 common_initialization();
470
471 // Pre-allocate c-bar vectors for forward and backward buckets.
472 std::vector<double> forward_cbar(fw_buckets.size());
473 std::vector<double> backward_cbar(bw_buckets.size());
474
475 // Run the labeling algorithms.
476 if constexpr (SYM == Symmetry::Asymmetric) {
477 run_labeling_algorithms<S, Full::Partial>(forward_cbar, backward_cbar);
478 } else {
479 forward_cbar =
480 labeling_algorithm<Direction::Forward, S, Full::Partial>();
481 }
482
483 // Acquire an initial best label from the forward label pool.
484 auto best_label = label_pool_fw->acquire();
485 if constexpr (SYM == Symmetry::Asymmetric) {
486 // Compute the best label if the forward and backward labels are
487 // feasible.
488 if (check_feasibility(fw_best_label, bw_best_label)) {
489 best_label = compute_label<S>(fw_best_label, bw_best_label);
490 } else {
491 best_label->cost = 0.0;
492 best_label->real_cost = std::numeric_limits<double>::infinity();
493 best_label->nodes_covered.clear();
494 }
495 } else {
496 best_label->cost = best_label->real_cost;
497 best_label->real_cost = std::numeric_limits<double>::infinity();
498 best_label->nodes_covered.clear();
499 }
500 merged_labels.push_back(best_label);
501
502 // Declare atomics to track the best cost and non-dominated label
503 // count.
504 std::atomic<double> best_cost{0.0};
505 std::atomic<size_t> non_dominated_labels_per_bucket{0};
506
507 // === Parallel Bucket Processing ===
508 // Choose a chunk size based on hardware concurrency.
509 const size_t chunk_size =
510 fw_buckets_size / MERGE_SCHED_CONCURRENCY; // 1/4 of buckets per core
511 const size_t n_chunks = (fw_buckets_size + chunk_size - 1) / chunk_size;
512
513 auto bulk_sender = stdexec::bulk(
514 stdexec::just(), n_chunks,
515 [this, chunk_size, &best_cost,
516 &non_dominated_labels_per_bucket](std::size_t chunk_idx) {
517 const size_t start_bucket = chunk_idx * chunk_size;
518 const size_t end_bucket =
519 std::min(start_bucket + chunk_size,
520 static_cast<size_t>(fw_buckets_size));
521
522 // Process each bucket within the current chunk.
523 for (size_t bucket = start_bucket; bucket < end_bucket; ++bucket) {
524 // Process non-dominated labels in the current bucket.
525 const auto &bucket_labels = fw_buckets[bucket].get_labels();
526 for (const Label *L : bucket_labels) {
527 if (L->is_dominated) continue;
528
529 non_dominated_labels_per_bucket.fetch_add(
530 1, std::memory_order_relaxed);
531
532 // Retrieve the outgoing arcs for the current label.
533 // const auto &to_arcs =
534 // nodes[L->node_id].get_arcs<Direction::Forward>();
535 const auto &to_arcs =
536 fw_buckets[bucket]
537 .template get_bucket_arcs<Direction::Forward>();
538 for (const auto &arc : to_arcs) {
539 const int to_node = fw_buckets[arc.to_bucket].node_id;
540 if constexpr (S >= Stage::Three ||
541 S == Stage::Eliminate) {
542 if (is_arc_fixed(L->node_id, to_node)) continue;
543 }
544
545 // Extend the label along the current arc.
546 auto Ls =
547 Extend<Direction::Forward, S, ArcType::Bucket,
548 Mutability::Const, Full::Reverse>(L, arc);
549
550 // Process each extended label.
551 if (!Ls.empty()) {
552 for (Label *L_prime : Ls) {
553 int bucket_to_process =
554 L_prime->vertex; // mutable copy if needed
555
556 // Concatenate the label with bucket
557 // information, updating best_cost.
558 ConcatenateLabel<S, SYM>(L, bucket_to_process,
559 best_cost);
560 }
561 }
562 }
563 }
564 }
565 });
566
567 // Submit the bulk work to the merge scheduler and wait for
568 // completion.
569 auto work = stdexec::starts_on(merge_sched, bulk_sender);
570 stdexec::sync_wait(std::move(work));
571
572 // === Post-processing and Sorting ===
573 // Sort merged labels based on cost (ascending order).
574 pdqsort(merged_labels.begin(), merged_labels.end(),
575 [](const Label *a, const Label *b) { return a->cost < b->cost; });
576
577#ifdef SCHRODINGER
578 // If using the Schrödinger approach, process additional paths.
579 if (merged_labels.size() > N_ADD) {
580 std::vector<Path> paths;
581 const int labels_size = merged_labels.size();
582 const int end_idx = std::min(N_ADD + N_ADD, labels_size);
583 paths.reserve(end_idx - N_ADD);
584 for (int i = N_ADD; i < end_idx; ++i) {
585 if (merged_labels[i]->nodes_covered.size() <= 3) continue;
586 paths.emplace_back(merged_labels[i]->nodes_covered,
587 merged_labels[i]->real_cost);
588 }
589 sPool.add_paths(paths);
590 sPool.iterate();
591 }
592#endif
593
594#ifdef RIH
595 // If using RIH stage modifications, submit the top labels to ILS.
596 if constexpr (S == Stage::Four) {
597 std::vector<Label *> top_labels;
598 top_labels.reserve(N_ADD);
599 const int n_candidates =
600 std::min(N_ADD, static_cast<int>(merged_labels.size()));
601 for (int i = 0; i < n_candidates; ++i) {
602 if (merged_labels[i]->nodes_covered.size() <= 3) continue;
603 top_labels.push_back(merged_labels[i]);
604 }
605 ils->submit_task(top_labels, nodes);
606 }
607#endif
608
609 // Store the cost of the best (first) merged label as the inner
610 // objective.
611 inner_obj = merged_labels[0]->cost;
612 return merged_labels;
613}

619template <Direction D, Stage S, ArcType A, Mutability M, Full F>
620
621inline std::vector<Label *> BucketGraph::Extend(
622 const std::conditional_t<M == Mutability::Mut, Label *, const Label *>
623 L_prime,
624 const std::conditional_t<
625 A == ArcType::Bucket, BucketArc,
626 std::conditional_t<A == ArcType::Jump, JumpArc, Arc>> &gamma,
627 std::vector<Label *> *output_buffer, int depth) noexcept {
628 // Pre-allocated result vector to avoid allocations for the common case of
629 // 0-1 results
630 static thread_local std::vector<Label *> result_vector;
631 std::vector<Label *> &result =
632 output_buffer ? *output_buffer : result_vector;
633 result.clear();
634
635 // Prepare temporary resource storage (resize to current resources size)
636 static thread_local std::vector<double> new_resources;
637 const size_t n_resources = options.resources.size();
638 new_resources.resize(n_resources);
639
640 // Cache initial values from L_prime
641 const int initial_node_id = L_prime->node_id;
642 const auto &initial_resources =
643 L_prime->resources; // Use reference instead of copy
644 const double initial_cost = L_prime->cost;
645
646 // Determine target node ID based on arc type
647 int node_id;
648 if constexpr (A == ArcType::Bucket) {
649 node_id =
650 assign_buckets<D>(fw_buckets, bw_buckets)[gamma.to_bucket].node_id;
651 } else if constexpr (A == ArcType::Node) {
652 node_id = gamma.to;
653 } else { // Jump arc
654 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
655 node_id = buckets[gamma.jump_bucket].node_id;
656 const auto &bucket = buckets[gamma.jump_bucket];
657
658 // Fast copy the resources
659 std::memcpy(new_resources.data(), initial_resources.data(),
660 n_resources * sizeof(double));
661
662 // Update resource bounds based on jump bucket properties
663 for (size_t i = 0; i < n_resources; ++i) {
664 if constexpr (D == Direction::Forward) {
665 new_resources[i] = std::max(new_resources[i],
666 static_cast<double>(bucket.lb[i]));
667 } else {
668 new_resources[i] = std::min(new_resources[i],
669 static_cast<double>(bucket.ub[i]));
670 }
671 }
672 }
673
674 // Early rejection: if we loop back or the node was already visited
675 if (node_id == initial_node_id ||
676 is_node_visited(L_prime->visited_bitmap, node_id))
677 return result;
678
679 // For stages where fixed arcs matter
680 if constexpr (S >= Stage::Three || S == Stage::Eliminate) {
681 if ((D == Direction::Forward &&
682 is_arc_fixed(initial_node_id, node_id)) ||
683 (D == Direction::Backward &&
684 is_arc_fixed(node_id, initial_node_id)))
685 return result;
686 }
687
688 // Process resources (feasibility check)
689 if constexpr (F != Full::TSP) {
690 if constexpr (A != ArcType::Jump) {
691 // Only copy resources for non-jump arcs (jump arcs already copied)
692 std::memcpy(new_resources.data(), initial_resources.data(),
693 n_resources * sizeof(double));
694
695 if (!process_all_resources<D>(new_resources, initial_resources,
696 gamma, nodes[node_id], n_resources))
697 return result;
698 } else {
699 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
700 const auto &bucket = buckets[gamma.jump_bucket];
701 for (size_t i = 0; i < n_resources; ++i) {
702 if constexpr (D == Direction::Forward) {
703 new_resources[i] =
704 std::max(new_resources[i] + gamma.resource_increment[i],
705 static_cast<double>(bucket.lb[i]));
706 if (numericutils::gt(new_resources[i], bucket.ub[i])) {
707 return {};
708 }
709
710 } else {
711 new_resources[i] =
712 std::min(new_resources[i] - gamma.resource_increment[i],
713 static_cast<double>(bucket.ub[i]));
714 if (numericutils::lt(new_resources[i], bucket.lb[i])) {
715 return {};
716 }
717 }
718 }
719 }
720 }
721
722 // For PSTEP/TSP: Check path length constraints
723 if constexpr (F == Full::PSTEP || F == Full::TSP) {
724 int n_visited = 0;
725 for (const auto &bitmap : L_prime->visited_bitmap)
726 n_visited += __builtin_popcountll(bitmap);
727 if (n_visited > options.max_path_size ||
728 (n_visited == options.max_path_size &&
729 node_id != options.end_depot))
730 return result;
731 }
732
733 // Get bucket number for the new label
734 int to_bucket = -1;
735 if constexpr (A == ArcType::Bucket) {
736 to_bucket = get_bucket_number<D>(node_id, new_resources);
737 } else if constexpr (A == ArcType::Jump) {
738 to_bucket = gamma.jump_bucket;
739 } else if constexpr (A == ArcType::Node) {

927template <Direction D, Stage S>
928inline bool BucketGraph::is_dominated(const Label *__restrict new_label,
929 const Label *__restrict label) noexcept {
930 // A label cannot dominate if its cost is higher.
931 // const double cost_diff = label->cost - new_label->cost;
932 if (numericutils::gt(label->cost, new_label->cost)) {
933 return false;
934 }
935
936 // Compare resource vectors.
937 const auto *__restrict new_res = new_label->resources.data();
938 const auto *__restrict lbl_res = label->resources.data();
939 const size_t n_res = options.resources.size();
940
941 if constexpr (D == Direction::Forward) {
942 // In the Forward direction, each resource value of 'label' must
943 // not exceed that of 'new_label'
944 for (size_t i = 0; i < n_res; ++i) {
945 if (numericutils::gt(lbl_res[i], new_res[i])) {
946 return false;
947 }
948 }
949 } else if constexpr (D == Direction::Backward) {
950 // In the Backward direction, each resource value of 'label'
951 // must not be less than that of 'new_label'
952 for (size_t i = 0; i < n_res; ++i) {
953 if (numericutils::lt(lbl_res[i], new_res[i])) {
954 return false;
955 }
956 }
957 }
958
959 // visits of label.
960 if constexpr (S == Stage::Three || S == Stage::Four ||
961 S == Stage::Enumerate) {
962 const size_t n_bitmap = label->visited_bitmap.size();
963 for (size_t i = 0; i < n_bitmap; ++i) {
964 // Every visited node in 'label' must also be visited in
965 // 'new_label'.
966 if ((label->visited_bitmap[i] & new_label->visited_bitmap[i]) !=
967 label->visited_bitmap[i]) {
968 return false;
969 }
970 }
971 }
972
973#ifdef SRC
974 // For Stage Four or Enumerate, apply additional SRC-based cost
975 // adjustments.
976 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
977 const double local_cost_diff = label->cost - new_label->cost;
978 const auto *__restrict lbl_srcs = label->SRCmap.data();
979 const auto *__restrict new_lbl_srcs = new_label->SRCmap.data();
980 const auto &active_cuts = cut_storage->getActiveCuts();
981 double dual_sum = 0.0;
982 const size_t n = active_cuts.size();
983 for (size_t i = 0; i < n; ++i) {
984 const auto &cut = active_cuts[i];
985 // Branch prediction hint: likely false.
986 if (__builtin_expect(lbl_srcs[cut.index] > new_lbl_srcs[cut.index],
987 0)) {
988 dual_sum += cut.dual_value;
989 if (__builtin_expect(
990 (numericutils::gt(local_cost_diff, dual_sum)), 0)) {
991 return false;
992 }
993 }
994 }
995 }
996#endif
997
998 return true;
999}

1036template <Direction D, Stage S>
1037inline bool BucketGraph::DominatedInCompWiseSmallerBuckets(
1038 const Label *__restrict__ L, int bucket,
1039 const std::vector<double> &__restrict__ c_bar,
1040 std::vector<uint64_t> &__restrict__ Bvisited, int &stat_n_dom) noexcept {
1041 // Cache direction-specific data
1042 const auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
1043 const auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
1044 const auto &uf = assign_buckets<D>(fw_union_find, bw_union_find);
1045 auto &union_cache = assign_buckets<D>(fw_union_cache, bw_union_cache);
1046
1047 // Cache frequently used label properties
1048 const int b_L = L->vertex;
1049 const double label_cost = L->cost;
1050 const size_t res_size = options.resources.size();
1051
1052 // Use static thread-local stack to avoid repeated allocations
1053 static thread_local std::vector<int> stack_buffer;
1054 if (stack_buffer.capacity() < R_SIZE) {
1055 stack_buffer.reserve(R_SIZE);
1056 }
1057 stack_buffer.clear();
1058 stack_buffer.push_back(bucket);
1059
1060 // Direct pointer to bit mask lookup for fast bit operations
1061 const uint64_t *const bit_mask_lookup_ptr = bit_mask_lookup.data();
1062
1063 // Inline dominance check function to avoid lambda overhead
1064 auto inline_check_dominance = [&](const std::vector<Label *> &labels,
1065 int &n_dom) -> bool {
1066#ifdef __AVX2__
1067 if (labels.size() >= AVX_LIM)
1068 return check_dominance_against_vector<D, S>(L, labels, cut_storage,
1069 res_size, n_dom);
1070#endif
1071 const size_t size = labels.size();
1072 size_t i = 0;
1073 // Loop unrolling: process 4 labels at a time
1074 for (; i + 3 < size; i += 4) {
1075 if (is_dominated<D, S>(L, labels[i]) ||
1076 is_dominated<D, S>(L, labels[i + 1]) ||
1077 is_dominated<D, S>(L, labels[i + 2]) ||
1078 is_dominated<D, S>(L, labels[i + 3])) {
1079 ++n_dom;
1080 return true;
1081 }
1082 }
1083 // Process remaining labels
1084 for (; i < size; ++i) {
1085 if (is_dominated<D, S>(L, labels[i])) {
1086 ++n_dom;
1087 return true;
1088 }
1089 }
1090 return false;
1091 };
1092
1093 // Main DFS traversal loop
1094 while (!stack_buffer.empty()) {
1095 const int current_bucket = stack_buffer.back();
1096 stack_buffer.pop_back();
1097
1098 const size_t segment = static_cast<size_t>(current_bucket) >> 6;
1099 const uint64_t bit = bit_mask_lookup_ptr[current_bucket & 63];
1100
1101 if (unlikely(Bvisited[segment] & bit)) continue;
1102
1103 // Mark as visited
1104 Bvisited[segment] |= bit;
1105
1106 // Fast path: if label cost is lower and bucket precedes L in the SCC
1107 // order, return early
1108 if (likely(label_cost < c_bar[current_bucket] &&
1109 ::precedes(current_bucket, b_L, uf, union_cache)))
1110 return false;
1111
1112 // Skip dominance check for L's own bucket
1113 if (b_L != current_bucket) {
1114 const auto &mother_bucket = buckets[current_bucket];
1115 if (mother_bucket.check_dominance(L, inline_check_dominance,
1116 stat_n_dom))
1117 return true;
1118 }
1119
1120 // Process neighbor buckets using pointer arithmetic and prefetching
1121 const auto &neighbors = Phi[current_bucket];
1122 const size_t n_neighbors = neighbors.size();
1123 const int *neighbor_ptr = neighbors.data();
1124 const int *neighbor_end = neighbor_ptr + n_neighbors;
1125 for (; neighbor_ptr != neighbor_end; ++neighbor_ptr) {
1126 const int b_prime = *neighbor_ptr;
1127 const size_t word_idx = static_cast<size_t>(b_prime) >> 6;
1128 const uint64_t mask = bit_mask_lookup_ptr[b_prime & 63];
1129 __builtin_prefetch(&Bvisited[word_idx], 0,
1130 1); // Prefetch Bvisited for this segment
1131 if (!(Bvisited[word_idx] & mask)) stack_buffer.push_back(b_prime);
1132 }
1133 }
1134
1135 return false;
1136}

1148template <Stage state, Full fullness>
1149void BucketGraph::run_labeling_algorithms(std::vector<double> &forward_cbar,
1150 std::vector<double> &backward_cbar) {
1151 // Schedule the forward labeling algorithm task on the scheduler.
1152 auto forward_task =
1153 stdexec::schedule(bi_sched) | stdexec::then([&]() {
1154 return labeling_algorithm<Direction::Forward, state, fullness>();
1155 });
1156
1157 // Schedule the backward labeling algorithm task on the scheduler.
1158 auto backward_task =
1159 stdexec::schedule(bi_sched) | stdexec::then([&]() {
1160 return labeling_algorithm<Direction::Backward, state, fullness>();
1161 });
1162
1163 // Combine the forward and backward tasks.
1164 auto combined_work =
1165 stdexec::when_all(std::move(forward_task), std::move(backward_task)) |
1166 stdexec::then([&](auto forward_result, auto backward_result) {
1167 // Store the results into the provided vectors.
1168 forward_cbar = std::move(forward_result);
1169 backward_cbar = std::move(backward_result);
1170 });
1171
1172 // Wait for all tasks to complete.
1173 stdexec::sync_wait(std::move(combined_work));
1174}

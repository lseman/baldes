89 static std::vector<std::vector<int>> computeSCCs(
90 const std::vector<VRPNode> &nodes) {
91 int V = nodes.size();
92 // Build adjacency list: vertices are nodes (using node.id) and an edge
93 // exists from u to v if there is a forward arc from node u to node v.
94 std::vector<std::vector<int>> graph(V);
95 for (const auto &node : nodes) {
96 for (const auto &arc : node.fw_arcs) {
97 // Assumes that node.id is in [0, V-1] and arc.to corresponds to
98 // another node's id.
99 graph[node.id].push_back(arc.to);
100 }
101 }
102
103 std::vector<int> disc(V, -1), low(V, -1);
104 std::vector<bool> inStack(V, false);
105 std::stack<int> st;
106 std::vector<std::vector<int>> sccs;
107 int time = 0;
108
109 // Tarjan's recursive helper.
110 std::function<void(int)> tarjanUtil = [&](int u) {
111 disc[u] = low[u] = time++;
112 st.push(u);
113 inStack[u] = true;
114
115 for (int v : graph[u]) {
116 if (disc[v] == -1) {
117 tarjanUtil(v);
118 low[u] = std::min(low[u], low[v]);
119 } else if (inStack[v]) {
120 low[u] = std::min(low[u], disc[v]);
121 }
122 }
123 if (low[u] == disc[u]) {
124 std::vector<int> scc;
125 while (true) {
126 int w = st.top();
127 st.pop();
128 inStack[w] = false;
129 scc.push_back(w);
130 if (w == u) break;
131 }
132 sccs.push_back(scc);
133 }
134 };
135
136 for (int i = 0; i < V; i++) {
137 if (disc[i] == -1) tarjanUtil(i);
138 }
139 return sccs;
140 }

165 // Helper lambda: compute a score for a given arc.
166 // The base score is defined as the negative cost (scaled) of the
167 // destination node. If the arc is found in arc_scores, add its
168 // bonus/penalty. If any cut applies (i.e. the arc is in the SRC set for
169 // the cut), override the score.
170 auto get_score = [&arc_scores, &nodes,
171 &cuts](const Arc &arc) -> double {
172 // Start with a base score as before.
173 double score = nodes[arc.to].cost / 10.0;
174 if (auto it = arc_scores.find(arc); it != arc_scores.end()) {
175 score += it->second;
176 }
177 for (const auto &cut : cuts) {
178 if (cut.type == CutType::ThreeRow &&
179 cut.isSRCset(arc.from, arc.to)) {
180 score += cut.dual_value / 10.0;
181 break;
182 }
183 }
184 // // Incorporate SCC information.
185 // // For example, if the source and destination are in different
186 // SCCs,
187 // // subtract a penalty.
188 // if (nodes[arc.from].sccId != nodes[arc.to].sccId) {
189 // // The penalty value can be tuned.
190 // score -= 5.0;
191 // } else {
192 // // Optionally, add a bonus if within the same SCC.
193 // score += 1.0;
194 // }
195 return score;
196 };

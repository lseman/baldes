77 void sort_arcs_by_scores(
78 const ankerl::unordered_dense::map<Arc, int, arc_hash> &arc_scores,
79 std::vector<VRPNode> &nodes, std::vector<ActiveCutInfo> &cuts) {
80 // Helper lambda to compute a score for a given arc.
81 // Note: 'cuts' is captured by reference to avoid copying.
82 auto get_score = [&arc_scores, &nodes,
83 &cuts](const Arc &arc) -> double {
84 // Base score: cost of the destination node divided by 100.
85 double score = -nodes[arc.to].cost / 100.0;
86
87 // Add additional score if arc is present in arc_scores.
88 if (auto it = arc_scores.find(arc); it != arc_scores.end()) {
89 score += it->second;
90 }
91
92 // If any cut applies to this arc, override score with the cut's
93 // dual value.
94 for (const auto &cut : cuts) {
95 if (cut.isSRCset(arc.from, arc.to)) {
96 score = -cut.dual_value / 100.0;
97 break;
98 }
99 }
100 return score;
101 };

104 if constexpr (D == Direction::Forward) {
105 for (auto &arcs : fw_arcs_scc) {
106 pdqsort(arcs.begin(), arcs.end(),
107 [&](const Arc &a, const Arc &b) {
108 return get_score(a) > get_score(b);

76 static std::vector<std::vector<int>> computeSCCs(const std::vector<VRPNode> &nodes) {
77 int V = nodes.size();
78 // Build adjacency list: vertices are nodes (using node.id) and an edge
79 // exists from u to v if there is a forward arc from node u to node v.
80 std::vector<std::vector<int>> graph(V);
81 for (const auto &node : nodes) {
82 for (const auto &arc : node.fw_arcs) {
83 // Assumes that node.id is in [0, V-1] and arc.to corresponds to
84 // another node's id.
85 graph[node.id].push_back(arc.to);
86 }
87 }
88
89 std::vector<int> disc(V, -1), low(V, -1);
90 std::vector<bool> inStack(V, false);
91 std::stack<int> st;
92 std::vector<std::vector<int>> sccs;
93 int time = 0;
94
95 // Tarjan's recursive helper.
96 std::function<void(int)> tarjanUtil = [&](int u) {
97 disc[u] = low[u] = time++;
98 st.push(u);
99 inStack[u] = true;
100
101 for (int v : graph[u]) {
102 if (disc[v] == -1) {
103 tarjanUtil(v);
104 low[u] = std::min(low[u], low[v]);
105 } else if (inStack[v]) {
106 low[u] = std::min(low[u], disc[v]);
107 }
108 }
109 if (low[u] == disc[u]) {
110 std::vector<int> scc;
111 while (true) {
112 int w = st.top();
113 st.pop();
114 inStack[w] = false;
115 scc.push_back(w);
116 if (w == u) break;
117 }
118 sccs.push_back(scc);
119 }
120 };
121
122 for (int i = 0; i < V; i++) {
123 if (disc[i] == -1) tarjanUtil(i);
124 }
125 return sccs;
126 }
127 int sccId = -1;

148 // Helper lambda: compute a score for a given arc.
149 // The base score is defined as the negative cost (scaled) of the
150 // destination node. If the arc is found in arc_scores, add its
151 // bonus/penalty. If any cut applies (i.e. the arc is in the SRC set for
152 // the cut), override the score.
153 auto get_score = [&arc_scores, &nodes, &cuts](const Arc &arc) -> double {
154 // Start with a base score as before.
155 double score = nodes[arc.to].cost / 10.0;
156 if (auto it = arc_scores.find(arc); it != arc_scores.end()) { score += it->second; }
157 for (const auto &cut : cuts) {
158 if (cut.type == CutType::ThreeRow && cut.isSRCset(arc.from, arc.to)) {
159 score += cut.dual_value / 10.0;
160 break;
161 }
162 }
163 // // Incorporate SCC information.
164 // // For example, if the source and destination are in different
165 // SCCs,
166 // // subtract a penalty.
167 // if (nodes[arc.from].sccId != nodes[arc.to].sccId) {
168 // // The penalty value can be tuned.
169 // score -= 5.0;
170 // } else {
171 // // Optionally, add a bonus if within the same SCC.
172 // score += 1.0;
173 // }
174 return score;

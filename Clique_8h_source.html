70 // Build and update the conflict graph using binary variables
71 void buildUpdateCg(const std::vector<std::vector<int>> &set, int binary_number) {
72 initCg(binary_number); // Initialize conflict graph for binary variables
73
74 // Create tasks for each combination of clique[i] and clique[j] for all cliques in `set`
75 std::vector<std::tuple<int, int>> tasks;
76 tasks.reserve(set.size() * (binary_number * (binary_number - 1)) / 2); // Preallocate task space for conflicts
77
78 for (const auto &clique : set) {
79 for (size_t i = 0; i < clique.size(); ++i) {
80 for (size_t j = i + 1; j < clique.size(); ++j) {
81 int var_i = clique[i];
82 int var_j = clique[j];
83
84 // Only process binary variables
85 if (var_i < binary_number && var_j < binary_number) { tasks.emplace_back(var_i, var_j); }
86 }
87 }
88 }
89
90 // Define chunk size for processing tasks in batches
91 const int chunk_size = 100; // Adjust based on performance needs
92
93 // Define a bulk sender to process tasks in parallel
94 auto bulk_sender =
95 stdexec::bulk(stdexec::just(), (tasks.size() + chunk_size - 1) / chunk_size,
96 [this, &tasks, chunk_size](std::size_t chunk_idx) {
97 size_t start_idx = chunk_idx * chunk_size;
98 size_t end_idx = std::min(start_idx + chunk_size, tasks.size());
99
100 // Process a chunk of tasks
101 for (size_t task_idx = start_idx; task_idx < end_idx; ++task_idx) {
102 const auto &[var_i, var_j] = tasks[task_idx];
103
104 // Mark conflict between var_i and var_j (and vice versa) in the conflict graph
105 {
106 std::lock_guard<std::mutex> lock(conflictGraph_mutex);
107 conflictGraph[var_i][var_j] = true; // Mark conflict between var_i and var_j
108 conflictGraph[var_j][var_i] = true; // Mark conflict between var_j and var_i
109 }
110 }
111 });
112 exec::static_thread_pool pool(std::thread::hardware_concurrency());
113 auto sched = pool.get_scheduler();
114 // Submit work to the scheduler for parallel execution
115 auto work = stdexec::starts_on(sched, bulk_sender);
116 stdexec::sync_wait(std::move(work));
117 }

119 void findCliques() {
120 cliques.clear(); // Clear previous cliques
121 ankerl::unordered_dense::map<int, std::vector<std::pair<double, int>>> rowToElements;
122
123 // Populate rowToElements for efficient row-wise access from SparseMatrix elements
124 for (const auto &elem : modelData.A_sparse.elements) {
125 int row = elem.row;
126 int col = elem.col;
127 double coeff = elem.value;
128
129 rowToElements[row].emplace_back(coeff, col);
130 }
131
132 // Prepare tasks for parallel processing. Each task represents a row to process.
133 std::vector<int> tasks(modelData.b.size());
134 std::iota(tasks.begin(), tasks.end(), 0); // Populate tasks with row indices
135
136 // Define chunk size for processing tasks in batches
137 const int chunk_size = 100; // Adjust based on performance needs
138
139 // Define a bulk sender to process tasks in parallel
140 auto bulk_sender =
141 stdexec::bulk(stdexec::just(), (tasks.size() + chunk_size - 1) / chunk_size,
142 [this, &tasks, &rowToElements, chunk_size](std::size_t chunk_idx) {
143 size_t start_idx = chunk_idx * chunk_size;
144 size_t end_idx = std::min(start_idx + chunk_size, tasks.size());
145
146 // Process a chunk of tasks (i.e., rows)
147 for (size_t task_idx = start_idx; task_idx < end_idx; ++task_idx) {
148 int row = tasks[task_idx]; // Get the row index
149 double rhs = modelData.b[row];
150
151 std::vector<std::pair<double, int>> binCoeffs;
152 binCoeffs.reserve(100); // Reserve space for binary coefficients
153
154 auto &rowElements = rowToElements[row];
155
156 // Adjust RHS and collect binary variable coefficients
157 for (const auto &elem : rowElements) {
158 double coeff = elem.first;
159 int col = elem.second;
160
161 if (modelData.vtype[col] != 'B') {
162 rhs -= (coeff > 0 ? coeff * modelData.ub[col] : 0);
163 } else {
164 binCoeffs.emplace_back(coeff, col);
165 }
166 }
167
168 if (binCoeffs.size() < 2) continue; // Skip rows with fewer than 2 binary coefficients
169
170 std::sort(binCoeffs.begin(), binCoeffs.end());
171
172 if (binCoeffs.back().first + binCoeffs[binCoeffs.size() - 2].first < rhs) continue;
173
174 int k = findFirstClique(binCoeffs, rhs);
175 if (k == -1) continue;
176
177 addCliqueFromIndex(binCoeffs, k);
178
179 processRemainingCliques(binCoeffs, k, rhs);
180 }
181 });
182
183 exec::static_thread_pool pool(std::thread::hardware_concurrency());
184 auto sched = pool.get_scheduler();
185 // Submit work to the scheduler for parallel execution
186 auto work = stdexec::starts_on(sched, bulk_sender);
187 stdexec::sync_wait(std::move(work));
188 }

196 int findFirstClique(const std::vector<std::pair<double, int>> &binCoeffs, double rhs) {
197 int left = 0;
198 int right = binCoeffs.size() - 2; // We are comparing two elements, hence size - 2.
199 int k = -1;
200
201 // Binary search for the first index where the sum of binCoeffs[mid] and binCoeffs[mid + 1] > rhs
202 while (left <= right) {
203 int mid = left + (right - left) / 2;
204
205 // Check if the sum of two consecutive coefficients exceeds rhs
206 if (binCoeffs[mid].first + binCoeffs[mid + 1].first > rhs) {
207 k = mid; // Found a valid clique start index
208 right = mid - 1; // Narrow down the search to find the earliest such index
209 } else {
210 left = mid + 1; // Move the search to the right
211 }
212 }
213
214 return k; // If no valid clique is found, k will remain -1.
215 }

217 void processRemainingCliques(const std::vector<std::pair<double, int>> &binCoeffs, int k, double rhs) {
218 // Process cliques starting from earlier indices before k
219 for (int o = k - 1; o >= 0; --o) {
220 // Find the smallest index f such that binCoeffs[o] + binCoeffs[f] > rhs
221 int left = o + 1;
222 int right = binCoeffs.size() - 1;
223 int f = -1;
224
225 while (left <= right) {
226 int mid = left + (right - left) / 2;
227
228 // If the sum of the two coefficients exceeds rhs, try to find the smallest valid f
229 if (binCoeffs[o].first + binCoeffs[mid].first > rhs) {
230 f = mid;
231 right = mid - 1;
232 } else {
233 left = mid + 1;
234 }
235 }
236
237 // If a valid f was found, we can form another clique
238 if (f != -1) {
239 std::vector<int> clique;
240 clique.push_back(binCoeffs[o].second); // Add the element at index o
241 for (int j = f; j < binCoeffs.size(); ++j) {
242 clique.push_back(binCoeffs[j].second); // Add elements starting from index f
243 }
244
245 // Add this new clique (assuming addClique is a function that stores the clique)
246 addClique(clique);
247 }
248 }

289 void extendCliques(const std::vector<std::vector<int>> &initialCliques, std::vector<std::vector<int>> &extCliques) {
290 // Iterate over each initial clique and attempt to extend it
291 for (const auto &clique : initialCliques) {
292 ankerl::unordered_dense::set<int> cliqueSet(clique.begin(),
293 clique.end()); // Store the clique for easy lookup
294 std::vector<int> extendedClique = clique; // Start with the current clique
295
296 bool extended = false; // Keep track if we successfully extend the clique
297
298 // Create a list of candidate vertices (not in the clique) and sort them by degree
299 std::vector<int> candidates;
300 for (size_t v = 0; v < conflictGraph.size(); ++v) {
301 if (cliqueSet.find(v) == cliqueSet.end()) { candidates.push_back(v); }
302 }
303
304 // Sort candidates by their degree (most connections to other vertices first)
305 std::sort(candidates.begin(), candidates.end(), [&](int a, int b) {
306 return std::count(conflictGraph[a].begin(), conflictGraph[a].end(), true) >
307 std::count(conflictGraph[b].begin(), conflictGraph[b].end(), true);
308 });
309
310 // Try to extend the clique with vertices sorted by their degree
311 for (int v : candidates) {
312 bool canExtend = true;
313 for (int u : clique) {
314 if (!conflictGraph[v][u]) {
315 canExtend = false; // If v is not connected to any member of the clique, break out
316 break;
317 }
318 }
319
320 if (canExtend) {
321 extendedClique.push_back(v);
322 cliqueSet.insert(v); // Add v to the clique set
323 extended = true;
324 }
325 }
326
327 // Add the extended clique to the result
328 if (extended) {
329 extCliques.push_back(extendedClique);
330 } else {
331 extCliques.push_back(clique); // If no extension was possible, keep the original clique
332 }
333 }

106 void updatePenaltyFunction(const std::vector<double> &nodeDuals) {
107 // Calculate average difference
108 double differenceSum = 0.0;
109 int numStabConstraints = 0;
110
111 for (size_t i = 0; i < numConstrs; i++) {
112 double value = std::abs(nodeDuals[i] - prevDuals[i]);
113 differenceSum += value;
114 numStabConstraints++;
115 }
116
117 averDifference = numStabConstraints > 0 ? differenceSum / numStabConstraints : 0.0;
118
119 // Three-piece penalty function update
120 if (savedInnerInterval == 0.0) {
121 savedInnerInterval = averDifference;
122 } else {
123 savedInnerInterval = savedInnerInterval + (averDifference - savedInnerInterval) * CURVATURE_RATE;
124 }
125
126 if (savedInnerInterval < std::numeric_limits<double>::infinity()) {
127 innerHalfInterval = savedInnerInterval * 0.5;
128 v = savedInnerInterval / c;
129 } else {
130 innerHalfInterval = 0.0;
131 v = 0.0;
132 }
133 }

152 double iterate(BNBNode *node, std::vector<double> &nodeDuals, double inner_obj, int stage) {
153 isInsideTrustRegion = true;
154 TRstop = false;
155
156 // Check if solution is inside trust region
157 for (int i = 0; i < numConstrs; i++) {
158 if (nodeDuals[i] < delta1[i] || nodeDuals[i] > delta2[i]) {
159 isInsideTrustRegion = false;
160 break;
161 }
162 }
163
164 double dualDiffSum = 0.0;
165 for (int i = 0; i < numConstrs; i++) { dualDiffSum += std::abs(nodeDuals[i] - prevDuals[i]); }
166
167 if (isInsideTrustRegion && dualDiffSum > EPSILON) {
168 double avg_diff = dualDiffSum / numConstrs;
169
170 // More conservative v updates
171 if (avg_diff < v) {
172 v = std::max(v * 0.9, avg_diff);
173 } else {
174 v = std::min(v * 1.1, avg_diff);
175 }
176
177 // Adjust kappa based on progress
178 double progress = (inner_obj - prev_inner_obj) / std::abs(prev_inner_obj + EPSILON);
179 adjustParameters(progress);
180
181 // More robust epsilon calculation
182 double new_eps = std::max(MIN_V, std::min(v / c, MAX_EPSILON));
183
184 // Update variable bounds and coefficients
185 updateVariables(nodeDuals, new_eps);
186 }
187
188 updateBounds(nodeDuals);
189
190 if (v <= MIN_V || stage == 4) {
191 cleanup(node);
192 return 0.0;
193 }
194
195 prevDuals = nodeDuals;
196 prev_inner_obj = inner_obj;
197 return v;
198 }

204 void updateVariables(const std::vector<double> &nodeDuals, double new_eps) {
205 for (int i = 0; i < numConstrs; i++) {
206 // Safeguard the dual values
207 double safeDual = safeguardValue(nodeDuals[i], -MAX_DUAL_CHANGE, MAX_DUAL_CHANGE);
208
209 // Update bounds with safeguards
210 w[i]->setUB(new_eps);
211 zeta[i]->setUB(new_eps);
212
213 // Update coefficients with safeguards
214 w[i]->setOBJ(std::max(0.0, safeDual - safeguardValue(v, 0.0, MAX_V)));
215 zeta[i]->setOBJ(safeDual + safeguardValue(v, 0.0, MAX_V));
216 }

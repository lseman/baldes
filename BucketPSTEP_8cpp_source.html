187inline void generateInitialPaths(int n, std::vector<std::vector<uint16_t>> &paths, std::vector<double> &path_costs,
188 std::vector<int> &firsts, std::vector<int> &lasts,
189 const std::vector<std::vector<double>> &cost_matrix) {
190 std::vector<bool> visited(n, false);
191 std::vector<uint16_t> initial_path;
192 int current_node = 0;
193 visited[current_node] = true;
194 initial_path.push_back(current_node);
195
196 auto cost = 0.0;
197
198 // Generate the path using the nearest neighbor heuristic
199 for (int step = 1; step < n; ++step) {
200 int nearest_node = -1;
201 double min_cost = std::numeric_limits<double>::max();
202
203 for (int j = 0; j < n; ++j) {
204 if (!visited[j] && cost_matrix[current_node][j] < min_cost) {
205 nearest_node = j;
206 min_cost = cost_matrix[current_node][j];
207 }
208 }
209
210 if (nearest_node != -1) {
211 initial_path.push_back(nearest_node);
212 visited[nearest_node] = true;
213 current_node = nearest_node;
214 }
215 cost += min_cost;
216 }
217 // Return to the starting node to complete the cycle
218 cost += cost_matrix[current_node][0];
219 initial_path.push_back(n);
220 paths.push_back(initial_path);
221 firsts.push_back(initial_path.front());
222 lasts.push_back(initial_path.back());
223 path_costs.push_back(cost);
224
225 fmt::print("Initial heuristic cost = {}\n", cost);
226}

228auto BucketGraph::solveTSP(std::vector<std::vector<uint16_t>> &paths, std::vector<double> &path_costs,
229 std::vector<int> &firsts, std::vector<int> &lasts,
230 std::vector<std::vector<double>> &cost_matrix, bool first_time) {
231 auto n = cost_matrix.size();
232 MIPProblem model = MIPProblem("tsp", 0, 0);
233 if (first_time) { generateInitialPaths(n - 1, paths, path_costs, firsts, lasts, cost_matrix); }
234 // Define variables
235 int R = paths.size(); // Number of p-steps
236 std::vector<baldesVarPtr> x(R); // Binary variables for each p-step
237 std::vector<baldesVarPtr> u(n); // Continuous variables for each node
238
239 // Create binary variables x_r for each p-step r
240 for (int i = 0; i < R; ++i) {
241 // fmt::print("path_costs[{}] = {}\n", i, path_costs[i]);
242 std::string varname = "x_" + std::to_string(i);
243 double custo = path_costs[i];
244 x[i] = model.add_variable(varname, VarType::Continuous, 0.0, 1.0, custo);
245 }
246
247 // Create continuous variables u_i for each node i
248 for (int i = 0; i < n; ++i) {
249 auto name = "u_" + std::to_string(i);
250 u[i] = model.add_variable(name, VarType::Continuous, 0.0, n - 1, 0.0);
251 }
252 model.setObjectiveSense(ObjectiveType::Minimize);
253
254 // Constraints (3.3) and (3.4):
255 std::vector<baldesCtrPtr> three_two(n);
256 std::vector<baldesCtrPtr> three_three(n);
257 for (int i = 1; i < n - 1; ++i) {
258 LinearExpression in_constraint;
259 LinearExpression out_constraint;
260 for (int r = 0; r < R; ++r) {
261 if (firsts[r] == i) { in_constraint += x[r]; }
262 if (lasts[r] == i) { in_constraint -= x[r]; }
263 if (std::find(paths[r].begin(), paths[r].end(), i) != paths[r].end() && i != lasts[r]) {
264 out_constraint += x[r];
265 }
266 }
267 three_two[i] = model.add_constraint(in_constraint == 0, "33_" + std::to_string(i));
268 three_three[i] = model.add_constraint(out_constraint == 1, "34_" + std::to_string(i));
269 }
270
271 // Constraints (3.5): Avoid sub-tours using MTZ constraints
272 std::vector<baldesCtrPtr> three_five_constraints;
273 std::vector<std::vector<baldesCtrPtr>> three_five_constraints_matrix(n, std::vector<baldesCtrPtr>(n));
274 for (int i = 0; i < n - 1; ++i) {
275 for (int j = 1; j < n; ++j) {
276 if (i != j) {
277 LinearExpression r_sum;
278 for (int r = 0; r < R; ++r) {
279 for (int k = 0; k < paths[r].size() - 1; ++k) {
280 if (paths[r][k] == i && paths[r][k + 1] == j) { r_sum += x[r]; }
281 }
282 }
283 auto ctr = u[i] - u[j] + (n - 1) * r_sum <= (n - 2);
284 three_five_constraints_matrix[i][j] =
285 model.add_constraint(ctr, "mtz_" + std::to_string(i) + "_" + std::to_string(j));
286 }
287 }
288 }
289
290 for (int i = 1; i < n; i++) {
291 model.add_constraint(u[i] >= 0, "u_lb_" + std::to_string(i));
292 model.add_constraint(u[i] <= n - 1, "u_ub_" + std::to_string(i));
293 }
294
295#ifdef IPM
296 IPSolver *ipSolver = new IPSolver();
297 auto matrix = model.extractModelDataSparse();
298 ipSolver->run_optimization(matrix, 1e-2);
299
300 auto duals = ipSolver->getDuals();
301#else
302 auto duals = std::vector<double>(N_SIZE, 0.0);
303#endif
304 std::vector<double> three_two_duals(n, 0.0);
305 std::vector<double> three_three_duals(n, 0.0);
306 std::vector<std::vector<double>> three_five_duals(n, std::vector<double>(n + 1, 0.0));
307 for (int i = 1; i < n - 1; ++i) {
308 three_two_duals[i] = options.three_two_sign * duals[three_two[i]->index()];
309 three_three_duals[i] = options.three_three_sign * duals[three_three[i]->index()];
310 }
311
312 // Extract duals for three_five constraints
313 for (int i = 0; i < n - 1; ++i) {
314 for (int j = 1; j < n; ++j) {
315 if (i != j) {
316 three_five_duals[i][j] =
317 options.three_five_sign * (n - 1) * duals[three_five_constraints_matrix[i][j]->index()];
318 }
319 }
320 }
321#ifdef IPM
322 auto lp_obj = ipSolver->getObjective();
323#else
324 auto lp_obj = 0.0;
325#endif
326 fmt::print("LP Objective: {}\n", lp_obj);
327 GRBEnv &env = GurobiEnvSingleton::getInstance();
328 GRBModel *modelGRB = model.toGurobiModel(env).release(); // Release ownership from unique_ptr
329 // set model verbose
330 modelGRB->set(GRB_IntParam_OutputFlag, 0);
331 // set model type as 2
332 modelGRB->set(GRB_IntParam_Method, 2);
333
334 // convert x vars to int and solve again
335 for (int i = 0; i < R; ++i) {
336 modelGRB->getVarByName("x_" + std::to_string(i)).set(GRB_CharAttr_VType, GRB_BINARY);
337 }
338 modelGRB->update();
339 modelGRB->optimize();
340
341 if (modelGRB->get(GRB_IntAttr_Status) == GRB_OPTIMAL) {
342 fmt::print("Integer solution found with value: {}\n", modelGRB->get(GRB_DoubleAttr_ObjVal));
343 for (int i = 0; i < R; ++i) {
344 if (modelGRB->getVarByName("x_" + std::to_string(i)).get(GRB_DoubleAttr_X) > 0.5) {
345 fmt::print("x_{} = {}\n", i, modelGRB->getVarByName("x_" + std::to_string(i)).get(GRB_DoubleAttr_X));
346 // print path nodes
347 for (auto node : paths[i]) { fmt::print("{} ", node); }
348 fmt::print("\n");
349 }
350 }
351 }
352
353 return std::make_tuple(three_two_duals, three_three_duals, three_five_duals);
354}

356std::vector<Label *> BucketGraph::solvePSTEP_by_MTZ() {
357 int JOBS = nodes.size();
358
359 std::vector<std::vector<uint16_t>> all_paths;

365 auto max_pstep_iterations = 10;
366 for (auto z = 0; z < max_pstep_iterations; ++z) {
367 fmt::print("--------------------------------------\n");
368 fmt::print("Iteration {}\n", z);
369 auto three_two_duals = std::get<0>(duals);
370 auto three_three_duals = std::get<1>(duals);
371 auto three_five_duals = std::get<2>(duals);
372 PSTEPDuals inner_pstep_duals;
373
374 // Convert three_two to vector of pairs
375 std::vector<std::pair<int, double>> three_two_tuples;
376 three_two_tuples.reserve(three_two_duals.size());
377 std::transform(three_two_duals.begin(), three_two_duals.end(), std::back_inserter(three_two_tuples),
378 [index = 0](double value) mutable { return std::make_pair(index++, value); });
379 inner_pstep_duals.setThreeTwoDualValues(std::move(three_two_tuples));
380
381 // Convert three_three to vector of pairs
382 std::vector<std::pair<int, double>> three_three_tuples;
383 three_three_tuples.reserve(three_three_duals.size());
384 std::transform(three_three_duals.begin(), three_three_duals.end(), std::back_inserter(three_three_tuples),
385 [index = 0](double value) mutable { return std::make_pair(index++, value); });
386 inner_pstep_duals.setThreeThreeDualValues(std::move(three_three_tuples));
387
388 // Convert three_five to vector of pairs of pairs
389 std::vector<std::pair<std::pair<int, int>, double>> arc_duals_tuples;
390 size_t total_size = 0;
391 for (const auto &row : three_five_duals) total_size += row.size();
392 arc_duals_tuples.reserve(total_size);
393
394 for (size_t i = 0; i < three_five_duals.size(); ++i) {
395 for (size_t j = 0; j < three_five_duals[i].size(); ++j) {
396 if (i != j) { // Skip diagonal or add specific condition if needed
397 arc_duals_tuples.emplace_back(std::pair{i, j}, three_five_duals[i][j]);
398 }
399 }
400 }
401 inner_pstep_duals.setArcDualValues(std::move(arc_duals_tuples));
402
403 auto sub_paths = solvePSTEP(inner_pstep_duals);
404 // print sub_paths.size
405 auto paths_added = 0;
406 for (auto path : sub_paths) {
407 if (path->cost >= 0) { continue; }
408 paths_added++;
409 all_paths.push_back(path->getRoute());
410 all_costs.push_back(path->real_cost);
411 all_firsts.push_back(path->nodes_covered.front());
412 all_lasts.push_back(path->nodes_covered.back());
413 }
414 if (paths_added == 0) { break; }
415
416 duals = solveTSP(all_paths, all_costs, all_firsts, all_lasts, distance_matrix);
417 }
418 return {};

421auto BucketGraph::solveTSPTW(std::vector<std::vector<uint16_t>> &paths, std::vector<double> &path_costs,
422 std::vector<int> &firsts, std::vector<int> &lasts,
423 std::vector<std::vector<double>> &cost_matrix, std::vector<double> &service_times,
424 std::vector<double> &time_windows_start, std::vector<double> &time_windows_end,
425 bool first_time) {
426 auto n = cost_matrix.size();
427 MIPProblem model = MIPProblem("tsptw", 0, 0);
428
429 if (first_time) {
430 generateInitialPathsTSPTW(n - 1, paths, path_costs, firsts, lasts, cost_matrix, time_windows_start,
431 time_windows_end, service_times);
432 }
433
434 // Variables
435 int R = paths.size();
436 std::vector<baldesVarPtr> x(R); // Binary variables for p-steps
437 std::vector<baldesVarPtr> omega(n); // Continuous variables for accumulated time
438
439 // Create binary variables x_r for each p-step r
440 for (int i = 0; i < R; ++i) {
441 x[i] = model.add_variable("x_" + std::to_string(i), VarType::Binary, 0.0, 1.0, path_costs[i]);
442 }
443
444 // Create continuous variables ω_i for time accumulation
445 for (int i = 0; i < n; ++i) {
446 omega[i] = model.add_variable("omega_" + std::to_string(i), VarType::Continuous, 0.0, 1e8, 0.0);
447 }
448 model.setObjectiveSense(ObjectiveType::Minimize);
449
450 std::vector<baldesCtrPtr> three_two(n);
451 std::vector<baldesCtrPtr> three_three(n);
452 // Constraints (4.11): Flow balance for p-steps
453 for (int i = 1; i < n - 1; ++i) {
454 LinearExpression flow_balance;
455 for (int r = 0; r < R; ++r) {
456 if (firsts[r] == i) flow_balance += x[r];
457 if (lasts[r] == i) flow_balance -= x[r];
458 }
459 three_two[i] = model.add_constraint(flow_balance == 0, "flow_" + std::to_string(i));
460 }
461
462 // Constraints (4.12): Visit each node at most once
463 for (int i = 1; i < n - 1; ++i) {
464 LinearExpression visit_once;
465 for (int r = 0; r < R; ++r) {
466 if (std::find(paths[r].begin(), paths[r].end(), i) != paths[r].end()) { visit_once += x[r]; }
467 }
468 three_three[i] = model.add_constraint(visit_once <= 1, "visit_" + std::to_string(i));
469 }
470
471 // Constraint (4.13): Visit source exactly once
472 LinearExpression source_visit;
473 for (int r = 0; r < R; ++r) {
474 if (firsts[r] == 0) { source_visit += x[r]; }
475 }
476 model.add_constraint(source_visit == 1, "source");
477
478 // Constraints (4.14): Time accumulation with big-M
479 std::vector<baldesCtrPtr> three_five_constraints;
480 std::vector<std::vector<baldesCtrPtr>> three_five_constraints_matrix(n, std::vector<baldesCtrPtr>(n));
481 for (int i = 0; i < n - 1; ++i) {
482 for (int j = 1; j < n; ++j) {
483 if (i != j) {
484 double M_ij = 1e8;
485 LinearExpression time_acc;
486 for (int r = 0; r < R; ++r) {
487 for (size_t k = 0; k < paths[r].size() - 1; ++k) {
488 if (paths[r][k] == i && paths[r][k + 1] == j) { time_acc += x[r]; }
489 }
490 }
491 three_five_constraints_matrix[i][j] = model.add_constraint(
492 omega[j] - omega[i] - (cost_matrix[i][j] + service_times[i]) * time_acc - M_ij * time_acc >= -M_ij,
493 "time_" + std::to_string(i) + "_" + std::to_string(j));
494 }
495 }
496 }
497
498 // define omega[0] = 0
499
500 // Constraints (4.15): Time window constraints
501 for (int i = 1; i < n - 1; ++i) {
502 LinearExpression node_used;
503 for (int r = 0; r < R; ++r) {
504 if (std::find(paths[r].begin(), paths[r].end(), i) != paths[r].end()) { node_used += x[r]; }
505 }
506 model.add_constraint(omega[i] - time_windows_start[i] * node_used >= 0, "tw_lb_" + std::to_string(i));
507 model.add_constraint(omega[i] - time_windows_end[i] * node_used <= 0, "tw_ub_" + std::to_string(i));
508 }
509
510// Solve the model
511#ifdef IPM
512 IPSolver *ipSolver = new IPSolver();
513 auto matrix = model.extractModelDataSparse();
514 ipSolver->run_optimization(matrix, 1e-2);
515
516 // Extract solution and duals
517 auto duals = ipSolver->getDuals();
518#else
519 auto duals = std::vector<double>(N_SIZE, 0.0);
520#endif
521 std::vector<double> three_two_duals(n, 0.0);
522 std::vector<double> three_three_duals(n, 0.0);
523 std::vector<std::vector<double>> three_five_duals(n, std::vector<double>(n + 1, 0.0));
524 for (int i = 1; i < n - 1; ++i) {
525 three_two_duals[i] = options.three_two_sign * duals[three_two[i]->index()];
526 three_three_duals[i] = options.three_three_sign * duals[three_three[i]->index()];
527 fmt::print("Duals for node {}: 3.2 = {}, 3.3 = {}\n", i, three_two_duals[i], three_three_duals[i]);
528 }
529
530 // Extract duals for three_five constraints
531 for (int i = 0; i < n - 1; ++i) {
532 for (int j = 1; j < n; ++j) {
533 if (i != j) {
534 three_five_duals[i][j] =
535 options.three_five_sign * (n - 1) * duals[three_five_constraints_matrix[i][j]->index()];
536 }
537 }
538 }
539
540 // Solution output and return
541 // fmt::print("LP Objective: {}\n", ipSolver->getObjective());
542
543 GRBEnv &env = GurobiEnvSingleton::getInstance();
544
545 GRBModel *modelGRB = model.toGurobiModel(env).release();
546 // set model verbose

72template <Stage S, Symmetry SYM>
73void BucketGraph::concatenate_from_forward_arc(const Label *label, const BucketArc &arc, BucketPricingPass &pass) {
74 const int to_node = arc.jump ? arc.jump_to_node : fw_buckets[arc.to_bucket].node_id;
75 if (to_node < 0) return;
76 if constexpr (S >= Stage::Three || S == Stage::Eliminate) {
77 if (is_arc_fixed(label->node_id, to_node)) return;
78 }
79
80 auto extended_bucket =
81 Extend<Direction::Forward, S, ArcType::Bucket, Mutability::Const, Full::Reverse>(label, arc);
82 if (extended_bucket == -1) return;
83
84 SpliceState splice_state;
85 splice_state.resources = label->resources;
86
87 std::array<double, R_SIZE> base_resources = label->resources;
88 if (arc.jump) {
89 const auto &jump_bucket = fw_buckets[arc.to_bucket];
90 for (size_t r = 0; r < options.resources.size(); ++r) {
91 base_resources[r] = std::max(base_resources[r], jump_bucket.lb[r]);
92 }
93 }
94
95 std::vector<double> splice_resources(options.resources.size());
96 if (!process_all_resources<Direction::Forward>(splice_resources, base_resources, arc, nodes[to_node],
97 options.resources.size())) {
98 return;
99 }
100
101 for (size_t r = 0; r < options.resources.size(); ++r) { splice_state.resources[r] = splice_resources[r]; }
102
103#if defined(SRC)
104 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
105 if (cut_storage && cut_storage->activeSize() > 0) {
106 splice_state.SRCmap = label->SRCmap;
107 splice_state.has_src = true;
108 }
109 }
110#endif
111
112 int bucket_to_process = extended_bucket;
113 concatenate_label_from_bucket<S, SYM>(label, bucket_to_process, pass.best_cost, &splice_state);
114}

116template <Stage S, Symmetry SYM>
117void BucketGraph::concatenate_pricing_pass(BucketPricingPass &pass) {
118 concatenation_labels_tested.store(0, std::memory_order_relaxed);
119 concatenation_labels_accepted.store(0, std::memory_order_relaxed);
120
121 const size_t chunk_size = std::max<size_t>(1, fw_buckets_size / MERGE_SCHED_CONCURRENCY);
122 const size_t n_chunks = (fw_buckets_size + chunk_size - 1) / chunk_size;
123
124 auto bulk_sender = stdexec::bulk(
125 stdexec::just(), n_chunks,
126 [this, chunk_size, &pass](std::size_t chunk_idx) {
127 const size_t start_bucket = chunk_idx * chunk_size;
128 const size_t end_bucket = std::min(start_bucket + chunk_size, static_cast<size_t>(fw_buckets_size));
129
130 for (size_t bucket = start_bucket; bucket < end_bucket; ++bucket) {
131 if constexpr (S == Stage::Four) {
132 if (pricing_truncated.load(std::memory_order_relaxed)) break;
133 }
134 const auto &bucket_labels = fw_buckets[bucket].get_labels();
135 for (const Label *L : bucket_labels) {
136 if constexpr (S == Stage::Four) {
137 if (pricing_truncated.load(std::memory_order_relaxed)) break;
138 }
139 if (L->is_dominated) continue;
140
141 pass.non_dominated_labels.fetch_add(1, std::memory_order_relaxed);
142
143 const auto &to_arcs = fw_buckets[bucket].template get_bucket_arcs<Direction::Forward>();
144 for (const auto &arc : to_arcs) {
145 if constexpr (S == Stage::Four) {
146 if (pricing_truncated.load(std::memory_order_relaxed)) break;
147 }
148 concatenate_from_forward_arc<S, SYM>(L, arc, pass);
149 }
150 }
151 }
152 });
153
154 auto work = stdexec::starts_on(merge_sched, bulk_sender);
155 stdexec::sync_wait(std::move(work));
156 last_concatenation_labels_tested = concatenation_labels_tested.load(std::memory_order_relaxed);
157 last_concatenation_labels_accepted = concatenation_labels_accepted.load(std::memory_order_relaxed);
158}

160template <Stage S>
161BucketGraph::BucketPricingResult BucketGraph::finalize_pricing_pass() {
162 pdqsort(merged_labels.begin(), merged_labels.end(),
163 [](const Label *a, const Label *b) { return a->cost < b->cost; });
164
165#ifdef SCHRODINGER
166 if (merged_labels.size() > N_ADD) {
167 std::vector<Path> paths;
168 const int labels_size = merged_labels.size();
169 const int end_idx = std::min(N_ADD + N_ADD, labels_size);
170 paths.reserve(end_idx - N_ADD);
171 for (int i = N_ADD; i < end_idx; ++i) {
172 const auto &route = merged_labels[i]->getRoute();
173 if (route.size() <= 3) continue;
174 paths.emplace_back(route, merged_labels[i]->real_cost);
175 }
176 sPool.add_paths(paths);
177 sPool.iterate();
178 }
179#endif
180
181#ifdef RIH
182 if constexpr (S == Stage::Four) {
183 std::vector<Label *> top_labels;
184 top_labels.reserve(N_ADD);
185 const int n_candidates = std::min(N_ADD, static_cast<int>(merged_labels.size()));
186 for (int i = 0; i < n_candidates; ++i) {
187 if (merged_labels[i]->nodes_covered.size() <= 3) continue;
188 top_labels.push_back(merged_labels[i]);
189 }
190 ils->submit_task(top_labels, nodes);
191 }
192#endif
193
194 inner_obj = merged_labels[0]->cost;
195 return make_bucket_pricing_result(merged_labels);
196}

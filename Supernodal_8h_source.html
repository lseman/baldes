117 void computeEliminationTree() {
118 const StorageIndex n = m_matrix.rows();
119 m_parent.assign(n, -1);
120
121 std::vector<StorageIndex> ancestor(n, -1);
122
123 for (StorageIndex k = 0; k < n; k++) {
124 ancestor[k] = k;
125
126 for (typename MatrixType::InnerIterator it(m_matrix, k); it; ++it) {
127 StorageIndex i = it.row();
128 if (i < k) continue; // Skip upper triangular part
129
130 StorageIndex j = k;
131 while (j != -1 && j < i) {
132 StorageIndex next = ancestor[j];
133 ancestor[j] = i;
134 if (next == j) {
135 m_parent[j] = i;
136 break;
137 }
138 j = next;
139 }
140 }

378 void solveInternal(VectorType& x) const {
379 const StorageIndex n = x.size();
380 std::cout << "\n=== Starting Solve Phase ===\n";
381
382 // Make a copy of the right-hand side
383 VectorType b = x;
384
385 // 1. Forward substitution (Ly = b)
386 std::cout << "\nForward substitution (Ly = b):\n";
387 std::cout << "Initial x: " << x.transpose() << "\n";
388
389 // Process one column at a time, not by supernodes
390 for (StorageIndex j = 0; j < n; j++) {
391 // Apply the current column's updates
392 Scalar xj = x(j);
393 for (typename MatrixType::InnerIterator it(m_L, j); it; ++it) {
394 if (it.row() > j) { // Only use strictly lower triangular part
395 x(it.row()) -= it.value() * xj;
396 }
397 }
398 std::cout << "After column " << j << ": " << x.transpose() << "\n";
399 }
400
401 // 2. Diagonal solve (Dz = y)
402 std::cout << "\nDiagonal solve (Dz = y):\n";
403 std::cout << "Before: " << x.transpose() << "\n";
404
405 // Process each diagonal entry
406 for (StorageIndex i = 0; i < n; i++) {
407 if (std::abs(m_D[i]) < 1e-14) {
408 throw std::runtime_error("Near-zero diagonal encountered");
409 }
410 x(i) /= m_D[i];
411 }
412 std::cout << "After: " << x.transpose() << "\n";
413
414 // 3. Backward substitution (L^T x = z)
415 std::cout << "\nBackward substitution (L^T x = z):\n";
416
417 // Process columns in reverse order
418 for (StorageIndex j = n - 1; j >= 0; j--) {
419 // Compute the dot product with already-computed values
420 Scalar sum = 0;
421 for (typename MatrixType::InnerIterator it(m_L, j); it; ++it) {
422 if (it.row() > j) {
423 sum += it.value() * x(it.row());
424 }
425 }
426 x(j) -= sum;
427
428 std::cout << "After column " << j << ": " << x.transpose() << "\n";
429 if (j == 0) break; // Handle unsigned underflow
430 }
431
432 // Verify solution quality
433 std::cout << "\n=== Solution Verification ===\n";
434
435 // Compute residual vector r = b - Ax
436 VectorType r = b;
437 for (StorageIndex j = 0; j < n; j++) {
438 for (typename MatrixType::InnerIterator it(m_matrix, j); it; ++it) {
439 r(it.row()) -= it.value() * x(j);
440 if (it.row() != j) { // Add symmetric contribution
441 r(j) -= it.value() * x(it.row());
442 }
443 }
444 }
445
446 // Compute various norms
447 Scalar r_norm = r.norm();
448 Scalar x_norm = x.norm();
449 Scalar b_norm = b.norm();
450 Scalar A_norm = 0; // Estimate matrix norm
451 for (StorageIndex j = 0; j < n; j++) {
452 Scalar col_sum = 0;
453 for (typename MatrixType::InnerIterator it(m_matrix, j); it; ++it) {
454 col_sum += std::abs(it.value());
455 if (it.row() != j) {
456 col_sum += std::abs(it.value()); // Add symmetric part
457 }
458 }
459 A_norm = std::max(A_norm, col_sum);
460 }
461
462 std::cout << "Matrix norm (1-norm estimate): " << A_norm << "\n";
463 std::cout << "Residual norm |b - Ax|: " << r_norm << "\n";
464 std::cout << "Solution norm |x|: " << x_norm << "\n";
465 std::cout << "Right-hand side norm |b|: " << b_norm << "\n";
466 std::cout << "Relative residual |b - Ax|/|b|: " << r_norm / b_norm
467 << "\n";
468 std::cout << "Scaled residual |b - Ax|/(|A|*|x|): "
469 << r_norm / (A_norm * x_norm) << "\n";
470
471 // Check component-wise errors
472 Scalar max_error = 0;
473 for (StorageIndex i = 0; i < n; i++) {
474 Scalar error = std::abs(x(i) - 1.0);
475 max_error = std::max(max_error, error);
476 std::cout << "x[" << i << "] = " << x(i) << " (error: " << error
477 << ")\n";
478 }
479 std::cout << "Maximum component-wise error: " << max_error << "\n";

483 for (StorageIndex i = 0; i < n; i++) {
484 std::cout << "r[" << i << "] = " << r(i) << "\n";

169 struct SegmentMasks {
170 constexpr static size_t n_segments = (N_SIZE + 63) / 64;
171 constexpr static size_t n_bit_positions = 64;
172 std::vector<std::vector<uint64_t>>
173 neighbor_masks; // [segment][bit_position]
174 std::vector<std::vector<uint64_t>>
175 base_set_masks; // [segment][bit_position]
176 size_t n_cuts; // Keep track of number of cuts for safety
177 uint64_t cut_limit_mask; // Mask to limit the number of cuts
178
179 std::vector<std::vector<uint64_t>>
180 valid_cut_masks; // [segment][bit_position] - NEW!

182 SegmentMasks() : n_cuts(0) {
183 neighbor_masks.resize(n_segments);
184 base_set_masks.resize(n_segments);
185 valid_cut_masks.resize(n_segments);
186 for (size_t i = 0; i < n_segments; ++i) {
187 neighbor_masks[i].resize(n_bit_positions, 0);
188 base_set_masks[i].resize(n_bit_positions, 0);
189 valid_cut_masks[i].resize(n_bit_positions, 0);
190 }

192 void precompute(const std::vector<ActiveCutInfo> &active_cuts) {
193 // Determine the number of cuts and compute the overall cut limit
194 // mask.
195 n_cuts = active_cuts.size();
196 cut_limit_mask = (n_cuts >= 64) ? ~0ULL : ((1ULL << n_cuts) - 1);
197
198 // Precompute individual bit masks for each cut.
199 std::vector<uint64_t> cut_bit_masks(n_cuts);
200 for (size_t i = 0; i < n_cuts; ++i) {
201 cut_bit_masks[i] = 1ULL << i;
202 }
203
204 // Resize and initialize mask matrices for all segments and bit
205 // positions.
206 neighbor_masks.assign(n_segments,
207 std::vector<uint64_t>(n_bit_positions, 0));
208 base_set_masks.assign(n_segments,
209 std::vector<uint64_t>(n_bit_positions, 0));
210 valid_cut_masks.assign(n_segments,
211 std::vector<uint64_t>(n_bit_positions, 0));
212
213 // Iterate over each segment.
214 for (size_t segment = 0; segment < n_segments; ++segment) {
215 // Determine the number of bits to process in this segment.
216 size_t bit_limit =
217 (segment == n_segments - 1 && (N_SIZE % 64) != 0)
218 ? (N_SIZE % 64)
219 : 64;
220
221 // Process each bit position within the segment.
222 for (size_t bit_pos = 0; bit_pos < bit_limit; ++bit_pos) {
223 const uint64_t bit_mask = 1ULL << bit_pos;
224 uint64_t neighbor_bits = 0;
225 uint64_t base_bits = 0;
226
227 // For each active cut, check if the bit is set in the
228 // neighbor or base set.
229 for (size_t cut_idx = 0; cut_idx < n_cuts; ++cut_idx) {
230 const auto &cut = *active_cuts[cut_idx].cut_ptr;
231 if (cut.neighbors[segment] & bit_mask) {
232 neighbor_bits |= cut_bit_masks[cut_idx];
233 }
234 if (cut.baseSet[segment] & bit_mask) {
235 base_bits |= cut_bit_masks[cut_idx];
236 }
237 }
238
239 // Apply the cut limit mask and store the computed values.
240 neighbor_masks[segment][bit_pos] =
241 neighbor_bits & cut_limit_mask;
242 base_set_masks[segment][bit_pos] =
243 base_bits & cut_limit_mask;
244 valid_cut_masks[segment][bit_pos] =
245 (neighbor_bits & base_bits) & cut_limit_mask;
246 }
247 }
248 }

323 // Update the IDs for remaining cuts: any cut with an ID greater than

325 for (auto &cut : cuts) {
326 if (cut.id > cutIndex) {
327 cut.id--;

331 // Remove the corresponding entry from the cutMaster_to_cut_map.

390 // Compute limited memory coefficients for a given set of cuts
391 std::vector<double> computeLimitedMemoryCoefficients(
392 const std::vector<uint16_t> &P) const {
393 std::vector<double> alphas;
394 alphas.reserve(cuts.size());
395
396 // Process each cut in the cuts collection.
397 for (const auto &cut : cuts) {
398 double alpha = 0.0;
399 int runningSum = 0;
400
401 // Extract constants from the current cut.
402 const int denominator = cut.p.den;
403 const auto &neighborsBitset =
404 cut.neighbors; // Bitset: neighbors indicator.
405 const auto &baseSetBitset =
406 cut.baseSet; // Bitset: base set indicator.
407 const auto &pValues = cut.p; // Contains numerator vector (p.num).
408 const auto &baseSetOrder =
409 cut.baseSetOrder; // Mapping from node id to its order.
410
411 // Iterate through internal nodes of the route (skip first and
412 // last).
413 for (size_t j = 1; j < P.size() - 1; ++j) {
414 const int nodeId = P[j];
415
416 // Bitwise check: is nodeId in the neighbor set?
417 if (!(neighborsBitset[nodeId / 64] & (1ULL << (nodeId % 64)))) {
418 runningSum = 0; // Reset the sum if the node is not in the
419 // neighbor set.
420 }
421 // Else, if nodeId belongs to the base set...
422 else if (baseSetBitset[nodeId / 64] & (1ULL << (nodeId % 64))) {
423 // Determine the position of nodeId in the base set using
424 // the order mapping.
425 const int pos = baseSetOrder[nodeId];
426 runningSum += pValues.num[pos];
427
428 // If running sum exceeds or equals the denominator, update
429 // alpha.
430 if (runningSum >= denominator) {
431 runningSum -= denominator;
432 alpha += 1.0;
433 }
434 }
435 }
436 alphas.push_back(alpha);
437 }
438
439 return alphas;
440 }

463 // number of cuts.

482 void updateRedCost(std::vector<Label *> labels) {
483 const auto active_cuts = getActiveCuts();
484
485 const auto n_cuts = activeSize();
486 const auto all_cuts = size();
487
488 if (n_cuts == 0) return;
489
490 for (auto label : labels) {
491 label->SRCmap.resize(all_cuts, 0.0);
492 for (auto node_id : label->nodes_covered) {
493 double total_cost_update = 0.0;
494 const size_t segment = node_id >> 6;
495 const size_t bit_position = node_id & 63;
496
497 auto &masks = getSegmentMasks();
498 const auto cut_limit_mask = masks.get_cut_limit_mask();
499 uint64_t valid_cuts =
500 masks.get_valid_cut_mask(segment, bit_position);
501 valid_cuts &= cut_limit_mask; // Safeguard
502
503 while (valid_cuts) {
504 const int cut_idx = __builtin_ctzll(valid_cuts);
505 const auto &active_cut = active_cuts[cut_idx];
506 const auto &cut = *active_cut.cut_ptr;
507 auto &src_map_value = label->SRCmap[active_cut.index];
508 src_map_value += cut.p.num[cut.baseSetOrder[node_id]];
509 if (src_map_value >= cut.p.den) {
510 src_map_value -= cut.p.den;
511 total_cost_update -= active_cut.dual_value;
512 }
513 valid_cuts &= (valid_cuts - 1);
514 }
515
516 uint64_t to_clear =
517 (~masks.get_neighbor_mask(segment, bit_position)) &
518 cut_limit_mask;
519 while (to_clear) {
520 const int cut_idx = __builtin_ctzll(to_clear);
521 if (cut_idx >= n_cuts) {
522 fmt::print(
523 "Warning: Invalid clear_idx {} >= n_cuts {}\n",
524 cut_idx, n_cuts);
525 break;
526 }
527 label->SRCmap[active_cuts[cut_idx].index] = 0.0;
528 to_clear &= (to_clear - 1);
529 }
530 label->cost += total_cost_update;
531 }
532 }

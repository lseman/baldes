110 if (it != cutMaster_to_cut_map.end()) {

183 auto computeLimitedMemoryCoefficients(const std::vector<int> &P) {
184 // iterate over cuts
185 std::vector<double> alphas;
186 alphas.reserve(cuts.size());
187 for (auto c : cuts) {
188 double alpha = 0.0;
189 double S = 0;
190 auto AM = c.neighbors;
191 auto C = c.baseSet;
192 auto p = c.multipliers;
193 auto order = c.baseSetOrder;
194
195 for (size_t j = 1; j < P.size() - 1; ++j) {
196 int vj = P[j];
197
198 // Check if the node vj is in AM (bitwise check)
199 if (!(AM[vj / 64] & (1ULL << (vj % 64)))) {
200 S = 0; // Reset S if vj is not in AM
201 }
202 if (C[vj / 64] & (1ULL << (vj % 64))) {
203 // Get the position of vj in C by counting the set bits up to vj
204 int pos = order[vj];
205 S += p[pos];
206 if (S >= 1) {
207 S -= 1;
208 alpha += 1;
209 }
210 }
211 }
212
213 alphas.push_back(alpha);
214 }
215 return alphas;
216 }

190 struct SegmentMasks {
191 constexpr static size_t n_segments = (N_SIZE + 63) / 64;
192 constexpr static size_t n_bit_positions = 64;
193 std::vector<std::vector<uint64_t>> neighbor_masks; // [segment][bit_position]
194 std::vector<std::vector<uint64_t>> base_set_masks; // [segment][bit_position]
195 size_t n_cuts; // Keep track of number of cuts for safety
196 uint64_t cut_limit_mask; // Mask to limit the number of cuts
197
198 std::vector<std::vector<uint64_t>> valid_cut_masks; // [segment][bit_position] - NEW!
199
200 SegmentMasks() : n_cuts(0) {
201 neighbor_masks.resize(n_segments);
202 base_set_masks.resize(n_segments);
203 valid_cut_masks.resize(n_segments);
204 for (size_t i = 0; i < n_segments; ++i) {
205 neighbor_masks[i].resize(n_bit_positions, 0);
206 base_set_masks[i].resize(n_bit_positions, 0);
207 valid_cut_masks[i].resize(n_bit_positions, 0);
208 }
209 }
210 void precompute(const std::vector<ActiveCutInfo> &active_cuts) {
211 // Determine the number of cuts and compute the overall cut limit
212 // mask.
213 n_cuts = active_cuts.size();
214 cut_limit_mask = (n_cuts >= 64) ? ~0ULL : ((1ULL << n_cuts) - 1);
215
216 // Precompute individual bit masks for each cut.
217 std::vector<uint64_t> cut_bit_masks(n_cuts);
218 for (size_t i = 0; i < n_cuts; ++i) { cut_bit_masks[i] = 1ULL << i; }
219
220 // Resize and initialize mask matrices for all segments and bit
221 // positions.
222 neighbor_masks.assign(n_segments, std::vector<uint64_t>(n_bit_positions, 0));
223 base_set_masks.assign(n_segments, std::vector<uint64_t>(n_bit_positions, 0));
224 valid_cut_masks.assign(n_segments, std::vector<uint64_t>(n_bit_positions, 0));
225
226 // Iterate over each segment.
227 for (size_t segment = 0; segment < n_segments; ++segment) {
228 // Determine the number of bits to process in this segment.
229 const size_t bit_limit = (segment == n_segments - 1 && (N_SIZE % 64) != 0) ? (N_SIZE % 64) : 64;
230
231 // Cache references to the current segment's mask vectors.
232 auto &nbr_mask_vec = neighbor_masks[segment];
233 auto &base_mask_vec = base_set_masks[segment];
234 auto &valid_mask_vec = valid_cut_masks[segment];
235
236 // Process each bit position within the segment.
237 for (size_t bit_pos = 0; bit_pos < bit_limit; ++bit_pos) {
238 const uint64_t bit_mask = 1ULL << bit_pos;
239 uint64_t neighbor_bits = 0;
240 uint64_t base_bits = 0;
241
242 // Loop over each active cut.
243 for (size_t cut_idx = 0; cut_idx < n_cuts; ++cut_idx) {
244 // Cache pointer to current cut.
245 const auto &cut = *active_cuts[cut_idx].cut_ptr;
246 if (cut.neighbors[segment] & bit_mask) neighbor_bits |= cut_bit_masks[cut_idx];
247 if (cut.baseSet[segment] & bit_mask) base_bits |= cut_bit_masks[cut_idx];
248 }
249
250 // Apply the cut limit mask and store computed values.
251 nbr_mask_vec[bit_pos] = neighbor_bits & cut_limit_mask;
252 base_mask_vec[bit_pos] = base_bits & cut_limit_mask;
253 valid_mask_vec[bit_pos] = (neighbor_bits & base_bits) & cut_limit_mask;
254 }
255 }
256 }

401 std::vector<double> computeLimitedMemoryCoefficients(const std::vector<uint16_t> &P) const {
402 const size_t P_size = P.size();
403 std::vector<double> alphas;
404 alphas.reserve(cuts.size());
405
406 const uint16_t *P_data = P.data();
407
408 for (const auto &cut : cuts) {
409 double alpha = 0.0;
410 int runningSum = 0;
411 const int denominator = cut.p.den;
412 const uint64_t *neighborsBitset = cut.neighbors.data();
413 const uint64_t *baseSetBitset = cut.baseSet.data();
414 const auto &pValues = cut.p.num;
415 const auto &baseSetOrder = cut.baseSetOrder;
416
417#if defined(SRC_MEMORY_MODE_ARC)
418 for (size_t j = 1; j < P_size - 1; ++j) {
419 const int nodeId = P_data[j];
420 const int prevNode = P_data[j - 1];
421 const size_t word = nodeId >> 6;
422 const uint64_t bit_mask = 1ULL << (nodeId & 63);
423 const size_t prev_word = prevNode >> 6;
424 const uint64_t prev_mask = 1ULL << (prevNode & 63);
425
426 if (!(neighborsBitset[word] & bit_mask) || !(neighborsBitset[prev_word] & prev_mask)) {
427 runningSum = 0;
428 }
429 if (baseSetBitset[word] & bit_mask) {
430 const int pos = baseSetOrder[nodeId];
431 runningSum += pValues[pos];
432 if (runningSum >= denominator) {
433 runningSum -= denominator;
434 alpha += 1.0;
435 }
436 }
437 }
438#else
439 for (size_t j = 1; j < P_size - 1; ++j) {
440 const int nodeId = P_data[j];
441 const size_t word = nodeId >> 6;
442 const uint64_t bit_mask = 1ULL << (nodeId & 63);
443
444 if (!(neighborsBitset[word] & bit_mask)) {
445 runningSum = 0;
446 } else if (baseSetBitset[word] & bit_mask) {
447 const int pos = baseSetOrder[nodeId];
448 runningSum += pValues[pos];
449 if (runningSum >= denominator) {
450 runningSum -= denominator;
451 alpha += 1.0;
452 }
453 }
454 }
455#endif
456 alphas.push_back(alpha);
457 }
458 return alphas;

619 const size_t segment = static_cast<size_t>(node_id) >> 6;
620 const uint64_t bit = 1ULL << (node_id & 63);
621
622 auto &updates = src_node_updates[node_id];
623 auto &clears = src_node_clears[node_id];
624 updates.reserve(active_cuts.size());
625 clears.reserve(active_cuts.size());
626
627 for (const auto &active_cut : active_cuts) {
628 const auto &cut = *active_cut.cut_ptr;
629 if (!(cut.neighbors[segment] & bit)) {
630 clears.push_back(static_cast<uint16_t>(active_cut.index));
631 continue;
632 }
633 if (cut.baseSet[segment] & bit) {
634 updates.push_back({static_cast<uint16_t>(active_cut.index),
635 static_cast<uint16_t>(cut.p.num[cut.baseSetOrder[node_id]]),
636 static_cast<uint16_t>(cut.p.den), active_cut.dual_value});
637 }
638 }
639 }

650 double computeSelectionScore(size_t cut_idx) const {
651 if (cut_idx >= cuts.size()) return 0.0;
652
653 const Cut &cut = cuts[cut_idx];
654 const double abs_dual = (cut_idx < SRCDuals.size()) ? std::abs(SRCDuals[cut_idx]) : 0.0;
655
656 const double current_dual_strength = std::log1p(abs_dual);
657 const double historical_dual_strength = std::log1p(cut.avg_dual_magnitude);
658 const double violation_strength =
659 std::log1p(std::max({0.0, cut.last_violation, cut.max_violation, cut.total_violation * 0.1}));
660 const double activity_bonus = 1.0 + 0.08 * std::log1p(static_cast<double>(cut.active_count)) +
661 0.04 * std::log1p(static_cast<double>(cut.inclusion_count));
662
663 const int64_t last_use_epoch = std::max(
664 {cut.creation_epoch, cut.last_separation_epoch, cut.last_nonzero_dual_epoch, cut.last_active_epoch});
665 const int64_t age = std::max<int64_t>(0, current_epoch_ - last_use_epoch);
666 const double age_penalty = (age == 0) ? 1.0 : std::pow(age_decay_alpha_, static_cast<double>(age));
667
668 return (4.0 * current_dual_strength + historical_dual_strength + violation_strength) * activity_bonus *
669 age_penalty;
670 }

679 void updateActiveCutsProbabilistic() {
680 constexpr size_t max_active_cuts = std::min<size_t>(MAX_SRC_CUTS, 64);
681
682 active_cuts.clear();
683 active_cuts.reserve(std::min(cuts.size(), max_active_cuts));
684
685 // Collect candidates with scores
686 std::vector<std::pair<size_t, double>> scored_candidates;
687 scored_candidates.reserve(cuts.size());
688
689 for (size_t i = 0; i < cuts.size(); ++i) {
690 if (i < SRCDuals.size() && std::abs(SRCDuals[i]) > SRC_DUAL_TOLERANCE) {
691 double score = computeSelectionScore(i);
692 scored_candidates.emplace_back(i, score);
693 }
694 }
695
696 if (scored_candidates.empty()) {
697 segment_masks.precompute(active_cuts);
698 precomputeSRCNodeTables();
699 return;
700 }
701
702 // Sort by score descending
703 pdqsort(scored_candidates.begin(), scored_candidates.end(),
704 [](const auto &a, const auto &b) { return a.second > b.second; });
705
706 // Truncate to active cut limit
707 if (scored_candidates.size() > max_active_cuts) { scored_candidates.resize(max_active_cuts); }
708
709 // Build active cuts from top scored
710 for (auto &[cut_idx, score] : scored_candidates) {
711 active_cuts.emplace_back(active_cuts.size(), cut_idx, &cuts[cut_idx], SRCDuals[cut_idx],
712 cuts[cut_idx].type);
713 auto &cut = cuts[cut_idx];
714 cut.inclusion_count++;
715 cut.active_count++;
716 cut.last_active_epoch = current_epoch_;
717 cut.last_nonzero_dual_epoch = current_epoch_;
718 const double abs_dual = std::abs(SRCDuals[cut_idx]);
719 cut.avg_dual_magnitude =
720 (cut.avg_dual_magnitude <= 0.0)
721 ? abs_dual
722 : (1.0 - dual_ema_alpha_) * cut.avg_dual_magnitude + dual_ema_alpha_ * abs_dual;
723 }
724
725 segment_masks.precompute(active_cuts);
726 precomputeSRCNodeTables();
727 }

773 void enforcePoolSizeLimit() {
774 if (cuts.size() <= max_pool_size_) return;
775
776 // Compute scores for all cuts.
777 std::vector<std::pair<size_t, double>> scored;
778 scored.reserve(cuts.size());
779 for (size_t i = 0; i < cuts.size(); ++i) {
780 if (!cuts[i].added) { scored.emplace_back(i, computeSelectionScore(i)); }
781 }
782 if (scored.empty()) return;
783
784 // Sort by score ascending (lowest first = candidates for removal)
785 pdqsort(scored.begin(), scored.end(), [](const auto &a, const auto &b) { return a.second < b.second; });
786
787 // Remove excess pending cuts (lowest scoring first). Erase in
788 // descending index order so vector indices remain valid.
789 const size_t excess = std::min(cuts.size() - max_pool_size_, scored.size());
790 std::vector<int> remove_indices;
791 remove_indices.reserve(excess);
792 for (auto &[cut_idx, score] : scored) {
793 if (remove_indices.size() >= excess) break;
794 remove_indices.push_back(static_cast<int>(cut_idx));
795 }
796 std::sort(remove_indices.begin(), remove_indices.end(), std::greater<int>());
797 for (int idx : remove_indices) { removeCut(idx); }
798 }

848 void updateRedCost(std::vector<Label *> labels) {
849 const auto active_cuts = getActiveCuts();
850
851 const auto n_cuts = activeSize();
852
853 for (auto label : labels) {
854 label->SRCmap.assign(n_cuts, 0);
855 if (n_cuts == 0) continue;
856 int prev_node = -1;
857 for (auto node_id : label->getRoute()) {
858 double total_cost_update = 0.0;
859#if defined(SRC_MEMORY_MODE_ARC)
860 for (const auto &active_cut : active_cuts) {
861 const auto &cut = *active_cut.cut_ptr;
862 auto &src_map_value = label->SRCmap[active_cut.index];
863 if (!cut.isSRCMemoryArc(prev_node, node_id)) { src_map_value = 0; }
864 if (cut.isSRCset(node_id)) {
865 src_map_value += cut.srcMultiplier(node_id);
866 if (src_map_value >= cut.p.den) {
867 src_map_value -= cut.p.den;
868 total_cost_update -= active_cut.dual_value;
869 }
870 }
871 }
872#else
873 for (const auto &update : getSRCNodeUpdates(node_id)) {
874 auto &src_map_value = label->SRCmap[update.active_idx];
875 src_map_value += update.add;
876 if (src_map_value >= update.den) {
877 src_map_value -= update.den;
878 total_cost_update -= update.dual;
879 }
880 }
881
882 for (const auto active_idx : getSRCNodeClears(node_id)) { label->SRCmap[active_idx] = 0; }
883#endif
884 label->cost += total_cost_update;
885 prev_node = node_id;
886 }
887 }

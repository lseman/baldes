157 // Ensure the cutIndex is within bounds

159 std::cerr << "Cut index " << cutIndex << " is out of bounds." << std::endl;
160 return;
161 }

163 // Erase the cut from the cuts vector
164 cuts.erase(cuts.begin() + cutIndex);

166 for (auto &entry : cuts) {

168 entry.id--; // Decrement index to account for the removed cut

172 // Find the corresponding entry in the map

175
176 if (it != cutMaster_to_cut_map.end()) {
177 // Remove the entry for the removed cut
178 cutMaster_to_cut_map.erase(it);
179 } else {
180 std::cerr << "Cut index " << cutIndex << " not found in cutMaster_to_cut_map." << std::endl;
181 return; // No need to continue if the cut wasn't found in the map
182 }
183
184 // Reorganize the indices in the map for all subsequent cuts
185 for (auto &entry : cutMaster_to_cut_map) {
186 if (entry.second > cutIndex) {
187 entry.second--; // Decrement index to account for the removed cut
188 }

249 auto computeLimitedMemoryCoefficients(const std::vector<int> &P) {
250 // iterate over cuts
251 std::vector<double> alphas;
252 alphas.reserve(cuts.size());
253 for (auto c : cuts) {
254 double alpha = 0.0;
255 int S = 0;
256 auto den = c.p.den;
257 auto AM = c.neighbors;
258 auto C = c.baseSet;
259 auto p = c.p;
260 auto order = c.baseSetOrder;
261
262 for (size_t j = 1; j < P.size() - 1; ++j) {
263 int vj = P[j];
264
265 // Check if the node vj is in AM (bitwise check)
266 if (!(AM[vj / 64] & (1ULL << (vj % 64)))) {
267 S = 0; // Reset S if vj is not in AM
268 } else if (C[vj / 64] & (1ULL << (vj % 64))) {
269 // Get the position of vj in C by counting the set bits up to vj
270 int pos = order[vj];
271 S += p.num[pos];
272 if (S >= den) {
273 S -= den;
274 alpha += 1;
275 }
276 }
277 }
278
279 alphas.push_back(alpha);
280 }
281 return alphas;
282 }

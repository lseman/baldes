101// ---------------------------------------------------------------------
102// Serialization buffer
103// ---------------------------------------------------------------------
104class SerializationBuffer {
105 static constexpr std::size_t BUFFER_SIZE = 8192;
106 std::vector<char> buffer_;
107 std::size_t pos_ = 0;
108
109 public:
110 SerializationBuffer() : buffer_(BUFFER_SIZE) {}
111
112 void flush(std::ostream& os) {
113 if (pos_ > 0) {
114 os.write(buffer_.data(), pos_);
115 pos_ = 0;
116 }
117 }
118
119 void write(const char* data, size_t size, std::ostream& os) {
120 if (pos_ + size > BUFFER_SIZE) {
121 flush(os);
122 if (size > BUFFER_SIZE) {
123 os.write(data, size);
124 return;
125 }
126 }
127 std::memcpy(buffer_.data() + pos_, data, size);
128 pos_ += size;
129 }
130};

218 size_t size;
219 deserialize_value(size, is);
220 container.clear();
221 if constexpr (requires { container.reserve(size); }) {
222 container.reserve(size);
223 }
224 for (size_t i = 0; i < size; ++i) {
225 typename T::value_type item;
226 deserialize_value(item, is);

243// ---------------------------------------------------------------------
244// Generic serialization using reflection
245// ---------------------------------------------------------------------
246template <typename T>
247void serialize(const T& obj, std::ostream& os) {
248 obj.visit([&os](const char* name, const auto& member) {
249 size_t nameLen = std::strlen(name);
250 serialize_value(nameLen, os);
251 os.write(name, nameLen);
252 serialize_value(member, os);
253 });
254}

256template <typename T>
257void deserialize(T& obj, std::istream& is) {
258 obj.visit([&is](const char* /*name*/, auto& member) {
259 size_t nameLen;
260 deserialize_value(nameLen, is);
261 std::string dummy(nameLen, '\0');
262 is.read(&dummy[0], nameLen);
263 deserialize_value(member, is);
264 });
265}

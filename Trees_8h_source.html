81 void rotate(TreeNode* x) {
82 TreeNode* p = x->parent;
83 TreeNode* g = p->parent;
84 bool is_left = (p->left == x);
85
86 // Update the parent's child pointer
87 TreeNode* child = is_left ? x->right : x->left;
88 if (is_left) {
89 p->left = child;
90 x->right = p;
91 } else {
92 p->right = child;
93 x->left = p;
94 }
95 if (child) child->parent = p;
96
97 // Update parent pointers
98 x->parent = g;
99 p->parent = x;
100
101 // Update grandparent's child pointer
102 if (g) {
103 (g->left == p ? g->left : g->right) = x;
104 }
105 }

107 void splay(TreeNode* x) {
108 while (x->parent) {
109 TreeNode* p = x->parent;
110 TreeNode* g = p->parent;
111
112 if (!g) {
113 rotate(x); // Zig
114 } else if ((g->left == p) == (p->left == x)) {
115 rotate(p); // Zig-zig
116 rotate(x);
117 } else {
118 rotate(x); // Zig-zag
119 rotate(x);
120 }
121 }
122 root = x;
123 }

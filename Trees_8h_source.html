77class SplayTree {

80 void rotate(TreeNode *x) {
81 TreeNode *p = x->parent;
82 TreeNode *g = p->parent;
83
84 if (p->left == x) { // Left child
85 p->left = x->right;
86 if (x->right) x->right->parent = p;
87 x->right = p;
88 } else { // Right child
89 p->right = x->left;
90 if (x->left) x->left->parent = p;
91 x->left = p;
92 }
93
94 x->parent = g;
95 p->parent = x;
96
97 if (g) {
98 if (g->left == p)
99 g->left = x;
100 else
101 g->right = x;
102 }
103 }

105 void splay(TreeNode *x) {
106 while (x->parent != nullptr) {
107 TreeNode *p = x->parent;
108 TreeNode *g = p->parent;
109
110 if (g == nullptr) {
111 // Zig step (single rotation)
112 rotate(x);
113 } else if ((g->left == p && p->left == x) ||
114 (g->right == p && p->right == x)) {
115 // Zig-zig step (double rotation)
116 rotate(p); // First rotate parent
117 rotate(x); // Then rotate x
118 } else {
119 // Zig-zag step (rotating x twice in different directions)
120 rotate(x); // Rotate x first
121 rotate(x); // Then rotate x again
122 }
123 }
124 root = x;
125 }

130 TreeNode *find(const std::vector<double> &point) {
131 TreeNode *curr = root;
132
133 while (curr != nullptr) {
134 if (curr->contains(point)) {
135 splay(curr); // Splay only if we find the node
136 return curr;
137 } else if (curr->is_less_than(point)) {
138 curr = curr->right;
139 } else {
140 curr = curr->left;

144 // If not found, no splaying needed for the closest node

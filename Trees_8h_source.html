47 // Utility function to check if two doubles are nearly equal.
48 inline bool nearlyEqual(double a, double b,
49 double epsilon = EPSILON) const {
50 return std::fabs(a - b) < epsilon;
51 }

53 // Returns true if the point is inside the hyper-rectangle defined by [low,
54 // high].
55 inline bool contains(const std::vector<double>& point) const {
56 for (size_t i = 0; i < low.size(); ++i) {
57 // Allow a little wiggle room with EPSILON.
58 if (point[i] < low[i] - EPSILON || point[i] > high[i] + EPSILON) {
59 return false;
60 }
61 }
62 return true;
63 }
64
65 // Compares the current node's interval with a point.
66 // Here we use a lexicographical comparison on the high bounds.
67 // Adjust this function if you need a different comparison semantics.
68 inline bool is_less_than(const std::vector<double>& point) const {
69 // Use lexicographical comparison but with an epsilon tolerance.
70 for (size_t i = 0; i < high.size(); ++i) {
71 if (!nearlyEqual(high[i], point[i])) {
72 return high[i] < point[i];
73 }
74 }
75 return false; // They are nearly equal.
76 }

92class SplayTree {
93 private:
94 TreeNode* root;
95
96 // Rotate x up in the tree.
97 inline void rotate(TreeNode* x) {
98 TreeNode* p = x->parent;
99 TreeNode* g = p->parent;
100 bool is_left = (p->left == x);
101
102 // Adjust parent's child pointer.
103 TreeNode* child = is_left ? x->right : x->left;
104 if (is_left) {
105 p->left = child;
106 x->right = p;
107 } else {
108 p->right = child;
109 x->left = p;
110 }
111 if (child) {
112 child->parent = p;
113 }
114
115 // Update parent pointers.
116 x->parent = g;
117 p->parent = x;
118
119 // Update grandparent's child pointer.
120 if (g) {
121 if (g->left == p) {
122 g->left = x;
123 } else {
124 g->right = x;
125 }
126 }
127 }
128
129 // Splay x to the root.
130 inline void splay(TreeNode* x) {
131 while (x->parent) {
132 TreeNode* p = x->parent;
133 TreeNode* g = p->parent;
134 if (!g) {
135 // Zig step.
136 rotate(x);
137 } else if ((g->left == p) == (p->left == x)) {
138 // Zig-zig step.
139 rotate(p);
140 rotate(x);
141 } else {
142 // Zig-zag step.
143 rotate(x);
144 rotate(x);
145 }
146 }
147 root = x;
148 }
149
150 // Recursively destroy the tree. For very deep trees, an iterative version
151 // might be preferred.
152 void destroyTree(TreeNode* node) {
153 if (node) {
154 destroyTree(node->left);
155 destroyTree(node->right);
156 delete node;
157 }
158 }

160 // Recursively copy the tree.
161 TreeNode* copyTree(TreeNode* node, TreeNode* parent = nullptr) {
162 if (!node) return nullptr;
163 TreeNode* newNode =
164 new TreeNode(node->low, node->high, node->bucket_index);
165 newNode->parent = parent;
166 newNode->left = copyTree(node->left, newNode);
167 newNode->right = copyTree(node->right, newNode);
168 return newNode;
169 }

193 // Find a node whose interval contains the given point.
194 TreeNode* find(const std::vector<double>& point) {
195 TreeNode* curr = root;
196 TreeNode* last = nullptr;
197 while (curr) {
198 last = curr;
199 if (curr->contains(point)) {
200 splay(curr);
201 return curr;
202 }
203 // Move right if point is greater, else left.
204 curr = curr->is_less_than(point) ? curr->right : curr->left;
205 }
206 if (last) splay(last);
207 return nullptr;
208 }

235 // Insert a new interval into the splay tree.
236 void insert(const std::vector<double>& low, const std::vector<double>& high,
237 int bucket_index) {
238 if (!root) {
239 root = new TreeNode(low, high, bucket_index);
240 return;
241 }
242
243 TreeNode* curr = root;
244 while (true) {
245 if (low < curr->low) {
246 if (!curr->left) {
247 curr->left = new TreeNode(low, high, bucket_index);
248 curr->left->parent = curr;
249 splay(curr->left);
250 break;
251 }
252 curr = curr->left;
253 } else if (low > curr->low) {
254 if (!curr->right) {
255 curr->right = new TreeNode(low, high, bucket_index);
256 curr->right->parent = curr;
257 splay(curr->right);
258 break;
259 }
260 curr = curr->right;
261 } else {
262 splay(curr);
263 break; // Duplicate interval, do nothing.
264 }
265 }
266 }

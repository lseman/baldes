145void HighDimCutsGenerator::generatePermutations(ankerl::unordered_dense::map<int, int> &count_map,
146 std::vector<int> &result, std::vector<std::vector<int>> &results,
147 int remaining) {
148 if (remaining == 0) { // Base case: all elements used
149 results.push_back(result);
150 return;
151 }
152
153 for (auto &[key, count] : count_map) {
154 if (count > 0) { // Only proceed if there are remaining elements
155 result.push_back(key);
156 --count;
157
158 generatePermutations(count_map, result, results, remaining - 1);
159
160 ++count; // Revert the count
161 result.pop_back(); // Backtrack
162 }
163 }

385 baldesCtrPtr addBranchingbaldesCtr(
386 double rhs, const BranchingDirection &sense, const CandidateType &ctype,
387 std::optional<std::variant<int, std::pair<int, int>, std::vector<int>>> payload = std::nullopt) {
388 // Get the decision variables from the MIP problem
389 auto &variables = mip.getVars(); // Assumes mip is a pointer to MIPProblem
390 LinearExpression linExpr; // Initialize linear expression for MIP
391 std::string name;
392
393 if (ctype == CandidateType::Vehicle) {
394 for (auto var : variables) {
395 linExpr.add_or_update_term(var->get_name(),
396 1.0); // Add each variable with coefficient 1.0
397 name = "branching_vehicle_" + std::to_string(int(rhs));
398 }
399 } else if (ctype == CandidateType::Node) {
400 for (size_t i = 0; i < variables.size(); ++i) {
401 // Use the payload as int for Node
402 std::visit(
403 [&](auto &&arg) {
404 using T = std::decay_t<decltype(arg)>;
405 if constexpr (std::is_same_v<T, int>) {
406 linExpr.add_or_update_term(variables[i]->get_name(),
407 paths[i].contains(arg) ? 1.0 : 0.0); // Use int payload
408 name = "branching_node_" + std::to_string(arg) + "_" + std::to_string(int(rhs));
409 } else {
410 throw std::invalid_argument("Payload for Node must be an int.");
411 }
412 },
413 *payload);
414 }
415 } else if (ctype == CandidateType::Edge) {
416 for (size_t i = 0; i < variables.size(); ++i) {
417 // Use the payload as std::pair<int, int> for Edge
418 std::visit(
419 [&](auto &&arg) {
420 using T = std::decay_t<decltype(arg)>;
421 if constexpr (std::is_same_v<T, std::pair<int, int>>) {
422 linExpr.add_or_update_term(
423 variables[i]->get_name(),
424 paths[i].timesArc(arg.first, arg.second) ? 1.0 : 0.0); // Use pair payload
425 name = "branching_edge_" + std::to_string(arg.first) + "_" + std::to_string(arg.second) +
426 "_" + std::to_string(int(rhs));
427 } else {
428 throw std::invalid_argument("Payload for Edge must be a std::pair<int, "
429 "int>.");
430 }
431 },
432 *payload);
433 }
434 } else if (ctype == CandidateType::Cluster) {
435 for (size_t i = 0; i < variables.size(); ++i) {
436 // Use the payload as std::vector<int> for Cluster
437 std::visit(
438 [&](auto &&arg) {
439 using T = std::decay_t<decltype(arg)>;
440 if constexpr (std::is_same_v<T, std::vector<int>>) {
441 for (int cluster_ele : arg) {
442 // Assuming `contains` checks if the path
443 // contains `cluster_ele`
444 if (paths[i].contains(cluster_ele)) {
445 linExpr.add_or_update_term(variables[i]->get_name(),
446 1.0); // Add term with coefficient 1.0
447 }
448 name =
449 "branching_cluster_" + std::to_string(cluster_ele) + "_" + std::to_string(int(rhs));
450 }
451 } else {
452 throw std::invalid_argument("Payload for Cluster must be a "
453 "std::vector<int>.");
454 }
455 },
456 *payload);
457 }
458 }
459
460 // Add the constraint to MIP based on the branching direction
461 baldesCtrPtr constraint;
462 if (mip.constraint_exists(name)) {
463 constraint = mip.get_constraint(name);
464 return constraint;
465 }
466 if (sense == BranchingDirection::Greater) {
467 constraint = mip.add_constraint(linExpr, rhs, '>');
468 constraint->set_name(name);
469 } else if (sense == BranchingDirection::Less) {
470 constraint = mip.add_constraint(linExpr, rhs, '<');
471 constraint->set_name(name);
472 } else {
473 constraint = mip.add_constraint(linExpr, rhs, '=');
474 constraint->set_name(name);
475 }
476
477 // Update the MIP structure if necessary
478 // mip->update();
479 return constraint;
480 }

348 auto addBranchingConstraint(double rhs, const BranchingDirection &sense, const CandidateType &ctype,
349 std::optional<std::variant<int, std::pair<int, int>>> payload = std::nullopt) {
350
351 // Get the decision variables from the model
352 auto varsPtr = model->getVars(); // Assumes model is a pointer to GRBModel
353 // deference the pointer
354 auto vars = std::vector<GRBVar>(varsPtr, varsPtr + model->get(GRB_IntAttr_NumVars));
355
356 GRBLinExpr linExpr; // Initialize linExpr
357
358 if (ctype == CandidateType::Vehicle) {
359 for (auto i = 0; i < vars.size(); i++) { linExpr += vars[i]; }
360 } else if (ctype == CandidateType::Node) {
361 for (auto i = 0; i < vars.size(); i++) {
362 // Use the payload as int for Node
363 std::visit(
364 [&](auto &&arg) {
365 using T = std::decay_t<decltype(arg)>;
366 if constexpr (std::is_same_v<T, int>) {
367 linExpr += vars[i] * paths[i].contains(arg); // Use the int payload
368 } else {
369 throw std::invalid_argument("Payload for Node must be an int.");
370 }
371 },
372 *payload);
373 }
374 } else if (ctype == CandidateType::Edge) {
375 for (auto i = 0; i < vars.size(); i++) {
376 // Use the payload as std::pair<int, int> for Edge
377 std::visit(
378 [&](auto &&arg) {
379 using T = std::decay_t<decltype(arg)>;
380 if constexpr (std::is_same_v<T, std::pair<int, int>>) {
381 linExpr += vars[i] * paths[i].timesArc(arg.first, arg.second); // Use the std::pair payload
382 } else {
383 throw std::invalid_argument("Payload for Edge must be a std::pair<int, int>.");
384 }
385 },
386 *payload);
387 }
388 }
389
390 // Add the constraint based on the sense
391 GRBConstr constraint;
392 if (sense == BranchingDirection::Greater) {
393 constraint = model->addConstr(linExpr, GRB_GREATER_EQUAL, rhs);
394 } else if (sense == BranchingDirection::Less) {
395 constraint = model->addConstr(linExpr, GRB_LESS_EQUAL, rhs);
396 } else {
397 constraint = model->addConstr(linExpr, GRB_EQUAL, rhs);
398 }
399
400 // Update the model to reflect the changes
401 model->update();
402
403 return constraint;
404 }

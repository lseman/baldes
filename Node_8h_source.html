427 baldesCtrPtr addBranchingbaldesCtr(
428 double rhs, const BranchingDirection &sense, const CandidateType &ctype,
429 std::optional<std::variant<int, std::pair<int, int>, std::vector<int>>>
430 payload = std::nullopt) {
431 // Get the decision variables from the MIP problem
432 auto &variables =
433 mip.getVars(); // Assumes mip is a pointer to MIPProblem
434 LinearExpression linExpr; // Initialize linear expression for MIP
435 std::string name;
436
437 if (ctype == CandidateType::Vehicle) {
438 for (auto var : variables) {
439 linExpr.add_or_update_term(
440 var->get_name(),
441 1.0); // Add each variable with coefficient 1.0
442 name = "branching_vehicle_" + std::to_string(int(rhs));
443 }
444 } else if (ctype == CandidateType::Node) {
445 for (size_t i = 0; i < variables.size(); ++i) {
446 // Use the payload as int for Node
447 std::visit(
448 [&](auto &&arg) {
449 using T = std::decay_t<decltype(arg)>;
450 if constexpr (std::is_same_v<T, int>) {
451 linExpr.add_or_update_term(
452 variables[i]->get_name(),
453 paths[i].contains(arg)
454 ? 1.0
455 : 0.0); // Use int payload
456 name = "branching_node_" + std::to_string(arg) +
457 "_" + std::to_string(int(rhs));
458 } else {
459 throw std::invalid_argument(
460 "Payload for Node must be an int.");
461 }
462 },
463 *payload);
464 }
465 } else if (ctype == CandidateType::Edge) {
466 for (size_t i = 0; i < variables.size(); ++i) {
467 // Use the payload as std::pair<int, int> for Edge
468 std::visit(
469 [&](auto &&arg) {
470 using T = std::decay_t<decltype(arg)>;
471 if constexpr (std::is_same_v<T, std::pair<int, int>>) {
472 linExpr.add_or_update_term(
473 variables[i]->get_name(),
474 paths[i].timesArc(arg.first, arg.second)
475 ? 1.0
476 : 0.0); // Use pair payload
477 name = "branching_edge_" +
478 std::to_string(arg.first) + "_" +
479 std::to_string(arg.second) + "_" +
480 std::to_string(int(rhs));
481 } else {
482 throw std::invalid_argument(
483 "Payload for Edge must be a std::pair<int, "
484 "int>.");
485 }
486 },
487 *payload);
488 }
489 } else if (ctype == CandidateType::Cluster) {
490 for (size_t i = 0; i < variables.size(); ++i) {
491 // Use the payload as std::vector<int> for Cluster
492 std::visit(
493 [&](auto &&arg) {
494 using T = std::decay_t<decltype(arg)>;
495 if constexpr (std::is_same_v<T, std::vector<int>>) {
496 for (int cluster_ele : arg) {
497 // Assuming `contains` checks if the path
498 // contains `cluster_ele`
499 if (paths[i].contains(cluster_ele)) {
500 linExpr.add_or_update_term(
501 variables[i]->get_name(),
502 1.0); // Add term with coefficient 1.0
503 }
504 name = "branching_cluster_" +
505 std::to_string(cluster_ele) + "_" +
506 std::to_string(int(rhs));
507 }
508 } else {
509 throw std::invalid_argument(
510 "Payload for Cluster must be a "
511 "std::vector<int>.");
512 }
513 },
514 *payload);
515 }
516 }
517
518 // Add the constraint to MIP based on the branching direction
519 baldesCtrPtr constraint;
520 if (mip.constraint_exists(name)) {
521 constraint = mip.get_constraint(name);
522 return constraint;
523 }
524 if (sense == BranchingDirection::Greater) {
525 constraint = mip.add_constraint(linExpr, rhs, '>');
526 constraint->set_name(name);
527 } else if (sense == BranchingDirection::Less) {
528 constraint = mip.add_constraint(linExpr, rhs, '<');
529 constraint->set_name(name);
530 } else {
531 constraint = mip.add_constraint(linExpr, rhs, '=');
532 constraint->set_name(name);
533 }
534
535 // Update the MIP structure if necessary
536 // mip->update();
537 return constraint;
538 }

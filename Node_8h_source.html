389 baldesCtrPtr addBranchingbaldesCtr(
390 double rhs, const BranchingDirection &sense, const CandidateType &ctype,
391 std::optional<std::variant<int, std::pair<int, int>, std::vector<int>>> payload = std::nullopt) {
392
393 // Get the decision variables from the MIP problem
394 auto &variables = mip.getVars(); // Assumes mip is a pointer to MIPProblem
395 LinearExpression linExpr; // Initialize linear expression for MIP
396 std::string name;
397
398 if (ctype == CandidateType::Vehicle) {
399 for (auto var : variables) {
400 linExpr.add_or_update_term(var->get_name(), 1.0); // Add each variable with coefficient 1.0
401 name = "branching_vehicle_" + std::to_string(int(rhs));
402 }
403 } else if (ctype == CandidateType::Node) {
404 for (size_t i = 0; i < variables.size(); ++i) {
405 // Use the payload as int for Node
406 std::visit(
407 [&](auto &&arg) {
408 using T = std::decay_t<decltype(arg)>;
409 if constexpr (std::is_same_v<T, int>) {
410 linExpr.add_or_update_term(variables[i]->get_name(),
411 paths[i].contains(arg) ? 1.0 : 0.0); // Use int payload
412 name = "branching_node_" + std::to_string(arg) + "_" + std::to_string(int(rhs));
413 } else {
414 throw std::invalid_argument("Payload for Node must be an int.");
415 }
416 },
417 *payload);
418 }
419 } else if (ctype == CandidateType::Edge) {
420 for (size_t i = 0; i < variables.size(); ++i) {
421 // Use the payload as std::pair<int, int> for Edge
422 std::visit(
423 [&](auto &&arg) {
424 using T = std::decay_t<decltype(arg)>;
425 if constexpr (std::is_same_v<T, std::pair<int, int>>) {
426 linExpr.add_or_update_term(
427 variables[i]->get_name(),
428 paths[i].timesArc(arg.first, arg.second) ? 1.0 : 0.0); // Use pair payload
429 name = "branching_edge_" + std::to_string(arg.first) + "_" + std::to_string(arg.second) +
430 "_" + std::to_string(int(rhs));
431 } else {
432 throw std::invalid_argument("Payload for Edge must be a std::pair<int, int>.");
433 }
434 },
435 *payload);
436 }
437 } else if (ctype == CandidateType::Cluster) {
438 for (size_t i = 0; i < variables.size(); ++i) {
439 // Use the payload as std::vector<int> for Cluster
440 std::visit(
441 [&](auto &&arg) {
442 using T = std::decay_t<decltype(arg)>;
443 if constexpr (std::is_same_v<T, std::vector<int>>) {
444 for (int cluster_ele : arg) {
445 // Assuming `contains` checks if the path contains `cluster_ele`
446 if (paths[i].contains(cluster_ele)) {
447 linExpr.add_or_update_term(variables[i]->get_name(),
448 1.0); // Add term with coefficient 1.0
449 }
450 name =
451 "branching_cluster_" + std::to_string(cluster_ele) + "_" + std::to_string(int(rhs));
452 }
453 } else {
454 throw std::invalid_argument("Payload for Cluster must be a std::vector<int>.");
455 }
456 },
457 *payload);
458 }
459 }
460
461 // Add the constraint to MIP based on the branching direction
462 baldesCtrPtr constraint;
463 if (mip.constraint_exists(name)) {
464 constraint = mip.get_constraint(name);
465 return constraint;
466 }
467 if (sense == BranchingDirection::Greater) {
468 constraint = mip.add_constraint(linExpr, rhs, '>');
469 constraint->set_name(name);
470 } else if (sense == BranchingDirection::Less) {
471 constraint = mip.add_constraint(linExpr, rhs, '<');
472 constraint->set_name(name);
473 } else {
474 constraint = mip.add_constraint(linExpr, rhs, '=');
475 constraint->set_name(name);
476 }
477
478 // Update the MIP structure if necessary
479 // mip->update();
480 return constraint;
481 }
482 using BranchingDualsPtr = std::shared_ptr<BranchingDuals>;
483 BranchingDualsPtr branchingDuals = std::make_shared<BranchingDuals>();

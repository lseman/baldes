127 void calculateRowActivities() {
128 activities.resize(A.num_rows);
129 for (size_t i = 0; i < activities.size(); ++i) {
130 RowActivity &activity = activities[i];
131 for (SparseMatrix::RowIterator it = A.rowIterator(i); it.valid(); it.next()) {
132 double coef = it.value();
133 int j = it.col();
134
135 // print lb and ub
136 fmt::print("lb[{}] = {}, ub[{}] = {}\n", j, lb[j], j, ub[j]);
137 if (ub[j] != NumericUtils::infty) {
138 coef < 0 ? activity.min += coef * ub[j] : activity.max += coef * ub[j];
139 } else {
140 coef < 0 ? ++activity.ninfmin : ++activity.ninfmax;
141 }
142
143 if (lb[j] != -NumericUtils::infty) {
144 coef < 0 ? activity.max += coef * lb[j] : activity.min += coef * lb[j];
145 } else {
146 coef < 0 ? ++activity.ninfmax : ++activity.ninfmin;
147 }

152 void processRowInformation() {
153 rowInfoVector.clear();
154 lhs_values.clear();
155 rhs_values.clear();

157 auto processBound = [&](char sense, double value) {
158 switch (sense) {
159 case '<':
160 // LHS is -inf, so set the kLhsInf flag
161 rowInfoVector.emplace_back(RowInfo::RowFlag::kLhsInf, -NumericUtils::infty, value);
162 lhs_values.push_back(-NumericUtils::infty);
163 rhs_values.push_back(value);
164 break;
165 case '>':
166 // RHS is +inf, so set the kRhsInf flag
167 rowInfoVector.emplace_back(RowInfo::RowFlag::kRhsInf, value, NumericUtils::infty);
168 lhs_values.push_back(value);
169 rhs_values.push_back(NumericUtils::infty);
170 break;
171 case '=':
172 // Neither LHS nor RHS are infinite, so no flags
173 rowInfoVector.emplace_back(RowInfo::RowFlag::None, value, value);
174 lhs_values.push_back(value);
175 rhs_values.push_back(value);
176 break;
177 default: std::cerr << "Invalid sense: " << sense << std::endl; return false;
178 }
179 return true;
180 };
181
182 // Iterate through the sense vector and process each bound
183 for (size_t i = 0; i < sense.size(); ++i) { processBound(sense[i], b[i]); }
184 }

203 // Convert a row to a knapsack form
204 void convert2Knapsack(int row) {
205 std::cout << "Converting row " << row << " to knapsack\n";
206 SparseMatrix::RowIterator rowIter = A.rowIterator(row);
207 std::vector<double> binCoeffs;
208
209 if (sense[row] != '<') {
210 return; // Only convert if the constraint is less-than-or-equal-to
211 }
212
213 std::cout << "Row " << row << " has rhs " << rhs_values[row] << "\n";
214
215 for (; rowIter.valid(); rowIter.next()) {
216 int j = rowIter.col();
217 double coeff = rowIter.value();
218
219 if (vtype[j] != 'B') {
220 // Move non-binary variables to the RHS
221 if (coeff < 0 && lb[j] != -NumericUtils::infty) {
222 rhs_values[row] -= coeff * lb[j];
223 } else if (ub[j] != NumericUtils::infty) {
224 rhs_values[row] -= coeff * ub[j];
225 }
226 } else {
227 // Handle binary variables
228 if (coeff > 0) {
229 binCoeffs.push_back(coeff);
230 } else {
231 rhs_values[row] -= coeff;
232 binCoeffs.push_back(-coeff);
233 }
234 }
235 }
236
237 std::cout << "Row " << row << " has new rhs " << rhs_values[row] << "\n";
238 }

240 void boundTightening() {
241 std::for_each(rowInfoVector.begin(), rowInfoVector.end(), [&](const RowInfo &rowInfo) {
242 size_t i = &rowInfo - &rowInfoVector[0]; // Get row index
243 SparseMatrix::RowIterator rowIter = A.rowIterator(i);
244 double alphaMax = 0.0, alphaMin = 0.0;
245
246 // Calculate min/max activities
247 for (; rowIter.valid(); rowIter.next()) {
248 double coeff = rowIter.value();
249 int j = rowIter.col();
250 alphaMin += coeff > 0 ? coeff * lb[j] : coeff * ub[j];
251 alphaMax += coeff > 0 ? coeff * ub[j] : coeff * lb[j];
252 }
253
254 if (lhs_values[i] > alphaMax || alphaMin > rhs_values[i]) {
255 std::cout << "baldesCtr " << i << " is infeasible.\n";
256 }
257 });
258 }

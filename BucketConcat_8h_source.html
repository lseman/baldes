11template <Stage S>
12void BucketGraph::publish_concatenation_candidates(const Label *forward_label,
13 std::span<const ConcatenationCandidate> candidates,
14 std::atomic<double> &best_cost) {
15 if (candidates.empty()) return;
16
17 std::lock_guard<std::mutex> lock(merge_mutex);
18
19 double current_best = best_cost.load(std::memory_order_relaxed);
20 for (const auto &candidate : candidates) {
21 const Label *candidate_label = candidate.label;
22 const double candidate_cost = candidate.cost;
23
24 if constexpr (S == Stage::Enumerate) {
25 const size_t route_limit = static_cast<size_t>(std::max(1, enumeration_policy.max_routes)) + 1;
26 const double cutoff = std::min(gap, enumeration_route_cutoff.load(std::memory_order_relaxed));
27 if (!numericutils::lt(candidate_cost, cutoff)) continue;
28
29 if (numericutils::lt(candidate_cost, current_best)) {
30 best_cost.store(candidate_cost, std::memory_order_relaxed);
31 current_best = candidate_cost;
32 }
33
34 if (merged_labels.size() < route_limit) {
35 auto pbest = compute_label<S>(forward_label, candidate_label, candidate_cost);
36 merged_labels.push_back(pbest);
37 std::push_heap(merged_labels.begin(), merged_labels.end(),
38 [](const Label *a, const Label *b) { return a->cost < b->cost; });
39 } else {
40 if (merged_labels.empty() || !numericutils::lt(candidate_cost, merged_labels.front()->cost)) continue;
41
42 auto pbest = compute_label<S>(forward_label, candidate_label, candidate_cost);
43 std::pop_heap(merged_labels.begin(), merged_labels.end(),
44 [](const Label *a, const Label *b) { return a->cost < b->cost; });
45 merged_labels.back() = pbest;
46 std::push_heap(merged_labels.begin(), merged_labels.end(),
47 [](const Label *a, const Label *b) { return a->cost < b->cost; });
48 }
49
50 if (merged_labels.size() >= route_limit && !merged_labels.empty()) {
51 enumeration_route_cutoff.store(merged_labels.front()->cost, std::memory_order_relaxed);
52 }
53 } else {
54 if (numericutils::lt(candidate_cost, current_best)) {
55 best_cost.store(candidate_cost, std::memory_order_relaxed);
56 current_best = candidate_cost;
57 }
58
59 if (numericutils::lte(candidate_cost, current_best)) {
60 auto pbest = compute_label<S>(forward_label, candidate_label, candidate_cost);
61 merged_labels.push_back(pbest);
62 }
63 }
64 }
65}

67inline void BucketGraph::push_unvisited_phi_neighbors(std::span<const int> neighbors,
68 ConcatenationScratch &scratch) const {
69 if (scratch.bucket_stack.capacity() < scratch.bucket_stack.size() + neighbors.size()) {
70 scratch.bucket_stack.reserve(scratch.bucket_stack.size() + neighbors.size());
71 }
72 for (int b_prime : neighbors) {
73 const size_t segment = static_cast<size_t>(b_prime) >> 6;
74 const uint64_t mask = bit_mask_lookup[b_prime & 63];
75 if (!(scratch.visited_buckets[segment] & mask)) { scratch.bucket_stack.push_back(b_prime); }
76 }
77}

79template <Stage S>
80bool BucketGraph::collect_concatenation_candidate(const Label *forward_label, const Label *backward_label,
81 double path_cost, const SpliceState *splice_state,
82 [[maybe_unused]] std::span<const ActiveCutInfo> active_cuts,
83 [[maybe_unused]] double max_src_splice_discount,
84 std::atomic<double> &best_cost, ConcatenationScratch &scratch) {
85 ++scratch.stats.labels_tested;
86 if constexpr (S == Stage::Four) {
87 constexpr uint64_t kStageFourLocalScanBudget = 25000;
88 if (scratch.stats.labels_tested > kStageFourLocalScanBudget) {
89 pricing_truncated.store(true, std::memory_order_relaxed);
90 return false;
91 }
92 }
93
94 const int forward_node_id = forward_label->node_id;
95 if (backward_label == nullptr || backward_label->is_dominated || backward_label->node_id == forward_node_id) {
96 return false;
97 }
98
99 double total_cost = path_cost + backward_label->cost;
100
101#if defined(SRC)
102 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
103 const double prune_limit =
104 (S == Stage::Enumerate) ? std::min(gap, enumeration_route_cutoff.load(std::memory_order_relaxed))
105 : best_cost.load(std::memory_order_relaxed);
106 if (!numericutils::lt(total_cost - max_src_splice_discount, prune_limit)) { return false; }
107
108 // In SRC stages the raw splice cost may be too pessimistic because
109 // concatenating the two partial paths can trigger additional SRC dual
110 // reductions. Apply the exact splice adjustment before the expensive
111 // bitmap/resource checks, then use the adjusted cost for safe pruning.
112 for (const auto &active_cut : active_cuts) {
113 const auto &cut = *active_cut.cut_ptr;
114 const size_t idx = active_cut.index;
115 const double dual = active_cut.dual_value;
116#if defined(SRC_MEMORY_MODE_ARC)
117 if (!cut.isSRCMemoryArc(forward_label->node_id, backward_label->node_id)) continue;
118#endif
119 const bool cut_hit = splice_state && splice_state->has_src
120 ? splice_state->SRCmap[idx] + backward_label->SRCmap[idx] >= cut.p.den
121 : forward_label->SRCmap[idx] + backward_label->SRCmap[idx] >= cut.p.den;
122 if (cut_hit) { total_cost -= dual; }
123 }
124 } else
125#endif
126 {
127 if (!numericutils::lt(total_cost, best_cost.load(std::memory_order_relaxed))) { return false; }
128 }
129
130 bool cost_acceptable;
131 if constexpr (S != Stage::Enumerate) {
132 cost_acceptable = numericutils::lt(total_cost, best_cost.load(std::memory_order_relaxed));
133 } else {
134 const double cutoff = std::min(gap, enumeration_route_cutoff.load(std::memory_order_relaxed));
135 cost_acceptable = numericutils::lt(total_cost, cutoff);
136 }
137
138 if (!cost_acceptable) { return false; }
139 if (visited_overlap(forward_label->visited_bitmap, backward_label->visited_bitmap)) { return false; }
140 if (!check_feasibility(forward_label, backward_label, splice_state)) { return false; }
141
142 scratch.candidates.push_back({backward_label, total_cost});
143 ++scratch.stats.labels_accepted;
144 if constexpr (S == Stage::Four) {
145 constexpr uint64_t kStageFourCandidateCap = static_cast<uint64_t>(N_ADD) + 2ULL;
146 const uint64_t hits = pricing_candidate_hits.fetch_add(1, std::memory_order_relaxed) + 1;
147 if (hits >= kStageFourCandidateCap) { pricing_truncated.store(true, std::memory_order_relaxed); }
148 }
149 return true;
150}

152template <Stage S, Symmetry SYM>
153void BucketGraph::concatenate_label_from_bucket(const Label *L, int b, std::atomic<double> &best_cost,
154 const SpliceState *splice_state) {
155 const size_t n_segments = (fw_buckets_size + 63) / 64;
156 static thread_local ConcatenationScratch scratch;
157 scratch.reset(n_segments, b);
158
159 auto &other_buckets = assign_symmetry<SYM>(fw_buckets, bw_buckets);
160 auto &other_c_bar = assign_symmetry<SYM>(fw_c_bar, bw_c_bar);
161 auto &other_phi = assign_symmetry<SYM>(Phi_fw, Phi_bw);
162
163 const int L_node_id = L->node_id;
164 const double L_cost = L->cost;
165 const bool has_branching = !branching_duals->empty();
166
167 std::span<const ActiveCutInfo> active_cuts;
168#if defined(SRC)
169 if constexpr (S > Stage::Three) { active_cuts = cut_storage->getActiveCuts(); }
170#endif
171
172 double max_src_splice_discount = 0.0;
173#if defined(SRC)
174 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
175 const SRCMap &src_map = (splice_state && splice_state->has_src) ? splice_state->SRCmap : L->SRCmap;
176 for (const auto &active_cut : active_cuts) {
177 if (active_cut.dual_value <= 0.0) continue;
178 if (src_map[active_cut.index] == 0) continue;
179 max_src_splice_discount += active_cut.dual_value;
180 }
181 }
182#endif
183
184 while (!scratch.bucket_stack.empty()) {
185 if constexpr (S == Stage::Four) {
186 if (pricing_truncated.load(std::memory_order_relaxed)) break;
187 }
188
189 const int current_bucket = scratch.bucket_stack.back();
190 scratch.bucket_stack.pop_back();
191
192 const size_t segment = current_bucket >> 6;
193 const uint64_t bit_mask = bit_mask_lookup[current_bucket & 63];
194 scratch.visited_buckets[segment] |= bit_mask;
195
196 const int bucketLprimenode = other_buckets[current_bucket].node_id;
197 double travel_cost = getcij(L_node_id, bucketLprimenode);
198
199#if defined(RCC) || defined(EXACT_RCC)
200 if constexpr (S == Stage::Four) { travel_cost -= arc_duals.getDual(L_node_id, bucketLprimenode); }
201#endif
202
203 if (has_branching) { travel_cost -= branching_duals->getDual(L_node_id, bucketLprimenode); }
204
205 const double path_cost = L_cost + travel_cost + (splice_state ? splice_state->cost_delta : 0.0);
206 const double bound = other_c_bar[current_bucket];
207
208 double prune_limit = best_cost.load(std::memory_order_relaxed);
209 if constexpr (S == Stage::Enumerate) {
210 prune_limit = std::min(gap, enumeration_route_cutoff.load(std::memory_order_relaxed));
211 }
212
213 double bucket_bound_cost = path_cost + bound;
214#if defined(SRC)
215 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
216 bucket_bound_cost -= max_src_splice_discount;
217 }
218#endif
219
220 if (numericutils::gte(bucket_bound_cost, prune_limit)) { continue; }
221
222 const auto &bucket = other_buckets[current_bucket];
223 const auto &labels = bucket.get_sorted_labels();
224 const auto &extra_labels = bucket.get_extra_labels();
225 const auto label_count = labels.size() + extra_labels.size();
226 if (label_count == 0) continue;
227 scratch.prepare_candidates(label_count);
228
229 for (const Label *L_bw : labels) {
230 if constexpr (S == Stage::Four) {
231 if (pricing_truncated.load(std::memory_order_relaxed)) break;
232 }
233 collect_concatenation_candidate<S>(L, L_bw, path_cost, splice_state, active_cuts, max_src_splice_discount,
234 best_cost, scratch);
235 }
236 for (const Label *L_bw : extra_labels) {
237 if constexpr (S == Stage::Four) {
238 if (pricing_truncated.load(std::memory_order_relaxed)) break;
239 }
240 collect_concatenation_candidate<S>(L, L_bw, path_cost, splice_state, active_cuts, max_src_splice_discount,
241 best_cost, scratch);
242 }
243
244 publish_concatenation_candidates<S>(L, scratch.candidates, best_cost);
245 push_unvisited_phi_neighbors(other_phi[current_bucket], scratch);
246 }
247
248 concatenation_labels_tested.fetch_add(scratch.stats.labels_tested, std::memory_order_relaxed);
249 concatenation_labels_accepted.fetch_add(scratch.stats.labels_accepted, std::memory_order_relaxed);
250}

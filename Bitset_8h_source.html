2 * @file Bitset.h
3 * @brief This file contains the definition of the Bitset class.
4 *
5 * This file contains the definition of the Bitset class, which represents a fixed-size bitset.
6 * The Bitset class provides methods to set, clear, test, toggle, count, check if all bits are set,
7 * check if no bits are set, perform bitwise AND, OR, XOR, and NOT operations, and compare bitsets.
8 * It also provides methods to convert to an integer, hash the bitset, and stream the bitset.
9 * The Bitset class is implemented using a std::array of uint64_t to store the bits.
10 *
11 */

18template <size_t N_BITS>
19class Bitset {
20private:
21 static constexpr size_t NUM_WORDS = (N_BITS + 63) / 64;
22
23 // Helper functions for bit index calculations
24 static constexpr size_t wordIndex(size_t bit) { return bit / 64; }
25
26 static constexpr size_t bitOffset(size_t bit) { return bit % 64; }
27
28public:
29 std::array<uint64_t, NUM_WORDS> bits_{};

31 // Bit proxy class for enabling read/write access with `[]`
32 class BitReference {
33 private:
34 uint64_t &word_;
35 uint64_t mask_;
36
37 public:
38 BitReference(uint64_t &word, uint64_t mask) : word_(word), mask_(mask) {}
39
40 // Assignment operator
41 BitReference &operator=(bool value) {
42 if (value)
43 word_ |= mask_;
44 else
45 word_ &= ~mask_;
46 return *this;
47 }
48
49 // Conversion to bool for reading the bit
50 operator bool() const { return (word_ & mask_) != 0; }
51
52 // Assignment from another BitReference
53 BitReference &operator=(const BitReference &other) { return *this = bool(other); }
54 };
55
56 // Subscript operator for read/write access
57 BitReference operator[](size_t bit) { return BitReference(bits_[wordIndex(bit)], 1ULL << bitOffset(bit)); }
58
59 // Const subscript operator for read-only access
60 bool operator[](size_t bit) const { return (bits_[wordIndex(bit)] >> bitOffset(bit)) & 1ULL; }
61
62 // Other member functions from before...
63
64 void set(size_t bit) { bits_[wordIndex(bit)] |= (1ULL << bitOffset(bit)); }
65
66 void clear(size_t bit) { bits_[wordIndex(bit)] &= ~(1ULL << bitOffset(bit)); }
67
68 bool test(size_t bit) const { return (bits_[wordIndex(bit)] >> bitOffset(bit)) & 1ULL; }
69
70 void toggle(size_t bit) { bits_[wordIndex(bit)] ^= (1ULL << bitOffset(bit)); }
71
72 size_t count() const {
73 return std::accumulate(bits_.begin(), bits_.end(), 0ULL,
74 [](size_t total, uint64_t word) { return total + __builtin_popcountll(word); });
75 }
76
77 // define .none() and .all() functions
78 bool none() const {
79 // Return true if no word contains a set bit (i.e., all are zero)
80 return std::none_of(bits_.begin(), bits_.end(), [](uint64_t word) { return word != 0; });
81 }
82
83 bool all() const {
84 // Return true if all words are fully set (i.e., all are UINT64_MAX)
85 return std::all_of(bits_.begin(), bits_.end(), [](uint64_t word) { return word == UINT64_MAX; });
86 }

88 // Bitwise AND operator
89 Bitset operator&(const Bitset &other) const {
90 Bitset result;

92 return result;
93 }

97 Bitset result;

99 return result;

104 Bitset result;

111 Bitset result;

113 return result;
114 }

116 // Logical operators (`&=`, `|=`, `^=`)
117 Bitset &operator&=(const Bitset &other) {
118 for (size_t i = 0; i < NUM_WORDS; ++i) { bits_[i] &= other.bits_[i]; }
119 return *this;
120 }

130 }

132 // Constructor to initialize from an integer (like 0)
133 Bitset(uint64_t value = 0) {
134 bits_.fill(0); // Initialize all bits to 0
135 if (value != 0) {
136 bits_[0] = value; // If non-zero, set only the first word
137 }
138 }

140 // Copy assignment from an integer to clear or partially set the bitset
141 Bitset &operator=(uint64_t value) {
142 bits_.fill(0);
143 if (value != 0) { bits_[0] = value; }
144 return *this;
145 }
146
147 bool operator==(const Bitset &other) const { return bits_ == other.bits_; }
148

152 // define << operator
153 friend std::ostream &operator<<(std::ostream &os, const Bitset &bitset) {
154 for (size_t i = 0; i < N_BITS; ++i) { os << bitset[i]; }
155 return os;

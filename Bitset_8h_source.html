24template <size_t N_BITS>
25class Bitset {
26 private:
27 static constexpr size_t NUM_WORDS = (N_BITS + 63) / 64;
28
29 static constexpr size_t wordIndex(size_t bit) { return bit / 64; }
30 static constexpr size_t bitOffset(size_t bit) { return bit % 64; }
31
32 public:
33 alignas(32) std::array<uint64_t, NUM_WORDS> bits_{};
34
35 constexpr Bitset(uint64_t value = 0) {
36 bits_.fill(0);
37 if (value != 0) {
38 bits_[0] = value;
39 }
40 }
41
42 constexpr void set(size_t bit) noexcept {
43 bits_[wordIndex(bit)] |= (1ULL << bitOffset(bit));
44 }
45 constexpr void clear(size_t bit) noexcept {
46 bits_[wordIndex(bit)] &= ~(1ULL << bitOffset(bit));
47 }
48 constexpr bool test(size_t bit) const noexcept {
49 return (bits_[wordIndex(bit)] >> bitOffset(bit)) & 1ULL;
50 }
51 constexpr void toggle(size_t bit) noexcept {
52 bits_[wordIndex(bit)] ^= (1ULL << bitOffset(bit));
53 }
54 constexpr void reset() noexcept { bits_.fill(0); }

56 // Const subscript operator for read-only access
57 constexpr bool operator[](size_t bit) const noexcept {
58 return (bits_[wordIndex(bit)] >> bitOffset(bit)) & 1ULL;
59 }

75 size_t count() const noexcept {

71void initRMP(MIPProblem *model, VRProblem *problem,
72 std::vector<std::vector<int>> &heuristicRoutes) {
73 auto instance = problem->instance;
74 int Um = problem->instance.nV; // Max number of vehicles
75
76 std::vector<MIPColumn> lambdaCols;
77 std::vector<double> lb, ub, obj;
78 std::vector<std::string> names;
79 std::vector<VarType> vtypes;
80
81 // Lambda function to compute the cost of a route
82 auto costCalc = [&](const std::vector<int> &route) {
83 double distance = 0.0;
84 for (size_t i = 0; i < route.size() - 1; ++i) {
85 const auto &source = route[i];
86 const auto &target = route[i + 1];
87 distance += instance.getcij(source, target);
88 }
89 distance += instance.getcij(route.back(), 0); // Return to the depot
90 return distance;
91 };
92
93 // Create a decision variable for each heuristic-generated route
94 int id = 0;
95 for (const auto &column : heuristicRoutes) {
96 int routeID = id;
97 std::string name = "x[" + std::to_string(routeID) + "]";
98 double cost = costCalc(column);
99
100 // Store the column and variable data
101 lb.push_back(0.0);
102 ub.push_back(1.0);
103 obj.push_back(cost);
104 names.push_back(name);
105 vtypes.push_back(
106 VarType::Continuous); // Assuming VarType::Continuous is used
107
108 id++;
109 }
110
111 // Add variables to the model
112 model->addVars(lb.data(), ub.data(), obj.data(), vtypes.data(),
113 names.data(), lb.size());
114
115 // First set of constraints: Each vertex should be visited at least once
116 for (int i = 1; i < instance.getNbVertices(); ++i) {
117 LinearExpression lhs;
118 for (size_t j = 0; j < heuristicRoutes.size(); ++j) {
119 // Count the occurrences of vertex i in heuristicRoutes[j]
120 int occurrences = std::count(heuristicRoutes[j].begin(),
121 heuristicRoutes[j].end(), i);
122 if (occurrences > 0) {
123 lhs.addTerm(model->getVar(j), occurrences);
124 }
125 }
126 model->add_constraint(lhs, 1.0,
127 '>'); // lhs >= 1 (visit the node at least once)
128 }
129
130 // Second part: Ensure the number of vehicles does not exceed the maximum
131 LinearExpression vehicle_constraint_lhs;
132 for (size_t j = 0; j < heuristicRoutes.size(); ++j) {
133 vehicle_constraint_lhs.addTerm(model->getVar(j), 1.0);
134 }
135 model->add_constraint(vehicle_constraint_lhs, static_cast<double>(Um),
136 '<'); // baldesCtr: sum(lambda) <= Um
137
138 // Set the objective to minimize
139 model->setObjectiveSense(ObjectiveType::Minimize);
140}

41void BucketGraph::UpdateBucketsSet(const double theta, const Label *label,
42 ankerl::unordered_dense::set<int> &Bbidi,
43 int &current_bucket,
44 std::vector<uint64_t> &Bvisited) {
45 // Precompute values and assign references for the direction-specific data
46
47 auto &Phi_opposite = assign_buckets<D>(Phi_bw, Phi_fw);
48 auto &buckets_opposite = assign_buckets<D>(bw_buckets, fw_buckets);
49 auto &c_bar_opposite = assign_buckets<D>(bw_c_bar, fw_c_bar);
50
51 const int bucketLnode = label->node_id;
52 std::vector<int> bucket_stack;
53 bucket_stack.reserve(10); // Pre-allocate space for the stack
54 bucket_stack.push_back(current_bucket);
55
56 // Lambda to compare bitmaps
57 auto bitmaps_conflict = [&](const Label *L1, const Label *L2) {
58 for (size_t i = 0; i < L1->visited_bitmap.size(); ++i) {
59 if ((L1->visited_bitmap[i] & L2->visited_bitmap[i]) != 0) {
60 return true;
61 }
62 }
63 return false;
64 };
65
66 while (!bucket_stack.empty()) {
67 int curr_bucket = bucket_stack.back();
68 bucket_stack.pop_back();
69 const size_t segment =
70 curr_bucket >> 6; // Determine the segment for the current bucket
71 const size_t bit_position =
72 curr_bucket & 63; // Determine the bit position within the segment
73 Bvisited[segment] |=
74 (1ULL << bit_position); // Set the bit corresponding to the current
75 // bucket as visited
76 // print current bucket
77 const int bucketLprimenode = buckets_opposite[curr_bucket].node_id;
78 double cost = getcij(bucketLnode, bucketLprimenode);
79
80 RCC_MODE_BLOCK(auto arc_dual =
81 arc_duals.getDual(bucketLnode, bucketLprimenode);
82 cost -= arc_dual;)
83 // Stop exploring if the cost exceeds the threshold
84 if (label->cost + cost + c_bar_opposite[curr_bucket] >= theta) {
85 continue;
86 }
87
88 // Only process if bucket is not already in Bbidi
89 if (Bbidi.insert(curr_bucket).second) {
90 const auto opposite_labels =
91 buckets_opposite[curr_bucket].get_labels();
92 for (const auto &L : opposite_labels) {
93 if (label->node_id == L->node_id ||
94 bitmaps_conflict(label, L)) {
95 continue; // Skip labels with the same node ID or
96 // conflicting bitmaps
97 }
98
99 const VRPNode &L_last_node = (D == Direction::Forward)
100 ? nodes[label->node_id]
101 : nodes[L->node_id];
102
103 bool violated = false;
104 for (auto r = 0; r < options.resources.size(); ++r) {
105 if (options.resources[r] == "time") {
106 const double time_constraint =
107 (D == Direction::Forward)
108 ? label->resources[TIME_INDEX] + cost +
109 L_last_node.duration
110 : L->resources[TIME_INDEX] + cost +
111 L_last_node.duration;
112
113 if (time_constraint >
114 ((D == Direction::Forward)
115 ? L->resources[TIME_INDEX]
116 : label->resources[TIME_INDEX])) {
117 violated = true;
118 break;
119 }
120 } else {
121 const double resource_constraint =
122 (D == Direction::Forward)
123 ? label->resources[r] +
124 L_last_node.consumption[r]
125 : L->resources[r] + L_last_node.consumption[r];
126
127 if (resource_constraint > ((D == Direction::Forward)
128 ? L->resources[r]
129 : label->resources[r])) {
130 violated = true;
131 break;
132 }
133 }
134 }
135
136 if (violated) {
137 continue;
138 }
139
140 // Insert the current bucket into Bbidi if the total cost is
141 // below theta
142 if (label->cost + cost + L->cost < theta) {
143 Bbidi.emplace(curr_bucket);
144 }
145 }
146 }
147
148 // Add neighboring buckets to the stack, only if not visited
149 for (int b_prime : Phi_opposite[curr_bucket]) {
150 const size_t segment_prime =
151 b_prime >>
152 6; // Determine the segment for the neighboring bucket
153 const size_t bit_position_prime =
154 b_prime & 63; // Determine the bit position within the segment
155
156 // If the neighboring bucket hasn't been visited, push it onto the
157 // stack
158 if ((Bvisited[segment_prime] & (1ULL << bit_position_prime)) == 0) {
159 bucket_stack.push_back(b_prime);
160 }
161 }
162 }
163}

340void BucketGraph::ObtainJumpBucketArcs() {
341 // Assign forward or backward buckets, fixed buckets, bucket indices, and
342 // Phi (adjacency list) based on direction
343 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
344 auto &fixed_buckets = assign_buckets<D>(fw_fixed_buckets, bw_fixed_buckets);
345 auto &num_buckets_index =
346 assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
347 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
348 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
349 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
350
351 int arc_counter = 0; // Counter for jump arcs added
352 int missing_counter = 0; // Counter for missing paths
353
354 // iterate over all nodes and clear jump arcs
355 for (auto &node : nodes) {
356 node.clear_jump_arcs<D>();
357 }
358
359 // Loop through all buckets
360 for (int b = 0; b < buckets_size; ++b) {
361 std::vector<int> B_bar; // Temporary storage for valid bucket indices
362
363 // Cache the current bucket's node ID and arcs
364 const int current_node_id = buckets[b].node_id;
365 const auto &arcs = buckets[b].template get_bucket_arcs<D>();
366 const auto &original_arcs =
367 nodes[current_node_id].template get_arcs<D>();
368
369 if (arcs.empty()) {
370 continue;
371 } // Skip if no arcs in the current bucket
372
373 // Process each original arc for the current node
374 for (const auto &orig_arc : original_arcs) {
375 const int from_node = orig_arc.from;
376 const int to_node = orig_arc.to;
377 bool have_path = false;
378
379 // Check if the path exists in the current bucket arcs
380 for (const auto &gamma : arcs) {
381 if (is_bucket_fixed<D>(gamma.from_bucket, gamma.to_bucket)) {
382 continue;
383 }
384
385 const int from_node_b = buckets[gamma.from_bucket].node_id;
386 const int to_node_b = buckets[gamma.to_bucket].node_id;
387
388 if (from_node == from_node_b && to_node == to_node_b) {
389 have_path = true;
390 break; // Path found, exit early
391 }
392 }
393
394 // If no path was found, find a jump arc
395 if (!have_path) {
396 ++missing_counter; // Increment missing path counter
397
398 // Cache the starting bucket and the number of buckets for this
399 // node
400 const int start_bucket = num_buckets_index[current_node_id];
401 const int node_buckets = num_buckets[current_node_id];
402
403 auto process_bucket = [&](int b, int b_prime) {
404 // Check if b_prime is adjacent to b in Phi
405 const auto &phi_b_prime = Phi[b_prime];
406 if (std::find(phi_b_prime.begin(), phi_b_prime.end(), b) ==
407 phi_b_prime.end()) {
408 return; // Not adjacent, skip
409 }
410
411 // Check arcs in the adjacent bucket b_prime
412 const auto &arcs_prime =
413 buckets[b_prime].template get_bucket_arcs<D>();
414 for (const auto &gamma_prime : arcs_prime) {
415 if (is_bucket_fixed<D>(gamma_prime.from_bucket,
416 gamma_prime.to_bucket)) {
417 continue;
418 }
419
420 const int from_node_prime =
421 buckets[gamma_prime.from_bucket].node_id;
422 const int to_node_prime =
423 buckets[gamma_prime.to_bucket].node_id;
424
425 // If the arc matches, add a jump arc
426 if (from_node == from_node_prime &&
427 to_node == to_node_prime) {
428 B_bar.push_back(
429 b_prime); // Add the valid bucket to B_bar
430
431 // Wrap the resource increment in a vector
432 const std::vector<double> res =
433 gamma_prime.resource_increment;
434 const double cost = gamma_prime.cost_increment;
435
436 buckets[b].template add_jump_arc<D>(b, b_prime, res,
437 cost);
438 nodes[buckets[b].node_id].template add_jump_arc<D>(
439 b, b_prime, res, cost, to_node);
440 ++arc_counter; // Increment the jump arc counter
441 }
442 }
443 };
444
445 if constexpr (D == Direction::Forward) {
446 // Forward direction: incrementing buckets
447 for (int b_prime = b + 1;
448 b_prime < start_bucket + node_buckets; ++b_prime) {
449 process_bucket(b, b_prime);
450 }
451 } else {
452 // Backward direction: decrementing buckets
453 // for (int b_prime = b - 1; b_prime >= start_bucket;
454 // --b_prime) { process_bucket(b, b_prime); }
455 for (int b_prime = b + 1;
456 b_prime < start_bucket + node_buckets; ++b_prime) {
457 process_bucket(b, b_prime);
458 }
459 }
460 }
461 }
462 }
463 CONDITIONAL(D, print_info("[Fw] added {} jump arcs\n", arc_counter),
464 print_info("[Bw] added {} jump arcs\n", arc_counter));
465}

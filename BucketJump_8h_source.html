41void BucketGraph::UpdateBucketsSet(const double theta, const Label *label,
42 ankerl::unordered_dense::set<int> &Bbidi,
43 int &current_bucket,
44 std::vector<uint64_t> &Bvisited) {
45 // Select direction-specific data.
46 auto &Phi_opposite = assign_buckets<D>(Phi_bw, Phi_fw);
47 auto &buckets_opposite = assign_buckets<D>(bw_buckets, fw_buckets);
48 auto &c_bar_opposite = assign_buckets<D>(bw_c_bar, fw_c_bar);

50 // Cache label values to avoid repeated member lookups.
51 const int bucketLnode = label->node_id;
52 const double label_cost = label->cost;
53 const auto &label_bitmap = label->visited_bitmap;
54 const size_t n_bitmap = label_bitmap.size();

56 // Precompute a flag for direction (avoid repeated ternary operators).
57 constexpr bool forward = (D == Direction::Forward);
58
59 // Lambda to check if the visited bitmaps of L and L_bw overlap.
60 auto overlapsVisited = [n_bitmap, this](const Label *L,
61 const Label *L_bw) -> bool {
62 // Make a local copy of the visited bitmap.
63 auto visited_bitmap = L->visited_bitmap;
64
65 // check if visited_bitmap overlaps at all
66 for (size_t i = 0; i < n_bitmap; ++i) {
67 if ((visited_bitmap[i] & L_bw->visited_bitmap[i]) != 0) {
68 return true;
69 }
70 }
71 // Iterate in reverse over nodes_covered without allocating a new
72 // vector.
73 // for (auto it = L_bw->nodes_covered.rbegin();
74 // it != L_bw->nodes_covered.rend(); ++it) {
75 // uint16_t node_id = *it;
76 // if (node_id == L->node_id ||
77 // is_node_visited(visited_bitmap, node_id))
78 // return true;
79 // // Update the bitmap for each 64-bit word.
80 // for (size_t i = 0; i < visited_bitmap.size(); ++i) {
81 // uint64_t &current_visited = visited_bitmap[i];
82 // if (current_visited != 0)
83 // current_visited &= neighborhoods_bitmap[node_id][i];
84 // }
85 // set_node_visited(visited_bitmap, node_id);
86 // }
87 return false;
88 };

90 // Inline helper function for bitmap conflict check.
91 auto bitmapsConflict = [n_bitmap](const Label *L1,
92 const Label *L2) -> bool {
93 // Assume both L1 and L2 have the same sized visited_bitmap.
94 for (size_t i = 0; i < n_bitmap; ++i) {
95 if ((L1->visited_bitmap[i] & L2->visited_bitmap[i]) != 0) {
96 return true;
97 }
98 }
99 return false;
100 };

102 // Use a local DFS stack to traverse buckets.
103 std::vector<int> bucket_stack;
104 bucket_stack.reserve(10);
105 bucket_stack.push_back(current_bucket);
106
107 // Process buckets until the stack is empty.
108 while (!bucket_stack.empty()) {
109 const int curr_bucket = bucket_stack.back();
110 bucket_stack.pop_back();
111
112 // Mark the current bucket as visited.
113 const size_t segment = curr_bucket >> 6;
114 const size_t bit_pos = curr_bucket & 63;
115 Bvisited[segment] |= (1ULL << bit_pos);
116
117 // Get the opposite bucket’s label and node id.
118 const int bucketLprimenode = buckets_opposite[curr_bucket].node_id;
119 double cost = getcij(bucketLnode, bucketLprimenode);
120
121 // Optionally adjust cost with arc duals if RCC mode is enabled.
122 RCC_MODE_BLOCK(auto arc_dual =
123 arc_duals.getDual(bucketLnode, bucketLprimenode);
124 cost -= arc_dual;)
125
126 // Skip if the accumulated cost plus lower bound exceeds theta.
127 if (numericutils::gt(label_cost + cost + c_bar_opposite[curr_bucket],
128 theta))
129 continue;
130
131 // Process this bucket only if it hasn't been processed already.
132 if (Bbidi.insert(curr_bucket).second) {
133 const auto &opposite_labels =
134 buckets_opposite[curr_bucket].get_labels();
135 for (const auto &L_opposite : opposite_labels) {
136 // Skip labels from the same node or with conflicting visited
137 // bitmaps.
138 // if (bucketLnode == L_opposite->node_id ||
139 // bitmapsConflict(label, L_opposite))
140 // continue;
141 if (bucketLnode == L_opposite->node_id) continue;
142
143 if constexpr (D == Direction::Forward) {
144 // Skip if the visited bitmaps overlap.
145 if (overlapsVisited(label, L_opposite)) continue;
146 } else {
147 // Skip if the visited bitmaps overlap.
148 if (overlapsVisited(L_opposite, label)) continue;
149 }
150
151 // Determine the reference node for resource checks.
152 const VRPNode &ref_node =
153 forward ? nodes[bucketLnode] : nodes[L_opposite->node_id];
154
155 bool violated = false;
156 // Use a local copy of resources if needed.
157 const size_t num_resources = options.resources.size();
158 for (size_t r = 0; r < num_resources; ++r) {
159 if (options.resources[r] == "time") {
160 // For time, add cost and the node's duration.
161 const double time_constraint =
162 forward ? label->resources[TIME_INDEX] + cost +
163 ref_node.duration
164 : L_opposite->resources[TIME_INDEX] + cost +
165 ref_node.duration;
166 if (numericutils::gt(
167 time_constraint,
168 forward ? L_opposite->resources[TIME_INDEX]
169 : label->resources[TIME_INDEX])) {
170 violated = true;
171 break;
172 }
173 } else {
174 const double resource_constraint =
175 forward
176 ? label->resources[r] + ref_node.consumption[r]
177 : L_opposite->resources[r] +
178 ref_node.consumption[r];
179 if (numericutils::gt(resource_constraint,
180 forward ? L_opposite->resources[r]
181 : label->resources[r])) {
182 violated = true;
183 break;
184 }
185 }
186 }
187
188 if (violated) continue;
189
190 // If the merged cost is below theta, record this bucket.
191 if (numericutils::lt(label_cost + cost + L_opposite->cost,
192 theta)) {
193 Bbidi.emplace(curr_bucket);
194 }
195 }
196 }
197
198 // Add unvisited neighbor buckets.
199 for (int b_prime : Phi_opposite[curr_bucket]) {
200 const size_t seg_prime = b_prime >> 6;
201 const size_t bit_prime = b_prime & 63;
202 if ((Bvisited[seg_prime] & (1ULL << bit_prime)) == 0)
203 bucket_stack.push_back(b_prime);
204 }
205 }
206}

218template <Direction D>
219void BucketGraph::BucketArcElimination(double theta) {
220 // Select direction-specific containers.
221 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
222 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
223 auto &Phi_opposite = assign_buckets<D>(Phi_bw, Phi_fw);
224 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
225 auto &fixed_buckets_bitmap =
226 assign_buckets<D>(fw_fixed_buckets_bitmap, bw_fixed_buckets_bitmap);
227
228 // Reset fixed buckets bitmap.
229 fixed_buckets_bitmap.assign(fixed_buckets_bitmap.size(), 0);
230 const int n_buckets = buckets_size;
231 // Compute the number of 64-bit segments for bitmaps.
232 const size_t n_segments = (buckets_size + 63) / 64;
233
234 // Define ArcMap to store arc information.
235 using ArcMap =
236 ankerl::unordered_dense::map<std::pair<std::pair<int, int>, int>,
237 ankerl::unordered_dense::set<int>,
238 arc_map_hash>;
239 ArcMap local_B_Ba_b;
240 int removed_arcs = 0;
241
242 // Helper: create an arc key from two node IDs and a bucket.
243 auto create_arc_key = [](int from, int to, int b) {
244 return std::make_pair(std::make_pair(from, to), b);
245 };
246
247 // Helper lambda to quickly reset a bitmap vector.
248 auto reset_bitmap = [&](std::vector<uint64_t> &bitmap) {
249 // Using memset since our vector is contiguous.
250 if (!bitmap.empty())
251 std::memset(bitmap.data(), 0, bitmap.size() * sizeof(uint64_t));
252 };
253
254 // Lambda: Process jump arcs for bucket 'b'.
255 auto process_jump_arcs = [&](int b) {
256 const auto &jump_arcs = buckets[b].template get_jump_arcs<D>();
257 if (jump_arcs.empty()) return;
258
259 std::vector<uint64_t> Bvisited(n_segments, 0);
260 auto labels = buckets[b].get_labels();
261
262 for (const auto &a : jump_arcs) {
263 auto increment = a.resource_increment;
264 auto arc_key = create_arc_key(a.from_bucket, a.to_bucket, b);
265 int b_opposite =
266 get_opposite_bucket_number<D>(a.to_bucket, increment);
267 auto &Bidi_map = local_B_Ba_b[arc_key];
268
269 // Process each label in the current bucket.
270 for (auto &L_item : labels) {
271 Bidi_map.insert(b);
272 // Reset Bvisited for each UpdateBucketsSet call.
273 reset_bitmap(Bvisited);
274 UpdateBucketsSet<D>(theta, L_item, Bidi_map, b_opposite,
275 Bvisited);
276 }
277 }
278 };
279
280 // Lambda: Process bucket arcs for bucket 'b'.
281 auto process_bucket_arcs = [&](int b) {
282 const auto &bucket_arcs = buckets[b].template get_bucket_arcs<D>();
283 if (bucket_arcs.empty()) return;
284
285 auto labels = buckets[b].get_labels();
286 std::vector<uint64_t> Bvisited(n_segments, 0);
287
288 for (const auto &a : bucket_arcs) {
289 // Build the resource increment vector.
290 std::vector<double> increment(options.resources.size(), 0.0);
291 for (size_t r = 0; r < options.resources.size(); ++r) {
292 if constexpr (D == Direction::Forward)
293 increment[r] = buckets[b].lb[r] + a.resource_increment[r];
294 else
295 increment[r] = buckets[b].ub[r] - a.resource_increment[r];
296 }
297
298 auto arc_key = create_arc_key(buckets[a.from_bucket].node_id,
299 buckets[a.to_bucket].node_id, b);
300 int b_opposite =
301 get_opposite_bucket_number<D>(a.to_bucket, increment);
302 auto &Bidi_map = local_B_Ba_b[arc_key];
303
304 // Process each label in bucket 'b'.
305 for (auto &L_item : labels) {
306 Bidi_map.insert(b);
307 reset_bitmap(Bvisited);
308 UpdateBucketsSet<D>(theta, L_item, Bidi_map, b_opposite,
309 Bvisited);
310 }
311
312 // Check if the opposite direction contains arcs.
313 if (auto it = local_B_Ba_b.find(arc_key);
314 it != local_B_Ba_b.end()) {
315 auto &Bidi_map_opposite = it->second;
316 bool contains = std::any_of(
317 Phi_opposite[b_opposite].begin(),
318 Phi_opposite[b_opposite].end(), [&](int b_prime) {
319 return Bidi_map_opposite.find(b_prime) !=
320 Bidi_map_opposite.end();
321 });
322 if (!contains && Bidi_map_opposite.find(b_opposite) ==
323 Bidi_map_opposite.end()) {
324 size_t bit_pos =
325 static_cast<size_t>(a.from_bucket) * n_buckets +
326 a.to_bucket;
327 fixed_buckets_bitmap[bit_pos / 64] |=
328 (1ULL << (bit_pos % 64));
329 ++removed_arcs;
330 }
331 }
332 }
333 };
334
335 // Process each bucket.
336 for (int b = 0; b < buckets_size; ++b) {
337 // Process node arcs.
338 const auto &node_arcs =
339 nodes[buckets[b].node_id].template get_arcs<D>();
340 for (const auto &a : node_arcs) {
341 auto arc_key = std::make_pair(std::make_pair(a.from, a.to), b);
342 auto &Bidi_map = local_B_Ba_b[arc_key];
343
344 if (!Phi[b].empty()) {
345 Bidi_map.insert(b);
346 // Merge neighbor bucket arcs.
347 for (const auto &b_prime : Phi[b]) {
348 auto neighbor_key =
349 std::make_pair(std::make_pair(a.from, a.to), b_prime);
350 if (auto it = local_B_Ba_b.find(neighbor_key);
351 it != local_B_Ba_b.end()) {
352 Bidi_map.insert(it->second.begin(), it->second.end());
353 }
354 }
355 }
356 }
357
358 // Process jump arcs and bucket arcs.
359 process_jump_arcs(b);
360 process_bucket_arcs(b);
361
362 // iterate over fixed buckets and remove them
363 for (int b_prime = 0; b_prime < buckets_size; ++b_prime) {
364 if (is_bucket_fixed<D>(b, b_prime)) {
365 buckets[b].template remove_bucket_arc<D>(b, b_prime);
366 }
367 }
368 }
369
370 // Print the total number of removed arcs.
371 if constexpr (D == Direction::Forward)
372 print_info("[Fw] removed arcs: {}\n", removed_arcs);
373 else
374 print_info("[Bw] removed arcs: {}\n", removed_arcs);
375}

389template <Direction D>
390void BucketGraph::ObtainJumpBucketArcs() {
391 // Select direction-specific containers.
392 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
393 auto &num_buckets_idx =
394 assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
395 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
396 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
397 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
398
399 int arc_counter = 0; // Count of jump arcs added
400 int missing_counter = 0; // Count of missing paths
401
402 // Clear jump arcs for all nodes.
403 for (auto &node : nodes) {
404 node.clear_jump_arcs<D>();
405 }
406
407 // Process each bucket.
408 for (int b = 0; b < buckets_size; ++b) {
409 std::vector<int> B_bar; // Temporary storage for valid adjacent
410 // bucket indices
411
412 const int current_node_id = buckets[b].node_id;
413 const auto &bucket_arcs = buckets[b].template get_bucket_arcs<D>();
414 const auto &orig_arcs = nodes[current_node_id].template get_arcs<D>();
415
416 // Skip if there are no bucket arcs.
417 if (bucket_arcs.empty()) {
418 continue;
419 }
420
421 // For each original arc from the current node...
422 for (const auto &orig_arc : orig_arcs) {
423 const int from_node = orig_arc.from;
424 const int to_node = orig_arc.to;
425 bool have_path = false;
426
427 // Check if the current bucket arcs contain the desired
428 // path.
429 for (const auto &gamma : bucket_arcs) {
430 if (is_bucket_fixed<D>(gamma.from_bucket, gamma.to_bucket)) {
431 continue;
432 }
433
434 const int from_node_b = buckets[gamma.from_bucket].node_id;
435 const int to_node_b = buckets[gamma.to_bucket].node_id;
436
437 if (from_node == from_node_b && to_node == to_node_b) {
438 have_path = true;
439 break; // Path found: no need to search further.
440 }
441 }
442
443 // If no valid path was found, try to obtain a jump arc.
444 if (!have_path) {
445 ++missing_counter;
446
447 // Determine bucket range for the current node.
448 const int start_bucket = num_buckets_idx[current_node_id];
449 const int node_buckets = num_buckets[current_node_id];
450
451 // Lambda to process candidate adjacent buckets.
452 auto process_bucket = [&](int current_b, int candidate_b) {
453 // Only proceed if candidate bucket is adjacent (per
454 // Phi).
455 const auto &phi_candidate = Phi[candidate_b];
456 if (std::find(phi_candidate.begin(), phi_candidate.end(),
457 current_b) == phi_candidate.end())
458 return;
459
460 // Check candidate bucket's arcs.
461 const auto &candidate_arcs =
462 buckets[candidate_b].template get_bucket_arcs<D>();
463 for (const auto &gamma_candidate : candidate_arcs) {
464 if (is_bucket_fixed<D>(gamma_candidate.from_bucket,
465 gamma_candidate.to_bucket))
466 continue;
467
468 const int from_node_candidate =
469 buckets[gamma_candidate.from_bucket].node_id;
470 const int to_node_candidate =
471 buckets[gamma_candidate.to_bucket].node_id;
472
473 // If the candidate arc matches the original
474 // arc...
475 if (from_node == from_node_candidate &&
476 to_node == to_node_candidate) {
477 B_bar.push_back(candidate_b);
478 std::vector<double> res =
479 gamma_candidate.resource_increment;
480
481 const double cost = gamma_candidate.cost_increment;
482
483 // Add jump arcs both in the bucket and at
484 // the node level.
485 buckets[current_b].template add_jump_arc<D>(
486 current_b, candidate_b, res, cost);
487 nodes[buckets[current_b].node_id]
488 .template add_jump_arc<D>(
489 current_b, candidate_b, res, cost, to_node);
490 buckets[current_b].template add_bucket_arc<D>(
491 current_b, candidate_b, res, cost, true);
492 ++arc_counter;
493 }
494 }
495 };
496
497 // Process candidate buckets. For now both directions
498 // use increasing index. (For backward direction, adjust
499 // the loop as needed.)
500 if constexpr (D == Direction::Forward) {
501 for (int candidate_b = b + 1;
502 candidate_b < start_bucket + node_buckets;
503 ++candidate_b) {
504 process_bucket(b, candidate_b);
505 }
506 } else {
507 // Uncomment and adjust if backward should iterate
508 // in reverse order.
509 // for (int candidate_b = b - 1; candidate_b >=
510 // start_bucket;
511 // --candidate_b) {
512 // process_bucket(b, candidate_b);
513 // }
514 for (int candidate_b = b + 1;
515 candidate_b < start_bucket + node_buckets;
516 ++candidate_b) {
517 process_bucket(b, candidate_b);
518 }
519 }
520 }
521 }
522 }
523
524 // Print summary of jump arcs added.
525 CONDITIONAL(D, print_info("[Fw] added {} jump arcs\n", arc_counter),
526 print_info("[Bw] added {} jump arcs\n", arc_counter));
527}

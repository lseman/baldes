41void BucketGraph::UpdateBucketsSet(const double theta, const Label *label, ankerl::unordered_dense::set<int> &Bbidi,
42 int &current_bucket, std::vector<uint64_t> &Bvisited) {
43 // Select direction-specific data.
44 auto &Phi_opposite = assign_buckets<D>(Phi_bw, Phi_fw);
45 auto &buckets_opposite = assign_buckets<D>(bw_buckets, fw_buckets);
46 auto &c_bar_opposite = assign_buckets<D>(bw_c_bar, fw_c_bar);
47
48 // Cache label values to avoid repeated member lookups.
49 const int bucketLnode = label->node_id;
50 const double label_cost = label->cost;
51 const auto &label_bitmap = label->visited_bitmap;
52 const size_t n_bitmap = label_bitmap.size();
53
54 // Precompute a flag for direction (avoid repeated ternary operators).
55 constexpr bool forward = (D == Direction::Forward);
56
57 // Lambda to check if the visited bitmaps of L and L_bw overlap.
58 auto overlapsVisited = [n_bitmap, this](const Label *L, const Label *L_bw) -> bool {
59 // Make a local copy of the visited bitmap.
60 auto visited_bitmap = L->visited_bitmap;
61
62 // check if visited_bitmap overlaps at all
63 for (size_t i = 0; i < n_bitmap; ++i) {
64 if ((visited_bitmap[i] & L_bw->visited_bitmap[i]) != 0) { return true; }
65 }
66 // Iterate in reverse over nodes_covered without allocating a new
67 // vector.
68 // for (auto it = L_bw->nodes_covered.rbegin();
69 // it != L_bw->nodes_covered.rend(); ++it) {
70 // uint16_t node_id = *it;
71 // if (node_id == L->node_id ||
72 // is_node_visited(visited_bitmap, node_id))
73 // return true;
74 // // Update the bitmap for each 64-bit word.
75 // for (size_t i = 0; i < visited_bitmap.size(); ++i) {
76 // uint64_t &current_visited = visited_bitmap[i];
77 // if (current_visited != 0)
78 // current_visited &= neighborhoods_bitmap[node_id][i];
79 // }
80 // set_node_visited(visited_bitmap, node_id);
81 // }
82 return false;
83 };
84
85 // Use a local DFS stack to traverse buckets.
86 static thread_local std::vector<int> bucket_stack;
87 bucket_stack.clear();
88 bucket_stack.reserve(32);
89 bucket_stack.push_back(current_bucket);
90
91 // Process buckets until the stack is empty.
92 while (!bucket_stack.empty()) {
93 const int curr_bucket = bucket_stack.back();
94 bucket_stack.pop_back();
95
96 // Skip if already visited.
97 const size_t segment = curr_bucket >> 6;
98 const size_t bit_pos = curr_bucket & 63;
99 if ((Bvisited[segment] & (1ULL << bit_pos)) != 0) { continue; }
100
101 // Mark the current bucket as visited.
102 Bvisited[segment] |= (1ULL << bit_pos);
103
104 // Get the opposite bucket’s label and node id.
105 const int bucketLprimenode = buckets_opposite[curr_bucket].node_id;
106 double cost = getcij(bucketLnode, bucketLprimenode);
107
108 // Optionally adjust cost with arc duals if RCC mode is enabled.
109 RCC_MODE_BLOCK(auto arc_dual = arc_duals.getDual(bucketLnode, bucketLprimenode); cost -= arc_dual;)
110
111 // Skip if the accumulated cost plus lower bound exceeds theta.
112 if (numericutils::gt(label_cost + cost + c_bar_opposite[curr_bucket], theta)) continue;
113
114 const auto &opposite_labels = buckets_opposite[curr_bucket].get_labels();
115 for (const auto &L_opposite : opposite_labels) {
116 // Skip labels from the same node.
117 if (bucketLnode == L_opposite->node_id) continue;
118
119 if constexpr (D == Direction::Forward) {
120 if (overlapsVisited(label, L_opposite)) continue;
121 } else {
122 if (overlapsVisited(L_opposite, label)) continue;
123 }
124
125 // Determine the reference node for resource checks.
126 const VRPNode &ref_node = forward ? nodes[bucketLnode] : nodes[L_opposite->node_id];
127
128 bool violated = false;
129 const size_t num_resources = options.resources.size();
130 for (size_t r = 0; r < num_resources; ++r) {
131 if (options.resources[r] == "time") {
132 const double time_constraint = forward
133 ? label->resources[TIME_INDEX] + cost + ref_node.duration
134 : L_opposite->resources[TIME_INDEX] + cost + ref_node.duration;
135 if (numericutils::gt(time_constraint,
136 forward ? L_opposite->resources[TIME_INDEX] : label->resources[TIME_INDEX])) {
137 violated = true;
138 break;
139 }
140 } else {
141 const double resource_constraint = forward ? label->resources[r] + ref_node.consumption[r]
142 : L_opposite->resources[r] + ref_node.consumption[r];
143 if (numericutils::gt(resource_constraint,
144 forward ? L_opposite->resources[r] : label->resources[r])) {
145 violated = true;
146 break;
147 }
148 }
149 }
150
151 if (violated) continue;
152
153 // Record only buckets that have at least one theta-compatible
154 // label pair.
155 if (numericutils::lt(label_cost + cost + L_opposite->cost, theta)) {
156 Bbidi.emplace(curr_bucket);
157 break;
158 }
159 }
160
161 // Add unvisited neighbor buckets.
162 for (int b_prime : Phi_opposite[curr_bucket]) {
163 const size_t seg_prime = b_prime >> 6;
164 const size_t bit_prime = b_prime & 63;
165 if ((Bvisited[seg_prime] & (1ULL << bit_prime)) == 0) bucket_stack.push_back(b_prime);
166 }
167 }

180template <Direction D>
181void BucketGraph::BucketArcElimination(double theta) {
182 // Select direction-specific containers.
183 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
184 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
185 auto &Phi_opposite = assign_buckets<D>(Phi_bw, Phi_fw);
186 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
187 auto &fixed_buckets_bitmap = assign_buckets<D>(fw_fixed_buckets_bitmap, bw_fixed_buckets_bitmap);
188
189 // Reset fixed buckets bitmap.
190 fixed_buckets_bitmap.assign(fixed_buckets_bitmap.size(), 0);
191 const int n_buckets = buckets_size;
192 // Compute the number of 64-bit segments for bitmaps.
193 const size_t n_segments = (buckets_size + 63) / 64;
194
195 // Define ArcMap to store arc information.
196 using ArcMap = ankerl::unordered_dense::map<std::pair<std::pair<int, int>, int>, ankerl::unordered_dense::set<int>,
197 arc_map_hash>;
198 ArcMap local_B_Ba_b;
199 int removed_arcs = 0;
200
201 // Helper: create an arc key from two node IDs and a bucket.
202 auto create_arc_key = [](int from, int to, int b) { return std::make_pair(std::make_pair(from, to), b); };
203
204 // Helper lambda to quickly reset a bitmap vector.
205 auto reset_bitmap = [&](std::vector<uint64_t> &bitmap) {
206 // Using memset since our vector is contiguous.
207 if (!bitmap.empty()) std::memset(bitmap.data(), 0, bitmap.size() * sizeof(uint64_t));
208 };
209
210 // Check if there exists b' in BB such that b' <= bb_arr in the preorder
211 // induced by Phi_opposite (i.e., reachable from bb_arr through Phi*).
212 auto has_compatible_predecessor = [&](int bb_arr, const ankerl::unordered_dense::set<int> &BB) -> bool {
213 if (BB.find(bb_arr) != BB.end()) return true;
214
215 std::vector<int> stack;
216 stack.push_back(bb_arr);
217 std::vector<uint64_t> visited(n_segments, 0ULL);
218
219 while (!stack.empty()) {
220 const int curr = stack.back();
221 stack.pop_back();
222 const size_t seg = static_cast<size_t>(curr) >> 6;
223 const uint64_t mask = 1ULL << (curr & 63);
224 if (visited[seg] & mask) continue;
225 visited[seg] |= mask;
226
227 if (BB.find(curr) != BB.end()) return true;
228
229 for (int pred : Phi_opposite[curr]) {
230 const size_t pseg = static_cast<size_t>(pred) >> 6;
231 const uint64_t pmask = 1ULL << (pred & 63);
232 if ((visited[pseg] & pmask) == 0) { stack.push_back(pred); }
233 }
234 }
235 return false;
236 };

238 // Lambda: Process jump arcs for bucket 'b'.
239 auto process_jump_arcs = [&](int b) {
240 const auto &bucket_arcs = buckets[b].template get_bucket_arcs<D>();
241 if (bucket_arcs.empty()) return;
242
243 std::vector<uint64_t> Bvisited(n_segments, 0);
244 const auto &labels = buckets[b].get_labels();
245
246 for (const auto &a : bucket_arcs) {
247 if (!a.jump) continue;
248 std::vector<double> increment(options.resources.size(), 0.0);
249 for (size_t r = 0; r < options.resources.size(); ++r) {
250 if constexpr (D == Direction::Forward)
251 increment[r] = buckets[b].lb[r] + a.resource_increment[r];
252 else
253 increment[r] = buckets[b].ub[r] - a.resource_increment[r];
254 }
255
256 auto arc_key = create_arc_key(buckets[a.from_bucket].node_id, buckets[a.to_bucket].node_id, b);
257 int b_opposite = get_opposite_bucket_number<D>(a.to_bucket, increment);
258 auto &Bidi_map = local_B_Ba_b[arc_key];
259
260 // Process each label in the current bucket.
261 for (auto &L_item : labels) {
262 // Reset Bvisited for each UpdateBucketsSet call.
263 reset_bitmap(Bvisited);
264 UpdateBucketsSet<D>(theta, L_item, Bidi_map, b_opposite, Bvisited);
265 }
266 }
267 };

269 // Lambda: Process bucket arcs for bucket 'b'.
270 auto process_bucket_arcs = [&](int b) {
271 const auto &bucket_arcs = buckets[b].template get_bucket_arcs<D>();
272 if (bucket_arcs.empty()) return;
273
274 const auto &labels = buckets[b].get_labels();
275 std::vector<uint64_t> Bvisited(n_segments, 0);
276
277 for (const auto &a : bucket_arcs) {
278 // Build the resource increment vector.
279 std::vector<double> increment(options.resources.size(), 0.0);
280 for (size_t r = 0; r < options.resources.size(); ++r) {
281 if constexpr (D == Direction::Forward)
282 increment[r] = buckets[b].lb[r] + a.resource_increment[r];
283 else
284 increment[r] = buckets[b].ub[r] - a.resource_increment[r];
285 }
286
287 auto arc_key = create_arc_key(buckets[a.from_bucket].node_id, buckets[a.to_bucket].node_id, b);
288 int b_opposite = get_opposite_bucket_number<D>(a.to_bucket, increment);
289 auto &Bidi_map = local_B_Ba_b[arc_key];
290
291 // Process each label in bucket 'b'.
292 for (auto &L_item : labels) {
293 reset_bitmap(Bvisited);
294 UpdateBucketsSet<D>(theta, L_item, Bidi_map, b_opposite, Bvisited);
295 }
296
297 // Fix gamma if there is no compatible predecessor bucket in BB.
298 if (!has_compatible_predecessor(b_opposite, Bidi_map)) {
299 const size_t bit_pos = static_cast<size_t>(a.from_bucket) * n_buckets + a.to_bucket;
300 fixed_buckets_bitmap[bit_pos / 64] |= (1ULL << (bit_pos % 64));
301 ++removed_arcs;
302 }
303 }
304 };
305
306 // Process each bucket.
307 for (int b = 0; b < buckets_size; ++b) {
308 // Process node arcs.
309 const auto &node_arcs = nodes[buckets[b].node_id].template get_arcs<D>();
310 for (const auto &a : node_arcs) {
311 auto arc_key = std::make_pair(std::make_pair(a.from, a.to), b);
312 auto &Bidi_map = local_B_Ba_b[arc_key];
313
314 if (!Phi[b].empty()) {
315 // Merge neighbor bucket arcs.
316 for (const auto &b_prime : Phi[b]) {
317 auto neighbor_key = std::make_pair(std::make_pair(a.from, a.to), b_prime);
318 if (auto it = local_B_Ba_b.find(neighbor_key); it != local_B_Ba_b.end()) {
319 Bidi_map.insert(it->second.begin(), it->second.end());
320 }
321 }
322 }
323 }
324
325 // Process jump arcs and bucket arcs.
326 process_jump_arcs(b);
327 process_bucket_arcs(b);
328
329 // iterate over fixed buckets and remove them
330 for (int b_prime = 0; b_prime < buckets_size; ++b_prime) {
331 if (is_bucket_fixed<D>(b, b_prime)) { buckets[b].template remove_bucket_arc<D>(b, b_prime); }
332 }
333 }
334
335 // Print the total number of removed arcs.
336 if constexpr (D == Direction::Forward)
337 print_info("[Fw] removed arcs: {}\n", removed_arcs);
338 else
339 print_info("[Bw] removed arcs: {}\n", removed_arcs);
340}

354template <Direction D>
355void BucketGraph::ObtainJumpBucketArcs() {
356 // Select direction-specific containers.
357 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
358 auto &num_buckets_idx = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
359 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
360 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
361 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
362
363 int arc_counter = 0; // Count of jump arcs added
364 int missing_counter = 0; // Count of missing paths
365
366 // Clear jump arcs for all nodes.
367 for (auto &node : nodes) { node.clear_jump_arcs<D>(); }
368
369 // Process each bucket.
370 for (int b = 0; b < buckets_size; ++b) {
371 std::vector<int> B_bar; // Temporary storage for valid adjacent
372 // bucket indices
373
374 const int current_node_id = buckets[b].node_id;
375 const auto &bucket_arcs = buckets[b].template get_bucket_arcs<D>();
376 const auto &orig_arcs = nodes[current_node_id].template get_arcs<D>();
377
378 // Skip if there are no bucket arcs.
379 if (bucket_arcs.empty()) { continue; }
380
381 // For each original arc from the current node...
382 for (const auto &orig_arc : orig_arcs) {
383 const int from_node = orig_arc.from;
384 const int to_node = orig_arc.to;
385 bool have_path = false;
386
387 // Check if the current bucket arcs contain the desired
388 // path.
389 for (const auto &gamma : bucket_arcs) {
390 if (is_bucket_fixed<D>(gamma.from_bucket, gamma.to_bucket)) { continue; }
391
392 const int from_node_b = buckets[gamma.from_bucket].node_id;
393 const int to_node_b = buckets[gamma.to_bucket].node_id;
394
395 if (from_node == from_node_b && to_node == to_node_b) {
396 have_path = true;
397 break; // Path found: no need to search further.
398 }
399 }
400
401 // If no valid path was found, try to obtain a jump arc.
402 if (!have_path) {
403 ++missing_counter;
404
405 // Determine bucket range for the current node.
406 const int start_bucket = num_buckets_idx[current_node_id];
407 const int node_buckets = num_buckets[current_node_id];
408
409 // Lambda to process candidate adjacent buckets.
410 auto process_bucket = [&](int current_b, int candidate_b) {
411 // Only proceed if candidate bucket is adjacent (per
412 // Phi).
413 const auto &phi_candidate = Phi[candidate_b];
414 if (std::find(phi_candidate.begin(), phi_candidate.end(), current_b) == phi_candidate.end()) return;
415
416 // Check candidate bucket's arcs.
417 const auto &candidate_arcs = buckets[candidate_b].template get_bucket_arcs<D>();
418 for (const auto &gamma_candidate : candidate_arcs) {
419 if (is_bucket_fixed<D>(gamma_candidate.from_bucket, gamma_candidate.to_bucket)) continue;
420
421 const int from_node_candidate = buckets[gamma_candidate.from_bucket].node_id;
422 const int to_node_candidate = buckets[gamma_candidate.to_bucket].node_id;
423
424 // If the candidate arc matches the original
425 // arc...
426 if (from_node == from_node_candidate && to_node == to_node_candidate) {
427 B_bar.push_back(candidate_b);
428 std::vector<double> res = gamma_candidate.resource_increment;
429
430 const double cost = gamma_candidate.cost_increment;
431
432 // Add jump arcs both in the bucket and at
433 // the node level.
434 buckets[current_b].template add_jump_arc<D>(current_b, candidate_b, res, cost);
435 nodes[buckets[current_b].node_id].template add_jump_arc<D>(current_b, candidate_b, res,
436 cost, to_node);
437 buckets[current_b].template add_bucket_arc<D>(current_b, candidate_b, res, cost, true);
438 ++arc_counter;
439 }
440 }
441 };
442
443 // Process successor candidates within the same node block.
444 // A valid candidate must still satisfy Phi-adjacency inside
445 // process_bucket.
446 for (int candidate_b = b + 1; candidate_b < start_bucket + node_buckets; ++candidate_b) {
447 process_bucket(b, candidate_b);
448 }
449 }
450 }
451 }
452
453 // Print summary of jump arcs added.
454 CONDITIONAL(D, print_info("[Fw] added {} jump arcs\n", arc_counter),
455 print_info("[Bw] added {} jump arcs\n", arc_counter));
456}

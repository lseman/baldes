206inline int VRPTW_mcd(int m, int n) {
207 int aux;
208
209 if (m < n) {
210 aux = n;
211 n = m;
212 m = aux;
213 }
214
215 if (n <= 1) return 1;
216
217 while (n != 0) {
218 aux = m;
219 m = n;
220 n = aux % n;

420inline int CVRP_read_instance(const std::string &file_name, InstanceData &instance) {
421 std::ifstream myfile(file_name);
422 if (!myfile.is_open()) {
423 std::cerr << "Failed to open file: " << file_name << std::endl;
424 return 0;
425 }
426
427 std::string line;
428 int dimension = 0;
429 bool node_coord_section = false, demand_section = false, depot_section = false;
430
431 while (std::getline(myfile, line)) {
432 std::istringstream iss(line);
433 std::string key;
434 iss >> key;
435
436 if (key == "NAME" || key == "COMMENT" || key == "TYPE" || key == "EDGE_WEIGHT_TYPE") {
437 // Ignore these fields
438 continue;
439 } else if (key == "DIMENSION") {
440 std::string colon;
441 iss >> colon >> dimension;
442 instance.nN = dimension + 1;
443 instance.x_coord.resize(instance.nN + 1); // 1-based indexing, so +1
444 instance.y_coord.resize(instance.nN + 1);
445 instance.demand.resize(instance.nN + 1);
446 instance.service_time.resize(instance.nN + 1);
447 instance.window_open.resize(instance.nN + 1);
448 instance.window_close.resize(instance.nN + 1);
449 instance.n_tw.resize(instance.nN + 1);
450 instance.distance.resize(instance.nN + 1, std::vector<double>(instance.nN + 1));
451 } else if (key == "CAPACITY") {
452 std::string colon;
453 iss >> colon >> instance.q; // Vehicle capacity
454 } else if (key == "NODE_COORD_SECTION") {
455 node_coord_section = true;
456 continue;
457 } else if (key == "DEMAND_SECTION") {
458 node_coord_section = false;
459 demand_section = true;
460 continue;
461 } else if (key == "DEPOT_SECTION") {
462 demand_section = false;
463 depot_section = true;
464 continue;
465 } else if (key == "EOF") {
466 break; // End of file
467 }
468
469 if (node_coord_section) {
470 int node;
471 double x, y;
472 // get the node, x and y given the format 1 764 255
473 std::istringstream coord_line(line); // Reinitialize stream for each line
474 if (!(coord_line >> node >> x >> y)) { // Attempt to parse node and coordinates
475 std::cerr << "Error reading NODE_COORD_SECTION at line: " << line << std::endl;
476 continue;
477 }
478 node -= 1; // Adjust to 0-based indexing
479 if (node >= 0 && node < dimension) {
480 instance.x_coord[node] = x;
481 instance.y_coord[node] = y;
482 } else {
483 std::cerr << "Node index out of range: " << node << std::endl;
484 }
485 } else if (demand_section) {
486 int node, demand;
487 std::istringstream demand_line(line); // Reinitialize stream for each line
488 if (!(demand_line >> node >> demand)) {
489 std::cerr << "Error reading DEMAND_SECTION at line: " << line << std::endl;
490 continue;
491 }
492 node -= 1; // Adjust to 0-based indexing
493
494 if (node >= 0 && node < dimension) {
495 instance.demand[node] = demand;
496 } else {
497 std::cerr << "Node index out of range: " << node << std::endl;
498 }
499 } else if (depot_section) {
500 int depot;
501 iss >> depot;
502 if (depot == -1) break; // End of DEPOT_SECTION
503 }
504 }
505
506 // Calculate Euclidean distances between nodes
507 int max_travel_time = 0;
508
509 // Set service time and window times
510 for (int i = 0; i < instance.nN; ++i) {
511 instance.service_time[i] = 0;
512 instance.window_open[i] = 0;
513 instance.window_close[i] = max_travel_time + max_travel_time / 2.0;
514 }
515
516 // define n_tw as 0
517 for (int i = 0; i < instance.nN; i++) { instance.n_tw[i] = 0; }
518
519 instance.x_coord[instance.nN - 1] = instance.x_coord[0];
520 instance.y_coord[instance.nN - 1] = instance.y_coord[0];
521 instance.demand[instance.nN - 1] = instance.demand[0];
522 instance.service_time[instance.nN - 1] = instance.service_time[0];
523 instance.window_open[instance.nN - 1] = instance.window_open[0];
524 instance.window_close[instance.nN - 1] = instance.window_close[0];
525 instance.n_tw[instance.nN - 1] = instance.n_tw[0];
526
527 // Calculate maximum travel time for setting window close time
528 for (int i = 0; i < instance.nN; ++i) {
529 for (int j = 0; j < instance.nN; ++j) {
530 int dx = instance.x_coord[i] - instance.x_coord[j];
531 int dy = instance.y_coord[i] - instance.y_coord[j];
532 int aux = static_cast<int>(10 * std::sqrt(dx * dx + dy * dy));
533 instance.distance[i][j] = aux / 10.0;
534
535 if (aux > max_travel_time) { max_travel_time = aux; }
536 }
537 }
538
539 // Assign the travel cost
540 instance.travel_cost = instance.distance;
541
542 instance.nV = 30;
543
544 for (int i = 1; i < instance.nN - 1; ++i) { instance.demand_sum += instance.demand[i]; }
545 instance.T_max = instance.window_close[0];
546 instance.problem_type = ProblemType::cvrp;
547
548 return 1; // Success
549}

14 fw_bucket_graph.clear();

18 bw_arcs.clear();

22 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);

25 const int num_resources = options.resources.size();
26 std::vector<std::vector<double>> node_base_intervals(nodes.size());
27 std::vector<std::vector<int>> node_split_counts(nodes.size());
28
29 for (size_t node_id = 0; node_id < nodes.size(); ++node_id) {
30 const auto &node = nodes[node_id];
31 node_base_intervals[node_id].resize(num_resources);
32 node_split_counts[node_id].resize(num_resources);
33 for (int r = 0; r < num_resources; ++r) {
34 double full_range = R_max[r] - R_min[r];
35 double node_range = node.ub[r] - node.lb[r];
36 int splits = 1;
37 if (std::fabs(full_range) > std::numeric_limits<double>::epsilon()) {
38 double splits_d = (node_range * intervals[r].interval) / full_range;
39 splits = std::max(1, static_cast<int>(std::round(splits_d)));
40 }
41 node_split_counts[node_id][r] = splits;
42 node_base_intervals[node_id][r] = (node.ub[r] - node.lb[r]) / static_cast<double>(splits);
43 }
44 }

46 for (auto &bucket : buckets) {
47 bucket.clear();
48 bucket.clear_arcs(D == Direction::Forward);

51 auto try_add_arc = [&](int from_bucket, const VRPNode &next_node, const std::vector<double> &res_inc,
52 double cost_inc) -> bool {
53 bool valid = true;
54 std::vector<double> head_resource(res_inc.size(), 0.0);
55 if constexpr (D == Direction::Forward) {
56 for (int r = 0; r < res_inc.size() && valid; ++r) {
57 if (numericutils::gt(buckets[from_bucket].lb[r] + res_inc[r], next_node.ub[r])) {
58 valid = false;
59 } else {
60 head_resource[r] = std::max(buckets[from_bucket].lb[r] + res_inc[r], next_node.lb[r]);
61 }
62 }
63 } else {
64 for (int r = 0; r < res_inc.size() && valid; ++r) {
65 if (numericutils::lt(buckets[from_bucket].ub[r] - res_inc[r], next_node.lb[r])) {
66 valid = false;
67 } else {
68 head_resource[r] = std::min(buckets[from_bucket].ub[r] - res_inc[r], next_node.ub[r]);
69 }
70 }
71 }
72 if (valid) {
73 const int to_bucket = get_bucket_number<D>(next_node.id, head_resource);
74 buckets[from_bucket].template add_bucket_arc<D>(from_bucket, to_bucket, res_inc, cost_inc, false);
75 }
76 return valid;
77 };

79 auto process_node = [&](int node_id) {
80 const auto &node = nodes[node_id];
81 std::vector<double> res_inc(num_resources);
82 const auto arcs = node.get_arcs<D>();
83
84 for (int i = 0; i < num_buckets[node.id]; ++i) {
85 int from_bucket = i + num_buckets_idx[node.id];
86 for (const auto &arc : arcs) {
87 const auto &next_node = nodes[arc.to];
88 if (node.id == next_node.id) continue;
89
90 const double travel_cost = getcij(node.id, next_node.id);
91 double cost_inc = travel_cost - next_node.cost;
92 for (int r = 0; r < num_resources; r++) {
93 res_inc[r] = node.consumption[r];
94 if (options.resources[r] == "time") { res_inc[r] += travel_cost; }
95 }
96
97 try_add_arc(from_bucket, next_node, res_inc, cost_inc);
98 }
99 }
100 };

102 std::vector<int> tasks(nodes.size());
103 std::iota(tasks.begin(), tasks.end(), 0);

105 constexpr int chunk_size = 10;
106 const unsigned int thread_count = std::max(1u, std::thread::hardware_concurrency() / 2);
107 exec::static_thread_pool pool(thread_count);
108 auto sched = pool.get_scheduler();
109
110 auto bulk_sender = stdexec::bulk(stdexec::just(), (tasks.size() + chunk_size - 1) / chunk_size,
111 [&, chunk_size](std::size_t chunk_idx) {
112 const size_t start_idx = chunk_idx * chunk_size;
113 const size_t end_idx = std::min(start_idx + chunk_size, tasks.size());
114 for (size_t idx = start_idx; idx < end_idx; ++idx) {
115 process_node(tasks[idx]);
116 }
117 });
118
119 auto work = stdexec::starts_on(sched, bulk_sender);
120 stdexec::sync_wait(std::move(work));

122 auto &bucket_graph = assign_buckets<D>(fw_bucket_graph, bw_bucket_graph);
123 auto &global_arcs = assign_buckets<D>(fw_arcs, bw_arcs);
124 bucket_graph.assign(buckets.size(), {});
125 global_arcs.clear();
126
127 size_t arc_count = 0;
128 for (int b = 0; b < static_cast<int>(buckets.size()); ++b) {
129 const auto &bucket_arcs = buckets[b].template get_bucket_arcs<D>();
130 arc_count += bucket_arcs.size();
131 }
132 global_arcs.reserve(arc_count);
133
134 for (int b = 0; b < static_cast<int>(buckets.size()); ++b) {
135 auto &neighbors = bucket_graph[b];
136 const auto &bucket_arcs = buckets[b].template get_bucket_arcs<D>();
137 neighbors.reserve(bucket_arcs.size());
138 for (const auto &arc : bucket_arcs) {
139 neighbors.push_back(arc.to_bucket);
140 global_arcs.push_back(arc);
141 }
142 }

145template <Direction D>
146Label *BucketGraph::get_best_label(const std::vector<int> &, const std::vector<double> &,
147 const std::vector<std::vector<int>> &) {
148 double best_cost = std::numeric_limits<double>::infinity();
149 Label *best_label = nullptr;
150 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
151
152 for (auto &bucket : buckets) {
153 Label *label = bucket.get_best_label();
154 if (!label) continue;
155 if (label->cost < best_cost) {
156 best_cost = label->cost;
157 best_label = label;
158 }
159 }
160 return best_label;
161}

163template <Direction D>
164void BucketGraph::SCC_handler() {
165 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
166 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
167 auto &bucket_graph = assign_buckets<D>(fw_bucket_graph, bw_bucket_graph);
168
169 std::vector<std::vector<int>> extended_bucket_graph = bucket_graph;
170 if (extended_bucket_graph.size() < buckets.size()) { extended_bucket_graph.resize(buckets.size()); }
171 for (size_t i = 0; i < buckets.size(); ++i) {
172 const auto &phi_set = Phi[i];
173 if (phi_set.empty()) continue;
174 for (auto phi_bucket : phi_set) {
175 if (phi_bucket >= 0 && static_cast<size_t>(phi_bucket) < extended_bucket_graph.size()) {
176 extended_bucket_graph[static_cast<size_t>(phi_bucket)].push_back(static_cast<int>(i));
177 }
178 }
179 }
180
181 SCC scc_finder;
182 scc_finder.convertFromAdjacency(extended_bucket_graph);
183 auto sccs = scc_finder.tarjanSCC();
184 auto topological_order = scc_finder.topologicalOrderOfSCCs(sccs);
185
186#ifdef VERBOSE
187 constexpr auto blue = "\033[34m";
188 constexpr auto reset = "\033[0m";
189 fmt::print((D == Direction::Forward) ? "FW SCCs:\n" : "BW SCCs:\n");
190 for (auto scc : topological_order) {
191 fmt::print("{}({}) -> {}", blue, scc, reset);
192 for (auto bucket : sccs[scc]) { fmt::print("{} ", bucket); }
193 fmt::print("\n");
194 }
195 fmt::print("\n");
196#endif
197
198 std::vector<std::vector<int>> ordered_sccs;
199 ordered_sccs.reserve(sccs.size());
200 for (int scc : topological_order) { ordered_sccs.push_back(sccs[scc]); }
201
202 auto sorted_sccs = sccs;
203 for (auto &scc : sorted_sccs) {
204 if constexpr (D == Direction::Forward) {
205 pdqsort(scc.begin(), scc.end(), [&buckets](int a, int b) { return buckets[a].lb < buckets[b].lb; });
206 } else {
207 pdqsort(scc.begin(), scc.end(), [&buckets](int a, int b) { return buckets[a].ub > buckets[b].ub; });
208 }
209 }
210
211 for (auto &node : nodes) {
212 if constexpr (D == Direction::Forward) {
213 node.fw_arcs_scc.resize(sccs.size());
214 } else {
215 node.bw_arcs_scc.resize(sccs.size());
216 }
217 }
218
219 auto process_bucket_arcs = [&](int bucket, int scc_index) {
220 int from_node_id = buckets[bucket].node_id;
221 VRPNode &node = nodes[from_node_id];
222 const auto &node_arcs = (D == Direction::Forward) ? node.fw_arcs : node.bw_arcs;
223 std::vector<Arc> &filtered_arcs =
224 (D == Direction::Forward) ? node.fw_arcs_scc[scc_index] : node.bw_arcs_scc[scc_index];
225
226 const auto &bucket_arcs = buckets[bucket].template get_bucket_arcs<D>();
227 for (const auto &arc : bucket_arcs) {
228 int to_node_id = buckets[arc.to_bucket].node_id;
229 auto it = std::find_if(node_arcs.begin(), node_arcs.end(),
230 [&to_node_id](const Arc &a) { return a.to == to_node_id; });
231 if (it != node_arcs.end()) { filtered_arcs.push_back(*it); }
232 }
233 };
234
235 int scc_ctr = 0;
236 for (const auto &scc : sccs) {
237 for (int bucket : scc) { process_bucket_arcs(bucket, scc_ctr); }
238 ++scc_ctr;
239 }
240
241 auto sortAndDedup = [](std::vector<Arc> &arcs) {
242 pdqsort(arcs.begin(), arcs.end(),
243 [](const Arc &a, const Arc &b) { return std::tie(a.from, a.to) < std::tie(b.from, b.to); });
244 auto last = std::unique(arcs.begin(), arcs.end(),
245 [](const Arc &a, const Arc &b) { return a.from == b.from && a.to == b.to; });
246 arcs.erase(last, arcs.end());
247 };
248
249 for (auto &node : nodes) {
250 if constexpr (D == Direction::Forward) {
251 for (auto &arcs : node.fw_arcs_scc) { sortAndDedup(arcs); }
252 } else {
253 for (auto &arcs : node.bw_arcs_scc) { sortAndDedup(arcs); }
254 }
255 }
256
257 std::vector<int> bucket_scc_rank(buckets.size(), -1);
258 for (size_t rank = 0; rank < ordered_sccs.size(); ++rank) {
259 for (int bucket : ordered_sccs[rank]) { bucket_scc_rank[bucket] = static_cast<int>(rank); }
260 }
261
262 UnionFind uf(ordered_sccs);
263 if constexpr (D == Direction::Forward) {
264 fw_ordered_sccs = ordered_sccs;
265 fw_topological_order = topological_order;
266 fw_sccs = sccs;
267 fw_sccs_sorted = sorted_sccs;
268 fw_bucket_scc_rank = std::move(bucket_scc_rank);
269 fw_union_find = uf;
270 } else {
271 bw_ordered_sccs = ordered_sccs;
272 bw_topological_order = topological_order;
273 bw_sccs = sccs;
274 bw_sccs_sorted = sorted_sccs;
275 bw_bucket_scc_rank = std::move(bucket_scc_rank);
276 bw_union_find = uf;
277 }

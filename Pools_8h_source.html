83 // Stop processing if the path is older than current_iteration + max_life

107 std::vector<Path> result;
108
109 // Remove the paths that are older than current_iteration + max_life or have a negative red_cost
110 auto it = paths.begin();
111 while (it != paths.end()) {
112 int iteration_added = std::get<0>(*it); // Get the iteration when the path was added
113 const Path &p = std::get<1>(*it);
114
115 // If the path is older than max_live_time, or has a negative red_cost
116 if (iteration_added + max_live_time < current_iteration || p.red_cost < 0) {
117 if (p.red_cost < 0) {
118 result.push_back(p); // Add paths with negative red_cost to the result
119 }
120 it = paths.erase(it); // Remove from paths and move iterator to the next element
121 } else {
122 ++it; // Only move the iterator if not erasing
123 }
124 }
125
126 // Sort the result based on red_cost

182 size_t block_index;
183 size_t offset;
184 };

189 void allocate_labels(size_t count) {
190 size_t blocks_needed = (count + MemoryBlock::BLOCK_SIZE - 1) / MemoryBlock::BLOCK_SIZE;
191 size_t current_block = memory_blocks.size();
192
193 available_labels.reserve(available_labels.size() + count);
194
195 for (size_t b = 0; b < blocks_needed; ++b) {
196 auto block = std::make_unique<MemoryBlock>();
197 size_t block_alloc = std::min(count - b * MemoryBlock::BLOCK_SIZE, MemoryBlock::BLOCK_SIZE);
198
199 for (size_t i = 0; i < block_alloc; ++i) {
200 available_labels.push_back({current_block, i});
201 }
202
203 block->used = block_alloc;
204 memory_blocks.push_back(std::move(block));
205 current_block++;
206 }
207 }
208
209 Label* get_label(const LabelRef& ref) {
210 assert(ref.block_index < memory_blocks.size());
211 assert(ref.offset < memory_blocks[ref.block_index]->used);
212 return &memory_blocks[ref.block_index]->data[ref.offset];
213 }
214
215public:
216 explicit LabelPool(size_t initial_pool_size, size_t max_pool_size = 5000000)
217 : pool_size(initial_pool_size), max_pool_size(max_pool_size) {
218 memory_blocks.reserve(initial_pool_size / MemoryBlock::BLOCK_SIZE + 1);
219 available_labels.reserve(initial_pool_size);
220 in_use_labels.reserve(initial_pool_size);
221 allocate_labels(pool_size);
222 }

224 Label* acquire() {
225 if (!available_labels.empty()) {
226 LabelRef ref = available_labels.back();
227 available_labels.pop_back();
228
229 Label* label = get_label(ref);
230 label->reset();
231
232 in_use_labels.push_back(ref);
233 return label;
234 }
235
236 // Need to allocate a new block
237 if (memory_blocks.empty() ||
238 memory_blocks.back()->used >= MemoryBlock::BLOCK_SIZE) {
239 size_t new_block_idx = memory_blocks.size();
240 auto block = std::make_unique<MemoryBlock>();
241 LabelRef ref = {new_block_idx, 0};
242 block->used = 1;
243
244 Label* label = block->data.get();
245 memory_blocks.push_back(std::move(block));
246 in_use_labels.push_back(ref);
247 return label;
248 }
249
250 // Use existing block
251 auto& block = memory_blocks.back();
252 LabelRef ref = {memory_blocks.size() - 1, block->used};
253 block->used++;
254
255 Label* label = get_label(ref);
256 in_use_labels.push_back(ref);
257 return label;
258 }

262 available_labels.reserve(available_labels.size() + in_use_labels.size());
263

268 std::make_move_iterator(in_use_labels.end())
269 );
270
271 in_use_labels.clear();
272 }

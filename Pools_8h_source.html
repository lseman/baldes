109 std::vector<Path> result;

111 // Remove the paths that are older than current_iteration + max_life or
112 // have a negative red_cost
113 auto it = paths.begin();
114 while (it != paths.end()) {
115 int iteration_added = std::get<0>(*it); // Get the iteration when the path was added
116 const Path &p = std::get<1>(*it);
117
118 // If the path is older than max_live_time, or has a negative
119 // red_cost
120 if (iteration_added + max_live_time < current_iteration || p.red_cost < 0) {
121 if (p.red_cost < 0) {

123 }
124 it = paths.erase(it); // Remove from paths and move iterator to
125 // the next element
126 } else {
127 ++it; // Only move the iterator if not erasing
128 }
129 }
130
131 // Sort the result based on red_cost
132 pdqsort(result.begin(), result.end(), [](const Path &a, const Path &b) { return a.red_cost < b.red_cost; });

159 // Align MemoryBlock to cache line for better performance
160 struct alignas(CACHE_LINE_SIZE) MemoryBlock {
161 std::unique_ptr<Label[]> data;
162 size_t used; // Number of labels allocated from this block
163
164 MemoryBlock() : data(std::make_unique<Label[]>(BLOCK_SIZE)), used(0) {}
165
166 // Disable copy, enable move
167 MemoryBlock(const MemoryBlock &) = delete;
168 MemoryBlock &operator=(const MemoryBlock &) = delete;
169 MemoryBlock(MemoryBlock &&) noexcept = default;
170 MemoryBlock &operator=(MemoryBlock &&) noexcept = default;
171 };

173 size_t pool_size;
174 size_t max_pool_size;

176 std::vector<MemoryBlock> memory_blocks;

178 // free_list holds pointers to labels that are available for reuse
179 // Use raw array with index-based management for better cache locality
180 std::vector<Label *> free_list;
181 size_t free_list_size; // Track size separately to avoid .size() calls
182
183 // Track in-use labels only when needed (removed for performance)
184 // Most use cases don't need this tracking
185
186 // Current block index for fast allocation
187 size_t current_block_idx;
188
189 // Allocate 'count' new labels and add them to the free list
190 void allocate_labels(size_t count) {
191 const size_t blocks_needed = (count + BLOCK_SIZE - 1) / BLOCK_SIZE;
192
193 // Reserve capacity once to avoid multiple reallocations
194 memory_blocks.reserve(memory_blocks.size() + blocks_needed);
195 free_list.reserve(free_list.size() + count);
196
197 for (size_t b = 0; b < blocks_needed; ++b) {
198 memory_blocks.emplace_back();
199 auto &block = memory_blocks.back();
200
201 // Calculate labels to allocate in this block
202 const size_t remaining_count = count - b * BLOCK_SIZE;
203 const size_t alloc_in_block = std::min(remaining_count, BLOCK_SIZE);
204
205 // Batch add pointers to free list
206 Label *base_ptr = block.data.get();
207 for (size_t i = 0; i < alloc_in_block; ++i) { free_list.push_back(base_ptr + i); }
208
209 block.used = alloc_in_block;
210 free_list_size += alloc_in_block;
211 }
212
213 current_block_idx = memory_blocks.size() - 1;
214 }

216public:
217 explicit LabelPool(size_t initial_pool_size, size_t max_pool_size = 5000000)
218 : pool_size(initial_pool_size), max_pool_size(max_pool_size), free_list_size(0), current_block_idx(0) {
219
220 // Pre-allocate with some headroom to reduce reallocations
221 const size_t reserve_size = initial_pool_size + (initial_pool_size >> 2); // +25%
222 memory_blocks.reserve((reserve_size + BLOCK_SIZE - 1) / BLOCK_SIZE);
223 free_list.reserve(reserve_size);
224
225 allocate_labels(pool_size);

228 // Fast acquire with minimal branching
229 [[nodiscard]] inline Label *acquire() noexcept {
230 Label *label;
231
232 // Fast path: reuse from free list
233 if (__builtin_expect(free_list_size > 0, 1)) {
234 --free_list_size;
235 label = free_list[free_list_size];
236 label->reset();
237 return label;
238 }
239
240 // Slow path: allocate from current block or create new block
241 auto &current_block = memory_blocks[current_block_idx];
242
243 if (__builtin_expect(current_block.used < BLOCK_SIZE, 1)) {
244 // Use next available label from current block
245 label = &current_block.data[current_block.used];
246 ++current_block.used;
247 label->reset();
248 return label;
249 }
250
251 // Reuse an existing block after fast_reset() before growing.
252 while (current_block_idx + 1 < memory_blocks.size()) {
253 ++current_block_idx;
254 auto &next_block = memory_blocks[current_block_idx];
255 if (next_block.used < BLOCK_SIZE) {
256 label = &next_block.data[next_block.used];
257 ++next_block.used;
258 label->reset();
259 return label;
260 }
261 }
262
263 // Need a new block
264 memory_blocks.emplace_back();
265 current_block_idx = memory_blocks.size() - 1;
266 auto &new_block = memory_blocks[current_block_idx];
267 label = &new_block.data[0];
268 new_block.used = 1;
269 label->reset();
270 return label;
271 }

273 // Batch acquire for better performance when multiple labels needed
274 inline void acquire_batch(Label **out_labels, size_t count) noexcept {
275 size_t acquired = 0;
276
277 // First, acquire from free list
278 if (free_list_size > 0) {
279 const size_t from_free = std::min(count, free_list_size);
280
281 // Copy pointers in batch
282 for (size_t i = 0; i < from_free; ++i) {
283 --free_list_size;
284 Label *label = free_list[free_list_size];
285 label->reset();
286 out_labels[i] = label;
287 }
288
289 acquired = from_free;
290 }
291
292 // Then allocate remaining from blocks
293 while (acquired < count) {
294 auto &current_block = memory_blocks[current_block_idx];
295
296 if (current_block.used < BLOCK_SIZE) {
297 const size_t available = BLOCK_SIZE - current_block.used;
298 const size_t to_allocate = std::min(count - acquired, available);
299
300 // Batch allocate from current block
301 Label *base_ptr = &current_block.data[current_block.used];
302 for (size_t i = 0; i < to_allocate; ++i) {
303 (base_ptr + i)->reset();
304 out_labels[acquired + i] = base_ptr + i;
305 }
306
307 current_block.used += to_allocate;
308 acquired += to_allocate;
309 } else {
310 // Reuse an existing block after fast_reset() before growing.
311 if (current_block_idx + 1 < memory_blocks.size()) {
312 ++current_block_idx;
313 } else {
314 memory_blocks.emplace_back();
315 current_block_idx = memory_blocks.size() - 1;
316 }
317 }
318 }

344 // Reset the pool - optimized version
345 inline void reset() noexcept {
346 size_t total_used = 0;
347 for (const auto &block : memory_blocks) { total_used += block.used; }
348
349 if (free_list.size() < total_used) { free_list.resize(total_used); }
350
351 free_list_size = 0;
352 for (auto &block : memory_blocks) {
353 Label *base_ptr = block.data.get();
354 for (size_t i = 0; i < block.used; ++i) { free_list[free_list_size++] = base_ptr + i; }
355 }

358 }

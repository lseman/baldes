76 void computeRC() {
77 for (auto &path : paths) {
78 int iteration_added = std::get<0>(path); // Get the iteration when the path was added
79
80 // Stop processing if the path is older than current_iteration +
81 // max_life
82 if (iteration_added + max_live_time < current_iteration) { break; }
83
84 Path &p = std::get<1>(path);
85 p.red_cost = p.cost;
86
87 if (p.size() > 3) {
88 for (int i = 1; i < p.size() - 1; i++) {
89 auto &node = (*nodes)[p[i]]; // Dereference nodes and access element

104 std::vector<Path> _get_paths_with_negative_red_cost() {
105 std::vector<Path> result;
106
107 // Remove the paths that are older than current_iteration + max_life or
108 // have a negative red_cost
109 auto it = paths.begin();
110 while (it != paths.end()) {
111 int iteration_added = std::get<0>(*it); // Get the iteration when the path was added
112 const Path &p = std::get<1>(*it);
113
114 // If the path is older than max_live_time, or has a negative
115 // red_cost
116 if (iteration_added + max_live_time < current_iteration || p.red_cost < 0) {
117 if (p.red_cost < 0) {
118 result.push_back(p); // Add paths with negative red_cost to the result
119 }
120 it = paths.erase(it); // Remove from paths and move iterator to
121 // the next element
122 } else {
123 ++it; // Only move the iterator if not erasing
124 }
125 }
126
127 // Sort the result based on red_cost
128 pdqsort(result.begin(), result.end(), [](const Path &a, const Path &b) { return a.red_cost < b.red_cost; });
129
130 return result;
131 }

152 static constexpr size_t labels_per_block = 64;
153 static constexpr size_t block_alignment = 64;

155 struct BucketArena {
156 std::vector<Label *> blocks;
157 size_t next = 0;
158 };
159
160 size_t max_pool_size{};
161 size_t labels_in_use{};
162 size_t allocated_labels{};
163 std::vector<BucketArena> bucket_arenas;
164 std::vector<Label *> recycled_labels;
165
166 BucketArena &arena_for(int bucket_id) {
167 const size_t index = bucket_id < 0 ? 0 : static_cast<size_t>(bucket_id) + 1;
168 if (index >= bucket_arenas.size()) bucket_arenas.resize(index + 1);
169 return bucket_arenas[index];
170 }
171
172 Label *allocate_block(BucketArena &arena) {
173 if (allocated_labels + labels_per_block > max_pool_size) std::abort();
174 auto *block = static_cast<Label *>(
175 ::operator new[](labels_per_block * sizeof(Label), std::align_val_t{block_alignment}));
176 for (size_t i = 0; i < labels_per_block; ++i) std::construct_at(block + i);
177 arena.blocks.push_back(block);
178 allocated_labels += labels_per_block;
179 return block;
180 }
181
182public:
183 explicit LabelPool(size_t initial_pool_size, size_t max_pool_size = 5000000)
184 : max_pool_size(std::max(initial_pool_size, max_pool_size)), bucket_arenas(1) {}
185
186 // Destination-aware allocation keeps labels of a bucket in contiguous
187 // cache-line-aligned blocks. bucket_id < 0 uses the fallback arena.
188 [[nodiscard]] inline Label *acquire(int bucket_id = -1) noexcept {
189 if (!recycled_labels.empty()) {
190 Label *label = recycled_labels.back();
191 recycled_labels.pop_back();
192 label->reset();
193 ++labels_in_use;
194 return label;
195 }
196
197 if (__builtin_expect(labels_in_use >= max_pool_size, 0)) std::abort();
198 BucketArena &arena = arena_for(bucket_id);
199 const size_t block_index = arena.next / labels_per_block;
200 const size_t slot = arena.next % labels_per_block;
201 if (block_index == arena.blocks.size()) allocate_block(arena);
202 Label *label = arena.blocks[block_index] + slot;
203 ++arena.next;
204 ++labels_in_use;
205 label->reset();
206 return label;
207 }
208
209 inline void acquire_batch(Label **out_labels, size_t count, int bucket_id = -1) noexcept {
210 for (size_t i = 0; i < count; ++i) out_labels[i] = acquire(bucket_id);
211 }
212
213 inline void release(Label *label) noexcept {
214 if (!label) return;

222 }

238 // Cleanup all allocated memory
239 void cleanup() noexcept {
240 for (auto &arena : bucket_arenas) {
241 for (Label *block : arena.blocks) {
242 for (size_t i = 0; i < labels_per_block; ++i) std::destroy_at(block + i);

244 }
245 }
246 bucket_arenas.clear();
247 recycled_labels.clear();

249 labels_in_use = 0;
250 }

13void BranchingDuals::computeDuals(BNBNode *model, double threshold) {
14 arcDuals_ = ArcDuals();
15 nodeDuals_ = NodeDuals();
16
17 const size_t size = std::min(branchingCandidates_.size(), branchingbaldesCtrs_.size());
18
19 // Compute dual contributions once per active branching constraint.
20 for (size_t i = 0; i < size; ++i) {
21 auto candidate = branchingCandidates_[i];
22 auto ctr = branchingbaldesCtrs_[i];
23
24 double dualValue = model->getDualVal(ctr->index());
25
26 // TODO: check sign direction
27 if (candidate->boundType == BranchingDirection::Less) { dualValue = -dualValue; }
28
29 if (std::abs(dualValue) < threshold) { continue; }
30
31 if (candidate->getCandidateType() == CandidateType::Edge) {
32 RawArc arc(candidate->sourceNode, candidate->targetNode);
33 arcDuals_.setOrIncrementDual(arc, dualValue);
34 } else if (candidate->getCandidateType() == CandidateType::Node) {
35 nodeDuals_.setOrIncrementDual(candidate->targetNode, dualValue);
36 }
37 }
38}

41Branching::evaluateWithCG(BNBNode *node, const std::vector<BranchingQueueItem> &phase1Candidates, VRProblem *problem) {
42 std::vector<BranchingQueueItem> results;
43 std::mutex results_mutex; // Mutex to protect shared results
44
45 const unsigned JOBS = std::max(1u, std::thread::hardware_concurrency());
46 exec::static_thread_pool pool(JOBS); // Pool with concurrent threads
47 auto sched = pool.get_scheduler();

49 // Define chunk size based on performance tuning (adjust as needed)
50 const int chunk_size = 10;

52
53 // Parallel bulk execution
54 auto bulk_sender = stdexec::bulk(
55 stdexec::just(), total_chunks,
56 [&results, &results_mutex, node, &phase1Candidates, problem, chunk_size](std::size_t chunk_idx) {
57 size_t start_idx = chunk_idx * chunk_size;
58 size_t end_idx = std::min(start_idx + chunk_size, phase1Candidates.size());
59
60 // Clone the problem for this thread
61 std::unique_ptr<VRProblem> problem_copy(problem->clone()); // Assuming clone method is available
62
63 std::vector<BranchingQueueItem> local_results; // Local storage to avoid locking too often
64
65 // Process the chunk of candidates
66 for (size_t idx = start_idx; idx < end_idx; ++idx) {
67 const auto &candidate = phase1Candidates[idx];
68
69 // Add branching constraints and create two child nodes
70 auto [childNode1, childNode2] =
71 applyBranchingbaldesCtrs(node, candidate, candidate.fractionalValue, candidate.candidateType);
72
73 // Solve CG and bound for each child node using the cloned problem
74 double deltaLB1, deltaLB2;
75 double feasLB1, feasLB2;
76 feasLB1 = problem_copy->heuristicCG(childNode1, 50); // Use the copy
77 deltaLB1 = problem_copy->bound(childNode1); // Use the copy
78 feasLB2 = problem_copy->heuristicCG(childNode2, 50); // Use the copy
79 deltaLB2 = problem_copy->bound(childNode2); // Use the copy
80
81 // Calculate product value for branching
82 double productValue = deltaLB1 * deltaLB2;
83
84 // Store the result locally
85 local_results.push_back({candidate.sourceNode, candidate.targetNode, candidate.fractionalValue,
86 productValue, candidate.g_m, candidate.g_m_v, candidate.g_v_vp,
87 candidate.candidateType, std::make_pair(feasLB1, feasLB2)});
88 }
89
90 // Add local results to shared results with mutex protection
91 {
92 std::lock_guard<std::mutex> lock(results_mutex);
93 results.insert(results.end(), local_results.begin(), local_results.end());
94 }
95 });
96
97 // Execute bulk task using stdexec
98 stdexec::sync_wait(stdexec::when_all(bulk_sender));
99
100 // Sort results by product value in descending order (highest product value first)
101 pdqsort(results.begin(), results.end(),
102 [](const BranchingQueueItem &a, const BranchingQueueItem &b) { return a.productValue > b.productValue; });
103
104 return results;

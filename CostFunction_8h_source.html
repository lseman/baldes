43class GenericParameters {
44private:
45 std::vector<double> doubles;
46 std::vector<int> ints;
47
48public:
49 template <typename T>
50 void set(const ParamType<T> &p) {
51 if constexpr (std::is_same_v<T, double>) {
52 if (doubles.size() <= p.index) doubles.resize(p.index + 1);
53 doubles[p.index] = p.value;
54 } else if constexpr (std::is_same_v<T, int>) {
55 if (ints.size() <= p.index) ints.resize(p.index + 1);
56 ints[p.index] = p.value;
57 }
58 }
59
60 template <typename T>
61 T get(size_t index) const {
62 if constexpr (std::is_same_v<T, double>) {
63 return index < doubles.size() ? doubles[index] : T{};
64 } else if constexpr (std::is_same_v<T, int>) {
65 return index < ints.size() ? ints[index] : T{};
66 }
67 return T{};
68 }

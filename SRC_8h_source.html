88 // assign 0 from last_path_idx to allPaths.size()
89 for (size_t i = 0; i < N_SIZE; ++i) {
90 vertex_route_map[i].resize(allPaths.size());
91 for (size_t j = last_path_idx; j < allPaths.size(); ++j) {

103 ++vertex_route_map[vertex][r];
104 }

182 void separateR1C1(const SparseMatrix &A, const std::vector<double> &x) {
183 // === Step 1: Extract Potential Cuts from allPaths ===
184 const size_t nPaths = allPaths.size();
185 const int JOBS = std::thread::hardware_concurrency();
186 const int chunk_size = (nPaths + JOBS - 1) / JOBS;
187 const size_t num_chunks = (nPaths + chunk_size - 1) / chunk_size;
188
189 // Reserve per-chunk storage to avoid locking.
190 std::vector<std::vector<std::pair<double, int>>> chunk_cuts(num_chunks);
191 for (auto &vec : chunk_cuts) {
192 vec.reserve(10); // Reserve an estimate; adjust as needed.
193 }
194
195 // Parallel execution in chunks over allPaths.
196 auto bulk_sender = stdexec::bulk(
197 stdexec::just(), num_chunks, [&](std::size_t chunk_idx) {
198 size_t start_idx = chunk_idx * chunk_size;
199 size_t end_idx = std::min(start_idx + chunk_size, nPaths);
200
201 // Local vector for storing cuts computed in this chunk.
202 std::vector<std::pair<double, int>> local_cuts;
203 local_cuts.reserve(chunk_size);
204 // Temporary map to count occurrences of nodes in a given path.
205
206 // Process each path in this chunk.
207 for (size_t i = start_idx; i < end_idx; ++i) {
208 for (auto node : allPaths[i].route) {
209 if (vertex_route_map[node][i] >= 2) {
210 // Calculate the lhs value for the current path.
211 double cut_value = x[i];
212 if (cut_value > 1e-3) { // Tolerance check.
213 local_cuts.emplace_back(cut_value, node);
214 }
215 }
216 }
217 }
218
219 // Save the local cuts into the per-chunk vector.
220 chunk_cuts[chunk_idx] = std::move(local_cuts);
221 });
222
223 // Submit the bulk work and wait for all tasks to complete.
224 auto work = stdexec::starts_on(sched, bulk_sender);
225 stdexec::sync_wait(std::move(work));
226
227 // Merge results from all chunks.
228 std::vector<std::pair<double, int>> tmp_cuts;
229 for (const auto &vec : chunk_cuts) {
230 tmp_cuts.insert(tmp_cuts.end(), vec.begin(), vec.end());
231 }
232
233 // Early return if no cuts were found.
234 if (tmp_cuts.empty()) return;
235
236 // === Step 2: Sort Extracted Cuts by Their Cut Value in Descending
237 // Order ===
238 pdqsort(
239 tmp_cuts.begin(), tmp_cuts.end(),
240 [](const std::pair<double, int> &a,
241 const std::pair<double, int> &b) { return a.first > b.first; });
242
243 // === Step 3: Generate and Add the Best Cut ===
244 // Only generate coefficients for the top cut.
245 const int cuts_to_apply =
246 std::min(static_cast<int>(tmp_cuts.size()), 3);
247 std::vector<double> coefficients_aux(allPaths.size(), 0.0);
248
249 // Setup a simple SRCPermutation for the cut.
250 SRCPermutation p;
251 p.num = {1};
252 p.den = 2;
253
254 for (int i = 0; i < cuts_to_apply; ++i) {
255 double cut_value = tmp_cuts[i].first;
256 int node = tmp_cuts[i].second;
257
258 // Initialize bitmask arrays for representing the cut.
259 std::array<uint64_t, num_words> C = {}; // Base set bitmask.
260 std::array<uint64_t, num_words> AM = {}; // Augmented mask.
261 std::vector<int> order(N_SIZE, 0);
262
263 // Define C: mark only the selected node.
264 C[node / 64] |= (1ULL << (node % 64));
265
266 // Define AM: for this simple cut, mark all nodes.
267 for (int node_idx = 0; node_idx <= N_SIZE; ++node_idx) {
268 AM[node_idx / 64] |= (1ULL << (node_idx % 64));
269 }
270
271 // Compute coefficients for each path based on the current cut.
272 coefficients_aux.assign(allPaths.size(), 0.0);
273 for (size_t j = 0; j < allPaths.size(); ++j) {
274 const auto &clients = allPaths[j].route;
275 coefficients_aux[j] =
276 computeLimitedMemoryCoefficient(C, AM, p, clients, order);
277 }
278
279 // Create a new cut using the computed coefficients.
280 Cut cut(C, AM, coefficients_aux, p);
281 cut.baseSetOrder = order;
282 cut.type = CutType::OneRow;
283
284 // Add the cut to the cut storage.
285 cutStorage.addCut(cut);
286 }
287 }

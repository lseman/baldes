73 void initializeVertexRouteMap() {
74 if (last_path_idx == 0) {
75 row_indices_map.clear();
76 row_indices_map.reserve(N_SIZE);
77 map_rank1_multiplier_dominance.assign(N_SIZE, {});
78 }
79
80 // Initialize the vertex_route_map with N_SIZE rows and allPaths.size()
81 // columns, all set to 0.
82 if (last_path_idx == 0) {
83 vertex_route_map.assign(N_SIZE, std::vector<int>(allPaths.size(), 0));
84 } else {
85 // assign 0 from last_path_idx to allPaths.size()
86 for (size_t i = 0; i < N_SIZE; ++i) {
87 vertex_route_map[i].resize(allPaths.size());
88 for (size_t j = last_path_idx; j < allPaths.size(); ++j) { vertex_route_map[i][j] = 0; }
89 }
90 }
91
92 // Populate the map: for each path, count the appearance of each vertex.
93 std::vector<unsigned char> seen_in_path(N_SIZE, 0);
94 std::vector<int> touched_vertices;
95 for (size_t r = last_path_idx; r < allPaths.size(); ++r) {
96 for (const auto &vertex : allPaths[r].route) {
97 // Only consider vertices that are not the depot (assumed at
98 // indices 0 and N_SIZE-1).
99 if (vertex > 0 && vertex < N_SIZE - 1) {
100 ++vertex_route_map[vertex][r];
101 if (!seen_in_path[vertex]) {
102 seen_in_path[vertex] = 1;
103 touched_vertices.push_back(vertex);
104 row_indices_map[vertex].push_back(static_cast<int>(r));
105 }
106 }
107 }
108 for (int vertex : touched_vertices) { seen_in_path[vertex] = 0; }
109 touched_vertices.clear();
110 }
111 last_path_idx = allPaths.size();
112 }

114 void initializeRank1HeuristicNeighbors() {
115 rank1_sep_heur_mem4_vertex.clear();
116 rank1_sep_heur_mem4_vertex.resize(N_SIZE);
117 if (high_rank_cuts.distances.size() != static_cast<size_t>(N_SIZE)) return;
118
119 for (int i = 1; i < N_SIZE - 1; ++i) {
120 std::vector<std::pair<double, int>> cost;
121 cost.reserve(N_SIZE - 2);
122 for (int j = 1; j < N_SIZE - 1; ++j) {
123 if (i == j) continue;
124 cost.emplace_back(high_rank_cuts.distances[i][j], j);
125 }
126 std::stable_sort(cost.begin(), cost.end(), [](const auto &a, const auto &b) { return a.first < b.first; });
127 int limit = std::min(MAX_HEURISTIC_SEP_ROW_RANK1, static_cast<int>(cost.size()));
128 for (int k = 0; k < limit; ++k) { rank1_sep_heur_mem4_vertex[i].push_back(cost[k].second); }
129 }
130 }

194 void separateR1C1(const SparseMatrix &A, const std::vector<double> &x) {
195 if (allPaths.empty()) return;
196
197 initializeRank1HeuristicNeighbors();
198
199 struct CandidateVertex {
200 double violation;
201 int node;
202 };
203
204 std::vector<CandidateVertex> tmp_cuts;
205 tmp_cuts.reserve(N_SIZE);
206 std::vector<double> vertex_violation(N_SIZE, 0.0);
207
208 for (int node = 1; node < N_SIZE - 1; ++node) {
209 if (row_indices_map[node].empty()) continue;
210
211 double violation = 0.0;
212 for (int path_idx : row_indices_map[node]) {
213 int visit_count = vertex_route_map[node][path_idx];
214 if (visit_count >= 2) {
215 int pair_count = visit_count / 2;
216 violation += pair_count * x[path_idx];
217 }
218 }
219 vertex_violation[node] = violation;
220 if (violation > 1e-3) { tmp_cuts.push_back({violation, node}); }
221 }
222
223 if (tmp_cuts.empty()) return;
224
225 // === Step 2: Sort vertices by total cut violation in descending order ===
226 pdqsort(tmp_cuts.begin(), tmp_cuts.end(),
227 [](const CandidateVertex &a, const CandidateVertex &b) { return a.violation > b.violation; });
228
229 std::vector<std::pair<double, int>> ordered_cuts;
230 ordered_cuts.reserve(tmp_cuts.size());
231 std::vector<char> seen(N_SIZE, 0);
232
233 for (const auto &candidate : tmp_cuts) {
234 if (static_cast<int>(ordered_cuts.size()) >= 3) break;
235 if (seen[candidate.node]) continue;
236 ordered_cuts.emplace_back(candidate.violation, candidate.node);
237 seen[candidate.node] = 1;
238 }
239 pdqsort(ordered_cuts.begin(), ordered_cuts.end(),
240 [](const std::pair<double, int> &a, const std::pair<double, int> &b) { return a.first > b.first; });
241 const int cuts_to_apply = std::min(static_cast<int>(ordered_cuts.size()), 3);
242 SRCPermutation p;
243 p.num = {1};
244 p.den = 2;
245
246 for (int i = 0; i < cuts_to_apply; ++i) {
247 int node = ordered_cuts[i].second;
248
249 std::array<uint64_t, num_words> C = {};
250 std::array<uint64_t, num_words> AM = {};
251 std::vector<int> order(N_SIZE, -1);
252
253 C[node / 64] |= (1ULL << (node % 64));
254 order[node] = 0;
255 const auto cut_key = cutStorage.compute_cut_key(C, p.num, p.den);
256 if (cutStorage.cutExists(cut_key).first >= 0) { continue; }
257
258 for (int node_idx = 0; node_idx < N_SIZE; ++node_idx) { AM[node_idx / 64] |= (1ULL << (node_idx % 64)); }
259
260 std::vector<int> coefficient_indices;
261 std::vector<double> coefficient_values;
262 const auto &support_paths = row_indices_map[node];
263 coefficient_indices.reserve(support_paths.size());
264 coefficient_values.reserve(support_paths.size());
265
266 double exact_violation = 0.0;
267 const double rhs = p.getRHS();
268 for (int path_idx : support_paths) {
269 if (path_idx < 0 || static_cast<size_t>(path_idx) >= allPaths.size() ||
270 static_cast<size_t>(path_idx) >= x.size())
271 continue;
272 const auto &clients = allPaths[path_idx].route;
273 const double coeff = computeLimitedMemoryCoefficient(C, AM, p, clients, order);
274 exact_violation += coeff * x[path_idx];
275 if (!numericutils::isZero(coeff)) {
276 coefficient_indices.push_back(path_idx);
277 coefficient_values.push_back(coeff);
278 }
279 }
280
281 if (exact_violation <= rhs + 1e-6) {
282 continue; // Skip cuts that are not actually violated by the current solution.
283 }
284
285 Cut cut(C, AM, {}, p);
286 cut.baseSetOrder = std::move(order);
287 cut.type = CutType::OneRow;
288 cut.coefficient_indices = std::move(coefficient_indices);
289 cut.coefficient_values = std::move(coefficient_values);
290
291 cutStorage.addCut(cut);
292 }
293 }

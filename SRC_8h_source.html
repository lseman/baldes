127 vis_map.clear();
128 for (const auto i : r.route) { ++vis_map[i]; }
129 for (const auto &[v, times] : vis_map) {
130 if (times > 1) {
131 // Calculate fractional cut value with integer division instead of `floor`
132 double cut_value = std::floor(times / 2.) * r.frac_x;
133 if (cut_value > 1e-3) { // Apply tolerance check
134 tmp_cuts.emplace_back(cut_value, v);
135 }
136 }
137 }
138 }
139 // print tmp_cuts.size()
140 if (tmp_cuts.empty()) return;
141 pdqsort(tmp_cuts.begin(), tmp_cuts.end(),
142 [](const std::pair<double, int> &a, const std::pair<double, int> &b) { return a.first > b.first; });
143
144 std::vector<double> coefficients_aux(allPaths.size(), 0.0);
145
146 auto cuts_to_apply = std::min(static_cast<int>(tmp_cuts.size()), 2);
147
148 for (int i = 0; i < cuts_to_apply; ++i) {
149 auto cut_value = tmp_cuts[i].first;
150 auto v = tmp_cuts[i].second;
151 // Create a new cut
152 std::array<uint64_t, num_words> C = {}; // Reset C for each cut
153 std::array<uint64_t, num_words> AM = {};
154 std::vector<int> order(N_SIZE, 0);
155 // define C
156 C[v / 64] |= (1ULL << (v % 64));
157 // define AM
158 for (int node = 0; node < N_SIZE; ++node) { AM[node / 64] |= (1ULL << (node % 64)); }

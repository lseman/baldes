119 // Dominance check method for Bucket
120 bool check_dominance(const Label *new_label,
121 const std::function<bool(const std::vector<Label *> &)> &dominance_func, int &stat_n_dom) {
122 if (!is_split) {
123 if (get_cb() > new_label->cost) return false;
124 if (dominance_func(get_labels())) {
125 ++stat_n_dom;
126 return true;
127 }
128 return false;
129 }
130
131 std::atomic<bool> dominance_found{false};
132 std::for_each(std::execution::par_unseq, sub_buckets.begin(), sub_buckets.end(), [&](Bucket &bucket) {
133 if (dominance_found || bucket.get_cb() > new_label->cost) return;
134 if (bucket.check_dominance(new_label, dominance_func, stat_n_dom)) { dominance_found = true; }
135 });
136
137 return dominance_found;
138 }

141 void split_into_sub_buckets() noexcept {
142 if (labels_vec.empty()) return;
143
144 // If range is too small, don't split
145 double total_range = ub[0] - lb[0];
146 if (total_range < min_split_range) return;
147
148 if (is_split) return;
149
150
151 // Sort labels by their values
152 std::vector<double> label_values;
153 label_values.reserve(labels_vec.size());
154 if (label_values.empty()) { return; }
155
156 is_split = true;
157 for (const auto *label : labels_vec) { label_values.push_back(label->resources[0]); }
158 std::sort(label_values.begin(), label_values.end());
159
160 // Find median value
161 size_t mid_idx = label_values.size() / 2;
162 double midpoint;
163 if (label_values.size() % 2 == 0) {
164 // If even number of labels, take average of middle two
165 midpoint = roundToTwoDecimalPlaces((label_values[mid_idx - 1] + label_values[mid_idx]) / 2.0);
166 } else {
167 // If odd number of labels, take middle value
168 midpoint = roundToTwoDecimalPlaces(label_values[mid_idx]);
169 }
170
171 // Ensure midpoint is within bucket bounds
172 midpoint = std::max(midpoint, lb[0]);
173 midpoint = std::min(midpoint, ub[0]);
174
175 sub_buckets.clear();
176 sub_buckets.reserve(2);
177
178 // Create first sub-bucket
179 Bucket first_sub_bucket;
180 first_sub_bucket.node_id = node_id;
181 first_sub_bucket.lb = {roundToTwoDecimalPlaces(lb[0])};
182 first_sub_bucket.ub = {midpoint};
183 first_sub_bucket.depth = depth + 1;
184 sub_buckets.push_back(first_sub_bucket);
185
186 // Create second sub-bucket
187 Bucket second_sub_bucket;
188 second_sub_bucket.node_id = node_id;
189 second_sub_bucket.lb = {midpoint};
190 second_sub_bucket.ub = {roundToTwoDecimalPlaces(ub[0])};
191 second_sub_bucket.depth = depth + 1;
192 sub_buckets.push_back(second_sub_bucket);
193
194 // Assign labels to sub-buckets
195 for (auto *label : labels_vec) { assign_label_to_sub_bucket(label); }
196 labels_vec.clear();
197 }

305 void add_sorted_label(Label *label) noexcept {
306 if (is_split) {
307 for (auto &bucket : sub_buckets) {
308 if (bucket.is_contained(label)) {
309 bucket.add_sorted_label(label);
310 return;
311 }
312 }
313 }
314
315 // Insert into current level's labels_vec
316 if (labels_vec.empty() || label->cost >= labels_vec.back()->cost) {
317 labels_vec.push_back(label);
318 } else if (label->cost <= labels_vec.front()->cost) {
319 labels_vec.insert(labels_vec.begin(), label);
320 } else {
321 auto it = std::lower_bound(labels_vec.begin(), labels_vec.end(), label,
322 [](const Label *a, const Label *b) { return a->cost < b->cost; });
323 labels_vec.insert(it, label);
324 }
325
326 if (!is_split && labels_vec.size() >= BUCKET_CAPACITY) { split_into_sub_buckets(); }
327 }

329 bool is_contained(const Label *label) {
330 // Check if each resource in label is within the bounds of sub_bucket
331 for (size_t i = 0; i < 1; ++i) {
332 if (label->resources[i] < lb[i] || label->resources[i] > ub[i]) {
333 return false; // Resource is out of bounds
334 }
335 }
336 return true; // All resources are within bounds

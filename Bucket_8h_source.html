117 bool check_dominance(const Label *new_label, const auto &dominance_func,
118 int &stat_n_dom) const noexcept {
119 if (get_cb() > new_label->cost) {
120 return false;
121 }
122 if (!is_split) {
123 if (dominance_func(get_labels(), stat_n_dom)) {
124 return true;
125 }
126 return false;
127 }
128 std::array<const Bucket *, 2> candidate_buckets;
129 size_t num_candidates = 0;
130 for (const auto &bucket : sub_buckets) {
131 if (bucket->get_cb() <= new_label->cost &&
132 num_candidates < candidate_buckets.size()) {
133 candidate_buckets[num_candidates++] = bucket;
134 }
135 }
136 std::ranges::sort(
137 std::ranges::subrange(candidate_buckets.begin(),
138 candidate_buckets.begin() + num_candidates),
139 [](const auto *a, const auto *b) noexcept {
140 return a->get_cb() < b->get_cb();
141 });
142 for (size_t i = 0; i < num_candidates; ++i) {
143 if (candidate_buckets[i]->check_dominance(new_label, dominance_func,
144 stat_n_dom)) {
145 return true;
146 }
147 }
148 return false;
149 }

151 void split_into_sub_buckets() noexcept {
152 if (labels_vec.empty() || is_split || depth >= MAX_BUCKET_DEPTH) {
153 return;
154 }
155 static constexpr size_t RESOURCE_INDEX = 0;
156 const double total_range = ub[RESOURCE_INDEX] - lb[RESOURCE_INDEX];
157 if (total_range < min_split_range) {
158 return;
159 }
160
161 std::span labels_span{labels_vec};
162 const size_t label_count = labels_span.size();
163 const size_t mid_idx = label_count / 2;
164
165 // Initialize sub-buckets on the stack
166 std::array<Bucket *, 2> new_buckets;
167 for (size_t i = 0; i < 2; ++i) {
168 auto bucket = new Bucket();
169 bucket->node_id = node_id;
170 bucket->depth = depth + 1;
171 bucket->min_split_range = min_split_range;
172
173 // Calculate bounds based on actual values at split point
174 if (i == 0) {
175 bucket->lb = {std::round(lb[RESOURCE_INDEX] * 100.0) / 100.0};
176 // Upper bound for first bucket is the resource value at mid_idx
177 bucket->ub = {
178 std::round(
179 labels_vec[mid_idx - 1]->resources[RESOURCE_INDEX] *
180 100.0) /
181 100.0};
182 } else {
183 // Lower bound for second bucket is the resource value at
184 // mid_idx
185 bucket->lb = {
186 std::round(labels_vec[mid_idx]->resources[RESOURCE_INDEX] *
187 100.0) /
188 100.0};
189 bucket->ub = {std::round(ub[RESOURCE_INDEX] * 100.0) / 100.0};
190 }
191 new_buckets[i] = bucket;
192 }
193
194#ifdef SORTED_LABELS
195 // Since we're splitting purely by position, we can directly split the
196 // vector
197 new_buckets[0]->labels_vec.reserve(mid_idx);
198 new_buckets[1]->labels_vec.reserve(label_count - mid_idx);
199
200 // Copy first half to first bucket
201 std::copy(labels_vec.begin(), labels_vec.begin() + mid_idx,
202 std::back_inserter(new_buckets[0]->labels_vec));
203
204 // Copy second half to second bucket
205 std::copy(labels_vec.begin() + mid_idx, labels_vec.end(),
206 std::back_inserter(new_buckets[1]->labels_vec));
207#else
208 // For unsorted labels, we still split purely by position
209 new_buckets[0]->labels_vec.reserve(mid_idx);
210 new_buckets[1]->labels_vec.reserve(label_count - mid_idx);
211
212 // Move first half to first bucket
213 std::ranges::move(std::ranges::subrange(labels_vec.begin(),
214 labels_vec.begin() + mid_idx),
215 std::back_inserter(new_buckets[0]->labels_vec));
216
217 // Move second half to second bucket
218 std::ranges::move(std::ranges::subrange(labels_vec.begin() + mid_idx,
219 labels_vec.end()),
220 std::back_inserter(new_buckets[1]->labels_vec));
221#endif
222
223 // Move new buckets into place
224 sub_buckets[0] = new_buckets[0];
225 sub_buckets[1] = new_buckets[1];
226
227 // Clear parent bucket's labels efficiently
228 labels_vec = std::vector<Label *>();
229 is_split = true;
230 shall_split = true;
231 }
232 void add_label(Label *label) noexcept {
233 // Early return for null label
234 if (!label) {
235 return;
236 }
237
238 if (is_split) {
239 // There should always be exactly 2 sub-buckets when split
240 assert(sub_buckets.size() == 2);
241
242 // Direct index access is faster than iteration for 2 buckets
243 auto &first_bucket = sub_buckets[0];
244 auto &second_bucket = sub_buckets[1];
245
246 if (first_bucket->is_contained(label)) {
247 if (first_bucket->depth >= MAX_BUCKET_DEPTH) {
248 first_bucket->labels_vec.push_back(label);
249 } else {
250 first_bucket->add_label(label);
251 }
252 return;
253 }
254
255 if (second_bucket->is_contained(label)) {
256 if (second_bucket->depth >= MAX_BUCKET_DEPTH) {
257 second_bucket->labels_vec.push_back(label);
258 } else {
259 second_bucket->add_label(label);
260 }
261 return;
262 }
263
264 // Handle out-of-bounds case
265#ifdef DEBUG
266 print_info(
267 "Warning: Label {:.2f} outside bucket range [{:.2f}, "
268 "{:.2f}]\n",
269 label->resources[0], lb[0], ub[0]);
270#endif
271
272 // Add to the closest sub-bucket instead of parent
273 const double mid_point =
274 (first_bucket->ub[0] + second_bucket->lb[0]) / 2.0;
275 if (label->resources[0] <= mid_point) {
276 first_bucket->labels_vec.push_back(label);
277 } else {
278 second_bucket->labels_vec.push_back(label);
279 }
280 return;
281 }
282
283 labels_vec.push_back(label);
284
285 // Check for split condition
286 if (depth < MAX_BUCKET_DEPTH && labels_vec.size() >= BUCKET_CAPACITY) {
287 split_into_sub_buckets();
288 }
289 }

405 void add_sorted_label(Label *label) noexcept {
406 if (!label) return;
407
408 // Split bucket case
409 if (is_split) {
410 // Try to add to appropriate sub-bucket
411 for (auto &bucket : sub_buckets) {
412 if (bucket->is_contained(label)) {
413 if (bucket->depth >= MAX_BUCKET_DEPTH) {
414 // At max depth, insert directly into bucket's vector
415 insert_sorted_label(bucket->labels_vec, label);
416 } else {
417 bucket->add_sorted_label(label);
418 }
419 return;
420 }
421 }
422 // Fallback: add to parent bucket if label doesn't fit in
423 // sub-buckets
424 insert_sorted_label(labels_vec, label);
425 return;
426 }
427
428 // Non-split bucket case
429 // Pre-reserve space if needed
430 if (labels_vec.size() == labels_vec.capacity()) {
431 labels_vec.reserve(std::max<size_t>(64, labels_vec.capacity() * 2));
432 }
433
434 // Handle empty vector case
435 if (labels_vec.empty()) {
436 labels_vec.push_back(label);
437 return;
438 }
439
440 // Fast path checks
441 if (label->cost >= labels_vec.back()->cost) {
442 labels_vec.push_back(label);
443 return;
444 }
445 if (label->cost <= labels_vec.front()->cost) {
446 labels_vec.insert(labels_vec.begin(), label);
447 return;
448 }
449
450 // Standard case: binary search and insert
451 insert_sorted_label(labels_vec, label);
452
453 // Check if we need to split
454 if (depth < MAX_BUCKET_DEPTH && labels_vec.size() >= BUCKET_CAPACITY) {
455 split_into_sub_buckets();
456 }
457 }

495 void remove_label(Label *label) noexcept {
496 if (!label) return;
497
498 if (is_split) {
499 // When split, directly access appropriate bucket based on resource
500 // value
501 constexpr size_t RESOURCE_INDEX = 0;
502 sub_buckets[label->resources[RESOURCE_INDEX] >
503 sub_buckets[0]->ub[RESOURCE_INDEX]]
504 ->remove_label(label);
505 return;
506 }
507
508 // For unsplit case, use a single optimized removal approach
509 auto &vec = labels_vec;
510 auto it = vec.size() <= 16
511 ? std::find(vec.begin(), vec.end(), label)
512 :
513#ifdef SORTED_LABELS
514 std::lower_bound(vec.begin(), vec.end(), label,
515 [](const Label *a, const Label *b) {
516 return a->cost < b->cost;
517 });
518#else
519 std::find(vec.begin(), vec.end(), label);
520#endif
521
522 if (it != vec.end() &&
523 (!std::is_sorted(vec.begin(), vec.end()) || *it == label)) {
524#ifdef SORTED_LABELS
525 vec.erase(it);
526#else
527 *it = std::move(vec.back());
528 vec.pop_back();
529#endif
530 }

629 Label *get_best_label() noexcept {
630 // Fast path for non-split buckets
631 if (!is_split) {

634 }

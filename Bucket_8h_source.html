80 // --- Virtual Splitting ---
81 // When the bucket reaches capacity, we virtually split it.
82 void virtual_split() noexcept {
83 if (labels.size() < 2 || is_virtual_split) return;
84
85 // Ensure labels are sorted by cost.
86 pdqsort(labels.begin(), labels.end(), [](const Label *a, const Label *b) { return a->cost < b->cost; });
87 // Set the virtual split index at the median.
88 virtual_split_index = labels.size() / 2;
89 if (virtual_split_index >= labels.size()) { virtual_split_index = labels.size() - 1; }

93 // --- Insertion ---
94 // Adds a label into the bucket in sorted order.
95 // This method will update the virtual split when needed.
96 void add_sorted_label(Label *label) noexcept {
97 if (!label) return;
98
99 size_t pos = 0;
100 if (labels.empty()) {
101 labels.push_back(label);
102 } else if (label->cost >= labels.back()->cost) {
103 pos = labels.size();
104 labels.push_back(label);
105 } else if (label->cost <= labels.front()->cost) {
106 labels.insert(labels.begin(), label);
107 pos = 0;
108 } else {
109 auto it = std::lower_bound(labels.begin(), labels.end(), label->cost,
110 [](const Label *a, double cost) { return a->cost < cost; });
111 pos = static_cast<size_t>(std::distance(labels.begin(), it));
112 labels.insert(it, label);
113 }
114
115 if (is_virtual_split) {
116 if (pos <= virtual_split_index) ++virtual_split_index;
117 return;
118 }
119
120 if (labels.size() >= BUCKET_CAPACITY) {
121 shall_split = true;
122 // add_sorted_label preserves ordering; avoid re-sorting here.
123 activate_virtual_split_assume_sorted();
124 }
125 }
126
127 // For cases where sorted order isn’t required on insertion.
128 void add_label(Label *label) noexcept {
129 if (!label) return;
130
131 if (is_virtual_split) {
132 const size_t split = std::min(virtual_split_index, labels.size());
133 if (split > 0 && split < labels.size() && label->cost <= labels[split]->cost) {
134 auto it = std::lower_bound(labels.begin(), labels.begin() + static_cast<std::ptrdiff_t>(split),
135 label->cost, [](const Label *a, double cost) { return a->cost < cost; });
136 labels.insert(it, label);
137 ++virtual_split_index;
138 } else {
139 auto it = std::lower_bound(labels.begin() + static_cast<std::ptrdiff_t>(split), labels.end(),
140 label->cost, [](const Label *a, double cost) { return a->cost < cost; });
141 labels.insert(it, label);
142 }
143 } else {
144 if (labels.empty() || label->cost >= labels.back()->cost) {
145 labels.push_back(label);
146 } else if (label->cost <= labels.front()->cost) {
147 labels.insert(labels.begin(), label);
148 } else {
149 auto it = std::lower_bound(labels.begin(), labels.end(), label->cost,
150 [](const Label *a, double cost) { return a->cost < cost; });
151 labels.insert(it, label);
152 }
153 if (labels.size() >= BUCKET_CAPACITY) shall_split = true;
154 if (labels.size() >= BUCKET_CAPACITY && depth < MAX_BUCKET_DEPTH) {
155 // Insertion above keeps full vector cost-sorted.
156 activate_virtual_split_assume_sorted();
157 }
158 }

178 // --- Dominance Check ---
179 // Checks whether a new label is dominated by any labels already in the
180 // bucket. The dominance_func is a lambda or function that performs the
181 // actual check on a given set of labels.
182 template <typename DominanceFunc>
183 bool check_dominance(const Label *new_label, DominanceFunc &&dominance_func, uint &stat_n_dom) const noexcept {
184 if (!new_label) return false;
185 if (labels.empty()) return false;
186 // Early exit: if the bucket's best cost is higher than the new label's
187 // cost, it cannot be dominated.
188 if (labels.front()->cost > new_label->cost) return false;
189
190 if (!is_virtual_split) {
191 return dominance_func(std::span<Label *const>(labels.data(), labels.size()), stat_n_dom);
192 } else {
193 // For a virtual split, check each half separately without copying.
194 auto *label_ptr = labels.data();
195 const size_t split = std::min(virtual_split_index, labels.size());
196 if (split == 0 || split >= labels.size()) {
197 return dominance_func(std::span<Label *const>(label_ptr, labels.size()), stat_n_dom);
198 }
199 if (dominance_func(std::span<Label *const>(label_ptr, split), stat_n_dom)) { return true; }
200 return dominance_func(std::span<Label *const>(label_ptr + split, labels.size() - split), stat_n_dom);
201 }
202 }

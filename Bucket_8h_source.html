85 if (soa_valid && soa_costs.size() == labels.size()) return;

88 soa_visited_signatures.resize(labels.size());

119 // --- Virtual Splitting ---
120 // When the bucket reaches capacity, we virtually split it.
121 void virtual_split() noexcept {
122 flush_extra_labels();
123 if (labels.size() < 2 || is_virtual_split) return;
124
125 // Ensure labels are sorted by cost.
126 pdqsort(labels.begin(), labels.end(), [](const Label *a, const Label *b) { return a->cost < b->cost; });
127 invalidate_label_cache();
128 // Set the virtual split index at the median.
129 virtual_split_index = labels.size() / 2;
130 if (virtual_split_index >= labels.size()) { virtual_split_index = labels.size() - 1; }
131 is_virtual_split = true;

134 void flush_extra_labels() {
135 if (extra_labels.empty()) return;
136
137 const auto old_size = labels.size();
138 labels.insert(labels.end(), extra_labels.begin(), extra_labels.end());
139#ifndef NDEBUG
140 assert(std::is_sorted(labels.begin(), labels.begin() + static_cast<std::ptrdiff_t>(old_size),
141 [](const Label *a, const Label *b) { return a->cost < b->cost; }));
142#endif
143 pdqsort(labels.begin() + static_cast<std::ptrdiff_t>(old_size), labels.end(),
144 [](const Label *a, const Label *b) { return a->cost < b->cost; });
145 std::inplace_merge(labels.begin(), labels.begin() + static_cast<std::ptrdiff_t>(old_size), labels.end(),
146 [](const Label *a, const Label *b) { return a->cost < b->cost; });
147#ifndef NDEBUG
148 assert(std::is_sorted(labels.begin(), labels.end(),
149 [](const Label *a, const Label *b) { return a->cost < b->cost; }));
150#endif
151 extra_labels.clear();
152 invalidate_label_cache();
153 min_cost = std::numeric_limits<double>::max();
154
155 is_virtual_split = false;
156 virtual_split_index = 0;
157 if (labels.size() >= BUCKET_CAPACITY) {
158 shall_split = true;
159 activate_virtual_split_assume_sorted();
160 }
161 }

172 size_t compact_dominated_labels() {
173 size_t removed = 0;
174
175 const auto compact = [&removed](auto &label_vec) {
176 const auto old_size = label_vec.size();
177 size_t write = 0;
178 for (size_t read = 0; read < label_vec.size(); ++read) {
179 Label *label = label_vec[read];
180 if (label == nullptr || label->is_dominated) continue;
181 if (write != read) label_vec[write] = label;
182 ++write;
183 }
184 label_vec.resize(write);
185 removed += old_size - label_vec.size();
186 };
187
188 compact(labels);
189 compact(extra_labels);
190 if (removed > 0) invalidate_label_cache();
191
192 min_cost = std::numeric_limits<double>::max();
193 for (Label *label : extra_labels) {
194 if (label && label->cost < min_cost) min_cost = label->cost;
195 }
196
197 is_virtual_split = false;
198 virtual_split_index = 0;
199 shall_split = labels.size() + extra_labels.size() >= BUCKET_CAPACITY;
200 if (labels.size() >= BUCKET_CAPACITY) { activate_virtual_split_assume_sorted(); }
201
202 return removed;
203 }

247 // --- Dominance Check ---
248 // Checks whether a new label is dominated by any labels already in the
249 // bucket. The dominance_func is a lambda or function that performs the
250 // actual check on a given set of labels.
251 template <typename SortedDominanceFunc, typename ExtraDominanceFunc>
252 bool check_dominance(const Label *new_label, SortedDominanceFunc &&sorted_dominance_func,
253 ExtraDominanceFunc &&extra_dominance_func, uint &stat_n_dom) const noexcept {
254 if (!new_label) return false;
255 if (labels.empty() && extra_labels.empty()) return false;
256 if (!labels.empty() && !is_virtual_split) {
257 if (sorted_dominance_func(std::span<Label *const>(labels.data(), labels.size()), stat_n_dom)) {
258 return true;
259 }
260 } else if (!labels.empty()) {
261 // For a virtual split, check each half separately without copying.
262 auto *label_ptr = labels.data();
263 const size_t split = std::min(virtual_split_index, labels.size());
264 if (split == 0 || split >= labels.size()) {
265 if (sorted_dominance_func(std::span<Label *const>(label_ptr, labels.size()), stat_n_dom)) {
266 return true;
267 }
268 } else {
269 if (sorted_dominance_func(std::span<Label *const>(label_ptr, split), stat_n_dom)) { return true; }
270 if (sorted_dominance_func(std::span<Label *const>(label_ptr + split, labels.size() - split),
271 stat_n_dom)) {
272 return true;
273 }
274 }
275 }
276 if (extra_labels.empty()) return false;
277 return extra_dominance_func(std::span<Label *const>(extra_labels.data(), extra_labels.size()), stat_n_dom);
278 }

280 template <typename SortedDominanceFunc, typename ExtraDominanceFunc>
281 bool check_dominance_soa(const Label *new_label, SortedDominanceFunc &&sorted_dominance_func,
282 ExtraDominanceFunc &&extra_dominance_func, uint &stat_n_dom) const {
283 if (!new_label) return false;
284 if (labels.empty() && extra_labels.empty()) return false;
285 if (!labels.empty()) {
286 ensure_label_cache();
287 if (!is_virtual_split) {
288 if (sorted_dominance_func(make_label_soa_view(0, labels.size()), stat_n_dom)) { return true; }
289 } else {
290 const size_t split = std::min(virtual_split_index, labels.size());
291 if (split == 0 || split >= labels.size()) {
292 if (sorted_dominance_func(make_label_soa_view(0, labels.size()), stat_n_dom)) { return true; }
293 } else {
294 if (sorted_dominance_func(make_label_soa_view(0, split), stat_n_dom)) { return true; }
295 if (sorted_dominance_func(make_label_soa_view(split, labels.size() - split), stat_n_dom)) {
296 return true;
297 }
298 }
299 }
300 }
301 if (extra_labels.empty()) return false;
302 return extra_dominance_func(std::span<Label *const>(extra_labels.data(), extra_labels.size()), stat_n_dom);
303 }

122 // Dominance check method for Bucket
123 bool check_dominance(const Label *new_label,
124 const std::function<bool(const std::vector<Label *> &)> &dominance_func, int &stat_n_dom) {
125 // Early return if cost bound exceeds new label's cost
126 if (get_cb() > new_label->cost) { return false; }
127
128 // Handle non-split case directly
129 if (!is_split) {
130 if (dominance_func(get_labels())) {
131 ++stat_n_dom;
132 return true;
133 }
134 return false;
135 }
136
137 // For small number of sub-buckets, use sequential processing
138 constexpr size_t PARALLEL_THRESHOLD = 4;
139 if (sub_buckets.size() <= PARALLEL_THRESHOLD) {
140 for (auto &bucket : sub_buckets) {
141 if (bucket.get_cb() <= new_label->cost &&
142 bucket.check_dominance(new_label, dominance_func, stat_n_dom)) {
143 return true;
144 }
145 }
146 return false;
147 }
148
149 // For larger sets, use parallel processing with optimizations
150 std::atomic<bool> dominance_found{false};
151 std::atomic<int> local_stat_n_dom{0};
152
153 // Pre-filter buckets that could potentially dominate
154 std::vector<Bucket *> candidate_buckets;
155 candidate_buckets.reserve(sub_buckets.size());
156
157 for (auto &bucket : sub_buckets) {
158 if (bucket.get_cb() <= new_label->cost) { candidate_buckets.push_back(&bucket); }
159 }
160
161 // If no candidates, return early
162 if (candidate_buckets.empty()) { return false; }
163
164 // Process candidates in parallel
165 std::for_each(std::execution::par_unseq, candidate_buckets.begin(), candidate_buckets.end(),
166 [&](Bucket *bucket) {
167 // Skip if dominance already found
168 if (dominance_found.load(std::memory_order_relaxed)) { return; }
169
170 int local_dom = 0;
171 if (bucket->check_dominance(new_label, dominance_func, local_dom)) {
172 dominance_found.store(true, std::memory_order_release);
173 local_stat_n_dom.fetch_add(local_dom, std::memory_order_relaxed);
174 }
175 });
176
177 // Update stats counter
178 stat_n_dom += local_stat_n_dom.load(std::memory_order_relaxed);
179
180 return dominance_found.load(std::memory_order_acquire);
181 }

183 void split_into_sub_buckets() noexcept {
184 // Early returns for invalid states
185 if (labels_vec.empty() || is_split || depth >= MAX_BUCKET_DEPTH) { return; }
186
187 constexpr size_t RESOURCE_INDEX = 0;
188 const double total_range = ub[RESOURCE_INDEX] - lb[RESOURCE_INDEX];
189
190 if (total_range < min_split_range) { return; }
191
192 // Pre-allocate vectors with exact sizes needed
193 size_t label_count = labels_vec.size();
194 std::vector<double> label_values;
195 label_values.reserve(label_count);
196
197 // Extract resource values
198 std::transform(labels_vec.begin(), labels_vec.end(), std::back_inserter(label_values),
199 [](const Label *label) { return label->resources[RESOURCE_INDEX]; });
200
201 // Find median using nth_element
202 const size_t mid_idx = label_count / 2;
203 std::nth_element(label_values.begin(), label_values.begin() + mid_idx, label_values.end());
204
205 const double midpoint = roundToTwoDecimalPlaces(label_values[mid_idx]);
206 const double clamped_midpoint = std::clamp(midpoint, lb[RESOURCE_INDEX], ub[RESOURCE_INDEX]);
207
208 // Pre-allocate sub-buckets
209 sub_buckets.clear();
210 sub_buckets.reserve(2);
211
212 // Initialize both sub-buckets at once
213 for (int i = 0; i < 2; ++i) {
214 sub_buckets.emplace_back();
215 auto &bucket = sub_buckets.back();
216 bucket.node_id = node_id;
217 bucket.depth = depth + 1;
218 bucket.min_split_range = min_split_range;
219
220 if (i == 0) {
221 bucket.lb = {roundToTwoDecimalPlaces(lb[RESOURCE_INDEX])};
222 bucket.ub = {clamped_midpoint};
223 } else {
224 bucket.lb = {clamped_midpoint};
225 bucket.ub = {roundToTwoDecimalPlaces(ub[RESOURCE_INDEX])};
226 }
227 }
228
229 // Partition labels more efficiently
230 auto &first = sub_buckets[0];
231 auto &second = sub_buckets[1];
232
233 // Reserve approximate space based on median
234 first.labels_vec.reserve(label_count / 2 + label_count / 10); // Add 10% buffer
235 second.labels_vec.reserve(label_count / 2 + label_count / 10);
236
237 // Use stable_partition for better cache utilization
238 auto partition_point = std::stable_partition(labels_vec.begin(), labels_vec.end(),
239 [RESOURCE_INDEX, clamped_midpoint](const Label *label) {
240 return label->resources[RESOURCE_INDEX] <= clamped_midpoint;
241 });
242
243 // Move labels to sub-buckets
244 first.labels_vec.assign(std::make_move_iterator(labels_vec.begin()), std::make_move_iterator(partition_point));
245 second.labels_vec.assign(std::make_move_iterator(partition_point), std::make_move_iterator(labels_vec.end()));
246
247 // Clear parent bucket's labels
248 labels_vec.clear();
249 labels_vec.shrink_to_fit();
250 is_split = true;
251
252#ifdef SORTED_LABELS
253 // Maintain sorting in sub-buckets if needed
254 // first.sort();
255 // second.sort();
256 pdqsort(first.labels_vec.begin(), first.labels_vec.end(),
257 [](const Label *a, const Label *b) { return a->cost < b->cost; });
258 pdqsort(second.labels_vec.begin(), second.labels_vec.end(),
259 [](const Label *a, const Label *b) { return a->cost < b->cost; });
260#endif
261 }

263 void add_label(Label *label) noexcept {
264 // Early return for null label
265 if (!label) { return; }
266
267 if (is_split) {
268 // There should always be exactly 2 sub-buckets when split
269 assert(sub_buckets.size() == 2);
270
271 // Direct index access is faster than iteration for 2 buckets
272 if (sub_buckets[0].is_contained(label)) {
273 if (sub_buckets[0].depth >= MAX_BUCKET_DEPTH) {
274 sub_buckets[0].labels_vec.push_back(label);
275 } else {
276 sub_buckets[0].add_label(label);
277 }
278 return;
279 }
280
281 if (sub_buckets[1].is_contained(label)) {
282 if (sub_buckets[1].depth >= MAX_BUCKET_DEPTH) {
283 sub_buckets[1].labels_vec.push_back(label);
284 } else {
285 sub_buckets[1].add_label(label);
286 }
287 return;
288 }
289
290// Handle out-of-bounds case
291#ifdef DEBUG
292 print_info("Warning: Label {:.2f} outside bucket range [{:.2f}, {:.2f}]\n", label->resources[0], lb[0],
293 ub[0]);
294#endif
295
296 // Add to the closest sub-bucket instead of parent
297 const double mid_point = (sub_buckets[0].ub[0] + sub_buckets[1].lb[0]) / 2.0;
298 if (label->resources[0] <= mid_point) {
299 sub_buckets[0].labels_vec.push_back(label);
300 } else {
301 sub_buckets[1].labels_vec.push_back(label);
302 }
303 return;
304 }
305
306 // Pre-reserve space if needed
307 if (labels_vec.size() == labels_vec.capacity()) {
308 labels_vec.reserve(std::max(size_t(64), labels_vec.capacity() * 2));
309 }
310
311 labels_vec.push_back(label);
312
313 // Check for split condition
314 if (depth < MAX_BUCKET_DEPTH && labels_vec.size() >= BUCKET_CAPACITY) { split_into_sub_buckets(); }
315 }

407 void add_sorted_label(Label *label) noexcept {
408 // Check for invalid label
409 if (!label) return;
410
411 if (is_split) {
412 // Find appropriate sub-bucket
413 for (auto &bucket : sub_buckets) {
414 if (bucket.is_contained(label)) {
415 // Only recurse if we haven't hit maximum depth
416 if (bucket.depth >= MAX_BUCKET_DEPTH) {
417 // Find insertion point even at max depth
418 auto insert_pos =
419 std::lower_bound(bucket.labels_vec.begin(), bucket.labels_vec.end(), label,
420 [](const Label *a, const Label *b) { return a->cost < b->cost; });
421 bucket.labels_vec.insert(insert_pos, label);
422 } else {
423 bucket.add_sorted_label(label);
424 }
425 return;
426 }
427 }
428 // Fallback: add to parent if label doesn't fit in any sub-bucket
429 auto insert_pos = std::lower_bound(labels_vec.begin(), labels_vec.end(), label,
430 [](const Label *a, const Label *b) { return a->cost < b->cost; });
431 labels_vec.insert(insert_pos, label);
432 return;
433 }
434
435 // Pre-reserve space if needed
436 if (labels_vec.size() == labels_vec.capacity()) {
437 labels_vec.reserve(std::max(size_t(64), labels_vec.capacity() * 2));
438 }
439
440 // Handle empty vector case
441 if (labels_vec.empty()) {
442 labels_vec.push_back(label);
443 return;
444 }
445
446 // Fast path for common cases
447 if (label->cost >= labels_vec.back()->cost) {
448 labels_vec.push_back(label);
449 return;
450 }
451
452 if (label->cost <= labels_vec.front()->cost) {
453 labels_vec.insert(labels_vec.begin(), label);
454 return;
455 }
456
457 // Binary search for insertion point
458 auto insert_pos = std::lower_bound(labels_vec.begin(), labels_vec.end(), label,
459 [](const Label *a, const Label *b) { return a->cost < b->cost; });
460 labels_vec.insert(insert_pos, label);
461
462 // Only split if we're not at max depth and have enough labels
463 if (depth < MAX_BUCKET_DEPTH && labels_vec.size() >= BUCKET_CAPACITY) { split_into_sub_buckets(); }
464 }

521 void remove_label(Label *label) noexcept {
522 if (!label) { return; }
523
524 if (!is_split) {
525 // For small vectors, linear search is faster than binary search
526 if (labels_vec.size() <= 16) {
527 auto it = std::find(labels_vec.begin(), labels_vec.end(), label);
528 if (it != labels_vec.end()) {
529 *it = std::move(labels_vec.back());
530 labels_vec.pop_back();
531 }
532 return;
533 }
534
535#ifdef SORTED_LABELS
536 // Use binary search for larger sorted vectors
537 auto it = std::lower_bound(labels_vec.begin(), labels_vec.end(), label,
538 [](const Label *a, const Label *b) { return a->cost < b->cost; });
539
540 if (it != labels_vec.end() && *it == label) { labels_vec.erase(it); }
541#else
542 // Use standard find for unsorted
543 auto it = std::find(labels_vec.begin(), labels_vec.end(), label);
544 if (it != labels_vec.end()) {
545 *it = std::move(labels_vec.back());
546 labels_vec.pop_back();
547 }
548#endif
549 return;

658 Label *get_best_label() noexcept {
659 if (!is_split) {
660 if (labels_vec.empty()) { return nullptr; }
661
662#ifdef SORTED_LABELS
663 return labels_vec[0];
664#else

671 assert(sub_buckets.size() == 2);

673 // Get best labels from both buckets
674 Label *first_best = sub_buckets[0].get_best_label();
675 Label *second_best = sub_buckets[1].get_best_label();
676
677 // Handle cases where one or both buckets might be empty
678 if (!first_best) { return second_best; }
679 if (!second_best) { return first_best; }
680
681 // Return the label with lower cost
682 return (first_best->cost <= second_best->cost) ? first_best : second_best;

117 // --- Insertion ---
118 // Adds a label into the bucket in sorted order.
119 // This method will update the virtual split when needed.
120 void add_sorted_label(Label *label) noexcept {
121 if (!label) return;
122
123 // Find the correct insertion point using binary search.
124 auto it = std::lower_bound(
125 labels.begin(), labels.end(), label,
126 [](const Label *a, const Label *b) { return a->cost < b->cost; });
127
128 // Calculate the position before inserting.
129 size_t pos = std::distance(labels.begin(), it);
130 labels.insert(it, label);
131
132 if (is_virtual_split) {
133 // If the new label is inserted before or at the split point,
134 // then shift the virtual split index to maintain the median.
135 if (pos <= virtual_split_index) {
136 ++virtual_split_index;
137 }
138 } else {
139 // Not split yet: Check if we have reached capacity to trigger a
140 // split.
141 if (labels.size() >= BUCKET_CAPACITY && !is_virtual_split)
142 virtual_split();
143 }
144 }
145
146 // For cases where sorted order isnâ€™t required on insertion.
147 void add_label(Label *label) noexcept {
148 if (!label) return;
149
150 if (is_virtual_split) {
151 // Since the bucket is virtually split, we assume labels are kept
152 // sorted. If the new label's cost is less than or equal to the
153 // label at the split, insert it into the lower half.
154 if (label->cost <= labels[virtual_split_index]->cost) {
155 labels.insert(labels.begin() + virtual_split_index, label);
156 ++virtual_split_index; // Adjust split index as lower half
157 // grows.
158 } else {
159 // Otherwise, append to the upper half.
160 labels.push_back(label);
161 }
162 } else {
163 // When not split, simply push back.
164 labels.push_back(label);
165 // Trigger virtual split if capacity is reached.
166 if (labels.size() >= BUCKET_CAPACITY && depth < MAX_BUCKET_DEPTH)
167 virtual_split();
168 }
169 }

187 // --- Dominance Check ---
188 // Checks whether a new label is dominated by any labels already in the
189 // bucket. The dominance_func is a lambda or function that performs the
190 // actual check on a given set of labels.
191 bool check_dominance(
192 const Label *new_label,
193 const std::function<bool(const std::pmr::vector<Label *> &, uint &)>
194 &dominance_func,
195 uint &stat_n_dom) const noexcept {
196 if (!new_label) return false;
197 // Early exit: if the bucket's best cost is higher than the new label's
198 // cost, it cannot be dominated.
199 if (get_cb() > new_label->cost) return false;
200 // print stat_n_dom
201
202 if (!is_virtual_split) {
203 return dominance_func(labels, stat_n_dom);
204 } else {
205 // For a virtual split, check each half separately.
206 std::pmr::vector<Label *> first_half(
207 labels.begin(), labels.begin() + virtual_split_index);
208 if (dominance_func(first_half, stat_n_dom)) return true;
209 std::pmr::vector<Label *> second_half(
210 labels.begin() + virtual_split_index, labels.end());
211 return dominance_func(second_half, stat_n_dom);
212 }

55 double get_cb() const {
56 // For empty unsplit buckets, return a high cost.
57 if (labels_vec.empty() && !is_split) {
58 return std::numeric_limits<double>::max();
59 }
60
61 if (!is_split) {

67 // label with the minimum cost.

73#endif

78 sub_buckets | std::views::transform([](const auto &sub_bucket) {
79 return sub_bucket->get_cb();
80 }));

119 bool check_dominance(const Label *new_label, const auto &dominance_func,
120 int &stat_n_dom) const noexcept {
121 // Early exit if the bucket's current best cost (cb) is greater than the
122 // new label's cost.
123 // if (numericutils::gt(get_cb(), new_label->cost)) {
124 // return false;
125 // }
126
127 // For non-split buckets, simply use the provided dominance function.
128 if (!is_split) {
129 return dominance_func(get_labels(), stat_n_dom);
130 }
131
132 // For split buckets, collect candidate sub-buckets whose best cost (cb)
133 // is not greater than the new label's cost.
134 std::array<const Bucket *, 2> candidate_buckets{};
135 size_t num_candidates = 0;
136 for (const auto &bucket : sub_buckets) {
137 // if (bucket->get_cb() <= new_label->cost &&
138 // num_candidates < candidate_buckets.size()) {
139 candidate_buckets[num_candidates++] = bucket;
140 }
141 // }
142
143 // Sort candidates by their best cost (cb) for early termination.
144 std::ranges::sort(
145 std::ranges::subrange(candidate_buckets.begin(),
146 candidate_buckets.begin() + num_candidates),
147 [](const Bucket *a, const Bucket *b) noexcept {
148 return a->get_cb() < b->get_cb();
149 });
150
151 // Check dominance recursively in candidate sub-buckets.
152 for (size_t i = 0; i < num_candidates; ++i) {
153 if (candidate_buckets[i]->check_dominance(new_label, dominance_func,
154 stat_n_dom)) {
155 return true;
156 }
157 }
158 return false;

161 // Splits the bucket into two sub-buckets based on the median resource
162 // value.
163 void split_into_sub_buckets() noexcept {
164 // Early exit if no labels, already split, or at max depth.
165 if (labels_vec.empty() || is_split || depth >= MAX_BUCKET_DEPTH) {
166 return;
167 }
168 static constexpr size_t RESOURCE_INDEX = 0;
169 const double total_range = ub[RESOURCE_INDEX] - lb[RESOURCE_INDEX];
170 if (total_range < min_split_range) {
171 return;
172 }
173
174 // Create a span over the labels vector.
175 std::span<Label *> labels_span{labels_vec};
176 const size_t label_count = labels_span.size();
177 const size_t mid_idx = label_count / 2;
178
179 // Initialize two new sub-buckets on the heap.
180 std::array<Bucket *, 2> new_buckets;
181 for (size_t i = 0; i < 2; ++i) {
182 auto bucket = new Bucket();
183 bucket->node_id = node_id;
184 bucket->depth = depth + 1;
185 bucket->min_split_range = min_split_range;
186 bucket->shall_split = false; // Default value
187
188 // Calculate new bounds based on the resource at the split point.
189 if (i == 0) {
190 // First bucket: lower bound from parent's lower bound,
191 // upper bound based on the resource value of the last label in
192 // the first half.
193 bucket->lb = {std::round(lb[RESOURCE_INDEX] * 100.0) / 100.0};
194 bucket->ub = {
195 std::round(
196 labels_vec[mid_idx - 1]->resources[RESOURCE_INDEX] *
197 100.0) /
198 100.0};
199 } else {
200 // Second bucket: lower bound based on the first label in the
201 // second half, upper bound from parent's upper bound.
202 bucket->lb = {
203 std::round(labels_vec[mid_idx]->resources[RESOURCE_INDEX] *
204 100.0) /
205 100.0};
206 bucket->ub = {std::round(ub[RESOURCE_INDEX] * 100.0) / 100.0};
207 }
208 new_buckets[i] = bucket;
209 }
210
211#ifdef SORTED_LABELS
212 // For sorted labels, reserve capacity and copy by position.
213 new_buckets[0]->labels_vec.reserve(mid_idx);
214 new_buckets[1]->labels_vec.reserve(label_count - mid_idx);
215 std::copy(labels_vec.begin(), labels_vec.begin() + mid_idx,
216 std::back_inserter(new_buckets[0]->labels_vec));
217 std::copy(labels_vec.begin() + mid_idx, labels_vec.end(),
218 std::back_inserter(new_buckets[1]->labels_vec));
219#else
220 // For unsorted labels, move by position.
221 new_buckets[0]->labels_vec.reserve(mid_idx);
222 new_buckets[1]->labels_vec.reserve(label_count - mid_idx);
223 std::ranges::move(std::ranges::subrange(labels_vec.begin(),
224 labels_vec.begin() + mid_idx),
225 std::back_inserter(new_buckets[0]->labels_vec));
226 std::ranges::move(std::ranges::subrange(labels_vec.begin() + mid_idx,
227 labels_vec.end()),
228 std::back_inserter(new_buckets[1]->labels_vec));
229#endif
230
231 // Store the new sub-buckets.
232 sub_buckets[0] = new_buckets[0];
233 sub_buckets[1] = new_buckets[1];
234
235 // Clear the parent's labels vector efficiently.
236 labels_vec.clear();
237 is_split = true;
238 // shall_split = true;
239 }

247 void add_label(Label *label) noexcept {
248 // Early return for null label
249 if (!label) {
250 return;
251 }
252
253 if (is_split) {
254 // There should always be exactly 2 sub-buckets when split
255 assert(sub_buckets.size() == 2);
256
257 // Direct index access is faster than iteration for 2 buckets
258 auto &first_bucket = sub_buckets[0];
259 auto &second_bucket = sub_buckets[1];
260
261 if (first_bucket->is_contained(label)) {
262 if (first_bucket->depth >= MAX_BUCKET_DEPTH) {
263 first_bucket->labels_vec.push_back(label);
264 } else {
265 first_bucket->add_label(label);
266 }
267 return;
268 }
269
270 if (second_bucket->is_contained(label)) {
271 if (second_bucket->depth >= MAX_BUCKET_DEPTH) {
272 second_bucket->labels_vec.push_back(label);
273 } else {
274 second_bucket->add_label(label);
275 }
276 return;
277 }
278
279 // Handle out-of-bounds case
280#ifdef DEBUG
281 print_info(
282 "Warning: Label {:.2f} outside bucket range [{:.2f}, "
283 "{:.2f}]\n",
284 label->resources[0], lb[0], ub[0]);
285#endif
286
287 // Add to the closest sub-bucket instead of parent
288 const double mid_point =
289 (first_bucket->ub[0] + second_bucket->lb[0]) / 2.0;
290 if (label->resources[0] <= mid_point) {
291 first_bucket->labels_vec.push_back(label);
292 } else {
293 second_bucket->labels_vec.push_back(label);
294 }
295 return;
296 }
297
298 labels_vec.push_back(label);
299
300 // Check for split condition
301 if (depth < MAX_BUCKET_DEPTH && labels_vec.size() >= BUCKET_CAPACITY) {
302 split_into_sub_buckets();
303 }
304 }

453 // Adds a label into the bucket (or its sub-buckets) in sorted order.
454 void add_sorted_label(Label *label) noexcept {
455 if (!label) return;
456
457 if (is_split) {
458 // For a split bucket, we know there are exactly two sub-buckets.
459 auto &first_bucket = sub_buckets[0];
460 auto &second_bucket = sub_buckets[1];
461
462 // Quick check for containment
463 if (label->resources[0] >= first_bucket->lb[0] &&
464 label->resources[0] <= first_bucket->ub[0]) {
465 if (first_bucket->depth >= MAX_BUCKET_DEPTH)
466 first_bucket->labels_vec.push_back(label);
467 else
468 first_bucket->add_sorted_label(label);
469 return;
470 }
471
472 if (label->resources[0] >= second_bucket->lb[0] &&
473 label->resources[0] <= second_bucket->ub[0]) {
474 if (second_bucket->depth >= MAX_BUCKET_DEPTH)
475 second_bucket->labels_vec.push_back(label);
476 else
477 second_bucket->add_sorted_label(label);
478 return;
479 }
480
481 // Fallback: add to the closest sub-bucket
482 const double mid_point =
483 (first_bucket->ub[0] + second_bucket->lb[0]) * 0.5;
484 if (label->resources[0] <= mid_point)
485 first_bucket->labels_vec.push_back(label);
486 else
487 second_bucket->labels_vec.push_back(label);
488 return;
489 }
490
491 // Non-split bucket case: optimize insertion by avoiding unnecessary
492 // shifting
493
494 // Reserve more capacity if needed, with a 50% growth factor to reduce
495 // reallocations
496 const size_t current_size = labels_vec.size();
497 if (current_size == labels_vec.capacity()) {
498 labels_vec.reserve(std::max<size_t>(
499 64, labels_vec.capacity() + (labels_vec.capacity() >> 1)));
500 }
501
502 // Fast paths for insertion at beginning or end
503 if (current_size == 0 || label->cost >= labels_vec.back()->cost) {
504 labels_vec.push_back(label);
505 return;
506 }
507
508 // Optimization: avoid calling front() which can be slower than direct
509 // access
510 if (label->cost <= labels_vec[0]->cost) {
511 // For insertion at beginning, use resize + shift instead of insert
512 // This is faster because it avoids one memory allocation and only
513 // does one shift
514 labels_vec.resize(current_size + 1);
515
516 // Shift all elements right by one position
517 for (size_t i = current_size; i > 0; --i) {
518 labels_vec[i] = labels_vec[i - 1];
519 }
520
521 // Insert at beginning
522 labels_vec[0] = label;
523 return;
524 }
525
526 // Use binary search to find the insertion point
527 auto it = std::lower_bound(
528 labels_vec.begin(), labels_vec.end(), label,
529 [](const Label *a, const Label *b) { return a->cost < b->cost; });
530
531 // If insertion is near the end, consider whether push_back + rotate is
532 // faster
533 const size_t pos = std::distance(labels_vec.begin(), it);
534 if (pos > current_size - 8) { // Heuristic: if inserting near the end
535 // Add to the end and rotate into position
536 labels_vec.push_back(label);
537 std::rotate(labels_vec.begin() + pos,
538 labels_vec.begin() + current_size, labels_vec.end());
539 } else {
540 // Use standard insert for positions not near the end
541 labels_vec.insert(it, label);
542 }
543
544 // Check if bucket size forces a split
545 if (depth < MAX_BUCKET_DEPTH && labels_vec.size() >= BUCKET_CAPACITY) {
546 // shall_split = true;
547 split_into_sub_buckets();
548 }
549 }

564 bool is_contained(const Label *label) const noexcept {
565 if (!label) {
566 return false;

588 void remove_label(Label *label) noexcept {
589 if (!label) return;
590
591 if (is_split) {
592 // When split, directly access appropriate bucket based on resource
593 // value
594 constexpr size_t RESOURCE_INDEX = 0;
595 sub_buckets[label->resources[RESOURCE_INDEX] >
596 sub_buckets[0]->ub[RESOURCE_INDEX]]
597 ->remove_label(label);
598 return;
599 }
600
601 // For unsplit case, use a single optimized removal approach
602 auto &vec = labels_vec;
603 auto it = vec.size() <= 16
604 ? std::find(vec.begin(), vec.end(), label)
605 :
606#ifdef SORTED_LABELS
607 std::lower_bound(vec.begin(), vec.end(), label,
608 [](const Label *a, const Label *b) {
609 return a->cost < b->cost;
610 });
611#else
612 std::find(vec.begin(), vec.end(), label);
613#endif
614
615 if (it != vec.end() &&
616 (!std::is_sorted(vec.begin(), vec.end()) || *it == label)) {
617#ifdef SORTED_LABELS
618 vec.erase(it);
619#else
620 *it = std::move(vec.back());
621 vec.pop_back();
622#endif
623 }
624 }

626 inline std::vector<Label *> get_labels() const {
627 if (!is_split) {
628 return labels_vec;
629 }
630 // Calculate total size (flattening recursively if needed)
631 size_t total_size = 0;
632 for (const auto &bucket : sub_buckets) {
633 if (bucket) {
634 total_size += bucket->is_split ? bucket->get_labels().size()
635 : bucket->labels_vec.size();
636 }
637 }
638 std::vector<Label *> combined;
639 combined.reserve(total_size);
640 // Insert labels from each sub-bucket directly
641 for (const auto &bucket : sub_buckets) {
642 if (bucket) {
643 if (bucket->is_split) {
644 // Call recursively; you might inline this loop if the
645 // recursion depth is low.
646 auto sub = bucket->get_labels();
647 combined.insert(combined.end(), sub.begin(), sub.end());
648 } else {
649 combined.insert(combined.end(), bucket->labels_vec.begin(),
650 bucket->labels_vec.end());
651 }
652 }
653 }
654 return combined;
655 }

802 // best label.
803 return *std::ranges::min_element(
804 labels_vec, [](const Label *a, const Label *b) {

806 });
807#endif

815 // lower cost.
816 if (!first_best) return second_best;
817 if (!second_best) return first_best;
818 return (first_best->cost <= second_best->cost) ? first_best
819 : second_best;
820 }

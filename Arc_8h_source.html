35// Custom hash function for Arc
36struct arc_hash {
37 std::size_t operator()(const Arc &arc) const {
38 XXH64_state_t *const state = XXH64_createState();
39 XXH64_reset(state, 0); // Use 0 as seed, or choose another seed
40
41 // Add all members to the hash state
42 XXH64_update(state, &arc.from, sizeof(arc.from));
43 XXH64_update(state, &arc.to, sizeof(arc.to));
44 XXH64_update(state, &arc.cost_increment, sizeof(arc.cost_increment));
45 XXH64_update(state, &arc.fixed, sizeof(arc.fixed));
46 XXH64_update(state, &arc.priority, sizeof(arc.priority));
47
48 // Add the resource_increment vector
49 if (!arc.resource_increment.empty()) {
50 XXH64_update(state, arc.resource_increment.data(), arc.resource_increment.size() * sizeof(double));
51 }
52
53 const std::size_t hash = XXH64_digest(state);
54 XXH64_freeState(state);
55 return hash;
56 }

129// Hash function for RawArc
130struct RawArcHash {
131 std::size_t operator()(const RawArc &arc) const {
132 // Take the address of 'arc.from' and 'arc.to' for hashing
133 XXH64_hash_t hash_value = XXH3_64bits(&arc.from, sizeof(arc.from)); // Hash the 'from' field
134 hash_value = XXH3_64bits_withSeed(&arc.to, sizeof(arc.to), hash_value); // Combine with 'to' field

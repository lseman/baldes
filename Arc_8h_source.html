47// Custom hash function for Arc
48struct arc_hash {
49 std::size_t operator()(const Arc &arc) const {
50 XXH64_state_t *const state = XXH64_createState();
51 XXH64_reset(state, 0); // Use 0 as seed, or choose another seed
52
53 // Add all members to the hash state
54 XXH64_update(state, &arc.from, sizeof(arc.from));
55 XXH64_update(state, &arc.to, sizeof(arc.to));
56 XXH64_update(state, &arc.cost_increment, sizeof(arc.cost_increment));
57 XXH64_update(state, &arc.fixed, sizeof(arc.fixed));
58 XXH64_update(state, &arc.priority, sizeof(arc.priority));
59
60 // Add the resource_increment vector
61 if (!arc.resource_increment.empty()) {
62 XXH64_update(state, arc.resource_increment.data(),
63 arc.resource_increment.size() * sizeof(double));
64 }
65
66 const std::size_t hash = XXH64_digest(state);
67 XXH64_freeState(state);
68 return hash;
69 }
70};

153// Hash function for RawArc
154struct RawArcHash {
155 std::size_t operator()(const RawArc &arc) const {
156 // Take the address of 'arc.from' and 'arc.to' for hashing
157 XXH64_hash_t hash_value =
158 XXH3_64bits(&arc.from, sizeof(arc.from)); // Hash the 'from' field
159 hash_value = XXH3_64bits_withSeed(
160 &arc.to, sizeof(arc.to), hash_value); // Combine with 'to' field
161
162 return static_cast<std::size_t>(
163 hash_value); // Return the final hash value

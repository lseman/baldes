75 CholmodSolver() {
76 cholmod_start(&c);
77 c.useGPU = 0;
78 c.nmethods = 1;
79 c.method[0].ordering = CHOLMOD_METIS;
80 c.postorder = true;
81
82#ifdef AUGMENTED
83 c.supernodal = CHOLMOD_SIMPLICIAL;
84#else
85 c.supernodal = CHOLMOD_SUPERNODAL;
86#endif
87
88 L = nullptr;
89 }
90
91 ~CholmodSolver() {
92 if (L) { cholmod_free_factor(&L, &c); }
93 cholmod_finish(&c);
94 }

96 void factorizeMatrix(const Eigen::SparseMatrix<double, Eigen::RowMajor, int> &matrix) {
97 cholmod_sparse *A = viewAsCholmod(matrix);
98
99 if (firstFactorization) {
100 if (L) { cholmod_free_factor(&L, &c); }
101 L = cholmod_analyze(A, &c);
102 firstFactorization = false;
103 }
104
105 cholmod_factorize(A, L, &c);
106 }
107
108 Eigen::VectorXd solve(const Eigen::VectorXd &rhs) {
109 cholmod_dense *b = viewAsCholmod(rhs);
110 cholmod_solve2(CHOLMOD_A, L, b, NULL, &X, NULL, &Y, &E, &c);
111 Eigen::VectorXd result = viewAsEigen(X);
112 return result;
113 }
114
115private:
116 static cholmod_sparse *viewAsCholmod(const Eigen::SparseMatrix<double, Eigen::RowMajor, int> &matrix) {
117 cholmod_sparse *result = new cholmod_sparse;
118 result->nrow = matrix.rows();
119 result->ncol = matrix.cols();
120 result->p = const_cast<int *>(matrix.outerIndexPtr());
121 result->i = const_cast<int *>(matrix.innerIndexPtr());
122 result->x = const_cast<double *>(matrix.valuePtr());
123 result->z = nullptr;
124 result->stype = -1;
125 result->itype = CHOLMOD_INT;
126 result->xtype = CHOLMOD_REAL;
127 result->dtype = CHOLMOD_DOUBLE;
128 result->sorted = 1;
129 result->packed = 1;
130 return result;
131 }
132
133 static cholmod_dense *viewAsCholmod(const Eigen::VectorXd &vector) {
134 cholmod_dense *result = new cholmod_dense;
135 result->nrow = vector.size();
136 result->ncol = 1;
137 result->nzmax = vector.size();
138 result->d = vector.size();
139 result->x = const_cast<double *>(vector.data());
140 result->z = nullptr;
141 result->xtype = CHOLMOD_REAL;
142 result->dtype = CHOLMOD_DOUBLE;
143 return result;
144 }
145
146 static Eigen::VectorXd viewAsEigen(cholmod_dense *vectorPtr) {
147 return Eigen::VectorXd::Map(reinterpret_cast<double *>(vectorPtr->x), vectorPtr->nrow);
148 }
149};

104 SparseSolver(SolverType type = LDLT) {
105 switch (type) {
106 case LDLT:
107 solver = new SolverWrapper<
108 Eigen::CustomSimplicialLDLT<Eigen::SparseMatrix<double>, Eigen::Lower, Eigen::AMDOrdering<int>>>();
109 break;
110 }
111 }
112
113 ~SparseSolver() { delete solver; }
114
115 void factorizeMatrix(const Eigen::SparseMatrix<double, Eigen::ColMajor, int> &matrix) {
116 solver->factorizeMatrix(matrix);
117 }
118
119 void reset() { solver->reset(); }
120
121 int info() { return solver->info(); }
122
123 Eigen::VectorXd solve(const Eigen::VectorXd &rhs) { return solver->solve(rhs); }

126 SolverType solverType;

128 struct SolverBase {
129 virtual void factorizeMatrix(const Eigen::SparseMatrix<double, Eigen::ColMajor, int> &matrix) = 0;
130 virtual Eigen::VectorXd solve(const Eigen::VectorXd &rhs) = 0;
131 virtual ~SolverBase() = default;
132 virtual void reset() = 0;
133 virtual int info() = 0;
134 };
135
136 template <typename Solver>
137 struct SolverWrapper : public SolverBase {
138 Solver solver;
139 void factorizeMatrix(const Eigen::SparseMatrix<double, Eigen::ColMajor, int> &matrix) override {
140 solver.factorizeMatrix(matrix);
141 }
142 Eigen::VectorXd solve(const Eigen::VectorXd &rhs) override { return solver.solve(rhs); }
143 void reset() override { solver.reset(); }
144 int info() override { return solver.info(); }
145 };
146
147 SolverBase *solver;
148};

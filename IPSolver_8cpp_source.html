48void IPSolver::convert_to_standard_form(const Eigen::SparseMatrix<double> &A, const Eigen::VectorXd &b,
49 const Eigen::VectorXd &c, const Eigen::VectorXd &lb, const Eigen::VectorXd &ub,
50 const Eigen::VectorXd &sense, Eigen::SparseMatrix<double> &As,
51 Eigen::VectorXd &bs, Eigen::VectorXd &cs) {
52
53 double infty = std::numeric_limits<double>::infinity();
54 int n = A.rows();
55 int m = A.cols();
56
57 // Operate directly on input vectors wherever possible to reduce copies
58 Eigen::VectorXd lo = lb;
59 Eigen::VectorXd hi = ub;
60
61 int n_free = 0, n_ubounds = 0, nv = A.cols();
62
63 // Counting bounds
64 for (int i = 0; i < lo.size(); ++i) {
65 if (lo[i] == -infty && hi[i] == infty) {
66 ++n_free;
67 } else if (std::isfinite(lo[i]) && std::isfinite(hi[i])) {
68 ++n_ubounds;
69 } else if (lo[i] == -infty && std::isfinite(hi[i])) {
70 // To be dealt with later
71 } else if (std::isfinite(lo[i]) && hi[i] == infty) {
72 // To be dealt with later
73 } else {
74 throw std::runtime_error("unexpected bounds");
75 }
76 }
77
78 int num_slacks = n - sense.sum();
79 cs.conservativeResize(c.size() + n_free + num_slacks);
80 cs.setZero();
81 cs.head(m) = c;
82
83 bs = b; // Direct assignment
84
85 std::vector<Eigen::Triplet<double>> triplets;
86 triplets.reserve(A.nonZeros() + num_slacks); // Reserve space to avoid reallocations
87
88 std::vector<int> ind_ub;
89 std::vector<double> val_ub;
90
91 int free = 0, ubi = 0;
92 for (int j = 0; j < lo.size(); ++j) {
93 double l = lo[j];
94 double h = hi[j];
95
96 for (Eigen::SparseMatrix<double>::InnerIterator it(A, j); it; ++it) {
97 int i = it.row(); // Row index
98 double v = it.value(); // Value at A(i, j)
99
100 if (l == -infty && h == infty) {
101 // Free variable
102 cs[j] = c[j];
103 cs[nv + free] = -c[j];
104 bs[i] -= 0; // No change to bs
105
106 triplets.emplace_back(i, j, v); // Add original value
107 triplets.emplace_back(i, nv + free, -v); // Add negated value
108
109 ++free;
110 } else if (std::isfinite(l) && std::isfinite(h)) {
111 // print l and h
112 // l <= x <= h
113 cs[j + free] = c[j];
114 bs[i] -= (v * l);
115
116 triplets.emplace_back(i, j, v); // Add value
117
118 ind_ub.push_back(j);
119 val_ub.push_back(h - l);
120 ++ubi;
121 } else if (l == -infty && std::isfinite(h)) {
122 // x <= h
123 cs[j] = -c[j];
124 bs[i] -= (-v * h);
125
126 triplets.emplace_back(i, j, -v); // Add negated value
127 } else if (std::isfinite(l) && h == infty) {
128 // l <= x
129 cs[j] = c[j];
130 bs[i] -= (v * l);
131
132 triplets.emplace_back(i, j, v); // Add value
133 } else {
134 throw std::runtime_error("Unexpected bounds");
135 }
136 }
137 }
138
139 // Adding slack variables
140 int slack_counter = 0;
141 for (int i = 0; i < sense.size(); ++i) {
142 if (sense(i) == 0) {
143 triplets.emplace_back(i, nv + n_free + slack_counter, 1.0);
144 ++slack_counter;
145 }
146 }
147
148 // Construct As from triplets
149 As.resize(bs.size(), cs.size());
150 As.setFromTriplets(triplets.begin(), triplets.end());
151 As.makeCompressed();
152}

176void IPSolver::update_residuals(Residuals &res, const Eigen::VectorXd &x, const Eigen::VectorXd &lambda,
177 const Eigen::VectorXd &s, const Eigen::VectorXd &v, const Eigen::VectorXd &w,
178 const Eigen::SparseMatrix<double> &A, const Eigen::VectorXd &b,
179 const Eigen::VectorXd &c, const Eigen::VectorXd &ubv, const Eigen::VectorXi &ubi,
180 const Eigen::VectorXd &vbv, const Eigen::VectorXi &vbi, double tau, double kappa) {
181 // Efficiently calculate rp (primal residual) using noalias to avoid unnecessary allocations
182 res.rp.noalias() = tau * b - A * x;
183 res.rpn = res.rp.norm(); // Primal residual norm
184
185 // Calculate ru (upper bound residual) with fewer iterations by combining operations
186 res.ru = -v; // Directly assign -v to res.ru
187 for (int i = 0; i < ubi.size(); ++i) { res.ru(ubi(i)) += tau * ubv(i) - x(ubi(i)); }
188
189 // Efficiently calculate rd (dual residual) using noalias to avoid temporary allocations
190 res.rd.noalias() = tau * c - A.transpose() * lambda - s;
191 for (int i = 0; i < ubi.size(); ++i) { res.rd(ubi(i)) += w(i); }
192
193 // Efficiently calculate rg (gap residual) by combining dot products
194 res.rg = kappa + c.dot(x) - b.dot(lambda) + ubv.dot(w);
195 res.rgn = std::abs(res.rg); // Since rg is scalar, its "norm" is just the absolute value
196}

258 SparseSolver &ls, const Eigen::VectorXd &theta_vw, const Eigen::VectorXi &ubi,
259 const Eigen::VectorXd &xi_p, const Eigen::VectorXd &xi_d, const Eigen::VectorXd &xi_u) {
260 // Initialize delta_z with zeros
261 delta_z.setZero(ubi.size());
262 // print xi_d size
263 // Use a dense vector and avoid creating an extra SparseVector when unnecessary
264 Eigen::VectorXd xi_d_mod = xi_d;
265 for (int i = 0; i < ubi.size(); ++i) { xi_d_mod(ubi(i)) -= xi_u(ubi(i)) * theta_vw(i); }

311void IPSolver::solve_newton_system(
312 Eigen::VectorXd &Delta_x, Eigen::VectorXd &Delta_lambda, Eigen::VectorXd &Delta_w, Eigen::VectorXd &Delta_s,
313 Eigen::VectorXd &Delta_v, double &Delta_tau, double &Delta_kappa, SparseSolver &ls, const Eigen::VectorXd &theta_vw,
314 const Eigen::VectorXd &b, const Eigen::VectorXd &c, const Eigen::VectorXi &ubi, const Eigen::VectorXd &ubv,
315 const Eigen::VectorXd &delta_x, const Eigen::VectorXd &delta_y, const Eigen::VectorXd &delta_w, double delta_0,
316 const Eigen::VectorXd &iter_x, const Eigen::VectorXd &iter_lambda, const Eigen::VectorXd &iter_w,
317 const Eigen::VectorXd &iter_s, const Eigen::VectorXd &iter_v, double iter_tau, double iter_kappa,
318 const Eigen::VectorXd &xi_p, const Eigen::VectorXd &xi_u, const Eigen::VectorXd &xi_d, double xi_g,
319 const Eigen::VectorXd &xi_xs, const Eigen::VectorXd &xi_vw, double xi_tau_kappa) {
320 Eigen::VectorXd xi_d_copy = xi_d - (xi_xs.array() / iter_x.array()).matrix();
321 Eigen::VectorXd xi_u_copy = xi_u - (xi_vw.array() / iter_w.array()).matrix();
322
323 // Call solve_augsys function here to update Delta_x, Delta_lambda, and
324 // Delta_w
325 solve_augsys(Delta_x, Delta_lambda, Delta_w, ls, theta_vw, ubi, xi_p, xi_d_copy, xi_u_copy);
326
327 // Avoid redundant calculations and simplify expressions where possible
328 Delta_tau = (xi_g + (xi_tau_kappa / iter_tau) + c.dot(Delta_x) - b.dot(Delta_lambda) + ubv.dot(Delta_w)) / delta_0;
329 Delta_kappa = (xi_tau_kappa - iter_kappa * Delta_tau) / iter_tau;
330
331 // Use in-place operations to update Delta_x, Delta_lambda, and Delta_w
332 Delta_x.array() += Delta_tau * delta_x.array();
333 Delta_lambda.array() += Delta_tau * delta_y.array();
334 Delta_w.array() += Delta_tau * delta_w.array();
335
336 // Simplify the Delta_s and Delta_v calculations
337 Delta_s = (xi_xs.array() - iter_s.array() * Delta_x.array()).cwiseQuotient(iter_x.array());
338 Delta_v = (xi_vw.array() - iter_v.array() * Delta_w.array()).cwiseQuotient(iter_w.array());
339}

354double IPSolver::max_alpha_single(const Eigen::VectorXd &v, const Eigen::VectorXd &dv) {
355
356 double alpha = std::numeric_limits<double>::infinity();
357 // #pragma omp parallel for reduction(min : alpha)
358 for (int i = 0; i < v.size(); ++i) {
359 if (dv(i) < 0) {
360 double potential_alpha = -v(i) / dv(i);
361 alpha = std::min(alpha, potential_alpha);
362 }
363 }
364
365 return alpha;
366}

390double IPSolver::max_alpha(const Eigen::VectorXd &x, const Eigen::VectorXd &dx, const Eigen::VectorXd &v,
391 const Eigen::VectorXd &dv, const Eigen::VectorXd &s, const Eigen::VectorXd &ds,
392 const Eigen::VectorXd &w, const Eigen::VectorXd &dw, double tau, double dtau, double kappa,
393 double dkappa) {
394 double alpha_tau = (dtau < 0) ? (-tau / dtau) : 1.0;
395 double alpha_kappa = (dkappa < 0) ? (-kappa / dkappa) : 1.0;

398 max_alpha_single(w, dw), alpha_tau, alpha_kappa});
399

685 lambda += alpha * Delta_lambda;
686 s += alpha * Delta_s;
687 v += alpha * Delta_v;
688 w += alpha * Delta_w;
689 tau += alpha * Delta_tau;
690 kappa += alpha * Delta_kappa;
691 }
692
693 int free_var = 0;
694 double inv_tau = 1.0 / tau;
695
696 Eigen::VectorXd original_x(As.cols());
697 for (int j = 0; j < lo.size(); ++j) {
698 double l = lo[j];
699 double h = hi[j];
700
701 if (l == -infty && h == infty) {
702 original_x[j] = (x[j + free_var] - x[nv_orig + free_var]) * inv_tau;
703 free_var += 1;
704 } else if (std::isfinite(l) && std::isfinite(h)) {
705 original_x[j] = l + x[j] * inv_tau;
706 } else if (l == -infty && std::isfinite(h)) {
707 original_x[j] = h - x[j] * inv_tau;
708 } else if (std::isfinite(l) && h == infty) {
709 original_x[j] = l + x[j] * inv_tau;
710 }
711 }
712
713 double objetivo = cs.dot(original_x);
714 lambda = lambda * inv_tau;
715
716 double dual_obj = b.dot(lambda);
717
718 // convert lambda to std::vector<double>
719 std::vector<double> lambda_vec(lambda.data(), lambda.data() + lambda.size());
720 std::vector<double> original_x_vec(original_x.data(), original_x.data() + original_x.size());
721
722 // return std::make_tuple(x, lambda, s, objetivo);
723 return std::make_tuple(objetivo, dual_obj, original_x_vec, lambda_vec);
724}

726OptimizationData IPSolver::extractOptimizationComponents(GRBModel &model) {
727 OptimizationData data;
728 int numConstrs = model.get(GRB_IntAttr_NumConstrs);
729 int numVars = model.get(GRB_IntAttr_NumVars);
730
731 std::vector<Eigen::Triplet<double>> triplets;
732 data.bs.resize(numConstrs);
733 data.cs.resize(numVars);
734 data.lo.resize(numVars);
735 data.hi.resize(numVars);
736 data.sense.resize(numConstrs);
737
738 // Extract the objective function
739 GRBVar *vars = model.getVars();
740 for (int j = 0; j < numVars; ++j) {
741 data.cs(j) = vars[j].get(GRB_DoubleAttr_Obj);
742 data.lo(j) = vars[j].get(GRB_DoubleAttr_LB);
743
744 if (vars[j].get(GRB_DoubleAttr_UB) == GRB_INFINITY) {
745 data.hi(j) = std::numeric_limits<double>::infinity();
746 } else {
747 data.hi(j) = vars[j].get(GRB_DoubleAttr_UB);
748 }
749 }
750
751 // Extract constraint coefficients and senses
752 for (int i = 0; i < numConstrs; ++i) {
753 GRBConstr constr = model.getConstr(i);
754 GRBLinExpr expr = model.getRow(constr);
755 data.bs(i) = constr.get(GRB_DoubleAttr_RHS);
756
757 //
758 data.sense(i) = (constr.get(GRB_CharAttr_Sense) == '=') ? 1 : 0;
759 for (int j = 0; j < expr.size(); ++j) {
760 // if constr.get(GRB_CharAttr_Sense) > flip the sign
761 double coef;
762 if (constr.get(GRB_CharAttr_Sense) == '>') {
763 coef = -expr.getCoeff(j);
764 data.bs(i) = -data.bs(i);
765 } else {
766 coef = expr.getCoeff(j);
767 }
768 GRBVar var = expr.getVar(j);
769 if (coef != 0.0) { triplets.push_back(Eigen::Triplet<double>(i, var.index(), coef)); }
770 }
771 }
772
773 // Build the sparse matrix As
774 data.As.resize(numConstrs, numVars);
775 data.As.setFromTriplets(triplets.begin(), triplets.end());
776
777 // save the model to a file, in the matricial form
778 data.As.makeCompressed();
779
780 return data;
781}

783OptimizationData IPSolver::convertToOptimizationData(const ModelData &modelData) {
784 OptimizationData optData;
785
786 // Convert SparseMatrix to Eigen::SparseMatrix using Eigen::Triplet
787 std::vector<Eigen::Triplet<double>> triplets;
788 auto sparseMatrix = modelData.A_sparse;
789
790 // Iterate over the CRS format of SparseMatrix to build triplets
791 sparseMatrix.forEachRow([&triplets](int row, int col, double value) { triplets.emplace_back(row, col, value); });
792
793 // Resize the Eigen sparse matrix
794 optData.As.resize(sparseMatrix.num_rows, sparseMatrix.num_cols);
795
796 // Set the values from the triplets
797 optData.As.setFromTriplets(triplets.begin(), triplets.end());
798
799 // Make the matrix compressed for efficient operations
800 optData.As.makeCompressed();
801
802 // Convert b to Eigen::VectorXd
803 optData.bs = Eigen::VectorXd::Map(modelData.b.data(), modelData.b.size());
804
805 // Convert c to Eigen::VectorXd
806 optData.cs = Eigen::VectorXd::Map(modelData.c.data(), modelData.c.size());
807
808 // Convert lb to Eigen::VectorXd
809 optData.lo = Eigen::VectorXd::Map(modelData.lb.data(), modelData.lb.size());
810
811 // Convert ub to Eigen::VectorXd
812 optData.hi = Eigen::VectorXd::Map(modelData.ub.data(), modelData.ub.size());
813
814 // Convert sense to Eigen::VectorXd (mapping '<' to 0, '=' to 1, '>' to 0 and flipping the corresponding row)
815 optData.sense.resize(modelData.sense.size());
816 for (size_t i = 0; i < modelData.sense.size(); ++i) {
817 if (modelData.sense[i] == '<') {
818 optData.sense[i] = 0.0;
819 } else if (modelData.sense[i] == '=') {
820 optData.sense[i] = 1.0;
821 } else if (modelData.sense[i] == '>') {
822 optData.sense[i] = 0.0;
823 optData.bs[i] = -optData.bs[i];
824 optData.As.row(i) *= -1; // Flip the row for '>'
825 }
826 }
827
828 return optData;
829}

831void IPSolver::update_linear_solver(SparseSolver &ls, const Eigen::VectorXd &theta, const Eigen::VectorXd &regP,
832 const Eigen::VectorXd &regD) {
833 // Update internal data
834 ls.theta = theta;
835 ls.regP = regP;
836 ls.regD = regD;
837
838#ifdef AUGMENTED
839 // Update S. S is stored as upper-triangular and only its diagonal changes.
840 Eigen::VectorXd combinedValues(ls.n + ls.m);
841 combinedValues.head(ls.n) = -theta - regP;
842 combinedValues.tail(ls.m) = regD;
843
844 // Efficiently update diagonal elements
845 for (int i = 0; i < combinedValues.size(); i++) { ls.S.coeffRef(i, i) = combinedValues[i]; }
846
847 // Refactorize
848 auto S = ls.S;
849 ls.factorizeMatrix(S);
850#else
851
852 // define lhs for normal equations
853 Eigen::SparseMatrix<double> lhs(ls.n + ls.m, ls.n + ls.m);
854 // define lhs as A (\Theta^{-1} + R_{p})^{-1} A^{\top} + R_{d}
855 Eigen::VectorXd d = 1.0 / (ls.theta.array() + ls.regP.array());
856 // set rhs as \xi_{p} + A (Θ^{-1} + R_{p})^{-1} \xi_{d}
857 Eigen::MatrixXd dDense = d.asDiagonal();
858 Eigen::SparseMatrix<double> dSparse = dDense.sparseView();
859
860 Eigen::MatrixXd regDDense = regD.asDiagonal();
861 Eigen::SparseMatrix<double> regDSparse = regDDense.sparseView();
862 Eigen::SparseMatrix<double> AD = ls.A * dSparse;
863 Eigen::SparseMatrix<double> ADA = AD * ls.A.transpose();
864 lhs = ADA + regDSparse;
865 ls.factorizeMatrix(lhs);
866
867#endif
868}

875void IPSolver::start_linear_solver(SparseSolver &ls, const Eigen::SparseMatrix<double> A) {
876 ls.A = A;
877 ls.m = A.rows();
878 ls.n = A.cols();
879 // print ls.A size
880
881#ifdef AUGMENTED
882 ls.theta = Eigen::VectorXd::Ones(ls.n);
883 ls.regP = Eigen::VectorXd::Ones(ls.n);
884 ls.regD = Eigen::VectorXd::Ones(ls.m);
885
886 Eigen::SparseMatrix<double> topRight = ls.A.transpose();
887 Eigen::SparseMatrix<double> bottomLeft = ls.A;
888 Eigen::SparseMatrix<double> topLeft = convertToSparseDiagonal(-ls.theta - ls.regP);
889 Eigen::SparseMatrix<double> bottomRight = convertToSparseDiagonal(ls.regD);
890
891 // S_ is known, reserve space for it
892 Eigen::SparseMatrix<double> S_(ls.n + ls.m, ls.n + ls.m);
893
894 // Reserving space for tripletList
895 int estimated_nonzeros = topLeft.nonZeros() + 2 * topRight.nonZeros() + bottomRight.nonZeros();
896 std::vector<Eigen::Triplet<double>> tripletList;
897 tripletList.reserve(estimated_nonzeros);
898
899 // Insert topLeft, topRight, bottomLeft, bottomRight matrices
900 auto insertBlock = [&](const Eigen::SparseMatrix<double> &block, int startRow, int startCol) {
901 for (int k = 0; k < block.outerSize(); ++k) {
902 for (Eigen::SparseMatrix<double>::InnerIterator it(block, k); it; ++it) {
903 tripletList.emplace_back(it.row() + startRow, it.col() + startCol, it.value());
904 }
905 }
906 };
907
908 insertBlock(topLeft, 0, 0);
909 insertBlock(topRight, 0, ls.n);
910 insertBlock(bottomLeft, ls.n, 0);
911 insertBlock(bottomRight, ls.n, ls.n);
912
913 // Finally, set the values from the triplets
914 S_.setFromTriplets(tripletList.begin(), tripletList.end());
915 // S_.makeCompressed();
916
917 ls.S = S_;
918 // Factorize
919 ls.factorizeMatrix(ls.S);
920#endif
921}

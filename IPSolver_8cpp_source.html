37void IPSolver::convert_to_standard_form(const Eigen::SparseMatrix<double> &A, const Eigen::VectorXd &b,
38 const Eigen::VectorXd &c, const Eigen::VectorXd &lb, const Eigen::VectorXd &ub,
39 const Eigen::VectorXd &sense, Eigen::SparseMatrix<double> &As,
40 Eigen::VectorXd &bs, Eigen::VectorXd &cs) {
41 double infty = std::numeric_limits<double>::infinity();
42 int n = A.rows();
43 int m = A.cols();
44
45 if (b.size() != n || c.size() != m) {
46 // print b.size(), n, c.size(), m
47 fmt::print("b.size(): {}, n: {}, c.size(): {}, m: {}\n", b.size(), n, c.size(), m);
48 throw std::invalid_argument("Size of b or c does not match the matrix A dimensions.");
49 }
50
51 // Operate directly on input vectors wherever possible to reduce copies
52 Eigen::VectorXd lo = lb;
53 Eigen::VectorXd hi = ub;
54
55 int n_free = 0, n_ubounds = 0, nv = A.cols();
56
57 // Counting bounds
58 for (int i = 0; i < lo.size(); ++i) {
59 if (lo[i] == -infty && hi[i] == infty) {
60 ++n_free;
61 } else if (std::isfinite(lo[i]) && std::isfinite(hi[i])) {
62 ++n_ubounds;
63 } else {
64 throw std::runtime_error("unexpected bounds");
65 }
66 }
67
68 int num_slacks = n - sense.sum();
69 cs.conservativeResize(c.size() + n_free + num_slacks);
70 cs.setZero();
71 cs.head(m) = c;
72
73 bs = b; // Direct assignment
74
75 std::vector<Eigen::Triplet<double>> triplets;
76 triplets.reserve(A.nonZeros() + num_slacks); // Reserve space to avoid reallocations
77
78 std::vector<int> ind_ub;
79 std::vector<double> val_ub;
80
81 int free = 0, ubi = 0;
82 int lo_size = lo.size();
83 triplets.reserve(lo_size * 2); // Estimate the triplet size more accurately
84
85 for (int j = 0; j < lo_size; ++j) {
86 double l = lo[j];
87 double h = hi[j];
88
89 bool is_free_var = (l == -infty && h == infty);
90 bool is_bounded_var = (std::isfinite(l) && std::isfinite(h));
91 bool is_upper_bound = (l == -infty && std::isfinite(h));
92 bool is_lower_bound = (std::isfinite(l) && h == infty);
93
94 for (Eigen::SparseMatrix<double>::InnerIterator it(A, j); it; ++it) {
95 int i = it.row(); // Row index
96 double v = it.value(); // Value at A(i, j)
97
98 if (i < 0 || i >= n || j < 0 || j >= m) {
99 std::cerr << "Error: Invalid matrix index (i=" << i << ", j=" << j << "). Matrix dimensions are " << n
100 << "x" << m << "." << std::endl;
101 throw std::out_of_range("Matrix index out of bounds");
102 }
103
104 if (is_free_var) {
105 if (nv + free >= cs.size()) {
106 std::cerr << "Error: Free variable index out of bounds. nv + free = " << nv + free
107 << ", cs size = " << cs.size() << std::endl;
108 throw std::out_of_range("Free variable index out of bounds");
109 }
110 // Free variable
111 cs[j] = c[j];
112 cs[nv + free] = -c[j];
113 triplets.emplace_back(i, j, v); // Add original value
114 triplets.emplace_back(i, nv + free, -v); // Add negated value
115 ++free;
116 } else if (is_bounded_var) {
117 // l <= x <= h
118 cs[j + free] = c[j];
119 bs[i] -= (v * l); // Compute once
120 triplets.emplace_back(i, j, v); // Add value
121
122 ind_ub.push_back(j);
123 val_ub.push_back(h - l);
124 ++ubi;
125 } else if (is_upper_bound) {
126 // x <= h
127 cs[j] = -c[j];
128 bs[i] -= (-v * h); // Compute once
129 triplets.emplace_back(i, j, -v); // Add negated value
130 } else if (is_lower_bound) {
131 // l <= x
132 cs[j] = c[j];
133 bs[i] -= (v * l); // Compute once
134 triplets.emplace_back(i, j, v); // Add value
135 } else {
136 throw std::runtime_error("Unexpected bounds");
137 }
138 }
139 }
140
141 // Adding slack variables
142 int slack_counter = 0;
143 for (int i = 0; i < sense.size(); ++i) {
144 if (sense(i) == 0) {
145 if (nv + n_free + slack_counter >= cs.size()) {
146 std::cerr << "Error: Slack variable index out of bounds. nv + n_free + slack_counter = "
147 << nv + n_free + slack_counter << ", cs size = " << cs.size() << std::endl;
148 throw std::out_of_range("Slack variable index out of bounds");
149 }
150 triplets.emplace_back(i, nv + n_free + slack_counter, 1.0);
151 ++slack_counter;
152 }
153 }
154
155 // Construct As from triplets
156 As.resize(bs.size(), cs.size());
157 As.setFromTriplets(triplets.begin(), triplets.end());
158 As.makeCompressed();
159}

168void IPSolver::update_residuals(Residuals &res, const Eigen::VectorXd &x, const Eigen::VectorXd &lambda,
169 const Eigen::VectorXd &s, const Eigen::VectorXd &v, const Eigen::VectorXd &w,
170 const Eigen::SparseMatrix<double> &A, const Eigen::VectorXd &b,
171 const Eigen::VectorXd &c, const Eigen::VectorXd &ubv, const Eigen::VectorXi &ubi,
172 const Eigen::VectorXd &vbv, const Eigen::VectorXi &vbi, double tau, double kappa) {
173 // Efficiently calculate rp (primal residual) using noalias to avoid unnecessary allocations
174 res.rp.noalias() = tau * b - A * x;
175 res.rpn = res.rp.norm(); // Primal residual norm
176
177 // Calculate ru (upper bound residual) with fewer iterations by combining operations
178 res.ru = -v; // Directly assign -v to res.ru
179
180 // Update res.ru more efficiently using Eigen's indexed operations
181 for (int i = 0; i < ubi.size(); ++i) {
182 int idx = ubi(i); // Cache index access
183 res.ru(idx) += tau * ubv(i) - x(idx);
184 }
185
186 // Efficiently calculate rd (dual residual) using noalias to avoid temporary allocations
187 res.rd.noalias() = tau * c - A.transpose() * lambda - s;
188
189 // Update res.rd similarly to ru, using Eigen's indexed access
190 for (int i = 0; i < ubi.size(); ++i) {
191 int idx = ubi(i); // Cache index access
192 res.rd(idx) += w(i);
193 }
194
195 // Efficiently calculate rg (gap residual) by combining dot products
196 res.rg = kappa + c.dot(x) - b.dot(lambda) + ubv.dot(w);
197 res.rgn = std::abs(res.rg); // Since rg is scalar, its "norm" is just the absolute value
198}

259void IPSolver::solve_augsys(Eigen::VectorXd &delta_x, Eigen::VectorXd &delta_y, Eigen::VectorXd &delta_z,
260 SparseSolver &ls, const Eigen::VectorXd &theta_vw, const Eigen::VectorXi &ubi,
261 const Eigen::VectorXd &xi_p, const Eigen::VectorXd &xi_d, const Eigen::VectorXd &xi_u) {
262 // Create a modifiable copy of xi_d
263 Eigen::VectorXd xi_d_mod = xi_d;
264
265 // Update the copy (xi_d_mod) instead of the original
266 for (int i = 0; i < ubi.size(); ++i) {
267 int idx = ubi(i); // Cache the index
268 xi_d_mod.coeffRef(idx) -= xi_u(idx) * theta_vw(i); // Modify the copy
269 }
270
271 // Call the function to solve the augmented system
272 solve_augmented_system(delta_x, delta_y, ls, xi_p, xi_d_mod);
273
274 // Update delta_z more efficiently with direct access
275 for (int i = 0; i < ubi.size(); ++i) {
276 int idx = ubi(i); // Cache the index
277 delta_z(i) = (delta_x(idx) - xi_u(i)) * theta_vw(i);
278 }
279}

288void IPSolver::solve_newton_system(
289 Eigen::VectorXd &Delta_x, Eigen::VectorXd &Delta_lambda, Eigen::VectorXd &Delta_w, Eigen::VectorXd &Delta_s,
290 Eigen::VectorXd &Delta_v, double &Delta_tau, double &Delta_kappa, SparseSolver &ls, const Eigen::VectorXd &theta_vw,
291 const Eigen::VectorXd &b, const Eigen::VectorXd &c, const Eigen::VectorXi &ubi, const Eigen::VectorXd &ubv,
292 const Eigen::VectorXd &delta_x, const Eigen::VectorXd &delta_y, const Eigen::VectorXd &delta_w, double delta_0,
293 const Eigen::VectorXd &iter_x, const Eigen::VectorXd &iter_lambda, const Eigen::VectorXd &iter_w,
294 const Eigen::VectorXd &iter_s, const Eigen::VectorXd &iter_v, double iter_tau, double iter_kappa,
295 const Eigen::VectorXd &xi_p, const Eigen::VectorXd &xi_u, const Eigen::VectorXd &xi_d, double xi_g,
296 const Eigen::VectorXd &xi_xs, const Eigen::VectorXd &xi_vw, double xi_tau_kappa) {
297 Eigen::VectorXd xi_d_copy = xi_d - (xi_xs.array() / iter_x.array()).matrix();
298 Eigen::VectorXd xi_u_copy = xi_u - (xi_vw.array() / iter_w.array()).matrix();
299
300 // Call solve_augsys function here to update Delta_x, Delta_lambda, and
301 // Delta_w
302 solve_augsys(Delta_x, Delta_lambda, Delta_w, ls, theta_vw, ubi, xi_p, xi_d_copy, xi_u_copy);
303
304 // Avoid redundant calculations and simplify expressions where possible
305 Delta_tau = (xi_g + (xi_tau_kappa / iter_tau) + c.dot(Delta_x) - b.dot(Delta_lambda) + ubv.dot(Delta_w)) / delta_0;
306 Delta_kappa = (xi_tau_kappa - iter_kappa * Delta_tau) / iter_tau;
307
308 // Use in-place operations to update Delta_x, Delta_lambda, and Delta_w
309 Delta_x.array() += Delta_tau * delta_x.array();
310 Delta_lambda.array() += Delta_tau * delta_y.array();
311 Delta_w.array() += Delta_tau * delta_w.array();
312
313 // Simplify the Delta_s and Delta_v calculations
314 Delta_s = (xi_xs.array() - iter_s.array() * Delta_x.array()).cwiseQuotient(iter_x.array());
315 Delta_v = (xi_vw.array() - iter_v.array() * Delta_w.array()).cwiseQuotient(iter_w.array());
316}

328double IPSolver::max_alpha_single(const Eigen::VectorXd &v, const Eigen::VectorXd &dv) {
329
330 double alpha = std::numeric_limits<double>::infinity();
331 // #pragma omp parallel for reduction(min : alpha)
332 for (int i = 0; i < v.size(); ++i) {
333 if (dv(i) < 0) {
334 double potential_alpha = -v(i) / dv(i);
335 alpha = std::min(alpha, potential_alpha);
336 }
337 }
338
339 return alpha;
340}

351double IPSolver::max_alpha(const Eigen::VectorXd &x, const Eigen::VectorXd &dx, const Eigen::VectorXd &v,
352 const Eigen::VectorXd &dv, const Eigen::VectorXd &s, const Eigen::VectorXd &ds,
353 const Eigen::VectorXd &w, const Eigen::VectorXd &dw, double tau, double dtau, double kappa,
354 double dkappa) {
355 double alpha_tau = (dtau < 0) ? (-tau / dtau) : 1.0;
356 double alpha_kappa = (dkappa < 0) ? (-kappa / dkappa) : 1.0;
357
358 double alpha = std::min({1.0, max_alpha_single(x, dx), max_alpha_single(v, dv), max_alpha_single(s, ds),
359 max_alpha_single(w, dw), alpha_tau, alpha_kappa});
360
361 return alpha;
362}

372std::tuple<double, double, std::vector<double>, std::vector<double>> IPSolver::run_optimization(ModelData &model,
373 const double tol) {
374
375 // auto componentes = extractOptimizationComponents(model);
376
377 auto componentes = convertToOptimizationData(model);
378
379 Eigen::SparseMatrix<double> As = componentes.As;
380 Eigen::VectorXd bs = componentes.bs;
381 Eigen::VectorXd cs = componentes.cs;
382 Eigen::VectorXd lo = componentes.lo;
383 Eigen::VectorXd hi = componentes.hi;
384 Eigen::VectorXd sense = componentes.sense;
385
386 // Convert to standard form;
387 Eigen::SparseMatrix<double> A;
388 Eigen::VectorXd b;
389 Eigen::VectorXd c;
390
391 int nv_orig = cs.size();
392
393 convert_to_standard_form(As, bs, cs, lo, hi, sense, A, b, c);
394
395 // preconditioner(A, b, c, lo, hi, sense);
396
397 int n = A.cols();
398 int m = A.rows();
399
400 // Output the initial results
401 // Tolerance and maximum iterations
402 int max_iter = 100;
403
404 // Initialize vectors and scalars
405 Eigen::VectorXd x = Eigen::VectorXd::Ones(n);
406 Eigen::VectorXd lambda = Eigen::VectorXd::Zero(m);
407 Eigen::VectorXd s = Eigen::VectorXd::Ones(n);
408
409 // initialize ubi and ubv as empty vectors
410 Eigen::VectorXi ubi;
411 Eigen::VectorXd ubv;
412
413 int count = 0; // Count of non-zero entries
414 for (int i = 0; i < hi.size(); i++) {
415 if (hi[i] != std::numeric_limits<double>::infinity()) { count++; }
416 }
417
418 Eigen::VectorXi tempUbi(count);
419 Eigen::VectorXd tempUbv(count);
420
421 double infty = std::numeric_limits<double>::infinity();
422 count = 0;
423 for (int i = 0; i < hi.size(); i++) {
424 if (hi[i] != infty) {
425 tempUbi[count] = i;
426 tempUbv[count] = hi[i];
427 count++;
428 }
429 }
430 ubi = tempUbi;
431 ubv = tempUbv;
432
433 Eigen::VectorXd v = Eigen::VectorXd::Ones(ubv.size());
434 Eigen::VectorXd w = Eigen::VectorXd::Ones(ubv.size());
435
436 // initialize vbi and vbv as empty vectors
437 Eigen::VectorXi vbi;
438 Eigen::VectorXd vbv;
439
440 double tau = 1.0;
441 double kappa = 1.0;
442
443 // Assuming lp.nv and lp.nc are the dimensions you need
444 Eigen::VectorXd regP = Eigen::VectorXd::Ones(n);
445 Eigen::VectorXd regD = Eigen::VectorXd::Ones(m);
446 double regG = 1.0;
447
448 SparseSolver ls;
449 start_linear_solver(ls, A);
450
451 int nc = A.rows(); // Assuming ls is the sparse matrix
452 int nv = A.cols();
453 int nu = ubi.size();
454
455 Eigen::VectorXd delta_x(nv), delta_y(nc), delta_z(nu);
456 // Residuals
457 Residuals res;
458
459 // Dimensions and constants
460 double r_min = std::sqrt(std::numeric_limits<double>::epsilon()); // approx 1e-8
461 int attempt = 0, ncor = 0;
462 // Residual related variables
463 double _p, _d, _g, mu;
464 // Step length and corrections
465 double alpha, alpha_c, alpha_;
466 // Damping factors
467 double beta, gamma, damping, oneMinusAlpha;
468 // Cross products and thresholds
469 double mu_l, mu_u, taukappa, t0;
470 // Theta values
471 Eigen::VectorXd theta_vw, theta_xs;
472 // Xi values
473 Eigen::VectorXd xi_p, xi_d, xi_u, xi_xs, xi_vw;
474 // Delta values
475 Eigen::VectorXd Delta_x(x.size()), Delta_lambda(lambda.size()), Delta_w(w.size()), Delta_s(s.size()),
476 Delta_v(v.size());
477 double Delta_tau, Delta_kappa;
478 // Corrected Delta values
479 Eigen::VectorXd Delta_x_c(x.size()), Delta_lambda_c(lambda.size()), Delta_w_c(w.size()), Delta_s_c(s.size()),
480 Delta_v_c(v.size());
481 double Delta_tau_c, Delta_kappa_c;
482 // Temporary values for corrections
483 Eigen::VectorXd xs, vw, t_xs, t_vw;
484 Eigen::ArrayXd t_xs_lower, t_xs_upper, t_vw_lower, t_vw_upper;
485 // Delta calculations
486 double delta_0, bl_dot_lambda, correction;
487
488 for (int k = 0; k < max_iter; ++k) {
489 // fmt::print("Iteration {}\n", k);
490 // Zero the necessary variables
491 ncor = 0;
492 beta = 0.1;
493 // Zero out the predictor search direction variables
494 delta_x.setZero();
495 delta_y.setZero();
496 delta_z.setZero();
497
498 Delta_x.setZero();
499 Delta_lambda.setZero();
500 Delta_w.setZero();
501 Delta_s.setZero();
502 Delta_v.setZero();
503 Delta_tau = 0.0;
504 Delta_kappa = 0.0;
505
506 // Update residuals
507 update_residuals(res, x, lambda, s, v, w, A, b, c, ubv, ubi, vbv, vbi, tau, kappa);
508 mu = (tau * kappa + x.dot(s) + v.dot(w)) / (n + ubi.size() + 1.0);
509
510 // Calculate _p, _d, and _g in parallel
511 // Calculate primal and dual residual norms using Infinity norm
512 double rp_norm = res.rp.lpNorm<Eigen::Infinity>();
513 double ru_norm = res.ru.lpNorm<Eigen::Infinity>();
514 double rd_norm = res.rd.lpNorm<Eigen::Infinity>();
515 double b_norm = b.lpNorm<Eigen::Infinity>();
516 double ubv_norm = ubv.lpNorm<Eigen::Infinity>();
517 double c_norm = c.lpNorm<Eigen::Infinity>();
518
519 // Avoid repeated calculations and improve readability
520 _p = std::fmax(rp_norm / (tau * (1.0 + b_norm)), ru_norm / (tau * (1.0 + ubv_norm)));
521 _d = rd_norm / (tau * (1.0 + c_norm));
522
523 // Calculate the dot product once for efficiency
524 double bl_dot_lambda = b.dot(lambda) - ubv.dot(w);
525 _g = std::abs(c.dot(x) - bl_dot_lambda) / (tau + std::abs(bl_dot_lambda));
526
527 // Check for optimality and infeasibility
528 if (_p <= 1e-10 && _d <= 1e-10 && _g <= tol) { break; }
529 // Scaling factors
530 theta_vw = w.cwiseQuotient(v);
531 theta_xs = s.cwiseQuotient(x);
532
533 for (int i = 0; i < ubi.size(); ++i) { theta_xs[ubi[i]] += theta_vw[i]; }
534 // Update regularizations
535 regP = (regP / 10.0).cwiseMax(r_min);
536 regD = (regD / 10.0).cwiseMax(r_min);
537 regG = std::max(r_min, regG / 10.0);
538
539 // Factorization with retries
540 for (int attempt = 0; attempt < 5; ++attempt) {
541 try {
542 // fmt::print("Attempt {}\n", attempt);
543 update_linear_solver(ls, theta_xs, regP, regD);
544 break;
545 } catch (std::runtime_error &) {
546 regP *= 100.0;
547 regD *= 100.0;
548 regG *= 100.0;
549 }
550 }
551
552 update_linear_solver(ls, theta_xs, regP, regD);
553
554 // Solve the augmented system
555 solve_augsys(delta_x, delta_y, delta_z, ls, theta_vw, ubi, b, c, ubv);
556 delta_0 = regG + kappa / tau - delta_x.dot(c) + delta_y.dot(b) - delta_z.dot(ubv);
557 // Solve the Newton system
558 solve_newton_system(Delta_x, Delta_lambda, Delta_w, Delta_s, Delta_v, Delta_tau, Delta_kappa, ls, theta_vw, b,
559 c, ubi, ubv, delta_x, delta_y, delta_z, delta_0, x, lambda, w, s, v, tau, kappa, res.rp,
560 res.ru, res.rd, res.rg, -x.cwiseProduct(s), -v.cwiseProduct(w), -tau * kappa);
561
562 // Calculate step length
563 alpha = max_alpha(x, Delta_x, v, Delta_v, s, Delta_s, w, Delta_w, tau, Delta_tau, kappa, Delta_kappa);
564
565 // Calculate gamma and damping
566 oneMinusAlpha = 1.0 - alpha;
567 gamma = std::fmax(oneMinusAlpha * oneMinusAlpha * std::fmin(beta, oneMinusAlpha), 0.1);
568 damping = 1.0 - gamma;
569
570 solve_newton_system(Delta_x, Delta_lambda, Delta_w, Delta_s, Delta_v, Delta_tau, Delta_kappa, ls, theta_vw, b,
571 c, ubi, ubv, delta_x, delta_y, delta_z, delta_0, x, lambda, w, s, v, tau, kappa,
572 damping * res.rp, damping * res.ru, damping * res.rd, damping * res.rg,
573 (-x.cwiseProduct(s)).array() + (gamma * mu) - Delta_x.cwiseProduct(Delta_s).array(),
574 (-v.cwiseProduct(w)).array() + (gamma * mu) - Delta_v.cwiseProduct(Delta_w).array(),
575 (-tau * kappa) + (gamma * mu) - Delta_tau * Delta_kappa);
576
577 alpha = max_alpha(x, Delta_x, v, Delta_v, s, Delta_s, w, Delta_w, tau, Delta_tau, kappa, Delta_kappa);
578
579 // High order corrections like Tulip
580 while ((ncor <= 3) && (alpha < 0.9995)) {
581 ncor += 1;
582 alpha_ = std::min(1.0, 2.0 * alpha);
583
584 mu_l = beta * mu * gamma;
585 mu_u = gamma * mu / beta;
586
587 // Perform in-place updates and avoid creating temporaries
588 xs = x + alpha_ * Delta_x;
589 xs.array() *= (s.array() + alpha_ * Delta_s.array());
590
591 vw = v + alpha_ * Delta_v;
592 vw.array() *= (w.array() + alpha_ * Delta_w.array());
593
594 // Use select only once to minimize temporary objects
595 t_xs = (xs.array() < mu_l).select(mu_l - xs.array(), (xs.array() > mu_u).select(mu_u - xs.array(), 0));
596 t_vw = (vw.array() < mu_l).select(mu_l - vw.array(), (vw.array() > mu_u).select(mu_u - vw.array(), 0));
597
598 // Direct calculation for taukappa and t0
599 taukappa = (tau + alpha_ * Delta_tau) * (kappa + alpha_ * Delta_kappa);
600 t0 = std::clamp(taukappa, mu_l, mu_u) - taukappa;
601
602 // Calculate sum_correction and subtract in-place
603 double sum_correction = (t_xs.sum() + t_vw.sum() + t0) / (nv + nu + 1);
604 t_xs.array() -= sum_correction;
605 t_vw.array() -= sum_correction;
606 t0 -= sum_correction;
607
608 Delta_x_c = Delta_x;
609 Delta_lambda_c = Delta_lambda;
610 Delta_w_c = Delta_w;
611 Delta_s_c = Delta_s;
612 Delta_v_c = Delta_v;
613 Delta_tau_c = Delta_tau;
614 Delta_kappa_c = Delta_kappa;
615
616 solve_newton_system(Delta_x_c, Delta_lambda_c, Delta_w_c, Delta_s_c, Delta_v_c, Delta_tau_c, Delta_kappa_c,
617 ls, theta_vw, b, c, ubi, ubv, delta_x, delta_y, delta_z, delta_0, x, lambda, w, s, v,
618 tau, kappa, Eigen::VectorXd::Zero(res.rp.size()), Eigen::VectorXd::Zero(res.ru.size()),
619 Eigen::VectorXd::Zero(res.rd.size()), 0, -t_xs, -t_vw, -t0);
620
621 alpha_c = max_alpha(x, Delta_x_c, v, Delta_v_c, s, Delta_s_c, w, Delta_w_c, tau, Delta_tau_c, kappa,
622 Delta_kappa_c);
623
624 if (alpha_c > alpha_) {
625 Delta_x = Delta_x_c;
626 Delta_lambda = Delta_lambda_c;
627 Delta_w = Delta_w_c;
628 Delta_s = Delta_s_c;
629 Delta_v = Delta_v_c;
630 Delta_tau = Delta_tau_c;
631 Delta_kappa = Delta_kappa_c;
632 alpha = alpha_c;
633 }
634
635 if (alpha_c < 1.1 * alpha_) { break; }
636 }
637
638 alpha *= 0.9995;
639
640 // Update iterates
641 x += alpha * Delta_x;
642 lambda += alpha * Delta_lambda;
643 s += alpha * Delta_s;
644 v += alpha * Delta_v;
645 w += alpha * Delta_w;
646 tau += alpha * Delta_tau;
647 kappa += alpha * Delta_kappa;
648 }
649
650 int free_var = 0;
651 double inv_tau = 1.0 / tau;
652
653 Eigen::VectorXd original_x(As.cols());
654 for (int j = 0; j < lo.size(); ++j) {
655 double l = lo[j];
656 double h = hi[j];
657
658 if (l == -infty && h == infty) {
659 original_x[j] = (x[j + free_var] - x[nv_orig + free_var]) * inv_tau;
660 free_var += 1;
661 } else if (std::isfinite(l) && std::isfinite(h)) {
662 original_x[j] = l + x[j] * inv_tau;
663 } else if (l == -infty && std::isfinite(h)) {
664 original_x[j] = h - x[j] * inv_tau;
665 } else if (std::isfinite(l) && h == infty) {
666 original_x[j] = l + x[j] * inv_tau;
667 }
668 }
669
670 double objetivo = cs.dot(original_x);
671 lambda = lambda * inv_tau;
672
673 double dual_obj = b.dot(lambda);
674
675 // convert lambda to std::vector<double>
676 std::vector<double> lambda_vec(lambda.data(), lambda.data() + lambda.size());
677 std::vector<double> original_x_vec(original_x.data(), original_x.data() + original_x.size());
678
679 dual_vals = lambda_vec;
680 primal_vals = original_x_vec;
681 objVal = objetivo;
682
683 // return std::make_tuple(x, lambda, s, objetivo);
684 return std::make_tuple(objetivo, dual_obj, original_x_vec, lambda_vec);
685}

799void IPSolver::update_linear_solver(SparseSolver &ls, const Eigen::VectorXd &theta, const Eigen::VectorXd &regP,
800 const Eigen::VectorXd &regD) {
801 // Update internal data
802 ls.theta = theta;
803 ls.regP = regP;
804 ls.regD = regD;
805
806#ifdef AUGMENTED
807 // Update S. S is stored as upper-triangular and only its diagonal changes.
808 Eigen::VectorXd combinedValues(ls.n + ls.m);
809 combinedValues.head(ls.n) = -theta - regP;
810 combinedValues.tail(ls.m) = regD;
811
812 // Efficiently update diagonal elements
813 for (int i = 0; i < combinedValues.size(); i++) { ls.S.coeffRef(i, i) = combinedValues[i]; }
814
815 // Refactorize
816 auto S = ls.S;
817 ls.factorizeMatrix(S);
818#else
819
820 // define lhs for normal equations
821 Eigen::SparseMatrix<double> lhs(ls.n + ls.m, ls.n + ls.m);
822 // define lhs as A (\Theta^{-1} + R_{p})^{-1} A^{\top} + R_{d}
823 Eigen::VectorXd d = 1.0 / (ls.theta.array() + ls.regP.array());
824 // set rhs as \xi_{p} + A (Θ^{-1} + R_{p})^{-1} \xi_{d}
825 Eigen::MatrixXd dDense = d.asDiagonal();
826 Eigen::SparseMatrix<double> dSparse = dDense.sparseView();
827
828 Eigen::MatrixXd regDDense = regD.asDiagonal();
829 Eigen::SparseMatrix<double> regDSparse = regDDense.sparseView();
830 Eigen::SparseMatrix<double> AD = ls.A * dSparse;
831 Eigen::SparseMatrix<double> ADA = AD * ls.A.transpose();
832 lhs = ADA + regDSparse;
833 ls.factorizeMatrix(lhs);
834
835#endif
836}

843void IPSolver::start_linear_solver(SparseSolver &ls, const Eigen::SparseMatrix<double> A) {
844 ls.A = A;
845 ls.m = A.rows();
846 ls.n = A.cols();
847 // print ls.A size
848
849#ifdef AUGMENTED
850 ls.theta = Eigen::VectorXd::Ones(ls.n);
851 ls.regP = Eigen::VectorXd::Ones(ls.n);
852 ls.regD = Eigen::VectorXd::Ones(ls.m);
853
854 Eigen::SparseMatrix<double> topRight = ls.A.transpose();
855 Eigen::SparseMatrix<double> bottomLeft = ls.A;
856 Eigen::SparseMatrix<double> topLeft = convertToSparseDiagonal(-ls.theta - ls.regP);
857 Eigen::SparseMatrix<double> bottomRight = convertToSparseDiagonal(ls.regD);
858
859 // S_ is known, reserve space for it
860 Eigen::SparseMatrix<double> S_(ls.n + ls.m, ls.n + ls.m);
861
862 // Reserving space for tripletList
863 int estimated_nonzeros = topLeft.nonZeros() + 2 * topRight.nonZeros() + bottomRight.nonZeros();
864 std::vector<Eigen::Triplet<double>> tripletList;
865 tripletList.reserve(estimated_nonzeros);
866
867 // Insert topLeft, topRight, bottomLeft, bottomRight matrices
868 auto insertBlock = [&](const Eigen::SparseMatrix<double> &block, int startRow, int startCol) {
869 for (int k = 0; k < block.outerSize(); ++k) {
870 for (Eigen::SparseMatrix<double>::InnerIterator it(block, k); it; ++it) {
871 tripletList.emplace_back(it.row() + startRow, it.col() + startCol, it.value());
872 }
873 }
874 };
875
876 insertBlock(topLeft, 0, 0);
877 insertBlock(topRight, 0, ls.n);
878 insertBlock(bottomLeft, ls.n, 0);
879 insertBlock(bottomRight, ls.n, ls.n);
880
881 // Finally, set the values from the triplets
882 S_.setFromTriplets(tripletList.begin(), tripletList.end());
883 // S_.makeCompressed();
884
885 ls.S = S_;
886 // Factorize
887 ls.factorizeMatrix(ls.S);
888#endif
889}

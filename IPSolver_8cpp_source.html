21ScalingFactors IPSolver::ruiz_scaling(Eigen::SparseMatrix<double> &A,
22 Eigen::VectorXd &b, Eigen::VectorXd &c,
23 Eigen::VectorXd &lo,
24 Eigen::VectorXd &hi) {
25 const int m = A.rows();
26 const int n = A.cols();
27 const int n_bounds = lo.size();
28
29 ScalingFactors factors;
30 factors.row_scaling = Eigen::VectorXd::Ones(m);
31 factors.col_scaling = Eigen::VectorXd::Ones(n);
32
33 // Parameters
34 const int max_iter = 10;
35 const double thresh = 1e-3; // Convergence threshold
36 const double alpha = 0.1; // Step size for updates
37
38 // Store initial norms for reference
39 double init_objective_norm = c.norm();
40 double init_constraint_norm = b.norm();
41
42 // Compute column norms for A and objective scaling
43 Eigen::VectorXd col_norms = Eigen::VectorXd::Zero(n);
44 Eigen::VectorXd row_norms = Eigen::VectorXd::Zero(m);
45
46 // Initial norms computation
47 for (int k = 0; k < A.outerSize(); ++k) {
48 for (Eigen::SparseMatrix<double>::InnerIterator it(A, k); it; ++it) {
49 double val = std::abs(it.value());
50 col_norms(it.col()) += val;
51 row_norms(it.row()) += val;
52 }
53 }
54
55 // Initial scaling to roughly same magnitude
56 double mean_norm =
57 (col_norms.sum() / (n * m) + std::abs(c.mean()) + std::abs(b.mean())) /
58 3.0;
59 if (mean_norm > 0) {
60 double log_scale = -std::log10(mean_norm);
61 double scale = std::pow(10.0, std::round(log_scale));
62
63 // Apply initial scaling
64 if (scale != 1.0 && scale > 1e-4 && scale < 1e4) {
65 A *= scale;
66 b *= scale;
67 c *= scale;
68
69 factors.row_scaling *= std::sqrt(scale);
70 factors.col_scaling *= std::sqrt(scale);
71 }
72 }
73
74 // Iterative equilibration
75 for (int iter = 0; iter < max_iter; iter++) {
76 // Compute gradient of the scaled problem
77 Eigen::VectorXd grad_scaling = Eigen::VectorXd::Zero(n);
78
79 // Consider objective contribution
80 grad_scaling = c.cwiseAbs();
81
82 // Add constraint contributions
83 for (int k = 0; k < A.outerSize(); ++k) {
84 for (Eigen::SparseMatrix<double>::InnerIterator it(A, k); it;
85 ++it) {
86 grad_scaling(it.col()) += std::abs(it.value());
87 }
88 }
89
90 // Compute scaling updates
91 double max_update = 0.0;
92 for (int j = 0; j < n; j++) {
93 if (grad_scaling(j) > 0) {
94 double target = 1.0;
95 double current = grad_scaling(j);
96 double log_ratio = std::log(target / current);
97
98 // Limit the update
99 double update =
100 std::min(std::max(log_ratio * alpha, -0.1), 0.1);
101 double scale = std::exp(update);
102
103 // Apply column scaling
104 if (scale != 1.0) {
105 for (Eigen::SparseMatrix<double>::InnerIterator it(A, j);
106 it; ++it) {
107 A.coeffRef(it.row(), j) *= scale;
108 }
109 c(j) *= scale;
110 factors.col_scaling(j) *= scale;
111
112 max_update = std::max(max_update, std::abs(update));
113 }
114 }
115 }
116
117 // Check convergence
118 if (max_update < thresh) break;
119 }
120
121 // Scale bounds
122 for (int j = 0; j < n_bounds && j < n; ++j) {
123 if (std::isfinite(lo(j))) {
124 lo(j) *= factors.col_scaling(j);
125 }
126 if (std::isfinite(hi(j))) {
127 hi(j) *= factors.col_scaling(j);
128 }
129 }
130
131 // Compute and print statistics
132 double scaled_objective_norm = c.norm();
133 double scaled_constraint_norm = b.norm();
134
135 std::cout << "\nGradient-based scaling statistics:\n";
136 std::cout << "Objective norm - Before: " << init_objective_norm
137 << ", After: " << scaled_objective_norm << "\n";
138 std::cout << "Constraint norm - Before: " << init_constraint_norm
139 << ", After: " << scaled_constraint_norm << "\n";
140 std::cout << "Column scaling range: [" << factors.col_scaling.minCoeff()
141 << ", " << factors.col_scaling.maxCoeff() << "]\n\n";
142
143 return factors;
144}

188void IPSolver::convert_to_standard_form(
189 const Eigen::SparseMatrix<double> &A, const Eigen::VectorXd &b,
190 const Eigen::VectorXd &c, const Eigen::VectorXd &lb,
191 const Eigen::VectorXd &ub, const Eigen::VectorXd &sense,
192 Eigen::SparseMatrix<double> &As, Eigen::VectorXd &bs, Eigen::VectorXd &cs) {
193 constexpr double infty = std::numeric_limits<double>::infinity();
194 const int n = A.rows();
195 const int m = A.cols();
196
197 if (b.size() != n || c.size() != m) {
198 fmt::print("b.size(): {}, n: {}, c.size(): {}, m: {}\n", b.size(), n,
199 c.size(), m);
200 throw std::invalid_argument(
201 "Size of b or c does not match the matrix A dimensions.");
202 }
203
204 Eigen::VectorXd lo = lb;
205 Eigen::VectorXd hi = ub;
206
207 int n_free = 0, n_ubounds = 0, nv = A.cols();
208
209 // Precompute bound categories
210 std::vector<bool> is_free(lo.size()), is_bounded(lo.size()),
211 is_upper(lo.size()), is_lower(lo.size());
212 for (int i = 0; i < lo.size(); ++i) {
213 if (lo[i] == -infty && hi[i] == infty) {
214 is_free[i] = true;
215 ++n_free;
216 } else if (std::isfinite(lo[i]) && std::isfinite(hi[i])) {
217 is_bounded[i] = true;
218 ++n_ubounds;
219 } else if (lo[i] == -infty && std::isfinite(hi[i])) {
220 is_upper[i] = true;
221 } else if (std::isfinite(lo[i]) && hi[i] == infty) {
222 is_lower[i] = true;
223 } else {
224 throw std::runtime_error("unexpected bounds");
225 }
226 }
227
228 // Precompute slack variables and reserve memory
229 int num_slacks = n - sense.sum();
230 cs.resize(c.size() + n_free + num_slacks);
231 cs.setZero();
232 cs.head(m) = c;
233
234 bs = b; // Direct assignment
235
236 std::vector<Eigen::Triplet<double>> triplets;
237 triplets.reserve(A.nonZeros() +
238 num_slacks); // Reserve space to avoid reallocations
239
240 std::vector<int> ind_ub;
241 std::vector<double> val_ub;
242
243 auto lb_extended = lb;
244 auto ub_extended = ub;
245 lb_extended.conservativeResize(nv + num_slacks + n_free);
246 ub_extended.conservativeResize(nv + num_slacks + n_free);
247 lb_extended.tail(num_slacks + n_free).setZero();
248 ub_extended.tail(num_slacks + n_free).setConstant(1.0);
249
250 int free = 0, ubi = 0;
251 for (int j = 0; j < lo.size(); ++j) {
252 double l = lo[j];
253 double h = hi[j];
254
255 for (Eigen::SparseMatrix<double>::InnerIterator it(A, j); it; ++it) {
256 int i = it.row();
257 double v = it.value();
258
259 if (is_free[j]) {
260 cs[j] = c[j];
261 cs[nv + free] = -c[j];
262 triplets.emplace_back(i, j, v);
263 triplets.emplace_back(i, nv + free, -v);
264 lb_extended[nv + free] = lb[j];
265 ub_extended[nv + free] = ub[j];
266 ++free;
267 } else if (is_bounded[j]) {
268 cs[j] = c[j];
269 bs[i] -= v * l;
270 triplets.emplace_back(i, j, v);
271 ind_ub.push_back(j);
272 val_ub.push_back(h - l);
273 ++ubi;
274 } else if (is_upper[j]) {
275 cs[j] = -c[j];
276 bs[i] -= (-v * h);
277 triplets.emplace_back(i, j, -v);
278 } else if (is_lower[j]) {
279 cs[j] = c[j];
280 bs[i] -= v * l;
281 triplets.emplace_back(i, j, v);
282 }
283 }
284 }
285
286 // Add slack variables
287 int slack_counter = 0;
288 for (int i = 0; i < sense.size(); ++i) {
289 if (sense(i) == 0) {
290 // print nv + n_free + slack_counter
291 triplets.emplace_back(i, nv + n_free + slack_counter, 1.0);
292 lb_extended[nv + n_free + slack_counter] = 0.0;
293 ub_extended[nv + n_free + slack_counter] = infty;
294 ++slack_counter;
295 }
296 }
297
298 // Construct sparse matrix As
299 As.resize(bs.size(), cs.size());
300 As.setFromTriplets(triplets.begin(), triplets.end());
301 As.makeCompressed();
302
303 n_slacks = num_slacks;
304}

314void IPSolver::update_residuals(
315 Residuals &res, const Eigen::VectorXd &x, const Eigen::VectorXd &lambda,
316 const Eigen::VectorXd &s, const Eigen::VectorXd &v,
317 const Eigen::VectorXd &w, const Eigen::SparseMatrix<double> &A,
318 const Eigen::VectorXd &b, const Eigen::VectorXd &c,
319 const Eigen::VectorXd &ubv, const Eigen::VectorXi &ubi, double tau,
320 double kappa) {
321 // Pre-compute tau-scaled vectors to avoid repeated multiplications
322 const Eigen::VectorXd tau_b = tau * b;
323 const Eigen::VectorXd tau_c = tau * c;
324
325 // Calculate primal residual (rp)
326 res.rp.noalias() = tau_b - A * x;
327 res.rpn = res.rp.norm();
328
329 // Update residual for upper bounds (ru)
330 res.ru = -v; // Direct assignment for better performance
331
332 // Use raw pointers for faster access in the loop
333 double *ru_data = res.ru.data();
334 const double *ubv_data = ubv.data();
335 const double *x_data = x.data();
336 const int *ubi_data = ubi.data();
337 const int ubi_size = ubi.size();
338
339 // Parallelize the loop for updating ru
340 for (int i = 0; i < ubi_size; ++i) {
341 const int idx = ubi_data[i];
342 ru_data[idx] += tau * ubv_data[i] - x_data[idx];
343 }
344
345 // Calculate dual residual (rd)
346 res.rd.noalias() = tau_c - A.transpose() * lambda - s;
347
348 // Update rd with w values
349 double *rd_data = res.rd.data();
350 const double *w_data = w.data();
351
352 // Parallelize the loop for updating rd
353 for (int i = 0; i < ubi_size; ++i) {
354 const int idx = ubi_data[i];
355 rd_data[idx] += w_data[i];
356 }
357
358 // Calculate gap residual (rg) using dot products
359 const double cx = c.dot(x);
360 const double blambda = b.dot(lambda);
361 const double wubv = ubv.dot(w);
362
363 res.rg = kappa + cx - blambda + wubv;
364 res.rgn = std::abs(res.rg);
365}

375 SparseSolver &ls,
376 const Eigen::VectorXd &xi_p,
377 const Eigen::VectorXd &xi_d) {

380 xi << xi_d, xi_p;
381
382 // Solve augmented system
383 Eigen::VectorXd d = ls.solve(xi);
384

386 dx = d.head(xi_d.size()); // Gets the first n elements
387 dy = d.tail(xi_p.size()); // Gets the last m elements

390void IPSolver::solve_augsys(Eigen::VectorXd &delta_x, Eigen::VectorXd &delta_y,
391 Eigen::VectorXd &delta_z, SparseSolver &ls,
392 const Eigen::VectorXd &theta_vw,
393 const Eigen::VectorXi &ubi,
394 const Eigen::VectorXd &xi_p,
395 const Eigen::VectorXd &xi_d,
396 const Eigen::VectorXd &xi_u) {
397 // Static allocation for frequently used vectors
398 static Eigen::VectorXd xi_d_mod;
399 if (xi_d_mod.size() != xi_d.size()) {
400 xi_d_mod.resize(xi_d.size());
401 }
402
403 // Use direct memory copy instead of assignment
404 std::memcpy(xi_d_mod.data(), xi_d.data(), xi_d.size() * sizeof(double));
405
406 // Get raw pointers for faster access
407 double *xi_d_data = xi_d_mod.data();
408 const double *theta_vw_data = theta_vw.data();
409 const double *xi_u_data = xi_u.data();
410 const int *ubi_data = ubi.data();
411 const int ubi_size = ubi.size();
412
413// Use SIMD vectorization for the update loop
414#pragma omp simd
415 for (int i = 0; i < ubi_size; ++i) {
416 const int idx = ubi_data[i];
417 xi_d_data[idx] -= xi_u_data[i] * theta_vw_data[i];
418 }
419
420 // Solve augmented system with optimized xi_d_mod
421 solve_augmented_system(delta_x, delta_y, ls, xi_p, xi_d_mod);
422
423 // Update delta_z efficiently
424 delta_z.resize(ubi_size);
425 double *delta_z_data = delta_z.data();
426 const double *delta_x_data = delta_x.data();
427
428// Use SIMD vectorization for final update
429#pragma omp simd
430 for (int i = 0; i < ubi_size; ++i) {
431 const int idx = ubi_data[i];
432 delta_z_data[i] = (delta_x_data[idx] - xi_u_data[i]) * theta_vw_data[i];
433 }
434}
443void IPSolver::solve_newton_system(
444 Eigen::VectorXd &Delta_x, Eigen::VectorXd &Delta_lambda,
445 Eigen::VectorXd &Delta_w, Eigen::VectorXd &Delta_s,
446 Eigen::VectorXd &Delta_v, double &Delta_tau, double &Delta_kappa,
447 SparseSolver &ls, const Eigen::VectorXd &theta_vw, const Eigen::VectorXd &b,
448 const Eigen::VectorXd &c, const Eigen::VectorXi &ubi,
449 const Eigen::VectorXd &ubv, const Eigen::VectorXd &delta_x,
450 const Eigen::VectorXd &delta_y, const Eigen::VectorXd &delta_w,
451 double delta_0, const Eigen::VectorXd &iter_x,
452 const Eigen::VectorXd &iter_lambda, const Eigen::VectorXd &iter_w,
453 const Eigen::VectorXd &iter_s, const Eigen::VectorXd &iter_v,
454 double iter_tau, double iter_kappa, const Eigen::VectorXd &xi_p,
455 const Eigen::VectorXd &xi_u, const Eigen::VectorXd &xi_d, double xi_g,
456 const Eigen::VectorXd &xi_xs, const Eigen::VectorXd &xi_vw,
457 double xi_tau_kappa) {
458 Eigen::VectorXd xi_d_copy =
459 xi_d - (xi_xs.array() / iter_x.array()).matrix();
460 Eigen::VectorXd xi_u_copy =
461 xi_u - (xi_vw.array() / iter_w.array()).matrix();
462
463 // Pre-compute frequently used values
464 const double inv_tau = 1.0 / iter_tau;
465 const double inv_kappa = 1.0 / iter_kappa;
466
467 // Pre-allocate vectors to avoid reallocations
468 static Eigen::VectorXd xi_d_mod;
469 if (xi_d_mod.size() != xi_d.size()) {
470 xi_d_mod.resize(xi_d.size());
471 }
472
473// Use vectorized operations for division
474#pragma omp parallel sections
475 {
476#pragma omp section
477 {
478 // Compute xi_d_mod = xi_d - xi_xs./iter_x efficiently
479 xi_d_mod = xi_d.array() - (xi_xs.array() / iter_x.array());
480 }
481#pragma omp section
482 {
483 // Pre-compute xi_u_copy = xi_u - xi_vw./iter_w
484 Eigen::VectorXd xi_u_copy =
485 xi_u - (xi_vw.array() / iter_w.array()).matrix();
486
487 // Call solve_augsys with pre-computed values
488 solve_augsys(Delta_x, Delta_lambda, Delta_w, ls, theta_vw, ubi,
489 xi_p, xi_d_mod, xi_u_copy);
490 }
491 }
492
493 // Compute Delta_tau using efficient dot products
494 const double c_dot_Delta_x = c.dot(Delta_x);
495 const double b_dot_Delta_lambda = b.dot(Delta_lambda);
496 const double ubv_dot_Delta_w = ubv.dot(Delta_w);
497
498 Delta_tau = (xi_g + xi_tau_kappa * inv_tau + c_dot_Delta_x -
499 b_dot_Delta_lambda + ubv_dot_Delta_w) /
500 delta_0;
501 Delta_kappa = (xi_tau_kappa - iter_kappa * Delta_tau) * inv_tau;
502
503// Update vectors using vectorized operations
504#pragma omp parallel sections
505 {
506#pragma omp section
507 {
508 Delta_x.array() += Delta_tau * delta_x.array();
509 Delta_lambda.array() += Delta_tau * delta_y.array();
510 Delta_w.array() += Delta_tau * delta_w.array();
511 }
512#pragma omp section
513 {
514 // Compute Delta_s and Delta_v using vectorized operations
515 Delta_s = (xi_xs.array() - iter_s.array() * Delta_x.array()) /
516 iter_x.array();
517 Delta_v = (xi_vw.array() - iter_v.array() * Delta_w.array()) /
518 iter_w.array();
519 }
520 }

532double IPSolver::max_alpha_single(const Eigen::VectorXd &v,
533 const Eigen::VectorXd &dv) {
534 double alpha = std::numeric_limits<double>::infinity();
535
536 // Get direct access to data
537 const double *v_data = v.data();
538 const double *dv_data = dv.data();
539 const Eigen::Index size = v.size();
540
541 // Process in chunks for better cache utilization
542 constexpr Eigen::Index CHUNK_SIZE = 4;
543 Eigen::Index i = 0;
544
545 // Process main chunks
546 for (; i + CHUNK_SIZE <= size; i += CHUNK_SIZE) {
547 if (dv_data[i] < 0) alpha = std::min(alpha, -v_data[i] / dv_data[i]);
548 if (dv_data[i + 1] < 0)
549 alpha = std::min(alpha, -v_data[i + 1] / dv_data[i + 1]);
550 if (dv_data[i + 2] < 0)
551 alpha = std::min(alpha, -v_data[i + 2] / dv_data[i + 2]);
552 if (dv_data[i + 3] < 0)
553 alpha = std::min(alpha, -v_data[i + 3] / dv_data[i + 3]);
554 }
555
556 // Handle remaining elements
557 for (; i < size; ++i) {
558 if (dv_data[i] < 0) {
559 alpha = std::min(alpha, -v_data[i] / dv_data[i]);
560 }
561 }
562
563 return alpha;
564}
576double IPSolver::max_alpha(const Eigen::VectorXd &x, const Eigen::VectorXd &dx,
577 const Eigen::VectorXd &v, const Eigen::VectorXd &dv,
578 const Eigen::VectorXd &s, const Eigen::VectorXd &ds,
579 const Eigen::VectorXd &w, const Eigen::VectorXd &dw,
580 double tau, double dtau, double kappa,
581 double dkappa) {
582 // Initialize alpha with first scalar check
583 double alpha = (dtau < 0) ? (-tau / dtau) : 1.0;
584
585 // Check kappa condition and update alpha if needed
586 if (dkappa < 0) {
587 double alpha_kappa = -kappa / dkappa;
588 if (alpha_kappa < alpha) {
589 alpha = alpha_kappa;
590 }
591 }
592
593 // Efficiently compute and update minimum alpha for each vector pair
594 double potential_alpha = max_alpha_single(x, dx);
595 if (potential_alpha < alpha) alpha = potential_alpha;
596
597 potential_alpha = max_alpha_single(v, dv);
598 if (potential_alpha < alpha) alpha = potential_alpha;
599
600 potential_alpha = max_alpha_single(s, ds);
601 if (potential_alpha < alpha) alpha = potential_alpha;
602
603 potential_alpha = max_alpha_single(w, dw);
604 if (potential_alpha < alpha) alpha = potential_alpha;
605
606 return alpha;
607}

714 // Adaptive tolerance parameters
715 const double initial_tol = 1e-6; // Initial tolerance
716 const double min_tol = 1e-10; // Minimum tolerance
717 const double scale_factor = 0.1; // Scaling factor for adaptive tolerance
718
719 for (int k = 0; k < max_iter; ++k) {
720 ncor = 0;
721 beta = 0.1;
722
723 // Batch zero initialization
724 delta_x.setZero();
725 delta_y.setZero();
726 delta_z.setZero();
727 Delta_x.setZero();
728 Delta_lambda.setZero();
729 Delta_w.setZero();
730 Delta_s.setZero();
731 Delta_v.setZero();
732 Delta_tau = Delta_kappa = 0.0;
733
734 // Update residuals without vbv and vbi
735 update_residuals(res, x, lambda, s, v, w, A, b, c, ubv, ubi, tau,
736 kappa);
737 mu = (tau * kappa + x.dot(s)) / (n + nu + 1.0);
738
739 // Efficient residual calculations
740 double rp_norm = res.rp.lpNorm<Eigen::Infinity>();
741 double ru_norm = res.ru.lpNorm<Eigen::Infinity>();
742 double rd_norm = res.rd.lpNorm<Eigen::Infinity>();
743 double b_norm = b.lpNorm<Eigen::Infinity>();
744 double ubv_norm = ubv.lpNorm<Eigen::Infinity>();
745 double c_norm = c.lpNorm<Eigen::Infinity>();
746
747 // Adaptive tolerance computation
748 double adaptive_tol = std::max(min_tol, initial_tol * scale_factor);
749
750 // Combined residual computations
751 bl_dot_lambda = b.dot(lambda) - ubv.dot(w);
752 _p = std::fmax(rp_norm / (tau * (1.0 + b_norm)),
753 ru_norm / (tau * (1.0 + ubv_norm)));
754 _d = rd_norm / (tau * (1.0 + c_norm));
755 _g = std::abs(c.dot(x) - bl_dot_lambda) /
756 (tau + std::abs(bl_dot_lambda));
757
758 // Save interior solution if residuals are small
759 if (!saved_interior_solution_bool &&
760 (_d <= adaptive_tol && _g <= adaptive_tol * 2)) {
761 save_interior_solution(x, lambda, w, s, v, tau, kappa);
762 saved_interior_solution_bool = true;
763 warm_start = true;
764 }
765
766 // Check convergence with adaptive tolerance
767 if (_p <= adaptive_tol && _d <= adaptive_tol && _g <= tol) break;
768
769 // Update adaptive tolerance for the next iteration
770 adaptive_tol = std::max(min_tol, adaptive_tol * scale_factor);
771
772 // Efficient scaling computations
773 theta_vw = w.cwiseQuotient(v);
774 theta_xs = s.cwiseQuotient(x);
775 for (int i = 0; i < ubi.size(); ++i) {
776 theta_xs[ubi[i]] += theta_vw[i];
777 }
778
779 // Update regularizations dynamically
780 for (int attempt = 0; attempt < 3; ++attempt) {
781 // Adjust regularization based on solver performance
782 regP = (regP / 10.0).cwiseMax(r_min);
783 regD = (regD / 10.0).cwiseMax(r_min);
784 regG = std::max(r_min, regG / 10.0);
785
786 // Update linear solver with new regularization
787 if (update_linear_solver(ls, theta_xs, regP, regD) == 0) break;
788
789 // Increase regularization if solver fails
790 regP *= 100.0;
791 regD *= 100.0;
792 regG *= 100.0;
793 }
794
795 solve_augsys(delta_x, delta_y, delta_z, ls, theta_vw, ubi, b, c, ubv);
796 delta_0 = regG + kappa / tau - delta_x.dot(c) + delta_y.dot(b) -
797 delta_z.dot(ubv);
798
799 // First Newton solve
800 solve_newton_system(Delta_x, Delta_lambda, Delta_w, Delta_s, Delta_v,
801 Delta_tau, Delta_kappa, ls, theta_vw, b, c, ubi,
802 ubv, delta_x, delta_y, delta_z, delta_0, x, lambda,
803 w, s, v, tau, kappa, res.rp, res.ru, res.rd, res.rg,
804 -x.cwiseProduct(s), -v.cwiseProduct(w),
805 -tau * kappa);
806
807 alpha = max_alpha(x, Delta_x, v, Delta_v, s, Delta_s, w, Delta_w, tau,
808 Delta_tau, kappa, Delta_kappa);
809
810 oneMinusAlpha = 1.0 - alpha;
811 gamma = std::fmax(
812 oneMinusAlpha * oneMinusAlpha * std::fmin(beta, oneMinusAlpha),
813 0.1);
814 damping = 1.0 - gamma;
815
816 // Second Newton solve with damping
817 solve_newton_system(
818 Delta_x, Delta_lambda, Delta_w, Delta_s, Delta_v, Delta_tau,
819 Delta_kappa, ls, theta_vw, b, c, ubi, ubv, delta_x, delta_y,
820 delta_z, delta_0, x, lambda, w, s, v, tau, kappa, damping * res.rp,
821 damping * res.ru, damping * res.rd, damping * res.rg,
822 (-x.cwiseProduct(s)).array() + (gamma * mu) -
823 Delta_x.cwiseProduct(Delta_s).array(),
824 (-v.cwiseProduct(w)).array() + (gamma * mu) -
825 Delta_v.cwiseProduct(Delta_w).array(),
826 (-tau * kappa) + (gamma * mu) - Delta_tau * Delta_kappa);
827
828 alpha = max_alpha(x, Delta_x, v, Delta_v, s, Delta_s, w, Delta_w, tau,
829 Delta_tau, kappa, Delta_kappa);
830
831 // High order corrections
832 while (ncor <= 2 && alpha < 0.9995) {
833 ncor++;
834 alpha_ = std::min(1.0, 2.0 * alpha);
835 mu_l = beta * mu * gamma;
836 mu_u = gamma * mu / beta;
837
838 // Compute products and corrections
839 xs = x + alpha_ * Delta_x;
840 xs.array() *= (s + alpha_ * Delta_s).array();
841 vw = v + alpha_ * Delta_v;
842 vw.array() *= (w + alpha_ * Delta_w).array();
843
844 t_xs =
845 (xs.array() < mu_l)
846 .select(mu_l - xs.array(),
847 (xs.array() > mu_u).select(mu_u - xs.array(), 0.0));
848 t_vw =
849 (vw.array() < mu_l)
850 .select(mu_l - vw.array(),
851 (vw.array() > mu_u).select(mu_u - vw.array(), 0.0));
852
853 taukappa =
854 (tau + alpha_ * Delta_tau) * (kappa + alpha_ * Delta_kappa);
855 t0 = std::clamp(taukappa, mu_l, mu_u) - taukappa;
856
857 const double sum_correction =
858 (t_xs.sum() + t_vw.sum() + t0) / (nv + nu + 1);
859 t_xs.array() -= sum_correction;
860 t_vw.array() -= sum_correction;
861 t0 -= sum_correction;
862
863 // Save current directions
864 Delta_x_c = Delta_x;
865 Delta_lambda_c = Delta_lambda;
866 Delta_w_c = Delta_w;
867 Delta_s_c = Delta_s;
868 Delta_v_c = Delta_v;
869 Delta_tau_c = Delta_tau;
870 Delta_kappa_c = Delta_kappa;
871
872 // Solve correction system
873 solve_newton_system(
874 Delta_x_c, Delta_lambda_c, Delta_w_c, Delta_s_c, Delta_v_c,
875 Delta_tau_c, Delta_kappa_c, ls, theta_vw, b, c, ubi, ubv,
876 delta_x, delta_y, delta_z, delta_0, x, lambda, w, s, v, tau,
877 kappa, Eigen::VectorXd::Zero(res.rp.size()),
878 Eigen::VectorXd::Zero(res.ru.size()),
879 Eigen::VectorXd::Zero(res.rd.size()), 0, -t_xs, -t_vw, -t0);
880
881 alpha_c =
882 max_alpha(x, Delta_x_c, v, Delta_v_c, s, Delta_s_c, w,
883 Delta_w_c, tau, Delta_tau_c, kappa, Delta_kappa_c);
884
885 if (alpha_c > alpha_) {
886 Delta_x = Delta_x_c;
887 Delta_lambda = Delta_lambda_c;
888 Delta_w = Delta_w_c;
889 Delta_s = Delta_s_c;
890 Delta_v = Delta_v_c;
891 Delta_tau = Delta_tau_c;
892 Delta_kappa = Delta_kappa_c;
893 alpha = alpha_c;
894 }
895
896 if (alpha_c < 1.1 * alpha_) break;
897 }
898
899 // Final update step
900 alpha *= 0.9995;
901 x += alpha * Delta_x;
902 lambda += alpha * Delta_lambda;
903 s += alpha * Delta_s;
904 v += alpha * Delta_v;
905 w += alpha * Delta_w;
906 tau += alpha * Delta_tau;
907 kappa += alpha * Delta_kappa;
908 }
909
910 // Final solution recovery
911 const double inv_tau = 1.0 / tau;
912 Eigen::VectorXd original_x(As.cols());
913 int free_var = 0;
914
915 for (int j = 0; j < lo.size(); ++j) {
916 const double l = lo[j];
917 const double h = hi[j];
918
919 if (l == -infty && h == infty) {
920 original_x[j] = (x[j + free_var] - x[nv_orig + free_var]) * inv_tau;
921 ++free_var;
922 } else if (std::isfinite(l) && std::isfinite(h)) {
923 original_x[j] = l + x[j] * inv_tau;
924 } else if (l == -infty && std::isfinite(h)) {
925 original_x[j] = h - x[j] * inv_tau;
926 } else if (std::isfinite(l) && h == infty) {
927 original_x[j] = l + x[j] * inv_tau;
928 }
929 }
930
931 // Final objective computation and output
932 objVal = cs.dot(original_x);
933 lambda *= inv_tau;
934 dual_vals.assign(lambda.data(), lambda.data() + lambda.size());
935 primal_vals.assign(original_x.data(),
936 original_x.data() + original_x.size());
937}

64 node.rank = 0;

70 if (scc.empty()) continue;

110 // Find with full path compression.
111 inline int find_and_compress(int x) noexcept {
112 int root = x;
113 // First pass: find the root.
114 while (nodes[root].parent != root) {
115 root = nodes[root].parent;
116 }
117 // Second pass: path compression.
118 int current = x;
119 while (current != root) {
120 int next = nodes[current].parent;
121 nodes[current].parent = root;
122 current = next;
123 }
124 // Update cache.
125 cache.element = x;
126 cache.root = root;
127 return root;
128 }

130 // Union two elements using union by rank and immediate path compression.
131 inline void unite_with_path_compression(int x, int y) noexcept {
132 int root_x = find_and_compress(x);
133 int root_y = find_and_compress(y);
134 if (root_x == root_y) return;
135
136 auto &node_x = nodes[root_x];
137 auto &node_y = nodes[root_y];
138
139 // Make the tree with higher rank the new root.
140 if (node_x.rank < node_y.rank) {
141 std::swap(root_x, root_y);
142 }
143 nodes[root_y].parent = root_x;
144 // Propagate subset index from the new root.
145 nodes[root_y].subset_index = node_x.subset_index;
146 if (node_x.rank == node_y.rank) {
147 node_x.rank++;
148 }
149 // Invalidate the cache.
150 cache.element = -1;
151 }

22template <Direction D>
23inline bool process_disposable_resource(double &new_resource,
24 double initial_resource,
25 double increment, double lb,
26 double ub) {
27 if constexpr (D == Direction::Forward) {
28 new_resource = std::max(initial_resource + increment, lb);
29 // new_resource = roundToTwoDecimalPlaces(new_resource);
30 return numericutils::lte(new_resource, ub);

48template <Direction D>
49inline double process_binary_resource(double increment) {
50 // For binary resource, return 1.0 if increment > 0 in forward direction,
51 // else 0.0, and vice versa in backward direction.
52 return (increment > 0) ? (D == Direction::Forward ? 1.0 : 0.0)
53 : (D == Direction::Forward ? 0.0 : 1.0);
54}

88template <Direction D>
89inline bool process_mtw_resource(double &new_resource, double initial_resource,
90 double increment, const VRPNode &theNode) {
91 if constexpr (D == Direction::Forward) {
92 for (size_t i = 0; i < theNode.mtw_lb.size(); ++i) {
93 new_resource =
94 std::max(initial_resource + increment, theNode.mtw_lb[i]);
95 if (new_resource <= theNode.ub[i]) return true;
96 }
97 } else {
98 for (size_t i = 0; i < theNode.mtw_ub.size(); ++i) {
99 new_resource =
100 std::min(initial_resource - increment, theNode.mtw_ub[i]);
101 if (new_resource >= theNode.lb[i]) return true;
102 }
103 }
104 return false;
105}

107// Main processing function for all resources.
108// new_resources: vector to be updated for the new resource levels.
109// initial_resources: initial values of the resources.
110// gamma: provides resource_increment among other things.
111// theNode: VRPNode with lower/upper bounds and other properties.
112// N: number of resources to process.
113template <Direction D, typename Gamma, typename VRPNode>
114bool BucketGraph::process_all_resources(
115 std::vector<double> &new_resources,
116 const std::array<double, R_SIZE> &initial_resources, const Gamma &gamma,
117 const VRPNode &theNode, size_t N) {
118 for (size_t I = 0; I < N; ++I) {
119 switch (static_cast<ResourceType>(options.resource_type[I])) {
120 [[likely]] case ResourceType::Disposable:
121 if (!process_disposable_resource<D>(
122 new_resources[I], initial_resources[I],
123 gamma.resource_increment[I], theNode.lb[I],
124 theNode.ub[I])) {
125 return false;
126 }
127 break;
128
129 case ResourceType::NonDisposable:
130 if (!process_non_disposable_resource<D>(
131 new_resources[I], initial_resources[I],
132 gamma.resource_increment[I], theNode.lb[I],
133 theNode.ub[I])) {
134 return false;
135 }
136 break;
137
138 case ResourceType::Binary:
139 new_resources[I] =
140 process_binary_resource<D>(gamma.resource_increment[I]);
141 break;
142
143 case ResourceType::Battery:
144 if (!process_battery_resource<D>(
145 new_resources[I], initial_resources[I],
146 theNode.is_station, options.battery_capacity)) {
147 return false;
148 }
149 break;
150
151 case ResourceType::RechargeCount:
152 if (!process_recharge_count<D>(
153 new_resources[I], initial_resources[I],
154 theNode.is_station, options.max_recharges)) {
155 return false;
156 }
157 break;
158
159 case ResourceType::RechargeTime:
160 if (!process_recharge_time<D>(
161 new_resources[I], initial_resources[I],
162 theNode.is_station, options.max_recharge_time)) {
163 return false;
164 }
165 break;
166
167 case ResourceType::MTW:
168 if (!process_mtw_resource<D>(
169 new_resources[I], initial_resources[I],
170 gamma.resource_increment[I], theNode)) {
171 return false;
172 }
173 break;
174 }
175 }
176 return true;

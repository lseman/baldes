62template <Direction D>
63static bool process_mtw(double& new_resource, double initial, double increment,
64 const VRPNode& node, const BucketOptions& opts) {
65 if constexpr (D == Direction::Forward) {
66 const auto adjusted = adjust_resource<D>(initial, increment);
67 for (size_t i = 0; i < node.mtw_lb.size(); ++i) {
68 new_resource = std::max(adjusted, node.mtw_lb[i]);
69 if (check_bounds<D>(new_resource, node.lb[i], node.ub[i]))
70 return true;
71 }
72 } else {
73 const auto adjusted = adjust_resource<D>(initial, increment);
74 for (size_t i = 0; i < node.mtw_ub.size(); ++i) {
75 new_resource = std::min(adjusted, node.mtw_ub[i]);
76 if (check_bounds<D>(new_resource, node.lb[i], node.ub[i]))
77 return true;
78 }
79 }
80 return false;

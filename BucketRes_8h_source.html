31template <Direction D>
32inline bool process_disposable_resource(double &new_resource, double initial_resource, double increment, double lb,
33 double ub) {
34 if constexpr (D == Direction::Forward) {
35 new_resource = std::max(initial_resource + increment, lb);
36 // new_resource = roundToTwoDecimalPlaces(new_resource);
37 return numericutils::lte(new_resource, ub);
38 } else {
39 new_resource = std::min(initial_resource - increment, ub);
40 // new_resource = roundToTwoDecimalPlaces(new_resource);
41 return numericutils::gte(new_resource, lb);
42 }
43}
44
45template <Direction D>
46inline bool process_non_disposable_resource(double &new_resource, double initial_resource, double increment, double lb,
47 double ub) {
48 new_resource = initial_resource + (D == Direction::Forward ? increment : -increment);
49 return new_resource >= lb && new_resource <= ub;
50}
51
52template <Direction D>
53inline double process_binary_resource(double increment) {
54 // For binary resource, return 1.0 if increment > 0 in forward direction,
55 // else 0.0, and vice versa in backward direction.
56 return (increment > 0) ? (D == Direction::Forward ? 1.0 : 0.0) : (D == Direction::Forward ? 0.0 : 1.0);
57}

59template <Direction D>
60inline bool process_battery_resource(double &new_resource, double initial_resource, bool is_station,
61 double battery_capacity) {
62 if (is_station) {
63 new_resource = battery_capacity;
64 return true;
65 }
66 new_resource = initial_resource;
67 return D == Direction::Forward ? new_resource >= 0 : new_resource <= battery_capacity;
68}

70template <Direction D>
71inline bool process_recharge_count(double &new_resource, double initial_resource, bool is_station,
72 double max_recharges) {
73 if (!is_station) {
74 new_resource = initial_resource;
75 return true;
76 }
77 new_resource = initial_resource + 1;
78 return new_resource <= max_recharges;
79}

81template <Direction D>
82inline bool process_recharge_time(double &new_resource, double initial_resource, bool is_station,
83 double max_recharge_time) {
84 new_resource = is_station ? 0 : initial_resource;
85 return new_resource <= max_recharge_time;
86}

88template <Direction D>
89inline bool process_mtw_resource(double &new_resource, double initial_resource, double increment,
90 const VRPNode &theNode) {
91 if constexpr (D == Direction::Forward) {
92 for (size_t i = 0; i < theNode.mtw_lb.size(); ++i) {
93 new_resource = std::max(initial_resource + increment, theNode.mtw_lb[i]);
94 if (numericutils::lte(new_resource, theNode.mtw_ub[i])) { return true; }
95 }
96 } else {
97 for (size_t i = 0; i < theNode.mtw_ub.size(); ++i) {
98 new_resource = std::min(initial_resource - increment, theNode.mtw_ub[i]);
99 if (numericutils::gte(new_resource, theNode.mtw_lb[i])) { return true; }
100 }
101 }
102 return false;
103}

105// Main processing function for all resources.
106// new_resources: vector to be updated for the new resource levels.
107// initial_resources: initial values of the resources.
108// gamma: provides resource_increment among other things.
109// theNode: VRPNode with lower/upper bounds and other properties.
110// N: number of resources to process.
111template <Direction D, typename Gamma, typename VRPNode>
112bool BucketGraph::process_all_resources(std::vector<double> &new_resources,
113 const std::array<double, R_SIZE> &initial_resources, const Gamma &gamma,
114 const VRPNode &theNode, size_t N) {
115 for (size_t I = 0; I < N; ++I) {
116 switch (static_cast<ResourceType>(options.resource_type[I])) {
117 [[likely]] case ResourceType::Disposable:
118 if (!process_disposable_resource<D>(new_resources[I], initial_resources[I], gamma.resource_increment[I],
119 theNode.lb[I], theNode.ub[I])) {
120 return false;
121 }
122 break;
123
124 case ResourceType::NonDisposable:
125 if (!process_non_disposable_resource<D>(new_resources[I], initial_resources[I], gamma.resource_increment[I],
126 theNode.lb[I], theNode.ub[I])) {
127 return false;
128 }
129 break;
130
131 case ResourceType::Binary: new_resources[I] = process_binary_resource<D>(gamma.resource_increment[I]); break;
132
133 case ResourceType::Battery:
134 if (!process_battery_resource<D>(new_resources[I], initial_resources[I], theNode.is_station,
135 options.battery_capacity)) {
136 return false;
137 }
138 break;
139
140 case ResourceType::RechargeCount:
141 if (!process_recharge_count<D>(new_resources[I], initial_resources[I], theNode.is_station,
142 options.max_recharges)) {
143 return false;
144 }
145 break;
146
147 case ResourceType::RechargeTime:
148 if (!process_recharge_time<D>(new_resources[I], initial_resources[I], theNode.is_station,
149 options.max_recharge_time)) {
150 return false;
151 }
152 break;
153
154 case ResourceType::MTW:
155 if (!process_mtw_resource<D>(new_resources[I], initial_resources[I], gamma.resource_increment[I],
156 theNode)) {
157 return false;
158 }
159 break;
160 }
161 }
162 return true;
163}

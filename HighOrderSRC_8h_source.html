179 void processSeedForPlan(int plan_idx, std::vector<Rank1MultiLabel> &local_pool) {
180 // Iterate through each `c, wc` pair in `c_N_noC`
181 for (auto &[c, wc] : c_N_noC) {
182 double vio, best_vio;
183 exactFindBestPermutationForOnePlan(c, plan_idx, vio);
184 if (vio < TOLERANCE) continue;
185
186 best_vio = vio;
187 char best_oper = 'o';
188 int add_j = -1, remove_j = -1;
189 std::pair<int, int> swap_i_j = {-1, -1};
190
191 // Perform add search and update best operation if necessary
192 addSearchCrazy(plan_idx, c, wc, vio, add_j);
193 if (vio > best_vio) {
194 best_vio = vio;
195 best_oper = 'a';
196 }
197
198 // Perform remove search and update best operation if necessary
199 removeSearchCrazy(plan_idx, c, vio, remove_j);
200 if (vio > best_vio) {
201 best_vio = vio;
202 best_oper = 'r';
203 }
204
205 // Perform swap search and update best operation if necessary
206 swapSearchCrazy(plan_idx, c, wc, vio, swap_i_j);
207 if (vio > best_vio) {
208 best_vio = vio;
209 best_oper = 's';
210 }
211
212 // Prepare `new_c` and `new_w_no_c` based on the best operation
213 std::vector<int> new_c, new_w_no_c;
214 applyBestOperation(best_oper, c, wc, add_j, remove_j, swap_i_j, new_c, new_w_no_c, plan_idx, best_vio);
215
216 // Only add to `local_pool` if an operation other than 'o' was performed
217 if (best_oper != 'o') { local_pool.emplace_back(new_c, new_w_no_c, plan_idx, best_vio, best_oper); }
218 }
219 }

221 void applyBestOperation(char best_oper, const std::vector<int> &c, const std::vector<int> &wc, int add_j,
222 int remove_j, const std::pair<int, int> &swap_i_j, std::vector<int> &new_c,
223 std::vector<int> &new_w_no_c, int plan_idx, double best_vio) {
224 switch (best_oper) {
225 case 'o': // No operation, directly add to target pool
226 if (c.size() >= 2 && c.size() <= max_row_rank1) {
227 cutLong tmp;
228 for (int i : c) tmp.set(i);
229 {
230 std::lock_guard<std::mutex> lock(pool_mutex);
231 generated_rank1_multi_pool[static_cast<int>(c.size())].emplace_back(tmp, plan_idx, best_vio);
232 }
233 }
234 break;
235
236 case 'a': // Add operation
237 new_c = c;
238 new_c.push_back(add_j);
239 new_w_no_c.reserve(wc.size() - 1);
240 std::copy_if(wc.begin(), wc.end(), std::back_inserter(new_w_no_c), [add_j](int w) { return w != add_j; });
241 break;
242
243 case 'r': // Remove operation
244 new_c.reserve(c.size() - 1);
245 std::copy_if(c.begin(), c.end(), std::back_inserter(new_c), [remove_j](int i) { return i != remove_j; });
246 new_w_no_c = wc;
247 break;
248
249 case 's': // Swap operation
250 new_c = c;
251 new_w_no_c = wc;
252 for (int &i : new_c) {
253 if (i == swap_i_j.first) i = swap_i_j.second;
254 }
255 for (int &i : new_w_no_c) {
256 if (i == swap_i_j.second) i = swap_i_j.first;
257 }
258 break;
259 }
260 }

264 void startSeedCrazy() {
265 num_label = 0;
266 std::vector<int> plan_indices = {0, 1, 2, 3, 4, 5, 6};
267
268 const int JOBS = plan_indices.size();
269 exec::static_thread_pool pool(JOBS);
270 auto sched = pool.get_scheduler();
271
272 auto bulk_sender = stdexec::bulk(stdexec::just(), JOBS, [&](std::size_t plan_idx) {
273 std::vector<Rank1MultiLabel> local_pool; // Local pool to collect results per thread
274 processSeedForPlan(plan_idx, local_pool);
275
276 // Lock and transfer data from local pool to shared pool
277 std::lock_guard<std::mutex> lock(pool_mutex);
278 for (auto &label : local_pool) {
279 if (num_label >= rank1_multi_label_pool.size()) {
280 rank1_multi_label_pool.resize(rank1_multi_label_pool.size() * 1.5);
281 }
282 rank1_multi_label_pool[num_label++] = (label);
283 }
284 });
285
286 auto work = stdexec::starts_on(sched, bulk_sender);
287 stdexec::sync_wait(std::move(work));
288 }

318 // For the frequency accumulation loop:
319 void accumulate_frequencies_simd(int *map_r_numbers, const auto &route_freqs, int route_size) {
320 const int simd_size = 8; // AVX2 processes 8 integers at once
321 int i = 0;
322
323 // Process 8 elements at a time using AVX2
324 for (; i + simd_size <= route_size; i += simd_size) {
325 __m256i freq = _mm256_loadu_si256((__m256i *)&route_freqs[i]);
326 __m256i curr = _mm256_loadu_si256((__m256i *)&map_r_numbers[i]);
327 __m256i mask = _mm256_cmpgt_epi32(freq, _mm256_setzero_si256());
328 freq = _mm256_and_si256(freq, mask);
329 curr = _mm256_add_epi32(curr, freq);
330 _mm256_storeu_si256((__m256i *)&map_r_numbers[i], curr);
331 }
332
333 // Handle remaining elements
334 for (; i < route_size; i++) {
335 const int frequency = route_freqs[i];
336 map_r_numbers[i] += frequency * (frequency > 0);
337 }
338 }

340 void exactFindBestPermutationForOnePlan(std::vector<int> &cut, const int plan_idx, double &vio) {
341 const int cut_size = static_cast<int>(cut.size());
342 const auto &plan = map_rank1_multiplier[cut_size][plan_idx];
343
344 if (get<1>(plan) == 0) {
345 vio = -std::numeric_limits<double>::max();
346 return;
347 }
348
349 // Use thread_local for frequently reused vectors to avoid allocations
350 static thread_local cutLong tmp;
351 tmp.reset();
352
353 // SIMD-friendly loop with no branching for setting bits
354 for (int i = 0; i < cut_size; ++i) {
355 const int it = cut[i];
356 tmp.set(it * (it >= 0 && it < v_r_map.size())); // Branchless using multiplication
357 }
358
359 // Quick cache lookup with shared lock
360 {
361 std::shared_lock<std::shared_mutex> lock(map_cut_plan_vio_mutex);
362 if (auto it = map_cut_plan_vio.find(tmp);
363 it != map_cut_plan_vio.end() && plan_idx < it->second.size() && !it->second[plan_idx].first.empty()) {
364 vio = it->second[plan_idx].second;
365 return;
366 }
367 }
368
369 // Upgrade to exclusive lock to modify
370 {
371 std::unique_lock<std::shared_mutex> lock(map_cut_plan_vio_mutex);
372 map_cut_plan_vio[tmp].resize(7);
373 }
374
375 const int denominator = get<1>(plan);
376 const double rhs = get<2>(plan);
377 const auto &coeffs = record_map_rank1_combinations[cut_size][plan_idx];
378
379 // Use thread_local vectors to avoid allocations
380 // static thread_local std::vector<int> map_r_numbers;
381 alignas(32) static thread_local std::vector<int, aligned_allocator<int>> map_r_numbers;
382
383 map_r_numbers.resize(sol.size());
384 std::fill(map_r_numbers.begin(), map_r_numbers.end(), 0);
385
386#ifndef __AVX2__
387 // Pre-compute route frequencies in a single pass
388 for (int idx : cut) {
389 if (idx >= 0 && idx < v_r_map.size()) {
390 const auto &route_freqs = v_r_map[idx];
391 const int route_size = static_cast<int>(route_freqs.size());
392
393 // SIMD-friendly loop for frequency accumulation
394 for (int route_idx = 0; route_idx < route_size; ++route_idx) {
395 const int frequency = route_freqs[route_idx];
396 map_r_numbers[route_idx] += frequency * (frequency > 0); // Branchless using multiplication
397 }
398 }
399 }
400#else
401 for (int idx : cut) {
402 if (idx >= 0 && idx < v_r_map.size()) {
403 const auto &route_freqs = v_r_map[idx];
404 const int route_size = static_cast<int>(route_freqs.size());
405 accumulate_frequencies_simd(map_r_numbers.data(), route_freqs, route_size);
406 }
407 }
408#endif

410 static thread_local std::vector<std::vector<int>> cut_num_times_vis_routes;
411
412 // Cache lookup with minimized critical section
413 bool cache_hit = false;
414 {
415 std::shared_lock<std::shared_mutex> lock(cut_cache_mutex);
416 if (auto cache_it = cut_cache.find(cut); cache_it != cut_cache.end()) {
417 cut_num_times_vis_routes = cache_it->second;
418 cache_hit = true;
419 }
420 }
421
422 if (!cache_hit) {
423 cut_num_times_vis_routes.clear();
424 cut_num_times_vis_routes.resize(cut_size);
425
426 // Pre-reserve space to avoid reallocations
427 for (auto &route_vec : cut_num_times_vis_routes) {
428 route_vec.reserve(num_valid_routes * 2); // Estimate capacity
429 }
430
431 for (int i = 0; i < cut_size; ++i) {
432 const int c = cut[i];
433 if (c >= 0 && c < v_r_map.size()) {
434 auto &current_route = cut_num_times_vis_routes[i];
435 const auto &route_freqs = v_r_map[c];
436
437 // Vectorizable loop for route processing
438 for (int route_idx = 0; route_idx < route_freqs.size(); ++route_idx) {
439 const int frequency = route_freqs[route_idx];
440 if (frequency > 0 && cached_map_old_new_routes[route_idx] != -1) {
441 const int val = cached_map_old_new_routes[route_idx];
442 current_route.insert(current_route.end(), frequency, val);
443 }
444 }
445 }
446 }
447
448 // Update cache
449 std::unique_lock<std::shared_mutex> lock(cut_cache_mutex);
450 cut_cache[cut] = cut_num_times_vis_routes;
451 }
452
453 // Use thread_local vector for accumulation
454 static thread_local std::vector<double> num_times_vis_routes;
455 num_times_vis_routes.resize(num_valid_routes);
456
457 double best_vio = -std::numeric_limits<double>::max();
458 int best_idx = -1;
459
460 // SIMD-friendly loop for coefficient processing
461 for (size_t cnt = 0; cnt < coeffs.size(); ++cnt) {
462 std::fill(num_times_vis_routes.begin(), num_times_vis_routes.end(), 0.0);
463
464 // Vectorizable nested loops
465 for (int i = 0; i < cut_size; ++i) {
466 const auto &routes = cut_num_times_vis_routes[i];
467 const double coeff = coeffs[cnt][i];
468
469 for (const int j : routes) { num_times_vis_routes[j] += coeff; }
470 }
471
472 // Compute violation score with SIMD-friendly loop
473 double vio_tmp = -rhs;
474 for (size_t i = 0; i < num_times_vis_routes.size(); ++i) {
475 const double scaled_value =
476 static_cast<int>(num_times_vis_routes[i] / denominator + tolerance) * cached_valid_routes[i];
477 vio_tmp += scaled_value;
478 }
479
480 if (vio_tmp > best_vio) {
481 best_vio = vio_tmp;
482 best_idx = static_cast<int>(cnt);
483 }
484 }
485 vio = best_vio;
486
487 // Final result preparation with thread_local storage
488 static thread_local std::vector<std::pair<int, int>> cut_coeff;
489 cut_coeff.resize(cut_size);
490
491 for (int i = 0; i < cut_size; ++i) {
492 if (best_idx >= 0 && best_idx < coeffs.size() && i < coeffs[best_idx].size()) {
493 cut_coeff[i] = {cut[i], coeffs[best_idx][i]};
494 }
495 }
496
497 pdqsort(cut_coeff.begin(), cut_coeff.end(), [](const auto &a, const auto &b) { return a.second > b.second; });
498
499 static thread_local std::vector<int> new_cut;
500 new_cut.resize(cut_size);
501 std::transform(cut_coeff.begin(), cut_coeff.end(), new_cut.begin(), [](const auto &a) { return a.first; });
502
503 // Update final results
504 std::unique_lock<std::shared_mutex> lock(map_cut_plan_vio_mutex);
505 if (plan_idx < map_cut_plan_vio[tmp].size()) { map_cut_plan_vio[tmp][plan_idx] = {new_cut, vio}; }
506 }

519 void generateSepHeurMem4Vertex() {
520 rank1_sep_heur_mem4_vertex.resize(dim);
521
522 // Precompute half-costs for nodes to avoid repeated divisions
523 std::vector<double> half_cost(dim);
524 for (int i = 0; i < dim; ++i) { half_cost[i] = nodes[i].cost / 2; }
525
526 for (int i = 0; i < dim; ++i) {
527 // Initialize and populate the `cost` vector directly for each `i`
528 std::vector<std::pair<int, double>> cost(dim);
529 cost[0] = {0, INFINITY};
530
531 for (int j = 1; j < dim - 1; ++j) { cost[j] = {j, cost_mat4_vertex[i][j] - (half_cost[i] + half_cost[j])}; }
532
533 // Use partial sort to get only the top `max_heuristic_sep_mem4_row_rank1` elements
534 std::partial_sort(cost.begin(), cost.begin() + max_heuristic_sep_mem4_row_rank1, cost.end(),
535 [](const auto &a, const auto &b) { return a.second < b.second; });
536
537 // Set bits in `vst2` for the smallest costs
538 cutLong &vst2 = rank1_sep_heur_mem4_vertex[i];
539 for (int k = 0; k < max_heuristic_sep_mem4_row_rank1; ++k) { vst2.set(cost[k].first); }
540 }

543 void constructVRMapAndSeedCrazy() {
544 // Resize `rank1_sep_heur_mem4_vertex` and initialize `v_r_map`
545 rank1_sep_heur_mem4_vertex.resize(dim);
546 v_r_map.assign(dim, std::vector<int>(sol.size(), 0));
547 const int max_dim_minus_1 = dim - 1;
548 const auto sol_size = sol.size();
549
550 // Populate `v_r_map` with route appearances for each vertex `i`
551#ifdef __cpp_lib_parallel_algorithm
552 std::for_each(std::execution::par_unseq, sol.begin(), sol.end(), [&](const auto &route_data) {
553 const auto r = &route_data - &sol[0]; // Get index without separate counter
554 for (const int i : route_data.route) {
555 if (i > 0 && i < max_dim_minus_1) { ++v_r_map[i][r]; }
556 }
557 });
558#else
559 for (size_t r = 0; r < sol_size; ++r) {
560 for (const int i : sol[r].route) {
561 if (i > 0 && i < max_dim_minus_1) { ++v_r_map[i][r]; }
562 }
563 }
564#endif
565
566 // Initialize `seed_map` with reserved space
567 ankerl::unordered_dense::map<cutLong, cutLong> seed_map;
568 seed_map.reserve(4096);
569
570 // Cache dimensions for efficiency
571
572 // Process vertices and build seed map
573 for (size_t i = 1; i < max_dim_minus_1; ++i) {
574 if (v_r_map[i].empty()) continue;
575
576 // Create working set for current vertex
577 cutLong working_set;
578 const auto &heur_mem = rank1_sep_heur_mem4_vertex[i];
579
580 // Build working set from routes containing vertex i
581 for (size_t route_idx = 0; route_idx < v_r_map[i].size(); ++route_idx) {

954 std::vector<ankerl::unordered_dense::set<int>> &segment,

957 const int sum = std::accumulate(vis.begin(), vis.end(), 0);
958 const int initial_mod = sum % denominator;
959
960 // Create key for memoization
961 auto key = vis;
962 key.push_back(initial_mod);
963
964 auto &cached_result = rank1_multi_mem_plan_map[key];
965
966 // Generate solutions if not cached
967 if (cached_result.empty()) {
968 std::deque<State> states;
969 states.emplace_back(0, initial_mod, vis, std::vector<int>{});
970
971 while (!states.empty()) {
972 auto [begin, target_rem, remaining, memory] = std::move(states.front());
973 states.pop_front();
974
975 int running_sum = 0;
976 for (size_t j = 0; j < remaining.size(); ++j) {
977 running_sum = (running_sum + remaining[j]) % denominator;
978
979 if (running_sum > 0) {
980 const size_t current_pos = begin + j;
981
982 // Try branching if possible
983 if (running_sum <= target_rem && j + 1 < remaining.size()) {
984 std::vector<int> new_remaining(remaining.begin() + j + 1, remaining.end());
985 states.emplace_back(current_pos + 1, target_rem - running_sum, std::move(new_remaining),
986 memory);
987 }
988
989 // Add to memory if not last element
990 if (current_pos != vis.size() - 1) { memory.push_back(current_pos); }
991 }
992 }
993 cached_result.push_back(std::move(memory));
994 }
995 }
996
997 // Process memory constraints
998 for (int i = 1; i < dim; ++i) {
999 if (mem[i]) {
1000 for (auto &seg : segment) { seg.erase(i); }
1001 }
1002 }
1003
1004 // Build visibility map
1005 std::vector<ankerl::unordered_dense::set<int>> vertex_visibility(dim);
1006 for (size_t plan_idx = 0; plan_idx < cached_result.size(); ++plan_idx) {
1007 for (int j : cached_result[plan_idx]) {
1008 for (int k : segment[j]) { vertex_visibility[k].insert(plan_idx); }
1009 }
1010 }
1011
1012 // Update memory based on visibility
1013 for (int i = 1; i < dim; ++i) {
1014 if (vertex_visibility[i].size() == cached_result.size()) {
1015 mem.set(i);
1016 for (auto &seg : segment) { seg.erase(i); }
1017 }
1018 }
1019

1021 std::vector<std::pair<cutLong, std::vector<int>>> memory_patterns;
1022 memory_patterns.reserve(cached_result.size());
1023
1024 for (const auto &memory_indices : cached_result) {
1025 cutLong pattern_mem;
1026
1027 // Build memory pattern
1028 for (int j : memory_indices) {
1029 for (int k : segment[j]) { pattern_mem.set(k); }
1030 }
1031
1032 // Check for empty pattern
1033 if (pattern_mem.none()) {
1034 plan.clear();
1035 return;
1036 }
1037
1038 // Find or add pattern
1039 auto it = std::find_if(memory_patterns.begin(), memory_patterns.end(), [&pattern_mem](const auto &entry) {
1040 return ((pattern_mem & entry.first) ^ pattern_mem).none();
1041 });
1042
1043 if (it != memory_patterns.end()) {
1044 it->first = std::move(pattern_mem);
1045 it->second = memory_indices;
1046 } else {
1047 memory_patterns.emplace_back(std::move(pattern_mem), memory_indices);
1048 }
1049 }
1050
1051 // Build final plan
1052 plan.resize(memory_patterns.size());

1054 [](const auto &entry) { return entry.second; });

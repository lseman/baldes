28struct PlanValidationResult {
29 // Pack booleans together to reduce memory footprint
30 bool is_valid : 1;
31 const std::tuple<std::vector<int>, int, int> *plan;
32
33 // Constructor for direct initialization
34 constexpr PlanValidationResult(
35 bool valid, const std::tuple<std::vector<int>, int, int> *p)
36 : is_valid(valid), plan(p) {}
37
38 // Static size validation to allow compile-time optimization
39 static constexpr bool isValidSize(int size) noexcept { return size >= 3; }
40
41 // Main validation function optimized for branch prediction
42 static PlanValidationResult validate(
43 const int plan_idx, const int size,
44 const ankerl::unordered_dense::map<
45 int, std::vector<std::tuple<std::vector<int>, int, int>>>
46 &map) noexcept {
47 // Early size check (likely to be predicted well by CPU)
48 if (unlikely(!isValidSize(size))) {
49 return {false, nullptr};
50 }
51
52 // Find size in map
53 auto size_it = map.find(size);
54 if (unlikely(size_it == map.end())) {
55 return {false, nullptr};
56 }
57
58 // Check plan index bounds
59 const auto &plans = size_it->second;
60 if (unlikely(static_cast<size_t>(plan_idx) >= plans.size())) {
61 return {false, nullptr};
62 }
63
64 // Return valid result with plan pointer
65 return {true, &plans[plan_idx]};
66 }
67
68 // Optional: Add cache-friendly batch validation
69 static std::vector<PlanValidationResult> validateBatch(
70 const std::vector<std::pair<int, int>> &plan_sizes,
71 const ankerl::unordered_dense::map<
72 int, std::vector<std::tuple<std::vector<int>, int, int>>>
73 &map) noexcept {
74 std::vector<PlanValidationResult> results;
75 results.reserve(plan_sizes.size()); // Avoid reallocation
76
77 for (const auto &[plan_idx, size] : plan_sizes) {
78 results.push_back(validate(plan_idx, size, map));
79 }
80
81 return results;
82 }

281 void startSeedCrazy() {
282 num_label = 0;
283 std::mutex rank1_mutex;
284
285 // Create thread pool
286 const int JOBS = std::thread::hardware_concurrency();
287 exec::static_thread_pool pool(JOBS);
288 auto sched = pool.get_scheduler();
289
290 // Create work items vector
291 std::vector<std::pair<int, std::reference_wrapper<const std::pair<
292 std::vector<int>, std::vector<int>>>>>
293 work_items;
294 work_items.reserve(
295 7 * c_N_noC.size()); // Preallocate to avoid reallocations
296
297 for (int plan_idx : {0, 1, 2, 3, 4, 5, 6}) {
298 for (const auto &pair : c_N_noC) {
299 work_items.emplace_back(plan_idx, std::ref(pair));
300 }
301 }
302
303 // Define chunk size to reduce parallelization overhead
304 const int chunk_size =
305 std::max(1, static_cast<int>(work_items.size()) / JOBS);
306
307 // Create bulk sender for parallel processing
308 auto bulk_sender = stdexec::bulk(
309 stdexec::just(), (work_items.size() + chunk_size - 1) / chunk_size,
310 [this, &work_items, chunk_size,
311 &rank1_mutex](std::size_t chunk_idx) {
312 size_t start_idx = chunk_idx * chunk_size;
313 size_t end_idx =
314 std::min(start_idx + chunk_size, work_items.size());
315
316 // Process a chunk of tasks
317 for (size_t task_idx = start_idx; task_idx < end_idx;
318 ++task_idx) {
319 const auto &[plan_idx, pair_ref] = work_items[task_idx];
320 const auto &pair = pair_ref.get();
321 const auto &[c, wc] = pair;
322
323 try {
324 if (c.empty() || wc.empty()) {
325 continue;
326 }
327
328 std::vector<int> c_mutable = c;
329 double initial_vio = 0.0;
330
331 exactFindBestPermutationForOnePlan(c_mutable, plan_idx,
332 initial_vio);
333
334 if (initial_vio < TOLERANCE) {
335 continue;
336 }
337
338 // Additional operations: add, remove, swap
339 int add_j, remove_j;
340 double add_vio = initial_vio, remove_vio = initial_vio;
341 std::pair<int, int> swap_i_j;
342 double swap_vio = initial_vio;
343
344 addSearchCrazy(plan_idx, c_mutable, wc, add_vio, add_j);
345 removeSearchCrazy(plan_idx, c_mutable, remove_vio,
346 remove_j);
347 swapSearchCrazy(plan_idx, c_mutable, wc, swap_vio,
348 swap_i_j);
349
350 double best_vio =
351 std::max({add_vio, remove_vio, swap_vio});
352 if (best_vio > initial_vio) {
353 std::vector<int> new_c, new_w_no_c;
354 char best_op = 'n';
355
356 if (best_vio == add_vio) {
357 best_op = 'a';
358 applyBestOperation('a', c_mutable, wc, add_j,
359 -1, std::make_pair(-1, -1),
360 new_c, new_w_no_c, plan_idx,
361 best_vio);
362 } else if (best_vio == remove_vio) {
363 best_op = 'r';
364 applyBestOperation(
365 'r', c_mutable, wc, -1, remove_j,
366 std::make_pair(-1, -1), new_c, new_w_no_c,
367 plan_idx, best_vio);
368 } else {
369 best_op = 's';
370 applyBestOperation('s', c_mutable, wc, -1, -1,
371 swap_i_j, new_c, new_w_no_c,
372 plan_idx, best_vio);
373 }
374
375 std::lock_guard<std::mutex> lock(rank1_mutex);
376 if (num_label + 1 > rank1_multi_label_pool.size()) {
377 rank1_multi_label_pool.resize(
378 std::max<std::size_t>(
379 rank1_multi_label_pool.size() * 2,
380 num_label + 1));
381 }
382 rank1_multi_label_pool[num_label++] =
383 Rank1MultiLabel(std::move(new_c),
384 std::move(new_w_no_c), plan_idx,
385 best_vio, best_op);
386 }
387 } catch (const std::exception &e) {
388 std::cerr << "Exception in task for plan_idx "
389 << plan_idx << ": " << e.what() << "\n";
390 } catch (...) {
391 std::cerr << "Unknown exception in task for plan_idx "
392 << plan_idx << "\n";
393 }
394 }
395 });

398 auto work = stdexec::starts_on(sched, bulk_sender);
399 stdexec::sync_wait(std::move(work));

508 // For the frequency accumulation loop:
509 void accumulate_frequencies_simd(int *__restrict__ map_r_numbers,
510 const int *__restrict__ route_freqs,
511 const int route_size) noexcept {
512 static constexpr int SIMD_SIZE =
513 8; // AVX2 processes 8 integers at once
514 static constexpr int PREFETCH_DISTANCE = 64; // Common cache line size
515
516 // Ensure proper alignment for best performance
517 alignas(32) const int *freq_ptr = route_freqs; // No .data() needed
518 alignas(32) int *map_ptr = map_r_numbers;
519
520 int i = 0;
521
522 // Process 8 elements at a time using AVX2
523 for (; i + SIMD_SIZE <= route_size; i += SIMD_SIZE) {
524 // Prefetch next cache lines
525 _mm_prefetch(reinterpret_cast<const char *>(freq_ptr + i +
526 PREFETCH_DISTANCE),
527 _MM_HINT_T0);
528 _mm_prefetch(
529 reinterpret_cast<const char *>(map_ptr + i + PREFETCH_DISTANCE),
530 _MM_HINT_T0);
531
532 // Load frequencies and current values
533 __m256i freq = _mm256_loadu_si256(
534 reinterpret_cast<const __m256i *>(&freq_ptr[i]));
535 __m256i curr = _mm256_loadu_si256(
536 reinterpret_cast<const __m256i *>(&map_ptr[i]));
537
538 // Mask frequencies greater than 0
539 __m256i zero = _mm256_setzero_si256();
540 __m256i mask = _mm256_cmpgt_epi32(freq, zero);
541 freq = _mm256_and_si256(freq, mask);
542
543 // Accumulate
544 curr = _mm256_add_epi32(curr, freq);
545 _mm256_storeu_si256(reinterpret_cast<__m256i *>(&map_ptr[i]), curr);
546 }
547
548 // Handle remaining elements with auto-vectorization hint
549#pragma omp simd
550 for (; i < route_size; i++) {
551 const int frequency = freq_ptr[i];
552 map_ptr[i] += frequency * (frequency > 0);
553 }
554 }

569 void exactFindBestPermutationForOnePlan(std::vector<int> &cut,
570 const int plan_idx, double &vio) {
571 const int cut_size = static_cast<int>(cut.size());
572 const auto &plan = map_rank1_multiplier[cut_size][plan_idx];
573
574 if (get<1>(plan) == 0) {
575 vio = -std::numeric_limits<double>::max();
576 return;
577 }
578
579 // Thread-local storage with aligned vectors
580 alignas(64) static thread_local cutLong tmp;
581 alignas(64) static thread_local std::vector<int, aligned_allocator<int>>
582 map_r_numbers;
583 alignas(64) static thread_local std::vector<double>
584 num_times_vis_routes;
585 alignas(64) static thread_local std::vector<std::vector<int>>
586 cut_num_times_vis_routes;
587 alignas(64) static thread_local std::vector<std::pair<int, int>>
588 cut_coeff;
589 alignas(64) static thread_local std::vector<int> new_cut;
590
591 tmp.reset();
592
593 // Filter valid indices
594 for (int it : cut) {
595 if (it >= 0 && static_cast<size_t>(it) < v_r_map.size()) {
596 tmp.set(it);
597 }
598 }
599 std::vector<std::pair<std::vector<int>, double>> cached_plans;
600 if (map_cut_plan_vio.find(tmp, cached_plans) &&
601 plan_idx < cached_plans.size() &&
602 !cached_plans[plan_idx].first.empty()) {
603 vio = cached_plans[plan_idx].second;
604 return;
605 }
606 map_cut_plan_vio.insert(
607 tmp, std::vector<std::pair<std::vector<int>, double>>(7));
608
609 const int denominator = get<1>(plan);
610 const double rhs = get<2>(plan);
611 const auto &coeffs = record_map_rank1_combinations[cut_size][plan_idx];
612
613 map_r_numbers.assign(sol.size(), 0);
614 auto indices = std::views::iota(0, cut_size);
615
616#ifdef __AVX2__
617 // Process frequencies using SIMD
618 for (int i = 0; i < cut_size; ++i) {
619 const int idx = cut[i];
620 if (idx >= 0 && static_cast<size_t>(idx) < v_r_map.size()) {
621 const auto &freqs = v_r_map[idx];
622 accumulate_frequencies_simd(map_r_numbers.data(), freqs.data(),
623 static_cast<int>(freqs.size()));
624 }
625 }
626#else
627 // Process frequencies
628 for (int i : indices) {
629 const int idx = cut[i];
630 if (idx >= 0 && static_cast<size_t>(idx) < v_r_map.size()) {
631 const auto &freqs = v_r_map[idx];
632 for (size_t j = 0; j < freqs.size(); ++j) {
633 const int freq = freqs[j];
634 map_r_numbers[j] += freq * (freq > 0);
635 }
636 }
637 }
638#endif
639
640 // Handle cache
641 std::vector<std::vector<int>> cached_routes;
642 if (cut_cache.find(cut, cached_routes)) {
643 cut_num_times_vis_routes = cached_routes;
644
645 } else {
646 cut_num_times_vis_routes.clear();
647 cut_num_times_vis_routes.resize(cut_size);
648
649 // Reserve space
650 for (auto &vec : cut_num_times_vis_routes) {
651 vec.reserve(num_valid_routes * 2);
652 }
653
654 for (int i : indices) {
655 const int c = cut[i];
656 if (c >= 0 && static_cast<size_t>(c) < v_r_map.size()) {
657 auto &current_route = cut_num_times_vis_routes[i];
658 const auto &route_freqs = v_r_map[c];
659
660 auto route_indices =
661 std::views::iota(size_t{0}, route_freqs.size());
662 for (size_t route_idx : route_indices) {
663 const int frequency = route_freqs[route_idx];
664 const int cached_route =
665 cached_map_old_new_routes[route_idx];

668 static_cast<size_t>(frequency),
669 cached_route);

678 // Process violations
679 num_times_vis_routes.resize(num_valid_routes);
680 std::vector<ViolationResult> violations;
681 violations.reserve(coeffs.size());
682
683 auto coeff_indices = std::views::iota(size_t{0}, coeffs.size());
684 for (size_t coeff_idx : coeff_indices) {
685 const auto &coeff_row = coeffs[coeff_idx];
686
687 constexpr size_t STACK_SIZE = 1024;
688 std::array<double, STACK_SIZE> local_times{};
689 std::unique_ptr<double[]> heap_times;
690 std::span<double> times_span;
691
692 if (num_valid_routes <= STACK_SIZE) {
693 times_span = std::span{local_times}.first(num_valid_routes);
694 } else {
695 heap_times = std::make_unique<double[]>(num_valid_routes);
696 times_span = std::span{heap_times.get(),
697 static_cast<size_t>(num_valid_routes)};

702 std::views::iota(size_t{0}, cut_num_times_vis_routes.size());
703 for (size_t i : route_indices) {
704 const double coeff = coeff_row[i];
705 const auto &routes = cut_num_times_vis_routes[i];
706 for (const int route : routes) {
707 times_span[route] += coeff;
708 }
709 }

711 double vio_tmp = -rhs;
712 for (size_t i = 0; i < num_valid_routes; ++i) {
713 vio_tmp +=
714 static_cast<int>(times_span[i] / denominator + TOLERANCE) *
715 cached_valid_routes[i];
716 }
717
718 violations.emplace_back(vio_tmp, static_cast<int>(coeff_idx));
719 }

721 const auto best_result = *std::ranges::max_element(
722 violations, [](const ViolationResult &a, const ViolationResult &b) {
723 return a.violation < b.violation;
724 });
725 vio = best_result.violation;
726
727 cut_coeff.resize(cut_size);
728 if (static_cast<size_t>(best_result.index) < coeffs.size()) {
729 const auto &best_coeffs = coeffs[best_result.index];
730
731 for (int i = 0; i < cut_size; ++i) {
732 cut_coeff[i] = {cut[i], i < static_cast<int>(best_coeffs.size())
733 ? static_cast<int>(best_coeffs[i])
734 : 0};
735 }
736 }
737
738 pdqsort(cut_coeff.begin(), cut_coeff.end(),
739 [](const std::pair<int, int> &a, const std::pair<int, int> &b) {
740 return a.second > b.second; // Sort in descending order
741 });
742
743 new_cut.resize(cut_size);
744 std::ranges::transform(cut_coeff, new_cut.begin(),
745 &std::pair<int, int>::first);
746
747 map_cut_plan_vio.upsert(tmp, [&](auto &value) {
748 value.resize(
749 std::max(value.size(), static_cast<size_t>(plan_idx + 1)));
750 value[plan_idx] = {new_cut, vio};
751 return true;
752 });

1291 void chooseCuts(const std::vector<R1c> &tmp_cuts,
1292 std::vector<R1c> &chosen_cuts, int numCuts) {

1294 numCuts = std::min(numCuts, static_cast<int>(tmp_cuts.size()));
1295 if (numCuts == 0) return;
1296 chosen_cuts.reserve(numCuts);
1297
1298 // Preallocate vectors to avoid repeated allocations
1299 std::vector<std::vector<int>> tmp_cut;
1300 tmp_cut.reserve(
1301 32); // Reasonable initial capacity for coefficient groups
1302 std::vector<int> new_cut;
1303 new_cut.reserve(1024); // Reasonable initial capacity for combined cuts
1304
1305 for (const auto &cut : tmp_cuts) {
1306 const auto &fst = cut.info_r1c.first;
1307 const auto &snd = cut.info_r1c.second;
1308 const int size = static_cast<int>(fst.size());
1309 const auto &coeff = get<0>(map_rank1_multiplier[size][snd]);

1311 // Find max coefficient to minimize resizing
1312 const int max_coeff = coeff[0];
1313
1314 // Clear and resize tmp_cut in one operation
1315 tmp_cut.clear();
1316 tmp_cut.resize(max_coeff + 1);
1317
1318 // Single-pass coefficient grouping
1319 for (int i = 0; i < size; ++i) {
1320 tmp_cut[coeff[i]].push_back(fst[i]);
1321 }
1322
1323 // Sort groups and combine in one pass
1324 new_cut.clear();
1325 for (int i = max_coeff; i >= 0; --i) {

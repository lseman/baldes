54 std::vector<double> computeRCCCoefficients(const std::vector<int> &label) {
55 std::vector<double> coeffs(cuts_.size(), 0);
56
57 // Precompute the pairs of consecutive nodes in the label
58 ankerl::unordered_dense::set<std::pair<int, int>> label_arcs;
59 for (int j = 0; j < label.size() - 1; j++) { label_arcs.insert({label[j], label[j + 1]}); }
60
61 // Iterate over cuts
62 for (int i = 0, n_cuts = cuts_.size(); i < n_cuts; i++) {
63 int coeff = 0;
64
65 // Iterate over arcs in the cut and check if they exist in the precomputed label arcs
66 for (const auto &arc : cuts_[i].arcs) { coeff += label_arcs.count({arc.from, arc.to}); }
67
68 coeffs[i] = coeff;
69 }
70
71 return coeffs;
72 }

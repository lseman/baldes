57 void markCutSeparated(RCCut &cut, double violation, double dual_mag) {
58 cut.separation_count++;
59 cut.dual_magnitude = dual_mag;

71 double computeSelectionScore(const RCCut &cut) const {
72 double strength = std::log1p(cut.avg_dual) * std::log1p(cut.avg_dual);
73 int64_t age = current_epoch_ - cut.creation_epoch;
74 double age_penalty = (age == 0) ? 1.0 : std::pow(age_decay_alpha_, static_cast<double>(age));
75 double inclusion_bonus = 1.0 + 0.1 * std::log1p(static_cast<double>(cut.inclusion_count));
76 return strength * age_penalty * inclusion_bonus;
77 }

82 void enforcePoolSizeLimit() {
83 if (cuts_.size() <= max_pool_size_) return;
84
85 std::vector<std::pair<size_t, double>> scored;
86 scored.reserve(cuts_.size());
87 for (size_t i = 0; i < cuts_.size(); ++i) { scored.emplace_back(i, computeSelectionScore(cuts_[i])); }
88
89 std::sort(scored.begin(), scored.end(), [](const auto &a, const auto &b) { return a.second < b.second; });
90
91 size_t excess = cuts_.size() - max_pool_size_;
92 size_t removed = 0;
93 for (auto &[idx, score] : scored) {
94 if (removed >= excess) break;
95 if (score > 0) {
96 cuts_.erase(cuts_.begin() + static_cast<int>(idx));
97 removed++;
98 }
99 }
100 }

106 std::vector<size_t> selectActiveCuts(size_t max_active = 64) {
107 std::vector<std::pair<size_t, double>> scored;
108 scored.reserve(cuts_.size());
109 for (size_t i = 0; i < cuts_.size(); ++i) {
110 if (cuts_[i].dual_magnitude > 1e-6) { scored.emplace_back(i, computeSelectionScore(cuts_[i])); }
111 }
112
113 std::sort(scored.begin(), scored.end(), [](const auto &a, const auto &b) { return a.second > b.second; });
114
115 if (scored.size() > max_active) scored.resize(max_active);
116
117 std::vector<size_t> active;
118 active.reserve(scored.size());
119 for (auto &[idx, score] : scored) {
120 active.push_back(idx);
121 markCutInclusion(cuts_[idx]);
122 }
123 return active;

151 std::vector<double> computeRCCCoefficients(const std::vector<uint16_t> &label) {
152 std::vector<double> coeffs(cuts_.size(), 0);
153 if (label.size() < 2 || cuts_.empty()) { return coeffs; }
154
155 // Precompute the pairs of consecutive nodes in the label
156 ankerl::unordered_dense::set<std::pair<int, int>> label_arcs;
157 label_arcs.reserve(label.size() - 1);
158 for (int j = 0; j < label.size() - 1; j++) { label_arcs.insert({label[j], label[j + 1]}); }
159
160 // Iterate over cuts
161 for (int i = 0, n_cuts = cuts_.size(); i < n_cuts; i++) {
162 int coeff = 0;
163
164 // Iterate over arcs in the cut and check if they exist in the precomputed label arcs
165 for (const auto &arc : cuts_[i].arcs) { coeff += label_arcs.count({arc.from, arc.to}); }
166
167 coeffs[i] = coeff;
168 }
169
170 return coeffs;
171 }

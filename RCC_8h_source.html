73 std::vector<double> computeRCCCoefficients(const std::vector<uint16_t> &label) {
74 std::vector<double> coeffs(cuts_.size(), 0);
75
76 // Precompute the pairs of consecutive nodes in the label
77 ankerl::unordered_dense::set<std::pair<int, int>> label_arcs;
78 for (int j = 0; j < label.size() - 1; j++) { label_arcs.insert({label[j], label[j + 1]}); }
79
80 // Iterate over cuts
81 for (int i = 0, n_cuts = cuts_.size(); i < n_cuts; i++) {
82 int coeff = 0;
83
84 // Iterate over arcs in the cut and check if they exist in the precomputed label arcs
85 for (const auto &arc : cuts_[i].arcs) { coeff += label_arcs.count({arc.from, arc.to}); }
86
87 coeffs[i] = coeff;
88 }
89
90 return coeffs;
91 }

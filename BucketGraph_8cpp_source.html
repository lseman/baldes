196std::vector<int> BucketGraph::computePhi(int &bucket_id, bool fw) {
197 std::vector<int> phi;
198
199 // Ensure bucket_id is within valid bounds
200 auto &buckets = fw ? fw_buckets : bw_buckets;
201 auto &fixed_buckets = fw ? fw_fixed_buckets : bw_fixed_buckets;
202 auto &node_interval_trees =
203 fw ? fw_node_interval_trees : bw_node_interval_trees;
204 auto &bucket_splits = fw ? fw_bucket_splits : bw_bucket_splits;
205 if (options.resources.size() > 1) {
206 if (bucket_id >= buckets.size() || bucket_id < 0) return phi;
207
208 std::vector<double> total_ranges(intervals.size());
209 std::vector<double> base_intervals(intervals.size());
210
211 for (int r = 0; r < intervals.size(); ++r) {
212 total_ranges[r] =
213 R_max[r] - R_min[r]; // Ensure integer type for total range
214 base_intervals[r] = total_ranges[r] / bucket_splits[bucket_id];
215 }
216
217 // Get the node ID and current bucket
218 int node_id = buckets[bucket_id].node_id;
219 auto &current_bucket = buckets[bucket_id];
220
221 // Retrieve the pre-built Splay Tree for this node
222 auto &node_tree = node_interval_trees[node_id];
223
224 // Search for matching intervals using the existing Splay Tree
225 if (fw) {
226 // Forward search: find the interval just below the current bucket
227 std::vector<double> target_low = current_bucket.lb;
228 for (int r = 0; r < intervals.size(); ++r) {
229 target_low[r] -=
230 base_intervals[r]; // Adjust for the base intervals
231 }
232
233 TreeNode *found_node = node_tree.find(target_low);
234 if (found_node != nullptr &&
235 buckets[found_node->bucket_index].node_id == node_id) {
236 // Check if the found bucket is fixed
237#ifdef FIX_BUCKETS
238 if (is_bucket_not_fixed<Direction::Forward>(
239 found_node->bucket_index, bucket_id))
240#endif
241 {
242 phi.push_back(found_node->bucket_index);
243 }
244 }
245 } else {
246 // Backward search: find the interval just above the current bucket
247 std::vector<double> target_high = current_bucket.ub;
248 for (int r = 0; r < intervals.size(); ++r) {
249 target_high[r] +=
250 base_intervals[r]; // Adjust for the base intervals
251 }
252
253 TreeNode *found_node = node_tree.find(target_high);
254 if (found_node != nullptr &&
255 buckets[found_node->bucket_index].node_id == node_id) {
256 // Check if the found bucket is fixed
257#ifdef FIX_BUCKETS
258 if (is_bucket_not_fixed<Direction::Forward>(
259 found_node->bucket_index, bucket_id))
260#endif
261 {
262 phi.push_back(found_node->bucket_index);
263 }
264 }
265 }
266
267 } else {
268 // Handle the case where R_SIZE == 1 with a simpler approach
269 int smaller = bucket_id - 1;
270
271 if (smaller >= 0 &&
272 buckets[smaller].node_id == buckets[bucket_id].node_id) {
273#ifdef FIX_BUCKETS
274 if (fw ? is_bucket_not_fixed_forward(smaller, bucket_id)
275 : is_bucket_not_fixed_backward(smaller, bucket_id))
276#endif
277 {
278 phi.push_back(smaller);
279 }
280 }
281 }
282 return phi;
283}

289void BucketGraph::calculate_neighborhoods(size_t num_closest) {
290 size_t num_nodes = nodes.size();
291
292 // Initialize the neighborhood bitmaps as vectors of uint64_t for forward
293 // and backward neighborhoods
294 neighborhoods_bitmap.resize(num_nodes); // Forward neighborhood
295
296 for (size_t i = 0; i < num_nodes; ++i) {
297 std::vector<std::pair<double, size_t>>
298 forward_distances; // Distances for forward neighbors
299
300 for (size_t j = 0; j < num_nodes; ++j) {
301 if (i != j) {
302 // Forward distance (i -> j)
303 double forward_distance = getcij(i, j);
304 forward_distances.push_back({forward_distance, j});
305 }
306 }
307
308 // Sort distances to find the closest nodes
309 pdqsort(forward_distances.begin(), forward_distances.end());
310
311 // Initialize the neighborhood bitmap vector for node i (forward and
312 // backward)
313 size_t num_segments = (num_nodes + 63) / 64;
314 neighborhoods_bitmap[i].resize(num_segments,
315 0); // Resizing for forward bitmap
316
317 // Include the node itself in both forward and backward neighborhoods
318 size_t segment_self = i >> 6;
319 size_t bit_position_self = i & 63;
320 neighborhoods_bitmap[i][segment_self] |=
321 (1ULL << bit_position_self); // Forward
322
323 // Map the top 'num_closest' closest nodes for forward and set them in
324 // the backward neighborhoods
325 for (size_t k = 0; k < num_closest && k < forward_distances.size();
326 ++k) {
327 size_t node_index = forward_distances[k].second;
328
329 // Determine the segment and the bit within the segment for the
330 // node_index (forward)
331 size_t segment = node_index >> 6;
332 size_t bit_position = node_index & 63;
333 neighborhoods_bitmap[i][segment] |=
334 (1ULL << bit_position); // Forward neighbor
335 }
336 }

349void BucketGraph::augment_ng_memories(std::vector<double> &solution,
350 std::vector<Path> &paths, bool aggressive,
351 int eta1, int eta2, int eta_max, int nC) {
352 // Pre-allocate with estimated size to avoid reallocations
353 std::vector<std::vector<uint16_t>> cycles;
354 cycles.reserve(paths.size() /
355 4); // Estimate 25% of paths might have cycles
356
357 // Detect cycles in fractional paths
358 for (size_t col = 0; col < paths.size(); ++col) {
359 const double epsilon = 1e-2;
360 if (solution[col] <= epsilon || solution[col] >= 1 - epsilon) continue;
361
362 const auto &path = paths[col];
363 ankerl::unordered_dense::map<uint16_t, int> visited_clients;
364 visited_clients.reserve(path.size());
365
366 for (size_t i = 0; i < path.size(); ++i) {
367 const uint16_t client = static_cast<uint16_t>(path[i]);
368 if (client == 0 || client == N_SIZE - 1) continue;
369
370 auto it = visited_clients.find(client);
371 if (it != visited_clients.end()) {
372 // Cycle found - extract it
373 std::vector<uint16_t> cycle;
374 cycle.reserve(i - it->second + 1);
375
376 for (size_t j = it->second; j <= i; ++j) {
377 cycle.push_back(static_cast<uint16_t>(path[j]));
378 }
379
380 cycles.push_back(cycle);
381 break;
382 }
383 visited_clients[client] = i;
384 }
385 }
386
387 if (cycles.empty()) return;
388
389 // Sort cycles by size
390 pdqsort(cycles.begin(), cycles.end(),
391 [](const auto &a, const auto &b) { return a.size() < b.size(); });
392
393 // Process cycles
394 int forbidden_count = 0;
395
396 // Lambda for counting bits in neighborhoods
397 auto count_neighborhood_bits = [this, eta_max](uint16_t node) -> int {
398 int count = 0;
399 for (const auto &segment : neighborhoods_bitmap[node]) {
400 count += __builtin_popcountll(segment);
401 if (count >= eta_max) return count;
402 }
403 return count;
404 };
405
406 for (const auto &cycle : cycles) {
407 if (cycle.empty()) continue;
408
409 // Check if cycle can be forbidden
410 bool can_forbid = true;
411 for (const uint16_t node : cycle) {
412 if (count_neighborhood_bits(node) >= eta_max) {
413 can_forbid = false;
414 break;
415 }
416 }
417
418 if (can_forbid && (cycle.size() <= eta1 || forbidden_count < eta2)) {
419 // Convert uint16_t vector to int vector for forbidCycle
420 std::vector<int> int_cycle(cycle.begin(), cycle.end());
421 forbidCycle(int_cycle, aggressive);
422 if (++forbidden_count >= eta2) break;
423 }
424 }

435void BucketGraph::forbidCycle(const std::vector<int> &cycle, bool aggressive) {
436 for (size_t i = 0; i < cycle.size() - 1; ++i) {
437 int v1 = cycle[i];
438 int v2 = cycle[i + 1];
439
440 // Update the bitmap to forbid v2 in the neighborhood of v1
441 size_t segment = v2 >> 6;
442 size_t bit_position = v2 & 63;
443 neighborhoods_bitmap[v1][segment] |= (1ULL << bit_position);
444
445 if (aggressive) {
446 segment = v1 >> 6;
447 bit_position = v1 & 63;
448 neighborhoods_bitmap[v2][segment] |= (1ULL << bit_position);
449 }
450 }

453void BucketGraph::set_adjacency_list_manual() {
454 // Clear existing arcs for each node
455 for (auto &node : nodes) {
456 node.clear_arcs(); // Remove any existing arcs associated with the node
457 }
458
459 // Step 1: Compute the clusters using MST-based clustering
460 MST mst_solver(nodes,
461 [&](int from, int to) { return this->getcij(from, to); });
462 double theta = 1.0; // Experiment with different values of θ
463 auto clusters = mst_solver.cluster(theta);
464
465 // Create a job-to-cluster mapping (cluster ID for each job/node)
466 std::vector<int> job_to_cluster(
467 nodes.size(), -1); // Mapping from job (node) to cluster ID
468 for (int cluster_id = 0; cluster_id < clusters.size(); ++cluster_id) {
469 for (int job : clusters[cluster_id]) {
470 job_to_cluster[job] = cluster_id;
471 }
472 }
473
474 // Step 2: Modify add_arcs_for_node to give priority based on cluster
475 // membership
476 auto add_arcs_for_node = [&](const VRPNode &node, int from_bucket,
477 std::vector<double> &res_inc) {
478 using Arc = std::tuple<double, int, std::vector<double>,
479 double>; // Arc: priority, to_node, resource
480 // increments, cost increment
481
482 std::vector<Arc> best_arcs;
483 best_arcs.reserve(nodes.size()); // Reserve space for forward arcs
484
485 std::vector<Arc> best_arcs_rev;
486 best_arcs_rev.reserve(nodes.size()); // Reserve space for reverse arcs
487
488 for (const auto &next_node : nodes) {
489 if (!manual_arcs.has_arc(node.id, next_node.id))
490 continue; // Skip arcs not in the manual list
491
492 if (next_node.id == options.depot || node.id == next_node.id)
493 continue; // Skip depot and same node
494
495 auto travel_cost =
496 getcij(node.id, next_node.id); // Calculate travel cost
497 double cost_inc =
498 travel_cost - next_node.cost; // Adjust cost increment by
499 // subtracting next node's cost
500
501 for (int r = 0; r < options.resources.size(); ++r) {
502 if (options.resources[r] == "time") {
503 res_inc[r] = travel_cost +
504 node.duration; // Update resource increment
505 // based on node duration
506 } else {
507 res_inc[r] = node.consumption[r];
508 }
509 }
510 // res_inc[TIME_INDEX] += travel_cost; // Add travel time to
511 // resource increment
512
513 int to_bucket = next_node.id;
514 if (from_bucket == to_bucket)
515 continue; // Skip arcs that loop back to the same bucket
516
517 bool feasible =
518 true; // Check feasibility based on resource constraints
519 for (int r = 0; r < options.resources.size(); ++r) {
520 if (node.lb[r] + res_inc[r] > next_node.ub[r]) {
521 feasible = false;
522 break;
523 }
524 }
525 if (!feasible) continue; // Skip infeasible arcs
526
527 // Step 3: Calculate priority based on cluster membership
528 double priority_value;
529 double reverse_priority_value;
530 if (job_to_cluster[node.id] == job_to_cluster[next_node.id]) {
531 // Higher priority if both nodes are in the same cluster
532 priority_value =
533 5.0 +
534 1.E-5 * next_node.start_time; // Adjust weight for
535 // same-cluster priority
536 reverse_priority_value =
537 1.0 + 1.E-5 * node.start_time; // Adjust weight for
538 // same-cluster priority
539 } else {
540 // Lower priority for cross-cluster arcs
541 priority_value =
542 1.0 +
543 1.E-5 * next_node.start_time; // Higher base value for
544 // cross-cluster arcs
545 reverse_priority_value =
546 5.0 + 1.E-5 * node.start_time; // Higher base value for
547 // cross-cluster arcs
548 }
549
550 best_arcs.emplace_back(priority_value, next_node.id, res_inc,
551 cost_inc); // Store the forward arc
552 best_arcs_rev.emplace_back(reverse_priority_value, next_node.id,
553 res_inc,
554 cost_inc); // Store the reverse arc
555 }
556
557 // Add forward arcs from the current node to its neighbors
558 for (const auto &arc : best_arcs) {
559 auto [priority_value, to_bucket, res_inc_local, cost_inc] = arc;
560 nodes[node.id].template add_arc<Direction::Forward>(
561 node.id, to_bucket, res_inc_local, cost_inc,
562 priority_value); // Add forward arc
563 }
564
565 // Add reverse arcs from neighboring nodes to the current node
566 for (const auto &arc : best_arcs_rev) {
567 auto [priority_value, to_bucket, res_inc_local, cost_inc] = arc;
568 nodes[to_bucket].template add_arc<Direction::Backward>(
569 to_bucket, node.id, res_inc_local, cost_inc,
570 priority_value); // Add reverse arc
571 }
572 };
573
574 // Step 4: Iterate over all nodes to set the adjacency list
575 for (const auto &VRPNode : nodes) {
576 if (VRPNode.id == options.end_depot)
577 continue; // Skip the last node (depot)
578
579 std::vector<double> res_inc(
580 intervals.size()); // Resource increment vector
581 add_arcs_for_node(VRPNode, VRPNode.id,
582 res_inc); // Add arcs for the current node
583 }
584}

589void BucketGraph::mono_initialization() {
590 // Clear previous data
591 merged_labels.clear();
592 merged_labels.reserve(100);
593 fw_c_bar.clear();
594 bw_c_bar.clear();
595
596 dominance_checks_per_bucket.assign(fw_buckets_size + 1, 0);
597 non_dominated_labels_per_bucket = 0;
598
599 // Resize cost vectors to match the number of buckets
600 fw_c_bar.resize(fw_buckets_size, std::numeric_limits<double>::infinity());
601 bw_c_bar.resize(bw_buckets_size, std::numeric_limits<double>::infinity());
602
603 auto &num_buckets =
604 assign_buckets<Direction::Forward>(num_buckets_fw, num_buckets_bw);
605 auto &num_bucket_index = assign_buckets<Direction::Forward>(
606 num_buckets_index_fw, num_buckets_index_bw);
607
608 int num_intervals = options.main_resources.size();
609 std::vector<double> total_ranges(num_intervals);
610 std::vector<double> base_intervals(num_intervals);
611
612 auto &VRPNode = nodes[0]; // Example for the first node
613
614 // Calculate base intervals and total ranges for each resource dimension
615 for (int r = 0; r < intervals.size(); ++r) {
616 base_intervals[r] =
617 (VRPNode.ub[r] - VRPNode.lb[r]) / intervals[r].interval;
618 }
619
620 // Clear forward and backward buckets
621 for (auto b = 0; b < fw_buckets.size(); b++) {
622 fw_buckets[b].clear();
623 bw_buckets[b].clear();
624 }
625
626 // Initialize forward buckets (generic for multiple dimensions)
627 std::vector<int> current_pos(num_intervals, 0);
628
629 std::vector<double> interval_starts(num_intervals);
630 std::vector<double> interval_ends(num_intervals);
631
632 // Initialize interval_starts to lower bounds and interval_ends to upper
633 // bounds
634 for (int r = 0; r < num_intervals; ++r) {
635 interval_starts[r] = VRPNode.lb[r];
636 interval_ends[r] = VRPNode.ub[r];
637 }
638 int offset_fw = 0;
639 int offset_bw = 0;
640
641 // Lambda to handle forward and backward direction initialization across
642 // combinations
643 auto initialize_intervals_combinations = [&](bool is_forward) {
644 int &offset = is_forward ? offset_fw : offset_bw;
645 auto &label_pool = is_forward ? label_pool_fw : label_pool_bw;
646 auto &buckets = is_forward ? fw_buckets : bw_buckets;
647 const auto &depot_id = is_forward ? options.depot : options.end_depot;
648 // print depot_id
649 int calculated_index_base =
650 is_forward ? num_buckets_index_fw[options.depot]
651 : num_buckets_index_bw[options.end_depot];
652
653 std::vector<int> current_pos(
654 num_intervals,
655 0); // Tracks current position in each interval dimension
656 std::vector<double> interval_bounds(num_intervals);
657
658 // Recursive lambda to generate all combinations of intervals
659 std::function<void(int)> generate_combinations = [&](int depth) {
660 if (depth == num_intervals) {
661 // All dimensions processed, now initialize for the current
662 // combination
663 auto depot = label_pool->acquire();
664 int calculated_index = calculated_index_base + offset;
665
666 // Set interval_bounds to current combination of interval starts
667 // or ends
668 for (int r = 0; r < num_intervals; ++r) {
669 interval_bounds[r] =
670 is_forward
671 ? interval_starts[r] +
672 current_pos[r] *
673 roundToTwoDecimalPlaces(base_intervals[r])
674 : interval_ends[r] -
675 current_pos[r] * roundToTwoDecimalPlaces(
676 base_intervals[r]);
677 }
678
679 // print interval
680 // Initialize depot with the current interval boundaries
681 depot->initialize(calculated_index, 0.0, interval_bounds,
682 depot_id);
683 depot->is_extended = false;
684 depot->cost += pstep_duals.getThreeTwoDualValue(depot_id);
685 depot->cost += pstep_duals.getThreeThreeDualValue(depot_id);
686 set_node_visited(depot->visited_bitmap, depot_id);
687 SRC_MODE_BLOCK(
688 depot->SRCmap.assign(cut_storage->SRCDuals.size(), 0);)
689 buckets[calculated_index].add_label(depot);
690 buckets[calculated_index].node_id = depot_id;
691
692 offset++; // Increment offset for each combination
693 return;
694 }
695
696 // Loop through each interval in the current dimension
697 for (int k = 0; k < intervals[depth].interval; ++k) {
698 current_pos[depth] = k;
699 generate_combinations(depth + 1);
700 }
701 };
702
703 // Start generating combinations from depth 0
704 generate_combinations(0);
705 };
706
707 // Call the lambda for both forward and backward directions, ensuring all
708 // combinations are processed
709 initialize_intervals_combinations(true); // Forward direction
710}

787void BucketGraph::generate_arcs() {

791 // Task for Forward Direction

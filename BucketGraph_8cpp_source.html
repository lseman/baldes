196std::vector<int> BucketGraph::computePhi(int &bucket_id, bool fw) {
197 std::vector<int> phi;
198
199 // Get references based on the direction.
200 auto &buckets = fw ? fw_buckets : bw_buckets;
201 auto &node_interval_trees =
202 fw ? fw_node_interval_trees : bw_node_interval_trees;
203
204 // Multi-dimensional resource case.
205 if (options.resources.size() > 1) {
206 if (bucket_id < 0 || bucket_id >= static_cast<int>(buckets.size()))
207 return phi;
208
209 const int n_dims = static_cast<int>(intervals.size());
210 std::vector<double> base_intervals(n_dims, 0.0);
211
212 // Get the node associated with the current bucket.
213 int node_id = buckets[bucket_id].node_id;
214 const VRPNode &node = nodes[node_id];
215
216 // Recompute splits and base intervals per dimension using the same
217 // logic as in define_buckets.
218 for (int r = 0; r < n_dims; ++r) {
219 double full_range = R_max[r] - R_min[r];
220 double node_range = node.ub[r] - node.lb[r];
221 int splits = 1;
222 if (std::fabs(full_range) >
223 std::numeric_limits<double>::epsilon()) {
224 double splits_d =
225 (node_range * intervals[r].interval) / full_range;
226 splits = std::max(1, static_cast<int>(std::round(splits_d)));
227 }
228 base_intervals[r] = node_range / static_cast<double>(splits);
229 }
230
231 // Start with the current bucket's baseline.
232 const auto &current_bucket = buckets[bucket_id];
233 std::vector<double> target = fw ? current_bucket.lb : current_bucket.ub;
234
235 // Adjust the target vector by one base interval for each dimension.
236 // For forward, shift left (i.e. subtract) but do not go below the
237 // node's lb. For backward, shift right (i.e. add) but do not exceed the
238 // node's ub.
239 for (int r = 0; r < n_dims; ++r) {
240 if (fw)
241 target[r] = std::max(node.lb[r], target[r] - base_intervals[r]);
242 else
243 target[r] = std::min(node.ub[r], target[r] + base_intervals[r]);
244 }
245
246 // Look up the candidate bucket in the node's interval tree.
247 TreeNode *found_node = node_interval_trees[node_id].find(target);
248 if (found_node != nullptr &&
249 buckets[found_node->bucket_index].node_id == node_id) {
250#ifdef FIX_BUCKETS
251 bool not_fixed = fw ? is_bucket_not_fixed_forward(
252 bucket_id, found_node->bucket_index)
253 : is_bucket_not_fixed_backward(
254 bucket_id, found_node->bucket_index);
255 // if (not_fixed)
256#endif
257 {
258 phi.push_back(found_node->bucket_index);
259 }
260 }
261 }
262 // Single-resource case (simpler).
263 else {
264 int neighbor = fw ? bucket_id - 1 : bucket_id - 1;
265 if (neighbor >= 0 && neighbor < static_cast<int>(buckets.size()) &&
266 buckets[neighbor].node_id == buckets[bucket_id].node_id) {
267#ifdef FIX_BUCKETS
268 bool not_fixed =
269 fw ? is_bucket_not_fixed_forward(neighbor, bucket_id)
270 : is_bucket_not_fixed_backward(neighbor, bucket_id);
271 // if (not_fixed)
272#endif
273 {
274 phi.push_back(neighbor);
275 }
276 }
277 }
278 return phi;
279}

285void BucketGraph::calculate_neighborhoods(size_t num_closest) {
286 const size_t num_nodes = nodes.size();
287
288 // Resize the neighborhoods_bitmap vector so that there's one bitmap per
289 // node.
290 neighborhoods_bitmap.resize(num_nodes);
291
292 // Pre-calculate the number of segments needed (each segment is 64 bits).
293 const size_t num_segments = (num_nodes + 63) / 64;
294
295 // Process each node to compute its neighborhood.
296 for (size_t i = 0; i < num_nodes; ++i) {
297 // Create a vector of (distance, node_index) pairs for node i.
298 std::vector<std::pair<double, size_t>> forward_distances;
299 forward_distances.reserve(num_nodes - 1);
300
301 for (size_t j = 0; j < num_nodes; ++j) {
302 if (i == j) continue;
303 double distance = getcij(i, j);
304 forward_distances.push_back({distance, j});
305 }
306
307 // Sort distances to obtain the closest nodes first.
308 pdqsort(forward_distances.begin(), forward_distances.end(),
309 [](const auto &a, const auto &b) { return a.first < b.first; });
310
311 // Initialize the neighborhood bitmap for node i with all segments set
312 // to zero.
313 neighborhoods_bitmap[i].assign(num_segments, 0ULL);
314
315 // Always include the node itself in its neighborhood.
316 size_t self_segment = i >> 6;
317 size_t self_bit = i & 63;
318 neighborhoods_bitmap[i][self_segment] |= (1ULL << self_bit);
319
320 // Add the top 'num_closest' closest nodes as forward neighbors.
321 for (size_t k = 0; k < num_closest && k < forward_distances.size();
322 ++k) {
323 size_t neighbor_index = forward_distances[k].second;
324 size_t segment = neighbor_index >> 6;
325 size_t bit_position = neighbor_index & 63;
326 neighborhoods_bitmap[i][segment] |= (1ULL << bit_position);
327 }
328 }

331void BucketGraph::prune_ng_cycles(int max_age, int min_usage,
332 int current_iteration) {
333 for (auto it = ng_cycles.begin(); it != ng_cycles.end();) {
334 if (current_iteration - it->last_used_iteration > max_age &&
335 it->usage_count < min_usage) {
336 // Remove the corresponding forbidden edges from the
337 // neighborhoods_bitmap.
338 for (size_t i = 0; i < it->cycle.size() - 1; ++i) {
339 int v1 = it->cycle[i];
340 int v2 = it->cycle[i + 1];
341 // Clear the bit corresponding to v2 in the neighborhood of
342 // v1.
343 size_t segment = static_cast<size_t>(v2) >> 6;
344 size_t bit_position = static_cast<size_t>(v2) & 63;
345 neighborhoods_bitmap[v1][segment] &=
346 ~bit_mask_lookup[bit_position];
347
348 // If your approach was aggressive, also clear the reverse.
349 segment = static_cast<size_t>(v1) >> 6;
350 bit_position = static_cast<size_t>(v1) & 63;
351 neighborhoods_bitmap[v2][segment] &=
352 ~bit_mask_lookup[bit_position];
353 }
354 it = ng_cycles.erase(it);
355 } else {
356 ++it;
357 }
358 }

361inline bool cycleMatchesCurrentSolution(const std::vector<uint16_t> &cycle,
362 const std::vector<double> &solution,
363 const std::vector<Path> &paths) {
364 const double epsilon = 1e-3;
365 // Iterate over all paths.
366 for (size_t pathIndex = 0; pathIndex < paths.size(); ++pathIndex) {
367 // Only consider paths with significant weight.
368 if (solution[pathIndex] <= epsilon) continue;
369
370 const auto &path = paths[pathIndex];
371
372 // Use std::search to see if the cycle appears as a contiguous
373 // subsequence in the path.
374 if (std::search(path.begin(), path.end(), cycle.begin(), cycle.end()) !=
375 path.end()) {
376 return true; // Cycle found in this path.
377 }
378 }
379 return false; // No matching path found for the cycle.
380}

392void BucketGraph::augment_ng_memories(std::vector<double> &solution,
393 std::vector<Path> &paths, bool aggressive,
394 int eta1, int eta2, int eta_max, int nC) {
395 // Pre-allocate an estimated number of cycles.
396 std::vector<CycleData> cycles;
397 cycles.reserve(paths.size() / 4);
398 ng_iteration_counter++;
399
400 // Detect cycles in fractional paths.
401 for (size_t pathIndex = 0; pathIndex < paths.size(); ++pathIndex) {
402 const double epsilon = 1e-3;
403 double sol_val = solution[pathIndex];
404 if (sol_val <= epsilon) continue;
405 const auto &path = paths[pathIndex];
406 ankerl::unordered_dense::map<uint16_t, int> visited_clients;
407 visited_clients.reserve(path.size());
408
409 // Iterate over the nodes in the path to detect cycles.
410 for (size_t i = 0; i < path.size(); ++i) {
411 uint16_t client = path[i];
412 // Skip depot nodes.
413 if (client == 0 || client == N_SIZE - 1) continue;
414 auto it = visited_clients.find(client);
415 if (it != visited_clients.end()) {
416 // Cycle found: extract cycle from first occurrence to current
417 // index.
418 std::vector<uint16_t> cycle;
419 cycle.reserve(i - it->second + 1);
420 for (size_t j = it->second; j <= i; ++j)
421 cycle.push_back(path[j]);
422
423 CycleData data;
424 data.cycle = std::move(cycle);
425 // Option 1: Use raw solution value as violation score.
426 // data.metric = sol_val;
427 // Option 2: Use fractionality (how far is sol_val from an
428 // integer).
429 data.metric = std::min(sol_val, 1.0 - sol_val);
430 cycles.push_back(std::move(data));
431 break; // Process only one cycle per path.
432 }
433 visited_clients[client] = i;
434 }
435 }
436 if (cycles.empty()) return;
437
438 // Sort cycles by descending metric value (i.e. cycles with highest
439 // violation or fractionality first).
440 pdqsort(cycles.begin(), cycles.end(),
441 [](const CycleData &a, const CycleData &b) {
442 return a.metric > b.metric;
443 });
444
445 int forbidden_count = 0;
446
447 // Cache neighborhood bit counts to avoid redundant __builtin_popcountll
448 // calls.
449 std::vector<int> cached_counts(N_SIZE, -1);
450 auto count_neighborhood_bits = [this, eta_max,
451 &cached_counts](uint16_t node) -> int {
452 if (cached_counts[node] >= 0) return cached_counts[node];
453 int count = 0;
454 for (const auto &segment : neighborhoods_bitmap[node]) {
455 count += __builtin_popcountll(segment);
456 if (count >= eta_max) break;
457 }
458 cached_counts[node] = count;
459 return count;
460 };
461
462 // Adapt thresholds based on the aggressive flag.
463 int effective_eta1 = eta1;
464 int effective_eta2 = eta2;
465
466 // Process each detected cycle.
467 for (const auto &data : cycles) {
468 const auto &cycle = data.cycle;
469 if (cycle.empty()) continue;
470
471 bool can_forbid = true;
472 // Check if any node in the cycle has too many neighbor bits.
473 for (uint16_t node : cycle) {
474 if (count_neighborhood_bits(node) >= eta_max) {
475 can_forbid = false;
476 break;
477 }
478 }
479
480 // If the cycle qualifies based on its size or our current count.
481 if (can_forbid && (static_cast<int>(cycle.size()) <= effective_eta1 ||
482 forbidden_count < effective_eta2)) {
483 std::vector<int> int_cycle(cycle.begin(), cycle.end());
484 forbidCycle(int_cycle, aggressive);
485 ++forbidden_count;
486 if (forbidden_count >= effective_eta2) break;
487 }
488 }
489 // If no cycle was added, try again without the effective_eta1 restriction.
490 if (forbidden_count == 0) {
491 for (const auto &data : cycles) {
492 const auto &cycle = data.cycle;
493 if (cycle.empty()) continue;
494
495 bool can_forbid = true;
496 for (uint16_t node : cycle) {
497 if (count_neighborhood_bits(node) >= eta_max) {
498 can_forbid = false;
499 break;
500 }
501 }
502 if (can_forbid && forbidden_count < effective_eta2) {
503 std::vector<int> int_cycle(cycle.begin(), cycle.end());
504 forbidCycle(int_cycle, aggressive);
505 ++forbidden_count;
506 if (forbidden_count >= effective_eta2) break;
507 }
508 }
509 }

521void BucketGraph::forbidCycle(const std::vector<int> &cycle, bool aggressive) {
522 // If the cycle is too short, there is nothing to forbid.
523 if (cycle.size() < 2) return;
524
525 // Process each adjacent pair in the cycle.
526 for (size_t i = 0; i < cycle.size() - 1; ++i) {
527 int v1 = cycle[i];
528 int v2 = cycle[i + 1];
529
530 // Forbid v2 in the neighborhood of v1.
531 size_t segment = static_cast<size_t>(v2) >> 6;
532 size_t bit_position = static_cast<size_t>(v2) & 63;
533 neighborhoods_bitmap[v1][segment] |= bit_mask_lookup[bit_position];
534
535 if (aggressive) {
536 // Additionally, forbid v1 in the neighborhood of v2.
537 segment = static_cast<size_t>(v1) >> 6;
538 bit_position = static_cast<size_t>(v1) & 63;
539 neighborhoods_bitmap[v2][segment] |= bit_mask_lookup[bit_position];
540 }
541 }

544void BucketGraph::set_adjacency_list_manual() {
545 // Clear existing arcs for each node
546 for (auto &node : nodes) {
547 node.clear_arcs(); // Remove any existing arcs associated with the
548 // node
549 }
550
551 // Step 1: Compute the clusters using MST-based clustering
552 MST mst_solver(nodes,
553 [&](int from, int to) { return this->getcij(from, to); });
554 double theta = 1.0; // Experiment with different values of
555 auto clusters = mst_solver.cluster(theta);
556
557 // Create a job-to-cluster mapping (cluster ID for each job/node)
558 std::vector<int> job_to_cluster(
559 nodes.size(), -1); // Mapping from job (node) to cluster ID
560 for (int cluster_id = 0; cluster_id < clusters.size(); ++cluster_id) {
561 for (int job : clusters[cluster_id]) {
562 job_to_cluster[job] = cluster_id;
563 }
564 }
565
566 // Step 2: Modify add_arcs_for_node to give priority based on cluster
567 // membership
568 auto add_arcs_for_node = [&](const VRPNode &node, int from_bucket,
569 std::vector<double> &res_inc) {
570 using Arc = std::tuple<double, int, std::vector<double>,
571 double>; // Arc: priority, to_node, resource
572 // increments, cost increment
573
574 std::vector<Arc> best_arcs;
575 best_arcs.reserve(nodes.size()); // Reserve space for forward arcs
576
577 std::vector<Arc> best_arcs_rev;
578 best_arcs_rev.reserve(nodes.size()); // Reserve space for reverse arcs
579
580 for (const auto &next_node : nodes) {
581 if (!manual_arcs.has_arc(node.id, next_node.id))
582 continue; // Skip arcs not in the manual list
583
584 if (next_node.id == options.depot || node.id == next_node.id)
585 continue; // Skip depot and same node
586
587 auto travel_cost =
588 getcij(node.id, next_node.id); // Calculate travel cost
589 double cost_inc =
590 travel_cost - next_node.cost; // Adjust cost increment by
591 // subtracting next node's cost
592
593 for (int r = 0; r < options.resources.size(); ++r) {
594 if (options.resources[r] == "time") {
595 res_inc[r] = travel_cost +
596 node.duration; // Update resource increment
597 // based on node duration
598 } else {
599 res_inc[r] = node.consumption[r];
600 }
601 }
602 // res_inc[TIME_INDEX] += travel_cost; // Add travel time to
603 // resource increment
604
605 int to_bucket = next_node.id;
606 if (from_bucket == to_bucket)
607 continue; // Skip arcs that loop back to the same bucket
608
609 bool feasible =
610 true; // Check feasibility based on resource constraints
611 for (int r = 0; r < options.resources.size(); ++r) {
612 if (node.lb[r] + res_inc[r] > next_node.ub[r]) {
613 feasible = false;
614 break;
615 }
616 }
617 if (!feasible) continue; // Skip infeasible arcs
618
619 // Step 3: Calculate priority based on cluster membership
620 double priority_value;
621 double reverse_priority_value;
622 if (job_to_cluster[node.id] == job_to_cluster[next_node.id]) {
623 // Higher priority if both nodes are in the same cluster
624 priority_value =
625 5.0 +
626 1.E-5 * next_node.start_time; // Adjust weight for
627 // same-cluster priority
628 reverse_priority_value =
629 1.0 + 1.E-5 * node.start_time; // Adjust weight for
630 // same-cluster priority
631 } else {
632 // Lower priority for cross-cluster arcs
633 priority_value =
634 1.0 +
635 1.E-5 * next_node.start_time; // Higher base value for
636 // cross-cluster arcs
637 reverse_priority_value =
638 5.0 + 1.E-5 * node.start_time; // Higher base value for
639 // cross-cluster arcs
640 }
641
642 best_arcs.emplace_back(priority_value, next_node.id, res_inc,
643 cost_inc); // Store the forward arc
644 best_arcs_rev.emplace_back(reverse_priority_value, next_node.id,
645 res_inc,
646 cost_inc); // Store the reverse arc
647 }
648
649 // Add forward arcs from the current node to its neighbors
650 for (const auto &arc : best_arcs) {
651 auto [priority_value, to_bucket, res_inc_local, cost_inc] = arc;
652 nodes[node.id].template add_arc<Direction::Forward>(
653 node.id, to_bucket, res_inc_local, cost_inc,
654 priority_value); // Add forward arc
655 }
656
657 // Add reverse arcs from neighboring nodes to the current node
658 for (const auto &arc : best_arcs_rev) {
659 auto [priority_value, to_bucket, res_inc_local, cost_inc] = arc;
660 nodes[to_bucket].template add_arc<Direction::Backward>(
661 to_bucket, node.id, res_inc_local, cost_inc,
662 priority_value); // Add reverse arc
663 }
664 };
665
666 // Step 4: Iterate over all nodes to set the adjacency list
667 for (const auto &VRPNode : nodes) {
668 if (VRPNode.id == options.end_depot)
669 continue; // Skip the last node (depot)
670
671 std::vector<double> res_inc(
672 intervals.size()); // Resource increment vector
673 add_arcs_for_node(VRPNode, VRPNode.id,
674 res_inc); // Add arcs for the current node
675 }
676}

681void BucketGraph::mono_initialization() {
682 // Clear previous data
683 merged_labels.clear();
684 merged_labels.reserve(100);
685 fw_c_bar.clear();
686 bw_c_bar.clear();
687
688 dominance_checks_per_bucket.assign(fw_buckets_size + 1, 0);
689 non_dominated_labels_per_bucket = 0;
690
691 // Resize cost vectors to match the number of buckets
692 fw_c_bar.resize(fw_buckets_size, std::numeric_limits<double>::infinity());
693 bw_c_bar.resize(bw_buckets_size, std::numeric_limits<double>::infinity());
694
695 auto &num_buckets =

698 num_buckets_index_fw, num_buckets_index_bw);
699

701 std::vector<double> total_ranges(num_intervals);
702 std::vector<double> base_intervals(num_intervals);
703
704 auto &VRPNode = nodes[0]; // Example for the first node
705
706 // Calculate base intervals and total ranges for each resource dimension
707 for (int r = 0; r < intervals.size(); ++r) {
708 base_intervals[r] =
709 (VRPNode.ub[r] - VRPNode.lb[r]) / intervals[r].interval;
710 }
711
712 // Clear forward and backward buckets
713 for (auto b = 0; b < fw_buckets.size(); b++) {
714 fw_buckets[b].clear();
715 bw_buckets[b].clear();
716 }
717
718 // Initialize forward buckets (generic for multiple dimensions)
719 std::vector<int> current_pos(num_intervals, 0);
720
721 std::vector<double> interval_starts(num_intervals);
722 std::vector<double> interval_ends(num_intervals);
723
724 // Initialize interval_starts to lower bounds and interval_ends to upper
725 // bounds
726 for (int r = 0; r < num_intervals; ++r) {
727 interval_starts[r] = VRPNode.lb[r];
728 interval_ends[r] = VRPNode.ub[r];
729 }
730 int offset_fw = 0;
731 int offset_bw = 0;
732
733 // Lambda to handle forward and backward direction initialization across
734 // combinations
735 auto initialize_intervals_combinations = [&](bool is_forward) {
736 int &offset = is_forward ? offset_fw : offset_bw;
737 auto &label_pool = is_forward ? label_pool_fw : label_pool_bw;
738 auto &buckets = is_forward ? fw_buckets : bw_buckets;
739 const auto &depot_id = is_forward ? options.depot : options.end_depot;
740 // print depot_id
741 int calculated_index_base =
742 is_forward ? num_buckets_index_fw[options.depot]
743 : num_buckets_index_bw[options.end_depot];
744
745 std::vector<int> current_pos(
746 num_intervals,
747 0); // Tracks current position in each interval dimension
748 std::vector<double> interval_bounds(num_intervals);
749
750 // Recursive lambda to generate all combinations of intervals
751 std::function<void(int)> generate_combinations = [&](int depth) {
752 if (depth == num_intervals) {
753 // All dimensions processed, now initialize for the current
754 // combination
755 auto depot = label_pool->acquire();
756 int calculated_index = calculated_index_base + offset;
757
758 // Set interval_bounds to current combination of interval
759 // starts or ends
760 for (int r = 0; r < num_intervals; ++r) {
761 interval_bounds[r] =
762 is_forward
763 ? interval_starts[r] +
764 current_pos[r] *
765 roundToTwoDecimalPlaces(base_intervals[r])
766 : interval_ends[r] -
767 current_pos[r] * roundToTwoDecimalPlaces(
768 base_intervals[r]);
769 }
770
771 // print interval
772 // Initialize depot with the current interval boundaries
773 depot->initialize(calculated_index, 0.0, interval_bounds,
774 depot_id);
775 depot->is_extended = false;
776 depot->cost += pstep_duals.getThreeTwoDualValue(depot_id);
777 depot->cost += pstep_duals.getThreeThreeDualValue(depot_id);
778 set_node_visited(depot->visited_bitmap, depot_id);
779 SRC_MODE_BLOCK(
780 depot->SRCmap.assign(cut_storage->SRCDuals.size(), 0);)
781 buckets[calculated_index].add_label(depot);
782 buckets[calculated_index].node_id = depot_id;
783
784 offset++; // Increment offset for each combination
785 return;
786 }
787
788 // Loop through each interval in the current dimension
789 for (int k = 0; k < intervals[depth].interval; ++k) {
790 current_pos[depth] = k;
791 generate_combinations(depth + 1);
792 }
793 };
794
795 // Start generating combinations from depth 0
796 generate_combinations(0);
797 };
798
799 // Call the lambda for both forward and backward directions, ensuring
800 // all combinations are processed
801 initialize_intervals_combinations(true); // Forward direction
802}

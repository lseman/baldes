141std::vector<int> BucketGraph::computePhi(int &bucket_id, bool fw) {
142 std::vector<int> phi;
143
144 // Ensure bucket_id is within valid bounds
145 auto &buckets = fw ? fw_buckets : bw_buckets;
146 auto &fixed_buckets = fw ? fw_fixed_buckets : bw_fixed_buckets;
147 auto &node_interval_trees = fw ? fw_node_interval_trees : bw_node_interval_trees;
148
149 if (options.resources.size() > 1) {
150 if (bucket_id >= buckets.size() || bucket_id < 0) return phi;
151
152 std::vector<double> total_ranges(intervals.size());
153 std::vector<double> base_intervals(intervals.size());
154
155 for (int r = 0; r < intervals.size(); ++r) {
156 total_ranges[r] = R_max[r] - R_min[r]; // Ensure integer type for total range
157 base_intervals[r] = total_ranges[r] / intervals[r].interval;
158 }
159
160 // Get the node ID and current bucket
161 int node_id = buckets[bucket_id].node_id;
162 auto &current_bucket = buckets[bucket_id];
163
164 // Retrieve the pre-built Splay Tree for this node
165 auto &node_tree = node_interval_trees[node_id];
166
167 // Search for matching intervals using the existing Splay Tree
168 if (fw) {
169 // Forward search: find the interval just below the current bucket
170 std::vector<double> target_low = current_bucket.lb;
171 for (int r = 0; r < intervals.size(); ++r) {
172 target_low[r] -= base_intervals[r]; // Adjust for the base intervals
173 }
174
175 TreeNode *found_node = node_tree.find(target_low);
176 if (found_node != nullptr && buckets[found_node->bucket_index].node_id == node_id) {
177 // Check if the found bucket is fixed
178#ifdef FIX_BUCKETS
179 if (fixed_buckets[found_node->bucket_index][bucket_id] == 0)
180#endif
181 {
182 phi.push_back(found_node->bucket_index);
183 }
184 }
185 } else {
186 // Backward search: find the interval just above the current bucket
187 std::vector<double> target_high = current_bucket.ub;
188 for (int r = 0; r < intervals.size(); ++r) {
189 target_high[r] += base_intervals[r]; // Adjust for the base intervals
190 }
191
192 TreeNode *found_node = node_tree.find(target_high);
193 if (found_node != nullptr && buckets[found_node->bucket_index].node_id == node_id) {
194 // Check if the found bucket is fixed
195#ifdef FIX_BUCKETS
196 if (fixed_buckets[found_node->bucket_index][bucket_id] == 0)
197#endif
198 {
199 phi.push_back(found_node->bucket_index);
200 }
201 }
202 }
203
204 } else {
205 // Handle the case where R_SIZE == 1 with a simpler approach
206 int smaller = bucket_id - 1;
207
208 if (smaller >= 0 && buckets[smaller].node_id == buckets[bucket_id].node_id) {
209#ifdef FIX_BUCKETS
210 if (fixed_buckets[smaller][bucket_id] == 0)
211#endif
212 {
213 phi.push_back(smaller);
214 }
215 }
216 }
217 return phi;

224void BucketGraph::calculate_neighborhoods(size_t num_closest) {
225 size_t num_nodes = nodes.size();
226
227 // Initialize the neighborhood bitmaps as vectors of uint64_t for forward and backward neighborhoods
228 neighborhoods_bitmap.resize(num_nodes); // Forward neighborhood
229
230 for (size_t i = 0; i < num_nodes; ++i) {
231 std::vector<std::pair<double, size_t>> forward_distances; // Distances for forward neighbors
232
233 for (size_t j = 0; j < num_nodes; ++j) {
234 if (i != j) {
235 // Forward distance (i -> j)
236 double forward_distance = getcij(i, j);
237 forward_distances.push_back({forward_distance, j});
238 }
239 }
240
241 // Sort distances to find the closest nodes
242 pdqsort(forward_distances.begin(), forward_distances.end());
243
244 // Initialize the neighborhood bitmap vector for node i (forward and backward)
245 size_t num_segments = (num_nodes + 63) / 64;
246 neighborhoods_bitmap[i].resize(num_segments, 0); // Resizing for forward bitmap
247
248 // Include the node itself in both forward and backward neighborhoods
249 size_t segment_self = i >> 6;
250 size_t bit_position_self = i & 63;
251 neighborhoods_bitmap[i][segment_self] |= (1ULL << bit_position_self); // Forward
252
253 // Map the top 'num_closest' closest nodes for forward and set them in the backward neighborhoods
254 for (size_t k = 0; k < num_closest && k < forward_distances.size(); ++k) {
255 size_t node_index = forward_distances[k].second;
256
257 // Determine the segment and the bit within the segment for the node_index (forward)
258 size_t segment = node_index >> 6;
259 size_t bit_position = node_index & 63;
260 neighborhoods_bitmap[i][segment] |= (1ULL << bit_position); // Forward neighbor
261 }
262 }

274void BucketGraph::augment_ng_memories(std::vector<double> &solution, std::vector<Path> &paths, bool aggressive,
275 int eta1, int eta2, int eta_max, int nC) {
276 std::set<std::pair<int, int>> forbidden_augmentations;
277 std::vector<std::vector<int>> cycles;
278
279 for (int col = 0; col < paths.size(); ++col) {
280
281 if (solution[col] > 1e-2 && solution[col] < 1 - 1e-2) {
282 ankerl::unordered_dense::map<int, int> visited_clients;
283 std::vector<int> cycle;
284 bool has_cycle = false;
285
286 for (int i = 0; i < paths[col].size(); ++i) {
287 int client = paths[col][i];
288 if (client == 0 || client == N_SIZE - 1) {
289 continue; // Ignore 0 in cycle detection
290 }
291 if (visited_clients.find(client) != visited_clients.end()) {
292 has_cycle = true;
293 // Start from the first occurrence of the repeated client to form the cycle
294 for (int j = visited_clients[client]; j <= i; ++j) { cycle.push_back(paths[col][j]); }
295 break; // Stop once the cycle is stored
296 }
297 visited_clients[client] = i;
298 }
299
300 if (has_cycle) { cycles.push_back(cycle); }
301 }
302 }
303
304 // Sort cycles by size to prioritize smaller cycles
305 pdqsort(cycles.begin(), cycles.end(),
306 [](const std::vector<int> &a, const std::vector<int> &b) { return a.size() < b.size(); });
307 int forbidden_count = 0;
308
309 for (const auto &cycle : cycles) {
310 // Check the current sizes of neighborhoods involved in the cycle
311 bool can_forbid = true;
312 for (const auto &node : cycle) {
313 // Count the number of 1s in neighborhoods_bitmap[node]
314 int count = 0;
315 // print size of neighborhoods_bitmap
316 for (const auto &segment : neighborhoods_bitmap[node]) {
317 count += __builtin_popcountll(segment); // Counts the number of set bits (1s)
318 if (count >= eta_max) {
319 can_forbid = false;
320 break;
321 }
322 }
323 if (!can_forbid) { break; }
324 }
325
326 if (can_forbid && (cycle.size() <= eta1 || (forbidden_count < eta2 && !cycle.empty()))) {
327 // Forbid the cycle
328 forbidCycle(cycle, aggressive);
329 forbidden_count++;
330 }
331
332 if (forbidden_count >= eta2) { break; }
333 }

344void BucketGraph::forbidCycle(const std::vector<int> &cycle, bool aggressive) {
345 for (size_t i = 0; i < cycle.size() - 1; ++i) {
346 int v1 = cycle[i];
347 int v2 = cycle[i + 1];
348
349 // Update the bitmap to forbid v2 in the neighborhood of v1
350 size_t segment = v2 >> 6;
351 size_t bit_position = v2 & 63;
352 neighborhoods_bitmap[v1][segment] |= (1ULL << bit_position);
353
354 if (aggressive) {
355 segment = v1 >> 6;
356 bit_position = v1 & 63;
357 neighborhoods_bitmap[v2][segment] |= (1ULL << bit_position);
358 }
359 }

362void BucketGraph::set_adjacency_list_manual() {
363 // Clear existing arcs for each node
364 for (auto &node : nodes) {
365 node.clear_arcs(); // Remove any existing arcs associated with the node
366 }
367
368 // Step 1: Compute the clusters using MST-based clustering
369 MST mst_solver(nodes, [&](int from, int to) { return this->getcij(from, to); });
370 double theta = 1.0; // Experiment with different values of θ
371 auto clusters = mst_solver.cluster(theta);
372
373 // Create a job-to-cluster mapping (cluster ID for each job/node)
374 std::vector<int> job_to_cluster(nodes.size(), -1); // Mapping from job (node) to cluster ID
375 for (int cluster_id = 0; cluster_id < clusters.size(); ++cluster_id) {
376 for (int job : clusters[cluster_id]) { job_to_cluster[job] = cluster_id; }
377 }
378
379 // Step 2: Modify add_arcs_for_node to give priority based on cluster membership
380 auto add_arcs_for_node = [&](const VRPNode &node, int from_bucket, std::vector<double> &res_inc) {
381 using Arc = std::tuple<double, int, std::vector<double>,
382 double>; // Arc: priority, to_node, resource increments, cost increment
383
384 std::vector<Arc> best_arcs;
385 best_arcs.reserve(nodes.size()); // Reserve space for forward arcs
386
387 std::vector<Arc> best_arcs_rev;
388 best_arcs_rev.reserve(nodes.size()); // Reserve space for reverse arcs
389
390 for (const auto &next_node : nodes) {
391 if (!manual_arcs.has_arc(node.id, next_node.id)) continue; // Skip arcs not in the manual list
392
393 if (next_node.id == options.depot || node.id == next_node.id) continue; // Skip depot and same node
394
395 auto travel_cost = getcij(node.id, next_node.id); // Calculate travel cost
396 double cost_inc = travel_cost - next_node.cost; // Adjust cost increment by subtracting next node's cost
397
398 for (int r = 0; r < options.resources.size(); ++r) {
399 if (options.resources[r] == "time") {
400 res_inc[r] = travel_cost + node.duration; // Update resource increment based on node duration
401 } else {
402 res_inc[r] = node.consumption[r];
403 }
404 }
405 // res_inc[TIME_INDEX] += travel_cost; // Add travel time to resource increment
406
407 int to_bucket = next_node.id;
408 if (from_bucket == to_bucket) continue; // Skip arcs that loop back to the same bucket
409
410 bool feasible = true; // Check feasibility based on resource constraints
411 for (int r = 0; r < options.resources.size(); ++r) {
412 if (node.lb[r] + res_inc[r] > next_node.ub[r]) {
413 feasible = false;
414 break;
415 }
416 }
417 if (!feasible) continue; // Skip infeasible arcs
418
419 // Step 3: Calculate priority based on cluster membership
420 double priority_value;
421 double reverse_priority_value;
422 if (job_to_cluster[node.id] == job_to_cluster[next_node.id]) {
423 // Higher priority if both nodes are in the same cluster
424 priority_value = 5.0 + 1.E-5 * next_node.start_time; // Adjust weight for same-cluster priority
425 reverse_priority_value = 1.0 + 1.E-5 * node.start_time; // Adjust weight for same-cluster priority
426 } else {
427 // Lower priority for cross-cluster arcs
428 priority_value = 1.0 + 1.E-5 * next_node.start_time; // Higher base value for cross-cluster arcs
429 reverse_priority_value = 5.0 + 1.E-5 * node.start_time; // Higher base value for cross-cluster arcs
430 }
431
432 best_arcs.emplace_back(priority_value, next_node.id, res_inc, cost_inc); // Store the forward arc
433 best_arcs_rev.emplace_back(reverse_priority_value, next_node.id, res_inc,
434 cost_inc); // Store the reverse arc
435 }
436
437 // Add forward arcs from the current node to its neighbors
438 for (const auto &arc : best_arcs) {
439 auto [priority_value, to_bucket, res_inc_local, cost_inc] = arc;
440 nodes[node.id].template add_arc<Direction::Forward>(node.id, to_bucket, res_inc_local, cost_inc,
441 priority_value); // Add forward arc
442 }
443
444 // Add reverse arcs from neighboring nodes to the current node
445 for (const auto &arc : best_arcs_rev) {
446 auto [priority_value, to_bucket, res_inc_local, cost_inc] = arc;
447 nodes[to_bucket].template add_arc<Direction::Backward>(to_bucket, node.id, res_inc_local, cost_inc,
448 priority_value); // Add reverse arc
449 }
450 };
451
452 // Step 4: Iterate over all nodes to set the adjacency list
453 for (const auto &VRPNode : nodes) {
454 if (VRPNode.id == options.end_depot) continue; // Skip the last node (depot)
455
456 std::vector<double> res_inc(intervals.size()); // Resource increment vector
457 add_arcs_for_node(VRPNode, VRPNode.id, res_inc); // Add arcs for the current node
458 }
459}

658void BucketGraph::mono_initialization() {
659 // Clear previous data
660 merged_labels.clear();
661 merged_labels.reserve(100);
662 fw_c_bar.clear();
663 bw_c_bar.clear();
664
665 dominance_checks_per_bucket.assign(fw_buckets_size + 1, 0);
666 non_dominated_labels_per_bucket = 0;
667
668 // Resize cost vectors to match the number of buckets
669 fw_c_bar.resize(fw_buckets_size, std::numeric_limits<double>::infinity());
670 bw_c_bar.resize(bw_buckets_size, std::numeric_limits<double>::infinity());
671
672 auto &num_buckets = assign_buckets<Direction::Forward>(num_buckets_fw, num_buckets_bw);
673 auto &num_bucket_index = assign_buckets<Direction::Forward>(num_buckets_index_fw, num_buckets_index_bw);
674
675 int num_intervals = options.main_resources.size();
676 std::vector<double> total_ranges(num_intervals);
677 std::vector<double> base_intervals(num_intervals);
678
679 auto &VRPNode = nodes[0]; // Example for the first node
680
681 // Calculate base intervals and total ranges for each resource dimension
682 for (int r = 0; r < intervals.size(); ++r) {
683 base_intervals[r] = (VRPNode.ub[r] - VRPNode.lb[r]) / intervals[r].interval;
684 }
685
686 // Clear forward and backward buckets
687 for (auto b = 0; b < fw_buckets.size(); b++) {
688 fw_buckets[b].clear();
689 bw_buckets[b].clear();
690 }
691
692 // Initialize forward buckets (generic for multiple dimensions)
693 std::vector<int> current_pos(num_intervals, 0);
694
695 std::vector<double> interval_starts(num_intervals);
696 std::vector<double> interval_ends(num_intervals);
697
698 // Initialize interval_starts to lower bounds and interval_ends to upper bounds
699 for (int r = 0; r < num_intervals; ++r) {
700 interval_starts[r] = VRPNode.lb[r];
701 interval_ends[r] = VRPNode.ub[r];
702 }
703 int offset_fw = 0;
704 int offset_bw = 0;
705
706 // Lambda to handle forward and backward direction initialization across combinations
707 auto initialize_intervals_combinations = [&](bool is_forward) {
708 int &offset = is_forward ? offset_fw : offset_bw;
709 auto &label_pool = is_forward ? label_pool_fw : label_pool_bw;
710 auto &buckets = is_forward ? fw_buckets : bw_buckets;
711 const auto &depot_id = is_forward ? options.depot : options.end_depot;
712 // print depot_id
713 int calculated_index_base =
714 is_forward ? num_buckets_index_fw[options.depot] : num_buckets_index_bw[options.end_depot];
715
716 std::vector<int> current_pos(num_intervals, 0); // Tracks current position in each interval dimension
717 std::vector<double> interval_bounds(num_intervals);
718
719 // Recursive lambda to generate all combinations of intervals
720 std::function<void(int)> generate_combinations = [&](int depth) {
721 if (depth == num_intervals) {
722 // All dimensions processed, now initialize for the current combination
723 auto depot = label_pool->acquire();
724 int calculated_index = calculated_index_base + offset;
725
726 // Set interval_bounds to current combination of interval starts or ends
727 for (int r = 0; r < num_intervals; ++r) {
728 interval_bounds[r] =
729 is_forward ? interval_starts[r] + current_pos[r] * roundToTwoDecimalPlaces(base_intervals[r])
730 : interval_ends[r] - current_pos[r] * roundToTwoDecimalPlaces(base_intervals[r]);
731 }
732
733 // print interval
734 // Initialize depot with the current interval boundaries
735 depot->initialize(calculated_index, 0.0, interval_bounds, depot_id);
736 depot->is_extended = false;
737 depot->cost += pstep_duals.getThreeTwoDualValue(depot_id);
738 depot->cost += pstep_duals.getThreeThreeDualValue(depot_id);
739 set_node_visited(depot->visited_bitmap, depot_id);
740 SRC_MODE_BLOCK(depot->SRCmap.assign(cut_storage->SRCDuals.size(), 0);)
741 buckets[calculated_index].add_label(depot);
742 buckets[calculated_index].node_id = depot_id;
743
744 offset++; // Increment offset for each combination
745 return;
746 }
747
748 // Loop through each interval in the current dimension
749 for (int k = 0; k < intervals[depth].interval; ++k) {
750 current_pos[depth] = k;
751 generate_combinations(depth + 1);
752 }
753 };
754
755 // Start generating combinations from depth 0
756 generate_combinations(0);
757 };
758
759 // Call the lambda for both forward and backward directions, ensuring all combinations are processed
760 initialize_intervals_combinations(true); // Forward direction
761}

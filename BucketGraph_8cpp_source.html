127std::vector<int> BucketGraph::computePhi(int &bucket_id, bool fw) {
128 std::vector<int> phi;
129
130 // Get references based on the direction.
131 auto &buckets = fw ? fw_buckets : bw_buckets;
132
133 // Multi-dimensional resource case.
134 if (options.resources.size() > 1) {
135 if (bucket_id < 0 || bucket_id >= static_cast<int>(buckets.size())) return phi;
136
137 const int n_dims = static_cast<int>(intervals.size());
138 std::vector<double> base_intervals(n_dims, 0.0);
139 std::vector<int> splits_per_dim(n_dims, 1);
140
141 // Get the node associated with the current bucket.
142 int node_id = buckets[bucket_id].node_id;
143 const VRPNode &node = nodes[node_id];
144
145 // Recompute splits and base intervals per dimension using the same
146 // logic as in define_buckets.
147 for (int r = 0; r < n_dims; ++r) {
148 double full_range = R_max[r] - R_min[r];
149 double node_range = node.ub[r] - node.lb[r];
150 int splits = 1;
151 if (std::fabs(full_range) > std::numeric_limits<double>::epsilon()) {
152 double splits_d = (node_range * intervals[r].interval) / full_range;
153 splits = std::max(1, static_cast<int>(std::round(splits_d)));
154 }
155 splits_per_dim[r] = splits;
156 base_intervals[r] = node_range / static_cast<double>(splits);
157 }
158
159 auto &num_buckets_index = fw ? num_buckets_index_fw : num_buckets_index_bw;
160 int local_bucket = bucket_id - num_buckets_index[node_id];
161
162 // Reconstruct the local multi-dimensional position vector for the
163 // current bucket using the same mixed-radix scheme as get_bucket_number.
164 std::vector<int> pos(n_dims, 0);
165 int rem = local_bucket;
166 for (int r = 0; r < n_dims; ++r) {
167 const int splits = splits_per_dim[r];
168 pos[r] = rem % splits;
169 rem /= splits;
170 }
171
172 // Build Phi in the paper's kappa order. Forward buckets have
173 // lb = l_v + kappa * d; backward buckets have ub = u_v - kappa * d.
174 // In both senses Phi_b contains adjacent component-wise smaller
175 // buckets, i.e. kappa' = kappa - e_r. For backward this is a higher
176 // physical resource interval, but still a smaller kappa.
177 for (int r = 0; r < n_dims; ++r) {
178 if (pos[r] == 0) continue;
179 std::vector<int> neighbor_pos = pos;
180 --neighbor_pos[r];
181
182 int candidate_bucket = num_buckets_index[node_id];
183 int multiplier = 1;
184 for (int s = 0; s < n_dims; ++s) {
185 const int splits = splits_per_dim[s];
186 candidate_bucket += neighbor_pos[s] * multiplier;
187 multiplier *= splits;
188 }
189
190 if (candidate_bucket >= 0 && candidate_bucket < static_cast<int>(buckets.size()) &&
191 buckets[candidate_bucket].node_id == node_id) {
192#ifdef FIX_BUCKETS
193 bool not_fixed = fw ? is_bucket_not_fixed_forward(bucket_id, candidate_bucket)
194 : is_bucket_not_fixed_backward(bucket_id, candidate_bucket);
195 // if (not_fixed)
196#endif
197 { phi.push_back(candidate_bucket); }
198 }
199 }
200 }
201 // Single-resource case (simpler).
202 else {
203 // Phi is kappa - 1 for both senses. Because backward buckets are stored
204 // in paper kappa order (kappa 0 is the interval touching u_v), this is
205 // also bucket_id - 1 even though physical resource values go up.
206 int neighbor = bucket_id - 1;
207 bool valid_neighbor = neighbor >= 0;
208 if (valid_neighbor && neighbor >= 0 && neighbor < static_cast<int>(buckets.size()) &&
209 buckets[neighbor].node_id == buckets[bucket_id].node_id) {
210#ifdef FIX_BUCKETS
211 bool not_fixed = fw ? is_bucket_not_fixed_forward(neighbor, bucket_id)
212 : is_bucket_not_fixed_backward(neighbor, bucket_id);
213 // if (not_fixed)
214#endif
215 { phi.push_back(neighbor); }
216 }
217 }
218 return phi;
219}

225void BucketGraph::calculate_neighborhoods(size_t num_closest) {
226 const size_t num_nodes = nodes.size();
227
228 // Resize the neighborhoods_bitmap vector so that there's one bitmap per
229 // node.
230 neighborhoods_bitmap.resize(num_nodes);
231
232 // Pre-calculate the number of segments needed (each segment is 64 bits).
233 const size_t num_segments = (num_nodes + 63) / 64;
234
235 // Process each node to compute its neighborhood.
236 for (size_t i = 0; i < num_nodes; ++i) {
237 // Create a vector of (distance, node_index) pairs for node i.
238 std::vector<std::pair<double, size_t>> forward_distances;
239 forward_distances.reserve(num_nodes - 1);
240
241 for (size_t j = 0; j < num_nodes; ++j) {
242 if (i == j) continue;
243 double distance = getcij(i, j);
244 forward_distances.push_back({distance, j});
245 }
246
247 // Sort distances to obtain the closest nodes first.
248 pdqsort(forward_distances.begin(), forward_distances.end(),
249 [](const auto &a, const auto &b) { return a.first < b.first; });
250
251 // Initialize the neighborhood bitmap for node i with all segments set
252 // to zero.
253 neighborhoods_bitmap[i].assign(num_segments, 0ULL);
254
255 // Always include the node itself in its neighborhood.
256 size_t self_segment = i >> 6;
257 size_t self_bit = i & 63;
258 neighborhoods_bitmap[i][self_segment] |= (1ULL << self_bit);
259
260 // Add the top 'num_closest' closest nodes as forward neighbors.
261 for (size_t k = 0; k < num_closest && k < forward_distances.size(); ++k) {
262 size_t neighbor_index = forward_distances[k].second;
263 size_t segment = neighbor_index >> 6;
264 size_t bit_position = neighbor_index & 63;
265 neighborhoods_bitmap[i][segment] |= (1ULL << bit_position);
266 }
267 }

270void BucketGraph::prune_ng_cycles(int max_age, int min_usage, int current_iteration) {
271 for (auto it = ng_cycles.begin(); it != ng_cycles.end();) {
272 if (current_iteration - it->last_used_iteration > max_age && it->usage_count < min_usage) {
273 // Remove the corresponding forbidden edges from the
274 // neighborhoods_bitmap.
275 for (size_t i = 0; i < it->cycle.size() - 1; ++i) {
276 int v1 = it->cycle[i];
277 int v2 = it->cycle[i + 1];
278 // Clear the bit corresponding to v2 in the neighborhood of
279 // v1.
280 size_t segment = static_cast<size_t>(v2) >> 6;
281 size_t bit_position = static_cast<size_t>(v2) & 63;
282 neighborhoods_bitmap[v1][segment] &= ~bit_mask_lookup[bit_position];
283
284 // If your approach was aggressive, also clear the reverse.
285 segment = static_cast<size_t>(v1) >> 6;
286 bit_position = static_cast<size_t>(v1) & 63;
287 neighborhoods_bitmap[v2][segment] &= ~bit_mask_lookup[bit_position];
288 }
289 it = ng_cycles.erase(it);
290 } else {
291 ++it;
292 }
293 }

296inline bool cycleMatchesCurrentSolution(const std::vector<uint16_t> &cycle, const std::vector<double> &solution,
297 const std::vector<Path> &paths) {
298 const double epsilon = 1e-3;
299 // Iterate over all paths.
300 for (size_t pathIndex = 0; pathIndex < paths.size(); ++pathIndex) {
301 // Only consider paths with significant weight.
302 if (solution[pathIndex] <= epsilon) continue;
303
304 const auto &path = paths[pathIndex];
305
306 // Use std::search to see if the cycle appears as a contiguous
307 // subsequence in the path.
308 if (std::search(path.begin(), path.end(), cycle.begin(), cycle.end()) != path.end()) {
309 return true; // Cycle found in this path.
310 }
311 }
312 return false; // No matching path found for the cycle.
313}

325void BucketGraph::augment_ng_memories(std::vector<double> &solution, std::vector<Path> &paths, bool aggressive,
326 int eta1, int eta2, int eta_max, int nC) {
327 // Pre-allocate an estimated number of cycles.
328 std::vector<CycleData> cycles;
329 cycles.reserve(paths.size() / 4);
330 ng_iteration_counter++;
331
332 // Detect cycles in fractional paths.
333 for (size_t pathIndex = 0; pathIndex < paths.size(); ++pathIndex) {
334 const double epsilon = 1e-3;
335 double sol_val = solution[pathIndex];
336 if (sol_val <= epsilon) continue;
337 const auto &path = paths[pathIndex];
338 ankerl::unordered_dense::map<uint16_t, int> visited_clients;
339 visited_clients.reserve(path.size());
340
341 // Iterate over the nodes in the path to detect cycles.
342 for (size_t i = 0; i < path.size(); ++i) {
343 uint16_t client = path[i];
344 // Skip depot nodes.
345 if (client == 0 || client == N_SIZE - 1) continue;
346 auto it = visited_clients.find(client);
347 if (it != visited_clients.end()) {
348 // Cycle found: extract cycle from first occurrence to current
349 // index.
350 std::vector<uint16_t> cycle;
351 cycle.reserve(i - it->second + 1);
352 for (size_t j = it->second; j <= i; ++j) cycle.push_back(path[j]);
353
354 CycleData data;
355 data.cycle = std::move(cycle);
356 // Option 1: Use raw solution value as violation score.
357 // data.metric = sol_val;
358 // Option 2: Use fractionality (how far is sol_val from an
359 // integer).
360 data.metric = std::min(sol_val, 1.0 - sol_val);
361 cycles.push_back(std::move(data));
362 break; // Process only one cycle per path.
363 }
364 visited_clients[client] = i;
365 }
366 }
367 if (cycles.empty()) return;
368
369 // Sort cycles by descending metric value (i.e. cycles with highest
370 // violation or fractionality first).
371 pdqsort(cycles.begin(), cycles.end(), [](const CycleData &a, const CycleData &b) { return a.metric > b.metric; });
372
373 int forbidden_count = 0;
374
375 // Cache neighborhood bit counts to avoid redundant __builtin_popcountll
376 // calls.
377 std::vector<int> cached_counts(N_SIZE, -1);
378 auto count_neighborhood_bits = [this, eta_max, &cached_counts](uint16_t node) -> int {
379 if (cached_counts[node] >= 0) return cached_counts[node];
380 int count = 0;
381 for (const auto &segment : neighborhoods_bitmap[node]) {
382 count += __builtin_popcountll(segment);
383 if (count >= eta_max) break;
384 }
385 cached_counts[node] = count;
386 return count;
387 };
388
389 // Adapt thresholds based on the aggressive flag.
390 int effective_eta1 = eta1;
391 int effective_eta2 = eta2;
392
393 // Process each detected cycle.
394 for (const auto &data : cycles) {
395 const auto &cycle = data.cycle;
396 if (cycle.empty()) continue;
397
398 bool can_forbid = true;
399 // Check if any node in the cycle has too many neighbor bits.
400 for (uint16_t node : cycle) {
401 if (count_neighborhood_bits(node) >= eta_max) {
402 can_forbid = false;
403 break;
404 }
405 }
406
407 // If the cycle qualifies based on its size or our current count.
408 if (can_forbid && (static_cast<int>(cycle.size()) <= effective_eta1 || forbidden_count < effective_eta2)) {
409 std::vector<int> int_cycle(cycle.begin(), cycle.end());
410 forbidCycle(int_cycle, aggressive);
411 ++forbidden_count;
412 if (forbidden_count >= effective_eta2) break;
413 }
414 }
415 // If no cycle was added, try again without the effective_eta1 restriction.
416 if (forbidden_count == 0) {
417 for (const auto &data : cycles) {
418 const auto &cycle = data.cycle;
419 if (cycle.empty()) continue;
420
421 bool can_forbid = true;
422 for (uint16_t node : cycle) {
423 if (count_neighborhood_bits(node) >= eta_max) {
424 can_forbid = false;
425 break;
426 }
427 }
428 if (can_forbid && forbidden_count < effective_eta2) {
429 std::vector<int> int_cycle(cycle.begin(), cycle.end());
430 forbidCycle(int_cycle, aggressive);
431 ++forbidden_count;
432 if (forbidden_count >= effective_eta2) break;
433 }
434 }
435 }
436}

447void BucketGraph::forbidCycle(const std::vector<int> &cycle, bool aggressive) {
448 // If the cycle is too short, there is nothing to forbid.
449 if (cycle.size() < 2) return;
450
451 // Process each adjacent pair in the cycle.
452 for (size_t i = 0; i < cycle.size() - 1; ++i) {
453 int v1 = cycle[i];
454 int v2 = cycle[i + 1];
455
456 // Forbid v2 in the neighborhood of v1.
457 size_t segment = static_cast<size_t>(v2) >> 6;
458 size_t bit_position = static_cast<size_t>(v2) & 63;
459 neighborhoods_bitmap[v1][segment] |= bit_mask_lookup[bit_position];
460
461 if (aggressive) {
462 // Additionally, forbid v1 in the neighborhood of v2.
463 segment = static_cast<size_t>(v1) >> 6;
464 bit_position = static_cast<size_t>(v1) & 63;
465 neighborhoods_bitmap[v2][segment] |= bit_mask_lookup[bit_position];
466 }
467 }

470void BucketGraph::set_adjacency_list_manual() {
471 // Clear existing arcs for each node
472 for (auto &node : nodes) {
473 node.clear_arcs(); // Remove any existing arcs associated with the
474 // node
475 }
476
477 // Step 1: Compute the clusters using MST-based clustering
478 MST mst_solver(nodes, [&](int from, int to) { return this->getcij(from, to); });
479 double theta = 1.0; // Experiment with different values of θ
480 auto clusters = mst_solver.cluster(theta);
481
482 // Create a job-to-cluster mapping (cluster ID for each job/node)
483 std::vector<int> job_to_cluster(nodes.size(), -1); // Mapping from job (node) to cluster ID
484 for (int cluster_id = 0; cluster_id < clusters.size(); ++cluster_id) {
485 for (int job : clusters[cluster_id]) { job_to_cluster[job] = cluster_id; }
486 }
487
488 // Step 2: Modify add_arcs_for_node to give priority based on cluster
489 // membership
490 auto add_arcs_for_node = [&](const VRPNode &node, int from_bucket, std::vector<double> &res_inc) {
491 using Arc = std::tuple<double, int, std::vector<double>,
492 double>; // Arc: priority, to_node, resource
493 // increments, cost increment
494
495 std::vector<Arc> best_arcs;
496 best_arcs.reserve(nodes.size()); // Reserve space for forward arcs
497
498 std::vector<Arc> best_arcs_rev;
499 best_arcs_rev.reserve(nodes.size()); // Reserve space for reverse arcs
500
501 for (const auto &next_node : nodes) {
502 if (!manual_arcs.has_arc(node.id, next_node.id)) continue; // Skip arcs not in the manual list
503
504 if (next_node.id == options.depot || node.id == next_node.id) continue; // Skip depot and same node
505
506 auto travel_cost = getcij(node.id, next_node.id); // Calculate travel cost
507 double cost_inc = travel_cost - next_node.cost; // Adjust cost increment by
508 // subtracting next node's cost
509
510 for (int r = 0; r < options.resources.size(); ++r) {
511 if (options.resources[r] == "time") {
512 res_inc[r] = travel_cost + node.duration; // Update resource increment
513 // based on node duration
514 } else {
515 res_inc[r] = node.consumption[r];
516 }
517 }
518 // res_inc[TIME_INDEX] += travel_cost; // Add travel time to
519 // resource increment
520
521 int to_bucket = next_node.id;
522 if (from_bucket == to_bucket) continue; // Skip arcs that loop back to the same bucket
523
524 bool feasible = true; // Check feasibility based on resource constraints
525 for (int r = 0; r < options.resources.size(); ++r) {
526 if (node.lb[r] + res_inc[r] > next_node.ub[r]) {
527 feasible = false;
528 break;
529 }
530 }
531 if (!feasible) continue; // Skip infeasible arcs
532
533 // Step 3: Calculate priority based on cluster membership
534 double priority_value;
535 double reverse_priority_value;
536 if (job_to_cluster[node.id] == job_to_cluster[next_node.id]) {
537 // Higher priority if both nodes are in the same cluster
538 priority_value = 5.0 + 1.E-5 * next_node.start_time; // Adjust weight for
539 // same-cluster priority
540 reverse_priority_value = 1.0 + 1.E-5 * node.start_time; // Adjust weight for
541 // same-cluster priority
542 } else {
543 // Lower priority for cross-cluster arcs
544 priority_value = 1.0 + 1.E-5 * next_node.start_time; // Higher base value for
545 // cross-cluster arcs
546 reverse_priority_value = 5.0 + 1.E-5 * node.start_time; // Higher base value for
547 // cross-cluster arcs
548 }
549
550 best_arcs.emplace_back(priority_value, next_node.id, res_inc,
551 cost_inc); // Store the forward arc
552 best_arcs_rev.emplace_back(reverse_priority_value, next_node.id, res_inc,
553 cost_inc); // Store the reverse arc
554 }
555
556 // Add forward arcs from the current node to its neighbors
557 for (const auto &arc : best_arcs) {
558 auto [priority_value, to_bucket, res_inc_local, cost_inc] = arc;
559 nodes[node.id].template add_arc<Direction::Forward>(node.id, to_bucket, res_inc_local, cost_inc,
560 priority_value); // Add forward arc
561 }
562
563 // Add reverse arcs from neighboring nodes to the current node
564 for (const auto &arc : best_arcs_rev) {
565 auto [priority_value, to_bucket, res_inc_local, cost_inc] = arc;
566 nodes[to_bucket].template add_arc<Direction::Backward>(to_bucket, node.id, res_inc_local, cost_inc,
567 priority_value); // Add reverse arc
568 }
569 };
570
571 // Step 4: Iterate over all nodes to set the adjacency list
572 for (const auto &VRPNode : nodes) {
573 if (VRPNode.id == options.end_depot) continue; // Skip the last node (depot)
574
575 std::vector<double> res_inc(intervals.size()); // Resource increment vector
576 add_arcs_for_node(VRPNode, VRPNode.id,
577 res_inc); // Add arcs for the current node
578 }
579}

584void BucketGraph::mono_initialization() {
585 // Clear previous data
586 merged_labels.clear();
587 merged_labels.reserve(100);
588 fw_c_bar.clear();
589 bw_c_bar.clear();
590
591 dominance_checks_per_bucket.assign(fw_buckets_size + 1, 0);

595 fw_c_bar.resize(fw_buckets_size, std::numeric_limits<double>::infinity());
596 bw_c_bar.resize(bw_buckets_size, std::numeric_limits<double>::infinity());

599 auto &num_bucket_index = assign_buckets<Direction::Forward>(num_buckets_index_fw, num_buckets_index_bw);

603 std::vector<double> base_intervals(num_intervals);

606
607 // Calculate base intervals and total ranges for each resource dimension
608 for (int r = 0; r < intervals.size(); ++r) {
609 base_intervals[r] = (VRPNode.ub[r] - VRPNode.lb[r]) / intervals[r].interval;
610 }
611
612 // Clear forward and backward buckets
613 for (auto b = 0; b < fw_buckets.size(); b++) {
614 fw_buckets[b].clear();
615 bw_buckets[b].clear();
616 }
617
618 // Initialize forward buckets (generic for multiple dimensions)
619 std::vector<int> current_pos(num_intervals, 0);
620
621 std::vector<double> interval_starts(num_intervals);
622 std::vector<double> interval_ends(num_intervals);
623
624 // Initialize interval_starts to lower bounds and interval_ends to upper
625 // bounds
626 for (int r = 0; r < num_intervals; ++r) {
627 interval_starts[r] = VRPNode.lb[r];
628 interval_ends[r] = VRPNode.ub[r];
629 }
630 int offset_fw = 0;
631 int offset_bw = 0;
632
633 // Lambda to handle forward and backward direction initialization across
634 // combinations
635 auto initialize_intervals_combinations = [&](bool is_forward) {
636 int &offset = is_forward ? offset_fw : offset_bw;
637 auto &label_pool = is_forward ? label_pool_fw : label_pool_bw;
638 auto &buckets = is_forward ? fw_buckets : bw_buckets;
639 const auto &depot_id = is_forward ? options.depot : options.end_depot;
640 // print depot_id
641 int calculated_index_base =
642 is_forward ? num_buckets_index_fw[options.depot] : num_buckets_index_bw[options.end_depot];
643
644 std::vector<int> current_pos(num_intervals,
645 0); // Tracks current position in each interval dimension
646 std::vector<double> interval_bounds(num_intervals);
647
648 // Recursive lambda to generate all combinations of intervals
649 std::function<void(int)> generate_combinations = [&](int depth) {
650 if (depth == num_intervals) {
651 // All dimensions processed, now initialize for the current
652 // combination
653 auto depot = label_pool->acquire();
654 int calculated_index = calculated_index_base + offset;
655
656 // Set interval_bounds to current combination of interval
657 // starts or ends
658 for (int r = 0; r < num_intervals; ++r) {
659 interval_bounds[r] =
660 is_forward ? interval_starts[r] + current_pos[r] * roundToTwoDecimalPlaces(base_intervals[r])
661 : interval_ends[r] - current_pos[r] * roundToTwoDecimalPlaces(base_intervals[r]);
662 }
663
664 // print interval
665 // Initialize depot with the current interval boundaries
666 depot->initialize(calculated_index, 0.0, interval_bounds, depot_id);
667 depot->is_extended = false;
668 depot->cost += pstep_duals.getThreeTwoDualValue(depot_id);
669 depot->cost += pstep_duals.getThreeThreeDualValue(depot_id);
670 depot->addNode(depot_id);
671 set_node_visited(depot->visited_bitmap, depot_id);
672 SRC_MODE_BLOCK(depot->SRCmap.assign(cut_storage->activeSize(), 0);)
673 buckets[calculated_index].add_label(depot);
674 buckets[calculated_index].node_id = depot_id;
675
676 offset++; // Increment offset for each combination
677 return;
678 }
679
680 // Loop through each interval in the current dimension
681 for (int k = 0; k < intervals[depth].interval; ++k) {
682 current_pos[depth] = k;
683 generate_combinations(depth + 1);
684 }
685 };
686
687 // Start generating combinations from depth 0
688 generate_combinations(0);
689 };
690
691 // Call the lambda for both forward and backward directions, ensuring
692 // all combinations are processed
693 initialize_intervals_combinations(true); // Forward direction
694}

52 ankerl::unordered_dense::set<CandidateSet, CandidateSetHasher, CandidateSetCompare> &candidates,
53 CandidateSet candidate) {
54 auto it = candidates.find(candidate);
55 if (it == candidates.end()) {
56 candidates.emplace(std::move(candidate));
57 return;
58 }
59 if (candidate.violation > it->violation + 1e-9) {
60 candidates.erase(it);
61 candidates.emplace(std::move(candidate));

164 if (v_r_map[i].find(r) != v_r_map[i].end()) continue;
165 candidate_nodes.clear();
166 candidate_nodes.push_back(i);
167 const auto &route = allPaths[r].route;

170 }
171 std::sort(candidate_nodes.begin(), candidate_nodes.end());
172 candidate_nodes.erase(std::unique(candidate_nodes.begin(), candidate_nodes.end()),
173 candidate_nodes.end());
174 if (candidate_nodes.size() < MIN_RANK || candidate_nodes.size() > MAX_RANK) continue;
175 memory_nodes.clear();
176 for (int vertex : rank1_sep_heur_mem4_vertex[i]) {
177 if (vertex > 0 && vertex < N_SIZE - 1 &&
178 !std::binary_search(candidate_nodes.begin(), candidate_nodes.end(), vertex)) {
179 memory_nodes.push_back(vertex);
180 }
181 }
182 if (!memory_nodes.empty()) {
183 std::sort(memory_nodes.begin(), memory_nodes.end());
184 memory_nodes.erase(std::unique(memory_nodes.begin(), memory_nodes.end()), memory_nodes.end());
185 }
186 auto &entry = seed_map[candidate_nodes];
187 if (entry.empty()) {
188 entry = memory_nodes;
189 } else {
190 for (auto v : memory_nodes) {
191 if (!std::binary_search(entry.begin(), entry.end(), v)) { entry.push_back(v); }

194 }
195 }
196 }

201 generateRouteSupportSeeds(const SparseMatrix &A, const std::vector<double> &x,

206 struct RouteSupport {
207 double weight;
208 int path_idx;
209 };
210 std::vector<RouteSupport> route_supports;
211 route_supports.reserve(nonzero_paths.size());
212 for (int path_idx : nonzero_paths) {
213 if (path_idx < 0 || static_cast<size_t>(path_idx) >= allPaths.size() || static_cast<size_t>(path_idx) >= x.size())
214 continue;
215 if (x[path_idx] > 1e-6) { route_supports.push_back({x[path_idx], path_idx}); }
216 }

218 [](const auto &a, const auto &b) { return a.weight > b.weight; });

220 constexpr int MAX_ROUTE_SEEDS = 60;
221 constexpr int MAX_ROUTE_NODE_POOL = 10;
222 constexpr int MAX_MEMORY_PER_SEED = 10;
223 constexpr int MAX_SEEDS_PER_ROUTE = 8;
224 const int n_routes_to_scan = std::min(MAX_ROUTE_SEEDS, static_cast<int>(route_supports.size()));
225
226 std::vector<char> in_candidate(N_SIZE, 0);
227 std::vector<int> route_nodes;
228 std::vector<int> candidate_nodes;
229 std::vector<int> memory_nodes;
230 for (int route_pos = 0; route_pos < n_routes_to_scan; ++route_pos) {
231 const auto &path = allPaths[route_supports[route_pos].path_idx];
232 route_nodes.clear();
233 for (int vertex : path.route) {
234 if (vertex <= 0 || vertex >= N_SIZE - 1) continue;
235 if (std::find(route_nodes.begin(), route_nodes.end(), vertex) == route_nodes.end()) {
236 route_nodes.push_back(vertex);
237 }
238 }
239 if (static_cast<int>(route_nodes.size()) < MIN_RANK) continue;
240
241 std::vector<std::pair<double, int>> scored_nodes;
242 scored_nodes.reserve(route_nodes.size());
243 for (int vertex : route_nodes) {
244 double score = -route_supports[route_pos].weight;
245 if (vertex < static_cast<int>(scores.size()) && !scores[vertex].empty()) {
246 score += 0.05 * scores[vertex].front().cost_score;
247 }
248 scored_nodes.emplace_back(score, vertex);
249 }
250 pdqsort(scored_nodes.begin(), scored_nodes.end(),
251 [](const auto &a, const auto &b) { return a.first < b.first; });
252 if (static_cast<int>(scored_nodes.size()) > MAX_ROUTE_NODE_POOL) {
253 scored_nodes.resize(MAX_ROUTE_NODE_POOL);
254 }
255
256 std::vector<int> node_pool;
257 node_pool.reserve(scored_nodes.size());
258 for (const auto &[_, vertex] : scored_nodes) { node_pool.push_back(vertex); }
259
260 int seeds_for_route = 0;
261 for (int rank = MAX_RANK; rank >= MIN_RANK && seeds_for_route < MAX_SEEDS_PER_ROUTE; --rank) {
262 if (rank > static_cast<int>(node_pool.size())) continue;
263 uint32_t mask = (1u << rank) - 1;
264 const uint32_t limit = (1u << node_pool.size());
265 while (mask < limit && seeds_for_route < MAX_SEEDS_PER_ROUTE) {
266 candidate_nodes.clear();
267 candidate_nodes.reserve(rank);
268 for (size_t bit = 0; bit < node_pool.size(); ++bit) {
269 if (mask & (1u << bit)) candidate_nodes.push_back(node_pool[bit]);
270 }
271 std::sort(candidate_nodes.begin(), candidate_nodes.end());
272
273 for (int node : candidate_nodes) { in_candidate[node] = 1; }
274 memory_nodes.clear();
275 memory_nodes.reserve(MAX_MEMORY_PER_SEED);
276 for (int node : route_nodes) {
277 if (static_cast<int>(memory_nodes.size()) >= MAX_MEMORY_PER_SEED) break;
278 if (!in_candidate[node]) memory_nodes.push_back(node);
279 }
280 for (int base : candidate_nodes) {
281 if (base >= static_cast<int>(rank1_sep_heur_mem4_vertex.size())) continue;
282 for (int neighbor : rank1_sep_heur_mem4_vertex[base]) {
283 if (static_cast<int>(memory_nodes.size()) >= MAX_MEMORY_PER_SEED) break;
284 if (neighbor <= 0 || neighbor >= N_SIZE - 1 || in_candidate[neighbor]) continue;
285 if (std::find(memory_nodes.begin(), memory_nodes.end(), neighbor) == memory_nodes.end()) {
286 memory_nodes.push_back(neighbor);
287 }
288 }
289 }
290 for (int node : candidate_nodes) { in_candidate[node] = 0; }
291
292 if (auto candidate = evaluateBaseThenMemory(candidate_nodes, A, x)) {
293 insertBestCandidate(seeds, std::move(*candidate));
294 }
295 ++seeds_for_route;
296
297 uint32_t c = mask & -mask;
298 uint32_t r = mask + c;
299 if (c == 0) break;
300 mask = (((r ^ mask) >> 2) / c) | r;
301 }
302 }
303 }

307 ankerl::unordered_dense::set<CandidateSet, CandidateSetHasher, CandidateSetCompare>
308 generateLiftedCutSeeds(const SparseMatrix &A, const std::vector<double> &x,
309 const std::vector<std::vector<NodeScore>> &scores) {
310 ankerl::unordered_dense::set<CandidateSet, CandidateSetHasher, CandidateSetCompare> seeds;
311 if (cutStorage == nullptr || cutStorage->empty()) return seeds;
312
313 constexpr int MAX_EXTENSION_POOL = 12;
314 constexpr int MAX_LIFTS_PER_CUT = 10;
315 constexpr int MAX_MEMORY_NODES = 12;
316
317 std::vector<int> base_nodes;
318 std::vector<int> extension_pool;
319 std::vector<int> candidate_nodes;
320 std::vector<int> memory_nodes;
321 std::vector<char> in_candidate(N_SIZE, 0);
322 for (const auto &cut : *cutStorage) {
323 base_nodes.clear();
324 memory_nodes.clear();
325 for (int node = 1; node < N_SIZE - 1; ++node) {
326 const size_t segment = static_cast<size_t>(node) >> 6;
327 const uint64_t bit = 1ULL << (node & 63);
328 if (cut.baseSet[segment] & bit) {
329 base_nodes.push_back(node);
330 } else if (cut.neighbors[segment] & bit) {
331 memory_nodes.push_back(node);
332 }
333 }
334 if (static_cast<int>(base_nodes.size()) < MIN_RANK ||
335 static_cast<int>(base_nodes.size()) >= MAX_RANK)
336 continue;
337
338 for (int node : base_nodes) { in_candidate[node] = 1; }
339 extension_pool.clear();
340 for (int base : base_nodes) {
341 if (base >= static_cast<int>(scores.size())) continue;
342 for (const auto &score : scores[base]) {
343 const int candidate = score.node;
344 if (candidate <= 0 || candidate >= N_SIZE - 1 || in_candidate[candidate]) continue;
345 if (std::find(extension_pool.begin(), extension_pool.end(), candidate) == extension_pool.end()) {
346 extension_pool.push_back(candidate);
347 if (static_cast<int>(extension_pool.size()) >= MAX_EXTENSION_POOL) break;
348 }
349 }
350 if (static_cast<int>(extension_pool.size()) >= MAX_EXTENSION_POOL) break;
351 }
352
353 int lifts_for_cut = 0;
354 const int max_extra = std::min(MAX_RANK - static_cast<int>(base_nodes.size()),
355 static_cast<int>(extension_pool.size()));
356 for (int extra = 1; extra <= max_extra && lifts_for_cut < MAX_LIFTS_PER_CUT; ++extra) {
357 uint32_t mask = (1u << extra) - 1;
358 const uint32_t limit = (1u << extension_pool.size());
359 while (mask < limit && lifts_for_cut < MAX_LIFTS_PER_CUT) {
360 candidate_nodes = base_nodes;
361 for (size_t bit = 0; bit < extension_pool.size(); ++bit) {
362 if (mask & (1u << bit)) candidate_nodes.push_back(extension_pool[bit]);
363 }
364 std::sort(candidate_nodes.begin(), candidate_nodes.end());
365
366 ankerl::unordered_dense::set<int> memory_set;
367 memory_set.reserve(MAX_MEMORY_NODES);
368 for (int node : memory_nodes) {
369 if (static_cast<int>(memory_set.size()) >= MAX_MEMORY_NODES) break;
370 if (!std::binary_search(candidate_nodes.begin(), candidate_nodes.end(), node)) {
371 memory_set.insert(node);
372 }
373 }
374 for (int node : extension_pool) {
375 if (static_cast<int>(memory_set.size()) >= MAX_MEMORY_NODES) break;
376 if (!std::binary_search(candidate_nodes.begin(), candidate_nodes.end(), node)) {
377 memory_set.insert(node);
378 }
379 }
380
381 if (auto candidate = evaluateBaseThenMemory(candidate_nodes, A, x)) {
382 insertBestCandidate(seeds, std::move(*candidate));
383 }
384 ++lifts_for_cut;
385
386 uint32_t c = mask & -mask;
387 uint32_t r = mask + c;
388 if (c == 0) break;
389 mask = (((r ^ mask) >> 2) / c) | r;
390 }
391 }
392 for (int node : base_nodes) { in_candidate[node] = 0; }
393 }
394 return seeds;
395 }

397 ankerl::unordered_dense::set<CandidateSet, CandidateSetHasher, CandidateSetCompare>
398 generateFractionalCooccurrenceSeeds(const SparseMatrix &A, const std::vector<double> &x,
399 const std::vector<std::vector<NodeScore>> &scores) {
400 ankerl::unordered_dense::set<CandidateSet, CandidateSetHasher, CandidateSetCompare> seeds;
401 if (allPaths.empty() || nonzero_paths.empty()) return seeds;
402
403 constexpr int MAX_SUPPORT_NODES = 48;
404 constexpr int MAX_ANCHORS = 24;
405 constexpr int MAX_EXTENSION_POOL = 12;
406 constexpr int MAX_CANDIDATES_ANCHOR = 18;
407 constexpr int MAX_MEMORY_NODES = 12;
408
409 std::vector<double> node_support(N_SIZE, 0.0);
410 for (int path_idx : nonzero_paths) {
411 if (path_idx < 0 || static_cast<size_t>(path_idx) >= allPaths.size() || static_cast<size_t>(path_idx) >= x.size())
412 continue;
413 const double x_val = x[path_idx];
414 if (numericutils::isZero(x_val)) continue;
415 for (int node : allPaths[path_idx].route) {
416 if (node > 0 && node < N_SIZE - 1) { node_support[node] += x_val; }
417 }
418 }
419
420 std::vector<int> support_nodes;
421 support_nodes.reserve(N_SIZE);
422 for (int node = 1; node < N_SIZE - 1; ++node) {
423 if (node_support[node] > 1e-6) support_nodes.push_back(node);
424 }
425 pdqsort(support_nodes.begin(), support_nodes.end(),
426 [&](int a, int b) { return node_support[a] > node_support[b]; });
427 if (static_cast<int>(support_nodes.size()) > MAX_SUPPORT_NODES) support_nodes.resize(MAX_SUPPORT_NODES);
428 if (static_cast<int>(support_nodes.size()) < MIN_RANK) return seeds;
429
430 std::vector<int> top_index(N_SIZE, -1);
431 for (int idx = 0; idx < static_cast<int>(support_nodes.size()); ++idx) { top_index[support_nodes[idx]] = idx; }
432
433 const int top_count = static_cast<int>(support_nodes.size());
434 std::vector<std::vector<double>> cooc(top_count, std::vector<double>(top_count, 0.0));
435 std::vector<int> route_top_nodes;
436 std::vector<char> seen_top(top_count, 0);
437 for (int path_idx : nonzero_paths) {
438 if (path_idx < 0 || static_cast<size_t>(path_idx) >= allPaths.size() || static_cast<size_t>(path_idx) >= x.size())
439 continue;
440 const double x_val = x[path_idx];
441 if (numericutils::isZero(x_val)) continue;
442 route_top_nodes.clear();
443 for (int node : allPaths[path_idx].route) {
444 if (node <= 0 || node >= N_SIZE - 1) continue;
445 const int idx = top_index[node];
446 if (idx < 0 || seen_top[idx]) continue;
447 seen_top[idx] = 1;
448 route_top_nodes.push_back(idx);
449 }
450 for (size_t i = 0; i < route_top_nodes.size(); ++i) {
451 for (size_t j = i + 1; j < route_top_nodes.size(); ++j) {
452 const int a = route_top_nodes[i];
453 const int b = route_top_nodes[j];
454 cooc[a][b] += x_val;
455 cooc[b][a] += x_val;
456 }
457 }
458 for (int idx : route_top_nodes) { seen_top[idx] = 0; }
459 }
460
461 std::vector<int> extension_pool;
462 std::vector<int> candidate_nodes;
463 std::vector<char> in_candidate(N_SIZE, 0);
464 const int anchor_count = std::min(MAX_ANCHORS, top_count);
465 for (int anchor_pos = 0; anchor_pos < anchor_count; ++anchor_pos) {
466 const int anchor = support_nodes[anchor_pos];
467 std::vector<std::pair<double, int>> neighbor_scores;
468 neighbor_scores.reserve(top_count - 1);
469 for (int pos = 0; pos < top_count; ++pos) {
470 if (pos == anchor_pos) continue;
471 const int node = support_nodes[pos];
472 double score = cooc[anchor_pos][pos] + 0.25 * node_support[node];
473 if (node < static_cast<int>(scores.size()) && !scores[node].empty()) {
474 score -= 0.01 * scores[node].front().cost_score;
475 }
476 if (score > 1e-6) neighbor_scores.emplace_back(score, node);
477 }
478 pdqsort(neighbor_scores.begin(), neighbor_scores.end(),
479 [](const auto &a, const auto &b) { return a.first > b.first; });
480 if (neighbor_scores.empty()) continue;
481
482 extension_pool.clear();
483 for (const auto &[_, node] : neighbor_scores) {
484 extension_pool.push_back(node);
485 if (static_cast<int>(extension_pool.size()) >= MAX_EXTENSION_POOL) break;
486 }
487
488 int candidates_for_anchor = 0;
489 for (int rank = MAX_RANK; rank >= 4 && candidates_for_anchor < MAX_CANDIDATES_ANCHOR; --rank) {
490 const int choose = rank - 1;
491 if (choose > static_cast<int>(extension_pool.size())) continue;
492 uint32_t mask = (1u << choose) - 1;
493 const uint32_t limit = (1u << extension_pool.size());
494 while (mask < limit && candidates_for_anchor < MAX_CANDIDATES_ANCHOR) {
495 candidate_nodes.clear();
496 candidate_nodes.reserve(rank);
497 candidate_nodes.push_back(anchor);
498 for (size_t bit = 0; bit < extension_pool.size(); ++bit) {
499 if (mask & (1u << bit)) candidate_nodes.push_back(extension_pool[bit]);
500 }
501 std::sort(candidate_nodes.begin(), candidate_nodes.end());
502
503 for (int node : candidate_nodes) { in_candidate[node] = 1; }
504 ankerl::unordered_dense::set<int> memory_set;
505 memory_set.reserve(MAX_MEMORY_NODES);
506 for (int node : extension_pool) {
507 if (static_cast<int>(memory_set.size()) >= MAX_MEMORY_NODES) break;
508 if (!in_candidate[node]) memory_set.insert(node);
509 }
510 for (int node : candidate_nodes) {
511 if (node >= static_cast<int>(rank1_sep_heur_mem4_vertex.size())) continue;
512 for (int neighbor : rank1_sep_heur_mem4_vertex[node]) {
513 if (static_cast<int>(memory_set.size()) >= MAX_MEMORY_NODES) break;
514 if (neighbor <= 0 || neighbor >= N_SIZE - 1 || in_candidate[neighbor]) continue;
515 memory_set.insert(neighbor);
516 }
517 }
518 for (int node : candidate_nodes) { in_candidate[node] = 0; }
519
520 if (auto candidate = evaluateBaseThenMemory(candidate_nodes, A, x)) {
521 insertBestCandidate(seeds, std::move(*candidate));
522 }
523 ++candidates_for_anchor;
524
525 uint32_t c = mask & -mask;
526 uint32_t r = mask + c;
527 if (c == 0) break;
528 mask = (((r ^ mask) >> 2) / c) | r;
529 }
530 }
531 }
532
533 return seeds;
534 }

537 generateSeedCandidates(const SparseMatrix &A, const std::vector<double> &x, const SeedMap &seed_map) {

540 const auto &candidate_nodes = seed_pair.first;
541 const auto &memory_nodes = seed_pair.second;
542 ankerl::unordered_dense::set<int> memory_set(memory_nodes.begin(), memory_nodes.end());
543 auto [violation, perm, rhs] = computeViolationWithBestPerm(candidate_nodes, memory_set, A, x);
544 if (violation > 1e-3) {
545 insertBestCandidate(seeds, CandidateSet(candidate_nodes, violation, perm, memory_set, rhs));
546 }
547 }
548 return seeds;
549 }

561 ankerl::unordered_dense::set<CandidateSet, CandidateSetHasher, CandidateSetCompare>
562 generateCandidates(const std::vector<std::vector<NodeScore>> &scores, const SparseMatrix &A,
563 const std::vector<double> &x) {
564 struct ViolationResult {
565 double violation = 0.0;
566 SRCPermutation perm;
567 double rhs = 0.0;
568 };
569
570 // Thread-local data for each thread.
571 struct ThreadLocalData {
572 ankerl::unordered_dense::set<CandidateSet, CandidateSetHasher, CandidateSetCompare> candidates;
573 ankerl::unordered_dense::map<CandidateSet, ViolationResult, CandidateSetHasher> candidate_cache;
574 ThreadLocalData() {
575 candidates.reserve(1000);
576 candidate_cache.reserve(1000);
577 }
578 };
579
580 const size_t numThreads = std::thread::hardware_concurrency();
581 std::vector<ThreadLocalData> threadData;
582 for (size_t t = 0; t < numThreads; ++t) { threadData.emplace_back(); }
583
584 ankerl::unordered_dense::set<CandidateSet, CandidateSetHasher, CandidateSetCompare> finalCandidates;
585 finalCandidates.reserve(numThreads * 1000);
586
587 // Use a lambda that accepts an explicit thread index.
588 auto process_vertex = [&](std::size_t i, size_t threadID) {
589 // Skip first and last vertices.
590 if (i == 0 || i == N_SIZE - 1) return;
591 auto &localData = threadData[threadID % numThreads];
592
593 // Build a lookup for heuristic memory from scores[i].
594 const auto &heuristic_memory_scores = scores[i];
595 ankerl::unordered_dense::set<int> heuristic_memory_lookup;
596 heuristic_memory_lookup.reserve(heuristic_memory_scores.size());
597 std::vector<double> node_score_lookup(N_SIZE, std::numeric_limits<double>::infinity());
598 for (const auto &score : heuristic_memory_scores) {
599 heuristic_memory_lookup.insert(score.node);
600 node_score_lookup[score.node] = score.cost_score;
601 }
602
603 // Build the working set from routes that contain vertex i.
604 std::vector<int> working_set;
605 working_set.reserve(std::min<size_t>(MAX_WORKING_SET_SIZE, allPaths.size()));
606 if (auto row_it = row_indices_map.find(static_cast<int>(i)); row_it != row_indices_map.end()) {
607 for (int r : row_it->second) {
608 for (const auto &v : allPaths[r].route) {
609 if (v <= 0 || v >= N_SIZE - 1 || v == static_cast<int>(i) ||
610 !heuristic_memory_lookup.contains(v))
611 continue;
612 if (std::find(working_set.begin(), working_set.end(), v) == working_set.end()) {
613 working_set.push_back(v);
614 }
615 }
616 }
617 }
618
619 // Add additional memory-based candidates for vertices with sparse
620 // route support.
621 if (working_set.size() < static_cast<size_t>(MIN_RANK - 1) &&
622 i < static_cast<size_t>(rank1_sep_heur_mem4_vertex.size())) {
623 for (int v : rank1_sep_heur_mem4_vertex[i]) {
624 if (static_cast<int>(working_set.size()) >= MAX_WORKING_SET_SIZE) break;
625 if (v == static_cast<int>(i) ||
626 std::find(working_set.begin(), working_set.end(), v) != working_set.end())
627 continue;
628 working_set.push_back(v);
629 }
630 }
631
632 // Limit working_set size using a heuristic if necessary.
633 if (working_set.size() > MAX_WORKING_SET_SIZE) {
634 std::vector<int> working_set_vec(working_set.begin(), working_set.end());
635 std::nth_element(working_set_vec.begin(), working_set_vec.begin() + MAX_WORKING_SET_SIZE,
636 working_set_vec.end(),
637 [&](int a, int b) { return node_score_lookup[a] < node_score_lookup[b]; });
638 working_set_vec.resize(MAX_WORKING_SET_SIZE);
639 working_set = std::move(working_set_vec);
640 }
641
642 // Helper to build candidate-specific memory from the working set.
643 auto build_candidate_memory = [&](const std::vector<int> &candidate_nodes) {
644 ankerl::unordered_dense::set<int> memory;
645 memory.reserve(working_set.size());
646 for (int v : working_set) {
647 if (std::find(candidate_nodes.begin(), candidate_nodes.end(), v) == candidate_nodes.end()) {
648 memory.insert(v);
649 }
650 }
651 return memory;
652 };
653
654 // Helper lambda to process a candidate set.
655 auto process_candidate = [&](const std::vector<int> &candidate_nodes) {
656 if (static_cast<int>(candidate_nodes.size()) < MIN_RANK ||
657 static_cast<int>(candidate_nodes.size()) > MAX_RANK)
658 return false;
659
660 auto candidate_memory = build_candidate_memory(candidate_nodes);
661 CandidateSet temp_candidate(candidate_nodes,
662 0.0, // dummy violation
663 SRCPermutation({}, 0), // dummy permutation
664 candidate_memory, 0.0);
665 // Check cache to avoid duplicate work.
666 auto cache_it = localData.candidate_cache.find(temp_candidate);
667 ViolationResult result;
668 if (cache_it != localData.candidate_cache.end()) {
669 result = cache_it->second;
670 } else {
671 std::tie(result.violation, result.perm, result.rhs) =
672 computeViolationWithBestPerm(candidate_nodes, candidate_memory, A, x);
673 localData.candidate_cache.emplace(temp_candidate, result);
674 }
675 if (result.violation > 1e-3) {
676 CandidateSet candidate_set(candidate_nodes, result.violation, result.perm, candidate_memory,
677 result.rhs);
678 insertBestCandidate(localData.candidates, std::move(candidate_set));
679 return true;
680 }
681 return false;
682 };
683
684 // Build the working set vector sorted by heuristic score.
685 std::vector<int> working_set_vec(working_set.begin(), working_set.end());
686 std::sort(working_set_vec.begin(), working_set_vec.end(),
687 [&](int a, int b) { return node_score_lookup[a] < node_score_lookup[b]; });
688
689 // --- 1. Generate promising root candidates from the best
690 // scored working set neighbors.
691 const int max_root_size = std::min(static_cast<int>(working_set_vec.size()), MAX_RANK - 1);
692 for (int root_size = MIN_RANK - 1; root_size <= max_root_size; ++root_size) {
693 std::vector<int> candidate_nodes;
694 candidate_nodes.reserve(root_size + 1);
695 candidate_nodes.push_back(static_cast<int>(i));
696 for (int idx = 0; idx < root_size; ++idx) { candidate_nodes.push_back(working_set_vec[idx]); }
697 std::sort(candidate_nodes.begin(), candidate_nodes.end());
698 process_candidate(candidate_nodes);
699 }
700
701 // --- 2. Generate additional candidate sets by combining vertices
702 // ---
703 if (working_set.size() > 1) {
704 size_t combination_count = 0;
705 // Make sure the working set is sorted by score.
706 // Generate combinations of size k.
707 for (size_t k = 2; k <= MAX_RANK && combination_count < MAX_COMBINATIONS; ++k) {
708 if (k > working_set_vec.size()) break;
709 // Initialize combination bitmask: k ones.
710 uint32_t mask = (1u << k) - 1;
711 while (mask < (1u << working_set_vec.size()) && combination_count < MAX_COMBINATIONS) {
712 std::vector<int> candidate_nodes;
713 candidate_nodes.reserve(k + 1);
714 candidate_nodes.push_back(static_cast<int>(i));
715 // Insert vertices corresponding to bits set in mask.
716 for (size_t bit = 0; bit < working_set_vec.size(); ++bit) {
717 if (mask & (1u << bit)) candidate_nodes.push_back(working_set_vec[bit]);
718 }
719 std::sort(candidate_nodes.begin(), candidate_nodes.end());
720 if (process_candidate(candidate_nodes)) ++combination_count;
721
722 // Gosper’s hack to get next combination.
723 uint32_t c = mask & -mask;
724 uint32_t r = mask + c;
725 if (c == 0) break; // safeguard against infinite loop
726 mask = (((r ^ mask) >> 2) / c) | r;
727 }
728 }
729 }
730 };
731
732 // Parallelize over vertices. Here we assign explicit thread indices.
733 std::vector<std::thread> threads;
734 std::atomic_size_t vertexIndex{1}; // starting from 1; skipping index 0
735 for (size_t t = 0; t < numThreads; ++t) {
736 threads.emplace_back([&, t]() {
737 while (true) {
738 size_t i = vertexIndex.fetch_add(1);
739 if (i >= N_SIZE - 1) break;
740 process_vertex(i, t);
741 }
742 });
743 }
744 for (auto &th : threads) th.join();
745
746 // Merge thread-local candidate sets into finalCandidates.
747 for (const auto &data : threadData) {
748 for (const auto &candidate : data.candidates) { insertBestCandidate(finalCandidates, candidate); }
749 }
750 return finalCandidates;
751 }

753 ankerl::unordered_dense::map<int, std::vector<SRCPermutation>> permutations_cache;
754
755 // ── Plan-group infrastructure (RouteOpt-style per-plan local search) ─────
756 // A "plan" is the equivalence class of all permutations sharing the same
757 // sorted numerator vector and denominator. Grouping allows us to restrict
758 // local-search evaluation to a single plan rather than retrying ALL
759 // permutations on every add/remove/swap step.
760 struct PlanGroup {
761 std::vector<int> sorted_num;
762 int den = 0;
763 std::vector<const SRCPermutation *> perms; // all orderings of this plan
764 };
765 // plan_groups_map[rank] = ordered list of PlanGroups for that rank
766 ankerl::unordered_dense::map<int, std::vector<PlanGroup>> plan_groups_map;
767
768 void buildPlanGroupsMap() {
769 plan_groups_map.clear();
770 for (const auto &[rank, perm_vec] : permutations_cache) {
771 // Use a sorted map so plan groups are in deterministic order.
772 std::map<std::pair<std::vector<int>, int>, std::vector<const SRCPermutation *>> tmp;
773 for (const auto &p : perm_vec) {
774 std::vector<int> sorted = p.num;
775 std::sort(sorted.begin(), sorted.end());
776 tmp[{sorted, p.den}].push_back(&p);
777 }
778 auto &groups = plan_groups_map[rank];
779 groups.reserve(tmp.size());
780 for (const auto &[key, ptrs] : tmp) {
781 PlanGroup pg;
782 pg.sorted_num = key.first;
783 pg.den = key.second;
784 pg.perms = ptrs;
785 groups.push_back(std::move(pg));
786 }
787 }
788 }

790 // Evaluate the best violation over all orderings of a SINGLE plan group.
791 // C, AM, and node_order must already be populated for the current node set.
792 // The caller must clean up node_order entries after the call.
793 double evalPlanGroup(const std::array<uint64_t, num_words> &C,
794 const std::array<uint64_t, num_words> &AM,
795 std::vector<int> &node_order,
796 const std::vector<int> &paths,
797 const PlanGroup &pg,
798 const std::vector<double> &x,
799 const SRCPermutation *&best_perm_out) const {
800 double best_vio = 0.0;
801 const SRCPermutation *best_perm = nullptr;
802 for (const auto *pp : pg.perms) {
803 const double rhs = pp->getRHS();
804 double lhs = 0.0;
805 for (int r : paths) {
806 if (static_cast<size_t>(r) >= x.size()) continue;
807 const double xv = x[r];
808 if (numericutils::isZero(xv)) continue;
809 lhs += static_cast<double>(
810 computeLimitedMemoryCoefficient(C, AM, *pp, allPaths[r].route, node_order)) *
811 xv;
812 }
813 const double vio = lhs - rhs;
814 if (vio > best_vio) {
815 best_vio = vio;
816 best_perm = pp;
817 }
818 }
819 best_perm_out = best_perm;
820 return best_vio;
821 }

823 // Find the plan group with the highest violation for a given node set.
824 // Internally sets up C/AM/node_order, evaluates all plan groups, cleans up.
825 // Returns {violation, plan_group_index(-1 if none), best_perm}.
826 std::tuple<double, int, const SRCPermutation *>
827 findBestPlanGroupIdx(const std::vector<int> &nodes,
828 const ankerl::unordered_dense::set<int> &mem,
829 const std::vector<int> &paths,
830 const std::vector<double> &x) const {
831 const int RANK = static_cast<int>(nodes.size());
832 auto it = plan_groups_map.find(RANK);
833 if (it == plan_groups_map.end()) return {0.0, -1, nullptr};
834 const auto &groups = it->second;
835
836 std::array<uint64_t, num_words> C{}, AM{};
837 thread_local std::vector<int> no;
838 if (no.size() < N_SIZE) no.assign(N_SIZE, -1);
839
840 int pos = 0;
841 for (int n : nodes) {
842 C[n / 64] |= bit_mask_lookup[n % 64];
843 AM[n / 64] |= bit_mask_lookup[n % 64];
844 no[n] = pos++;
845 }
846 for (int n : mem) AM[n / 64] |= bit_mask_lookup[n % 64];
847
848 double best_vio = 0.0;
849 int best_gi = -1;
850 const SRCPermutation *best_perm = nullptr;
851 for (int gi = 0; gi < static_cast<int>(groups.size()); ++gi) {
852 const SRCPermutation *perm = nullptr;
853 double vio = evalPlanGroup(C, AM, no, paths, groups[gi], x, perm);
854 if (vio > best_vio) {
855 best_vio = vio;
856 best_gi = gi;
857 best_perm = perm;
858 }
859 }
860 for (int n : nodes) no[n] = -1; // restore thread-local
861 return {best_vio, best_gi, best_perm};
862 }

864 // ── RouteOpt-style greedy high-dim cut search ─────────────────────────────
865 // For each seed the function runs a deterministic add/remove/swap greedy
866 // local search restricted to one plan group at a time.
867 //
868 // Performance notes (vs. previous version):
869 // • Seeds are processed in parallel via stdexec::bulk.
870 // • SWAP step uses a precomputed C_base/AM_base and flips only 2 bits per
871 // candidate (O(1)) instead of rebuilding O(rank) from scratch.
872 // • Working neighbourhood is a sorted std::vector for cache-friendly scan.
873 // • Seeds are built exactly once and reused from separate().
874 void getHighDimCutsRouteOpt(const std::vector<double> &x, const SeedMap &seed_map) {
875 if (allPaths.empty() || plan_groups_map.empty() || seed_map.empty()) return;
876
877 struct HarvestedCut {
878 double vio = 0.0;
879 std::vector<int> nodes;
880 ankerl::unordered_dense::set<int> mem;
881 const SRCPermutation *perm = nullptr;
882 };
883
884 // Flatten to vector so stdexec::bulk can address by index.
885 using SeedPair = std::pair<std::vector<int>, std::vector<int>>;
886 std::vector<SeedPair> seeds_vec(seed_map.begin(), seed_map.end());
887
888 std::vector<HarvestedCut> all_harvested;
889 all_harvested.reserve(seeds_vec.size());
890 std::mutex harvest_mutex;
891
892 // ── parallel over seeds ───────────────────────────────────────────────
893 auto bulk_sender = stdexec::bulk(
894 stdexec::just(), seeds_vec.size(),
895 [&](std::size_t idx) {
896 const auto &[nodes_vec, mem_vec] = seeds_vec[idx];
897 if (nodes_vec.empty()) return;
898 const int RANK = static_cast<int>(nodes_vec.size());
899 if (RANK < MIN_RANK || RANK > MAX_RANK) return;
900 if (plan_groups_map.find(RANK) == plan_groups_map.end()) return;
901
902 // Thread-local scratch storage (one copy per OS thread).
903 thread_local std::vector<int> cand_paths;
904 thread_local std::vector<int> no2;
905 if (no2.size() < N_SIZE) no2.assign(N_SIZE, -1);
906
907 ankerl::unordered_dense::set<int> cur_mem(mem_vec.begin(), mem_vec.end());
908 std::vector<int> cur_nodes = nodes_vec; // sorted
909
910 collectCandidatePathUnion(cur_nodes, cand_paths);
911 if (cand_paths.empty()) return;
912
913 auto [cur_vio, cur_gi, cur_perm] = findBestPlanGroupIdx(cur_nodes, cur_mem, cand_paths, x);
914 if (cur_vio <= 1e-4 || cur_gi < 0) return;
915
916 // ── working neighbourhood as sorted small-vector ──────────────
917 std::vector<int> w_no_c;
918 w_no_c.reserve(32);
919 const auto w_insert = [&](int n) {
920 auto it = std::lower_bound(w_no_c.begin(), w_no_c.end(), n);
921 if (it == w_no_c.end() || *it != n) w_no_c.insert(it, n);
922 };
923 const auto w_erase = [&](int n) {
924 auto it = std::lower_bound(w_no_c.begin(), w_no_c.end(), n);
925 if (it != w_no_c.end() && *it == n) w_no_c.erase(it);
926 };
927 for (int n : cur_nodes) {
928 if (n >= static_cast<int>(rank1_sep_heur_mem4_vertex.size())) continue;
929 for (int nb : rank1_sep_heur_mem4_vertex[n]) {
930 if (nb > 0 && nb < N_SIZE - 1 &&
931 !std::binary_search(cur_nodes.begin(), cur_nodes.end(), nb))
932 w_insert(nb);
933 }
934 }
935 for (int n : mem_vec) {
936 if (n > 0 && n < N_SIZE - 1 &&
937 !std::binary_search(cur_nodes.begin(), cur_nodes.end(), n))
938 w_insert(n);
939 }
940
941 // ── greedy loop ───────────────────────────────────────────────
942 constexpr int MAX_GREEDY_STEPS = 8;
943 for (int step = 0; step < MAX_GREEDY_STEPS; ++step) {
944 double best_delta = 1e-5;
945 int best_type = 0; // 0=stay, 1=add, 2=remove, 3=swap
946 int best_add = -1, best_rem = -1;
947 int best_swap_out = -1, best_swap_in = -1;
948 const int cur_rank = static_cast<int>(cur_nodes.size());
949
950 // ── SWAP (rank unchanged → same plan group) ───────────────
951 // Precompute C_base/AM_base once; flip 2 bits per candidate.
952 {
953 auto pgit = plan_groups_map.find(cur_rank);
954 if (pgit != plan_groups_map.end() && cur_gi >= 0 &&
955 cur_gi < static_cast<int>(pgit->second.size())) {
956 const auto &pg = pgit->second[cur_gi];
957
958 std::array<uint64_t, num_words> C_base{}, AM_base{};
959 for (int i = 0; i < cur_rank; ++i) {
960 const int n = cur_nodes[i];
961 C_base[n / 64] |= bit_mask_lookup[n % 64];
962 AM_base[n / 64] |= bit_mask_lookup[n % 64];
963 }
964 for (int n : cur_mem) AM_base[n / 64] |= bit_mask_lookup[n % 64];
965
966 for (int out_idx = 0; out_idx < cur_rank; ++out_idx) {
967 const int out_node = cur_nodes[out_idx];
968 for (const int in_node : w_no_c) {
969 // Build swapped node set
970 std::vector<int> new_nodes = cur_nodes;
971 new_nodes[out_idx] = in_node;
972 std::sort(new_nodes.begin(), new_nodes.end());
973
974 // C2/AM2: O(1) bit-flip from base
975 std::array<uint64_t, num_words> C2 = C_base;
976 C2[out_node / 64] &= ~bit_mask_lookup[out_node % 64];
977 C2[in_node / 64] |= bit_mask_lookup[in_node % 64];
978
979 std::array<uint64_t, num_words> AM2 = AM_base;
980 if (!cur_mem.count(out_node))
981 AM2[out_node / 64] &= ~bit_mask_lookup[out_node % 64];
982 AM2[in_node / 64] |= bit_mask_lookup[in_node % 64];
983
984 // no2: rebuild for new_nodes (O(rank), necessary)
985 int pos3 = 0;
986 for (int n : new_nodes) no2[n] = pos3++;
987
988 collectCandidatePathUnion(new_nodes, cand_paths);
989 const SRCPermutation *pp = nullptr;
990 const double vio2 =
991 evalPlanGroup(C2, AM2, no2, cand_paths, pg, x, pp);
992 for (int n : new_nodes) no2[n] = -1;
993
994 if (vio2 - cur_vio > best_delta) {
995 best_delta = vio2 - cur_vio;
996 best_type = 3;
997 best_swap_out = out_node;
998 best_swap_in = in_node;
999 }
1000 }
1001 }
1002 // No outer no2 cleanup needed — we never set it in the outer loop.
1003 }
1004 }
1005
1006 // ── ADD (rank increases → re-find plan) ───────────────────
1007 if (cur_rank < MAX_RANK) {
1008 for (const int nb : w_no_c) {
1009 std::vector<int> new_nodes = cur_nodes;
1010 new_nodes.push_back(nb);
1011 std::sort(new_nodes.begin(), new_nodes.end());
1012 collectCandidatePathUnion(new_nodes, cand_paths);
1013 auto [vio2, gi2, pp2] =
1014 findBestPlanGroupIdx(new_nodes, cur_mem, cand_paths, x);
1015 if (vio2 - cur_vio > best_delta) {
1016 best_delta = vio2 - cur_vio;
1017 best_type = 1;
1018 best_add = nb;
1019 }
1020 }
1021 }
1022
1023 // ── REMOVE (rank decreases → re-find plan) ────────────────
1024 if (cur_rank > MIN_RANK) {
1025 for (int ri = 0; ri < cur_rank; ++ri) {
1026 std::vector<int> new_nodes;
1027 new_nodes.reserve(cur_rank - 1);
1028 for (int j2 = 0; j2 < cur_rank; ++j2)
1029 if (j2 != ri) new_nodes.push_back(cur_nodes[j2]);
1030 collectCandidatePathUnion(new_nodes, cand_paths);
1031 auto [vio2, gi2, pp2] =
1032 findBestPlanGroupIdx(new_nodes, cur_mem, cand_paths, x);
1033 // Small bonus: simpler cut is cheaper in pricing.
1034 if (vio2 - cur_vio > best_delta + 5e-5) {
1035 best_delta = vio2 - cur_vio;
1036 best_type = 2;
1037 best_rem = cur_nodes[ri];
1038 }
1039 }
1040 }
1041
1042 if (best_type == 0) break; // converged
1043
1044 // ── apply best move ───────────────────────────────────────
1045 if (best_type == 1) { // ADD
1046 cur_nodes.push_back(best_add);
1047 std::sort(cur_nodes.begin(), cur_nodes.end());
1048 w_erase(best_add);
1049 if (best_add < static_cast<int>(rank1_sep_heur_mem4_vertex.size())) {
1050 for (int nb : rank1_sep_heur_mem4_vertex[best_add]) {
1051 if (nb > 0 && nb < N_SIZE - 1 &&
1052 !std::binary_search(cur_nodes.begin(), cur_nodes.end(), nb))
1053 w_insert(nb);
1054 }
1055 }
1056 collectCandidatePathUnion(cur_nodes, cand_paths);
1057 auto [vio2, gi2, pp2] = findBestPlanGroupIdx(cur_nodes, cur_mem, cand_paths, x);
1058 cur_vio = vio2; cur_gi = gi2; cur_perm = pp2;
1059
1060 } else if (best_type == 2) { // REMOVE
1061 auto it2 = std::find(cur_nodes.begin(), cur_nodes.end(), best_rem);
1062 if (it2 != cur_nodes.end()) {
1063 w_insert(best_rem);
1064 cur_nodes.erase(it2);
1065 }
1066 collectCandidatePathUnion(cur_nodes, cand_paths);
1067 auto [vio2, gi2, pp2] = findBestPlanGroupIdx(cur_nodes, cur_mem, cand_paths, x);
1068 cur_vio = vio2; cur_gi = gi2; cur_perm = pp2;
1069
1070 } else { // SWAP
1071 auto it2 = std::find(cur_nodes.begin(), cur_nodes.end(), best_swap_out);
1072 if (it2 != cur_nodes.end()) *it2 = best_swap_in;
1073 std::sort(cur_nodes.begin(), cur_nodes.end());
1074 w_erase(best_swap_in);
1075 w_insert(best_swap_out);
1076 // Same rank → stay in same plan group, re-eval incrementally
1077 const int new_rank = static_cast<int>(cur_nodes.size());

1257 std::tuple<double, SRCPermutation, double>
1258 computeViolationWithBestPermBits(std::vector<int> node_vec,
1259 std::array<uint64_t, num_words> augmented_memory_bits,
1260 const SparseMatrix &, const std::vector<double> &x) {
1261 static constexpr double EPSILON = 1e-6;
1262 std::sort(node_vec.begin(), node_vec.end());
1263 const int RANK = static_cast<int>(node_vec.size());
1264
1265 double best_violation = 0.0;
1266 double best_rhs = 0.0;
1267 const SRCPermutation *best_perm = nullptr;
1268
1269 std::array<uint64_t, num_words> candidate_bits{};
1270
1271 thread_local std::vector<int> node_order;
1272 if (node_order.size() < N_SIZE) {
1273 node_order.assign(N_SIZE, -1);
1274 } else {
1275 std::fill(node_order.begin(), node_order.end(), -1);
1276 }
1277
1278 int i = 0;
1279 for (auto node : node_vec) {
1280 const int word_idx = node / 64;
1281 const uint64_t bit_mask = bit_mask_lookup[node % 64];
1282
1283 candidate_bits[word_idx] |= bit_mask;
1284 augmented_memory_bits[word_idx] |= bit_mask;
1285 node_order[node] = i++;
1286 }
1287
1288 const auto &permutations = getPermutations(RANK);
1289
1290 thread_local std::vector<int> candidate_paths;
1291 collectCandidatePathUnion(node_vec, candidate_paths);
1292 if (candidate_paths.empty()) { return {0.0, SRCPermutation({}, 0), 0.0}; }
1293
1294 for (const auto &perm : permutations) {
1295 const double rhs = perm.getRHS();
1296 const SRCPermutation src_perm = {perm.num, perm.den};
1297 double lhs = 0.0;
1298
1299 for (int path_idx : candidate_paths) {
1300 if (static_cast<size_t>(path_idx) >= x.size()) continue;
1301 const double x_val = x[path_idx];
1302 if (numericutils::isZero(x_val)) continue;
1303
1304 const int coeff = computeLimitedMemoryCoefficient(candidate_bits, augmented_memory_bits, src_perm,
1305 allPaths[path_idx].route, node_order);
1306 lhs += coeff * x_val;
1307 }
1308
1309 const double violation = lhs - (rhs + EPSILON);
1310 if (violation > best_violation) {
1311 best_violation = violation;
1312 best_perm = &perm;
1313 best_rhs = rhs;
1314 }
1315 }
1316
1317 return {best_violation, best_perm ? *best_perm : SRCPermutation({}, 0), best_rhs};
1318 }

1345 bool addCutToCutStorage(const CandidateSet &candidate, const std::vector<double> &x) {
1346 // Initialize bit arrays for the candidate (C) and augmented memory
1347 // (AM).
1348 std::array<uint64_t, num_words> C = {};
1349 std::array<uint64_t, num_words> AM = {};
1350
1351 // Build a deterministic node order for the candidate.
1352 std::vector<int> node_ordered(candidate.nodes.begin(), candidate.nodes.end());
1353 std::sort(node_ordered.begin(), node_ordered.end());
1354 std::vector<int> order(N_SIZE, -1);
1355 int ordering = 0;
1356 for (int node : node_ordered) { order[node] = ordering++; }
1357
1358 // Set bits for candidate nodes.
1359 for (int node : node_ordered) {
1360 C[node / 64] |= bit_mask_lookup[node % 64];
1361 AM[node / 64] |= bit_mask_lookup[node % 64];
1362 }
1363 // Set bits for candidate neighbors into the augmented memory.
1364 for (auto node : candidate.neighbor) { AM[node / 64] |= bit_mask_lookup[node % 64]; }
1365
1366 // Prepare the permutation from the candidate.
1367 SRCPermutation p;
1368 p.num = candidate.perm.num;
1369 p.den = candidate.perm.den;
1370
1371 // Store only the nonzero coefficients. This keeps high-order cuts
1372 // lighter in memory while still allowing us to build the initial
1373 // master constraint exactly.
1374 std::vector<int> coefficient_indices;
1375 std::vector<double> coefficient_values;
1376 coefficient_indices.reserve(allPaths.size() / 8 + 1);
1377 coefficient_values.reserve(allPaths.size() / 8 + 1);
1378
1379 thread_local std::vector<int> candidate_paths;
1380 collectCandidatePathUnion(node_ordered, candidate_paths);
1381
1382 double exact_violation = 0.0;
1383 const double rhs = p.getRHS();
1384 for (const int path_idx : candidate_paths) {
1385 if (static_cast<size_t>(path_idx) >= allPaths.size() || static_cast<size_t>(path_idx) >= x.size())
1386 continue;
1387 const double coeff = computeLimitedMemoryCoefficient(C, AM, p, allPaths[path_idx].route, order);
1388 exact_violation += coeff * x[path_idx];
1389 if (!numericutils::isZero(coeff)) {
1390 coefficient_indices.push_back(path_idx);
1391 coefficient_values.push_back(coeff);
1392 }
1393 }
1394
1395 if (exact_violation <= rhs + 1e-6) { return false; }
1396
1397 // Construct the Cut object using the candidate data.
1398 Cut cut(C, AM, {}, p);
1399 cut.baseSetOrder = order;
1400 cut.coefficient_indices = std::move(coefficient_indices);
1401 cut.coefficient_values = std::move(coefficient_values);
1402
1403 // Set the cut type based on the number of candidate nodes.
1404 const auto cut_rank = candidate.nodes.size();
1405 if (cut_rank == 4) {
1406 cut.type = CutType::FourRow;
1407 // return;
1408 } else if (cut_rank == 5) {
1409 cut.type = CutType::FiveRow;
1410 }
1411
1412 // Add the cut to the global cut storage.
1413 const size_t previous_size = cutStorage->size();
1414 cutStorage->addCut(cut);
1415 return cutStorage->size() > previous_size || cut.updated;
1416 }

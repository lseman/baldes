40inline void combinations(const std::vector<T> &elements, int k, std::vector<std::vector<T>> &result) {
41 std::vector<int> indices(k);
42 std::iota(indices.begin(), indices.end(), 0);
43 while (true) {
44 std::vector<T> combination;
45 for (int idx : indices) { combination.push_back(elements[idx]); }
46 result.push_back(combination);
47
48 int i = k - 1;
49 while (i >= 0 && indices[i] == elements.size() - k + i) { --i; }
50 if (i < 0) break;
51
52 ++indices[i];
53 for (int j = i + 1; j < k; ++j) { indices[j] = indices[j - 1] + 1; }
54 }

64 struct NodeScore {
65 int node = 0;
66 double violation_score = 0.0;
67 double distance_score = 0.0;
68 double cost_score = 0.0;
69 double combined_score = 0.0;
70
71 NodeScore() = default;
72
73 NodeScore(int n, double v, double d, double c) : node(n), violation_score(v), distance_score(d), cost_score(c) {
74 // Combine scores with weights
75 // We want higher violation_score, lower distance_score, and lower cost_score
76 constexpr double VIOLATION_WEIGHT = 1.0;
77 constexpr double DISTANCE_WEIGHT = -0.3; // Negative because we want closer nodes
78 constexpr double COST_WEIGHT = -0.2; // Negative because we want lower costs
79
80 combined_score =
81 VIOLATION_WEIGHT * violation_score + DISTANCE_WEIGHT * distance_score + COST_WEIGHT * cost_score;
82 }
83
84 bool operator<(const NodeScore &other) const { return combined_score > other.combined_score; }
85 };

95 // Compute contribution scores for each node based on column overlap and x values
96 std::vector<std::vector<NodeScore>> computeNodeScores(const SparseMatrix &A, const std::vector<double> &x) {
97
98 std::vector<std::vector<NodeScore>> scores(N_SIZE);
99 std::vector<std::vector<int>> col_to_rows(A.num_cols);
100
101 // Build column to rows mapping
102 for (int idx = 0; idx < A.values.size(); ++idx) {
103 int row = A.rows[idx];
104 if (row > N_SIZE - 2) continue;
105 col_to_rows[A.cols[idx]].push_back(row + 1);
106 }
107
108 // For each node, compute scores with other nodes
109 for (int i = 1; i < N_SIZE - 1; ++i) {
110 std::vector<double> node_contributions(N_SIZE, 0.0);
111
112 // Find columns where node i appears
113 for (int idx = 0; idx < A.values.size(); ++idx) {
114 if (A.rows[idx] + 1 == i) {
115 int col = A.cols[idx];
116 double col_value = x[col];
117
118 // For each other node that shares this column
119 for (int other_row : col_to_rows[col]) {
120 if (other_row != i) { node_contributions[other_row] += col_value; }
121 }
122 }
123 }
124
125 // Convert to scores incorporating distance and cost
126 std::vector<NodeScore> node_scores;
127 for (int j = 1; j < N_SIZE - 1; ++j) {
128 if (j != i && node_contributions[j] > 0) {
129 double violation_score = node_contributions[j];
130 double distance_score = distances[i][j];
131 double cost_score = nodes[i].cost + nodes[j].cost;
132
133 // Additional cost-based filtering
134 if (cost_score < nodes[i].cost * 3) { // Only consider if total cost is reasonable
135 node_scores.emplace_back(j, violation_score, distance_score, cost_score);
136 }
137 }
138 }
139
140 // Sort by combined score
141 std::sort(node_scores.begin(), node_scores.end());
142
143 // Keep only the top candidates
144 if (node_scores.size() > MAX_CANDIDATES_PER_NODE) { node_scores.resize(MAX_CANDIDATES_PER_NODE); }
145
146 scores[i] = std::move(node_scores);
147 }
148
149 return scores;
150 }

152 // Generate promising candidate sets based on node scores
153 std::vector<CandidateSet> generateCandidates(const std::vector<std::vector<NodeScore>> &scores,
154 const std::vector<std::vector<int>> &row_indices_map,
155 const SparseMatrix &A, const std::vector<double> &x) {
156
157 std::vector<CandidateSet> candidates;
158 std::vector<bool> used(N_SIZE, false);
159
160 for (int start_node = 1; start_node < N_SIZE - 1; ++start_node) {
161 if (scores[start_node].empty()) continue;
162
163 std::vector<int> current_set = {start_node};
164 std::fill(used.begin(), used.end(), false);
165 used[start_node] = true;
166
167 // Build set of size RANK greedily
168 while (current_set.size() < RANK) {
169 double best_violation = 0.0;
170 int best_node = -1;
171 Permutation best_perm = getPermutations()[0]; // Default permutation
172
173 // Try adding each candidate node
174 for (const auto &score : scores[start_node]) {
175 int candidate = score.node;
176 if (!used[candidate]) {
177 current_set.push_back(candidate);
178 auto [violation, perm] = computeViolationWithBestPerm(current_set, row_indices_map, A, x);
179
180 if (violation > best_violation) {
181 best_violation = violation;
182 best_node = candidate;
183 best_perm = perm;
184 }
185 current_set.pop_back();
186 }
187 }
188
189 if (best_node == -1) break;
190 current_set.push_back(best_node);
191 used[best_node] = true;
192
193 // If we have a complete set with good violation, add it
194 if (current_set.size() == RANK && best_violation > 0.8) {
195 candidates.emplace_back(current_set, best_violation, best_perm);
196 if (candidates.size() >= MAX_CANDIDATES_PER_NODE) break;
197 }
198 }
199 }
200
201 return candidates;
202 }

213 std::pair<double, Permutation> computeViolationWithBestPerm(const std::vector<int> &nodes,
214 const std::vector<std::vector<int>> &row_indices_map,
215 const SparseMatrix &A, const std::vector<double> &x) {
216 std::vector<int> expanded(A.num_cols, 0);
217
218 for (int node : nodes) {
219 for (int idx : row_indices_map[node]) { expanded[A.cols[idx]] += 1; }
220 }
221
222 double best_violation = 0.0;
223 Permutation *best_perm = nullptr;
224 auto permutations = getPermutations();
225
226 for (auto &perm : permutations) {
227 double lhs = 0.0;
228 for (int idx = 0; idx < A.num_cols; ++idx) {
229 if (expanded[idx] >= 2) {
230 double coef = 0.0;
231 for (size_t i = 0; i < RANK; ++i) {
232 if (expanded[idx] > i) { coef += static_cast<double>(perm.num[i]) / perm.den; }
233 }
234 lhs += coef * x[idx];
235 }
236 }
237
238 if (lhs > best_violation) {
239 best_violation = lhs;
240 best_perm = &perm;
241 }
242 }
243
244 return {best_violation, best_perm ? *best_perm : permutations[0]};
245 }

247 CandidateSet localSearch(const CandidateSet &initial, const std::vector<std::vector<int>> &row_indices_map,
248 const SparseMatrix &A, const std::vector<double> &x, double temperature = 0.1,
249 int max_iterations = 100) {
250 CandidateSet best = initial;
251 CandidateSet current = initial;
252 int iterations_without_improvement = 0;
253
254 std::random_device rd;
255 std::mt19937 gen(rd());
256 std::uniform_real_distribution<> dis(0.0, 1.0);
257
258 for (int iter = 0; iter < max_iterations && iterations_without_improvement < 20; ++iter) {
259 // Generate neighbor by swapping one node
260 auto neighbor = current;
261 int pos_to_swap = gen() % RANK;
262 int original_node = neighbor.nodes[pos_to_swap];
263
264 // Try to find a better neighbor
265 bool improved = false;
266 for (int new_node = 1; new_node < N_SIZE - 1; ++new_node) {
267 if (std::find(neighbor.nodes.begin(), neighbor.nodes.end(), new_node) != neighbor.nodes.end()) {
268 continue;
269 }
270
271 neighbor.nodes[pos_to_swap] = new_node;
272 auto [new_violation, new_perm] = computeViolationWithBestPerm(neighbor.nodes, row_indices_map, A, x);
273
274 // Accept if better or with probability based on temperature
275 double delta = new_violation - current.violation;
276 if (delta > 0 || (temperature > 0 && dis(gen) < std::exp(delta / temperature))) {
277 current.nodes = neighbor.nodes;
278 current.violation = new_violation;
279 current.perm = new_perm;
280 improved = true;
281
282 if (new_violation > best.violation) {
283 best = current;
284 iterations_without_improvement = 0;
285 }
286 break;
287 }
288 }
289
290 if (!improved) {
291 neighbor.nodes[pos_to_swap] = original_node;
292 iterations_without_improvement++;
293 }
294
295 temperature *= 0.95; // Cooling schedule
296 }
297
298 return best;
299 }

333 void separate(const SparseMatrix &A, const std::vector<double> &x) {
334 // Compute promising node combinations based on column overlap
335 auto node_scores = computeNodeScores(A, x);
336
337 // Build row indices map for efficient access
338 std::vector<std::vector<int>> row_indices_map(N_SIZE);
339 for (int idx = 0; idx < A.values.size(); ++idx) {
340 int row = A.rows[idx];
341 if (row > N_SIZE - 2) continue;
342 row_indices_map[row + 1].push_back(idx);
343 }
344
345 // Generate promising candidates
346 auto candidates = generateCandidates(node_scores, row_indices_map, A, x);
347
348 // Parallel local search
349 const int JOBS = std::thread::hardware_concurrency();
350 exec::static_thread_pool pool(JOBS);
351 auto sched = pool.get_scheduler();
352
353 std::mutex candidates_mutex;
354 std::vector<CandidateSet> improved_candidates;
355
356 auto bulk_sender = stdexec::bulk(
357 stdexec::just(), candidates.size(),
358 [this, &row_indices_map, &A, &x, &candidates_mutex, &improved_candidates, &candidates](std::size_t idx) {
359 auto improved = localSearch(candidates[idx], row_indices_map, A, x);
360
361 if (improved.violation > 1.0 + 1e-3) {
362 std::lock_guard<std::mutex> lock(candidates_mutex);
363 improved_candidates.push_back(improved);
364 }
365 });
366
367 auto work = stdexec::starts_on(sched, bulk_sender);
368 stdexec::sync_wait(std::move(work));
369
370 // Sort and add best cuts
371 std::sort(improved_candidates.begin(), improved_candidates.end(),
372 [](const auto &a, const auto &b) { return a.violation > b.violation; });
373
374 const int max_cuts = 2;
375 for (int i = 0; i < std::min(max_cuts, static_cast<int>(improved_candidates.size())); ++i) {
376 std::vector<int> order(N_SIZE);
377 int ordering = 0;
378 for (int node : improved_candidates[i].nodes) { order[node] = ordering++; }
379 addCutToCutStorage(improved_candidates[i], order);
380 }
381 }

383 double computeLimitedMemoryCoefficient(const std::array<uint64_t, num_words> &C,
384 const std::array<uint64_t, num_words> &AM, const SRCPermutation &p,
385 const std::vector<uint16_t> &P, std::vector<int> &order) noexcept {
386 double alpha = 0.0;
387 int S = 0;
388 auto den = p.den;
389
390 for (size_t j = 1; j < P.size() - 1; ++j) {
391 int vj = P[j];
392
393 // Precompute bitshift values for reuse
394 uint64_t am_mask = (1ULL << (vj & 63));
395 uint64_t am_index = vj >> 6;
396
397 // Check if vj is in AM using precomputed values
398 if (!(AM[am_index] & am_mask)) {
399 S = 0; // Reset S if vj is not in AM
400 } else if (C[am_index] & am_mask) {
401 // Get the position of vj in C by counting the set bits up to vj
402 int pos = order[vj];
403
404 S += p.num[pos];
405 if (S >= den) {
406 S -= den;
407 alpha += 1;
408 }
409 }
410 }
411
412 return alpha;
413 }

127 // Use a lambda that accepts an explicit thread index.
128 auto process_vertex = [&](std::size_t i, size_t threadID) {
129 // Skip first and last vertices.
130 if (i == 0 || i == N_SIZE - 1) return;
131 auto &localData = threadData[threadID % numThreads];
132
133 // Build a lookup for heuristic memory from scores[i].
134 const auto &heuristic_memory = scores[i];
135 ankerl::unordered_dense::set<int> heuristic_memory_lookup(
136 heuristic_memory.begin(), heuristic_memory.end());

138 // Build the working set from routes that contain vertex i.
139 ankerl::unordered_dense::set<int> working_set;
140 working_set.reserve(
141 std::min<size_t>(MAX_WORKING_SET_SIZE, allPaths.size()));
142 for (size_t r = 0; r < allPaths.size(); ++r) {

148 }
149 }

151 // Limit working_set size using a heuristic if necessary.
152 if (working_set.size() > MAX_WORKING_SET_SIZE) {
153 std::vector<int> working_set_vec(working_set.begin(),
154 working_set.end());
155 std::nth_element(working_set_vec.begin(),
156 working_set_vec.begin() + MAX_WORKING_SET_SIZE,
157 working_set_vec.end(), [&](int a, int b) {
158 return localData.precomputedScoreSums[a] <
159 localData.precomputedScoreSums[b];
160 });
161 working_set_vec.resize(MAX_WORKING_SET_SIZE);
162 working_set = ankerl::unordered_dense::set<int>(
163 working_set_vec.begin(), working_set_vec.end());

166 // Helper lambda to process a candidate set.
167 auto process_candidate =
168 [&](const ankerl::unordered_dense::set<int> &candidate_nodes) {
169 if (candidate_nodes.size() < MIN_RANK ||
170 candidate_nodes.size() > MAX_RANK)
171 return false;
172 CandidateSet temp_candidate(
173 candidate_nodes,
174 0.0, // dummy violation
175 SRCPermutation({}, 0), // dummy permutation
176 heuristic_memory_lookup, 0.0);
177 // Check cache to avoid duplicate work.
178 auto cache_it =
179 localData.candidate_cache.find(temp_candidate);
180 double violation = 0.0;
181 SRCPermutation perm({}, 0);
182 double rhs = 0.0;
183 if (cache_it != localData.candidate_cache.end()) {
184 violation = cache_it->second;
185 } else {
186 std::tie(violation, perm, rhs) =
187 computeViolationWithBestPerm(
188 candidate_nodes, heuristic_memory_lookup, A, x);
189 localData.candidate_cache.emplace(
190 CandidateSet(candidate_nodes, violation, perm,
191 heuristic_memory_lookup, rhs),
192 violation);
193 }
194 if (violation > 1e-3) {
195 CandidateSet candidate_set(
196 candidate_nodes, violation, perm,
197 heuristic_memory_lookup, rhs);
198 localData.candidates.emplace(std::move(candidate_set));
199 return true;
200 }
201 return false;
202 };
203
204 // --- 1. Generate candidate sets for routes that do NOT contain
205 // vertex i ---
206 for (size_t r = 0; r < allPaths.size(); ++r) {
207 if (vertex_route_map[i][r] > 0)
208 continue; // Only process routes missing vertex i
209
210 // Start candidate set with vertex i.
211 ankerl::unordered_dense::set<int> candidate_nodes;
212 candidate_nodes.insert(static_cast<int>(i));
213 candidate_nodes.insert(working_set.begin(), working_set.end());
214
215 process_candidate(candidate_nodes);
216 }
217
218 // --- 2. Generate additional candidate sets by combining vertices
219 // ---
220 if (working_set.size() > 1) {
221 size_t combination_count = 0;
222 std::vector<int> working_set_vec(working_set.begin(),
223 working_set.end());
224 // Generate combinations of size k.
225 for (size_t k = 2;
226 k <= MAX_RANK && combination_count < MAX_COMBINATIONS;
227 ++k) {
228 if (k > working_set_vec.size()) break;
229 // Initialize combination bitmask: k ones.
230 uint32_t mask = (1u << k) - 1;
231 while (mask < (1u << working_set_vec.size()) &&
232 combination_count < MAX_COMBINATIONS) {
233 ankerl::unordered_dense::set<int> candidate_nodes;
234 candidate_nodes.insert(static_cast<int>(i));
235 // Insert vertices corresponding to bits set in mask.
236 for (size_t bit = 0; bit < working_set_vec.size();
237 ++bit) {
238 if (mask & (1u << bit))
239 candidate_nodes.insert(working_set_vec[bit]);
240 }
241 if (process_candidate(candidate_nodes))
242 ++combination_count;
243
244 // Gosper’s hack to get next combination.
245 uint32_t c = mask & -mask;
246 uint32_t r = mask + c;
247 if (c == 0) break; // safeguard against infinite loop
248 mask = (((r ^ mask) >> 2) / c) | r;
249 }
250 }
251 }
252 };
253
254 // Parallelize over vertices. Here we assign explicit thread indices.
255 std::vector<std::thread> threads;
256 std::atomic_size_t vertexIndex{1}; // starting from 1; skipping index 0
257 for (size_t t = 0; t < numThreads; ++t) {
258 threads.emplace_back([&, t]() {
259 while (true) {
260 size_t i = vertexIndex.fetch_add(1);
261 if (i >= N_SIZE - 1) break;
262 process_vertex(i, t);
263 }
264 });
265 }
266 for (auto &th : threads) th.join();
267
268 // Merge thread-local candidate sets into finalCandidates.
269 for (const auto &data : threadData) {
270 finalCandidates.insert(data.candidates.begin(),
271 data.candidates.end());
272 }
273 return finalCandidates;
274 }

308 // std::sort(node_vec.begin(), node_vec.end());

313 double best_rhs = 0.0;
314 const SRCPermutation *best_perm = nullptr;
315
316 // Initialize bit arrays for efficient set operations.
317 std::array<uint64_t, num_words>
318 candidate_bits{}; // all bits initially 0
319 std::array<uint64_t, num_words>
320 augmented_memory_bits{}; // all bits initially 0
321
322 // Create a node order lookup vector; initialize with -1 (invalid
323 // index).
324 std::vector<int> node_order(N_SIZE, -1);
325
326 // Populate candidate_bits and node_order based on candidate nodes.
327 int i = 0;
328 for (auto node : nodes) {
329 const int word_idx = node / 64;
330 const uint64_t bit_mask = bit_mask_lookup[node % 64];
331
332 candidate_bits[word_idx] |= bit_mask;
333 augmented_memory_bits[word_idx] |= bit_mask;
334 node_order[node] = i++;
335 }
336
337 // Add memory nodes to augmented_memory_bits.
338 for (const int node : memory) {
339 int word_idx = node / 64;
340 augmented_memory_bits[word_idx] |= (1ULL << (node % 64));
341 }
342
343 // Get all permutations for the given rank (compute once).
344 const auto &permutations = getPermutations(RANK);
345
346 // Pre-allocate a vector for storing cut coefficients.
347 std::vector<int> cut_coefficients;
348 cut_coefficients.reserve(allPaths.size());
349
350 // Evaluate each permutation.
351 for (const auto &perm : permutations) {
352 // Compute the RHS value from the permutation.
353 double rhs = perm.getRHS();
354 // Create an SRCPermutation from the current permutation.
355 const SRCPermutation src_perm{perm.num, perm.den};
356
357 // Reset coefficient vector and accumulator for LHS.
358 // cut_coefficients.clear();
359 double lhs = 0.0;
360
361 // Process each path from the global allPaths container.
362 for (size_t path_idx = 0; path_idx < allPaths.size(); ++path_idx) {
363 const double x_val = x[path_idx];
364
365 // If the path's x value is effectively zero, skip computation.
366 if (numericutils::isZero(x_val)) {
367 // cut_coefficients.push_back(0.0);
368 continue;
369 }
370
371 // Compute the coefficient using candidate_bits,
372 // augmented_memory_bits, current permutation, the route from
373 // the path, and the node ordering.
374 const int coeff = computeLimitedMemoryCoefficient(
375 candidate_bits, augmented_memory_bits, src_perm,
376 allPaths[path_idx].route, node_order);
377 // cut_coefficients.push_back(coeff);
378 lhs += coeff * x_val;
379 }
380
381 // Compute the violation: the difference between lhs and (rhs +
382 // EPSILON).
383 const double violation = lhs - (rhs + EPSILON);
384 if (violation > best_violation) {
385 best_violation = violation;
386 best_perm = &perm;
387 best_rhs = rhs;
388 }
389 }
390
391 // Return the best violation, corresponding permutation (or default if
392 // none), and best rhs.
393 return {best_violation, best_perm ? *best_perm : SRCPermutation({}, 0),
394 best_rhs};

586 int computeLimitedMemoryCoefficient(
587 const std::array<uint64_t, num_words> &C,
588 const std::array<uint64_t, num_words> &AM, const SRCPermutation &p,
589 const std::vector<uint16_t> &P, std::vector<int> &order) noexcept {
590 int alpha = 0.0;
591 int cumulative_sum = 0;
592 const int denominator = p.den;
593
594 // Loop over the positions in P, ignoring the first and last elements.
595 for (size_t j = 1; j < P.size() - 1; ++j) {
596 const int vertex = P[j];
597
598 // Determine the bit mask and word index for the vertex.
599 const size_t word_index = vertex >> 6;
600 const uint64_t bit_mask = 1ULL << (vertex & 63);
601
602 // If the augmented memory does not contain the vertex, reset
603 // cumulative sum.
604 if (!(AM[word_index] & bit_mask)) {
605 cumulative_sum = 0;
606 }
607 // Otherwise, if the candidate set contains the vertex,
608 // add the corresponding numerator from the permutation.
609 else if (C[word_index] & bit_mask) {
610 int pos = order[vertex];
611 cumulative_sum += p.num[pos];
612 // If cumulative sum exceeds or equals the denominator, update
613 // alpha.
614 if (cumulative_sum >= denominator) {
615 cumulative_sum -= denominator;
616 alpha += 1;
617 }
618 }
619 }
620
621 return alpha;
622 }

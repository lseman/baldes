6void MIPProblem::addVars(const double *lb, const double *ub, const double *obj,
7 const VarType *vtypes, const std::string *names,
8 const MIPColumn *cols, size_t count) {
9 // Loop over each variable to add.
10 for (size_t i = 0; i < count; ++i) {
11 // Add variable with its bounds, objective, and type.
12 add_variable(names[i], vtypes[i], lb[i], ub[i], obj[i]);
13 gurobiCache->addColumn(cols[i]);
14
15 // The new variable's index is the last element in 'variables'.
16 int col_index = variables.size() - 1;
17
18 // Get the nonzero terms from the MIP column.
19 auto terms = cols[i].getTerms();
20
21 // Reserve space for batch insertion based on the number of terms.
22 std::vector<int> batch_rows;
23 std::vector<int> batch_cols;
24 std::vector<double> batch_values;
25 batch_rows.reserve(terms.size());
26 batch_cols.reserve(terms.size());
27 batch_values.reserve(terms.size());
28
29 // Cache the number of constraints.
30 const size_t numCons = constraints.size();
31 // Create a vector of vectors for constraint updates.
32 std::vector<std::vector<std::pair<int, double>>> constraint_updates(
33 numCons);
34
35 // Collect all terms for this variable.
36 for (const auto &[row_index, value] : terms) {
37 batch_rows.push_back(row_index);
38 batch_cols.push_back(col_index);
39 batch_values.push_back(value);
40
41 if (row_index < numCons) { // Safety check.
42 constraint_updates[row_index].emplace_back(col_index, value);
43 }
44 }
45
46 // Perform batch insertion into the sparse matrix.
47 sparse_matrix.insert_batch(batch_rows, batch_cols, batch_values);
48
49 // Update constraints with the new variable's coefficients.
50 for (size_t row = 0; row < numCons; ++row) {
51 for (const auto &[col_idx, val] : constraint_updates[row]) {
52 constraints[row]->addTerm(variables[col_idx], val);
53 }
54 }
55 }

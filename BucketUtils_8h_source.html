83inline int BucketGraph::get_bucket_number(int node, std::vector<double> &resource_values_vec) noexcept {
84 const size_t num_resources = options.main_resources.size();
85
86 const auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
87
88 // Optionally adjust resource values (uncomment if you need epsilon
89 // adjustments)
90 for (size_t r = 0; r < num_resources; ++r) {
91 if constexpr (D == Direction::Forward) {
92 // resource_values_vec[r] += numericutils::eps;
93 } else { // Direction::Backward
94 // resource_values_vec[r] -= numericutils::eps;
95 }
96 }
97 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
98 // Start with the offset for this node’s buckets.
99 int bucket_number = num_buckets_index[node];
100 int multiplier = 1;
101
102 for (size_t r = 0; r < num_resources; ++r) {
103 // Node-specific resource bounds.
104 double node_lb = nodes[node].lb[r];
105 double node_ub = nodes[node].ub[r];
106
107 // Global bounds (if needed for computing splits).
108 double global_lb = R_min[r];
109 double global_ub = R_max[r];
110 double full_range = global_ub - global_lb;
111 double node_range = node_ub - node_lb;
112
113 // Determine the number of splits (i.e. intervals) for resource r.
114 int splits = std::max(1, static_cast<int>(std::round((node_range / full_range) * intervals[r].interval)));
115 double interval_width = node_range / splits;
116
117 int pos = 0;
118 if constexpr (D == Direction::Forward) {
119 // Compute the bucket "digit" from the lower bound.
120 pos =
121 std::min(splits - 1, static_cast<int>(std::floor((resource_values_vec[r] - node_lb) / interval_width)));
122 } else { // Direction::Backward
123 // For backward, we consider the distance from the upper bound.
124 pos =
125 std::min(splits - 1, static_cast<int>(std::floor((node_ub - resource_values_vec[r]) / interval_width)));
126 }
127
128 bucket_number += pos * multiplier;
129 multiplier *= splits;
130 }
131
132 // check if buckets[bucket_number] contains the resource_vec
133 if (unlikely(!buckets[bucket_number].contains(resource_values_vec))) {
134 // print resource_values_vec
135 for (auto val : resource_values_vec) { fmt::print("{}\n", val); }
136 // print bucket bounds
137 fmt::print("{}\n", buckets[bucket_number].lb[0]);
138 fmt::print("{}\n", buckets[bucket_number].ub[0]);
139
140 std::throw_with_nested(std::runtime_error("Resource values not contained in bucket"));
141 return -1;
142 }
143
144 return bucket_number;
145}

157// A constant representing the number of splits for a node that spans the full
158// range. For example, if a node covers the entire full range, it will get
159// FULL_RANGE_SPLITS buckets.
160void BucketGraph::define_buckets() {
161 // 1. Save the old buckets and fixed status.
162 std::vector<Bucket> old_buckets;
163 std::vector<bool> was_fixed;
164 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
165 size_t old_size = buckets.size();
166
167 // Map to store the fixed range (min_lb and max_ub) for each node pair.
168 ankerl::unordered_dense::map<int,
169 ankerl::unordered_dense::map<int, std::pair<std::vector<double>, std::vector<double>>>>
170 node_pair_fixed_ranges;
171
172 if (old_size > 0) {
173 old_buckets = buckets;
174 was_fixed.resize(old_size * old_size, false);
175 auto &old_bitmap = assign_buckets<D>(fw_fixed_buckets_bitmap, bw_fixed_buckets_bitmap);
176
177 // Mark fixed bucket pairs.
178 for (size_t i = 0; i < old_size; ++i) {
179 for (size_t j = 0; j < old_size; ++j) {
180 size_t pos = i * old_size + j;
181 if (pos / 64 < old_bitmap.size()) {
182 was_fixed[pos] = ((old_bitmap[pos / 64] & (1ULL << (pos % 64))) != 0);
183 }
184 }
185 }
186
187 const int num_intervals = options.main_resources.size();
188 // For each fixed bucket pair, update the fixed range for the
189 // corresponding node pair.
190 for (size_t i = 0; i < old_size; ++i) {
191 for (size_t j = 0; j < old_size; ++j) {
192 if (!was_fixed[i * old_size + j]) continue;
193 const auto &bucket_from = old_buckets[i];
194 const auto &bucket_to = old_buckets[j];
195 int from_node = bucket_from.node_id;
196 int to_node = bucket_to.node_id;
197 // Initialize if necessary.
198 if (node_pair_fixed_ranges[from_node].find(to_node) == node_pair_fixed_ranges[from_node].end()) {
199 node_pair_fixed_ranges[from_node][to_node] = {
200 std::vector<double>(num_intervals, std::numeric_limits<double>::max()),
201 std::vector<double>(num_intervals, std::numeric_limits<double>::lowest())};
202 }
203 auto &[min_lbs, max_ubs] = node_pair_fixed_ranges[from_node][to_node];
204 for (int r = 0; r < num_intervals; ++r) {
205 min_lbs[r] = std::min(min_lbs[r], bucket_from.lb[r]);
206 max_ubs[r] = std::max(max_ubs[r], bucket_from.ub[r]);
207 }
208 }
209 }
210 } // End if old buckets exist.
211
212 // 2. Clear the current buckets and fixed bitmap.
213 buckets.clear();
214 auto &fixed_buckets_bitmap = assign_buckets<D>(fw_fixed_buckets_bitmap, bw_fixed_buckets_bitmap);
215 fixed_buckets_bitmap.clear();
216
217 // 3. Prepare variables and containers for new bucket definition.
218 const int num_intervals = options.main_resources.size();
219 std::vector<double> total_ranges(num_intervals); // unused in current code
220 std::vector<double> base_intervals(num_intervals); // unused in current code
221
222 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
223 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
224 auto &node_interval_trees = assign_buckets<D>(fw_node_interval_trees, bw_node_interval_trees);
225 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
226 auto &bucket_splits = assign_buckets<D>(fw_bucket_splits, bw_bucket_splits);
227
228 const size_t num_nodes = nodes.size();
229 num_buckets.resize(num_nodes);
230 num_buckets_index.resize(num_nodes);
231 node_interval_trees.assign(num_nodes, SplayTree());
232
233 // Lambda to calculate an interval for one dimension.
234 auto calculate_interval = [&](double lb, double ub, double base_interval, int pos, int max_intervals,
235 bool is_forward) -> std::pair<double, double> {
236 double start, end;
237 if (is_forward) {
238 // For the first interval, start exactly at lb.
239 start = (pos == 0) ? lb : lb + pos * base_interval;
240 // For the last interval, end exactly at ub.
241 end = (pos == max_intervals - 1) ? ub : lb + (pos + 1) * base_interval;
242 } else {
243 // For Backward: first interval gets ub exactly, last gets lb
244 // exactly.
245 start = (pos == max_intervals - 1) ? lb : ub - (pos + 1) * base_interval;
246 end = (pos == 0) ? ub : ub - pos * base_interval;
247 }
248 start = roundToTwoDecimalPlaces(start);
249 end = roundToTwoDecimalPlaces(end);
250 return {start, end};
251 };
252
253 int cum_sum = 0;
254 int bucket_index = 0;
255 std::vector<double> interval_start(num_intervals);
256 std::vector<double> interval_end(num_intervals);
257 std::vector<int> pos(num_intervals,
258 0); // combination indices for multi-dimensions
259
260 // 4. Process each node to define new buckets.
261 for (const auto &vrp_node : nodes) {
262 std::vector<int> node_split_counts(num_intervals);
263 std::vector<double> node_base_interval(num_intervals);
264 // Determine the number of splits and base interval for each
265 // resource dimension.
266 for (int r = 0; r < num_intervals; ++r) {
267 double full_range = R_max[r] - R_min[r];
268 double node_range = vrp_node.ub[r] - vrp_node.lb[r];
269 int splits = 1;
270 if (std::fabs(full_range) > std::numeric_limits<double>::epsilon()) {
271 double splits_d = (node_range * intervals[r].interval) / full_range;
272 splits = std::max(1, static_cast<int>(std::round(splits_d)));
273 }
274 node_split_counts[r] = splits;
275 node_base_interval[r] = (vrp_node.ub[r] - vrp_node.lb[r]) / static_cast<double>(splits);
276 }
277 // Optionally store node-specific split counts.
278 bucket_splits[vrp_node.id] = node_split_counts;
279
280 int n_buckets = 0;
281 if (num_intervals == 1) {
282 // Single-dimensional case.
283 for (int j = 0; j < node_split_counts[0]; ++j) {
284 auto [int_start, int_end] = calculate_interval(vrp_node.lb[0], vrp_node.ub[0], node_base_interval[0], j,
285 node_split_counts[0], D == Direction::Forward);
286 if constexpr (D == Direction::Backward)
287 int_start = std::max(int_start, vrp_node.lb[0]);
288 else
289 int_end = std::min(int_end, vrp_node.ub[0]);
290 buckets.push_back(Bucket(vrp_node.id, std::vector<double>{int_start}, std::vector<double>{int_end}));
291 n_buckets++;
292 cum_sum++;
293 }
294 } else {
295 // Multi-dimensional: iterate over all combinations.
296 fmt::print("Multiple intervals\n");
297 std::fill(pos.begin(), pos.end(), 0);
298 while (true) {
299 for (int r = 0; r < num_intervals; ++r) {
300 auto [int_start, int_end] =
301 calculate_interval(vrp_node.lb[r], vrp_node.ub[r], node_base_interval[r], pos[r],
302 node_split_counts[r], D == Direction::Forward);
303 interval_start[r] = int_start;
304 interval_end[r] = int_end;
305 if constexpr (D == Direction::Backward)
306 interval_start[r] = std::max(interval_start[r], R_min[r]);
307 else
308 interval_end[r] = std::min(interval_end[r], R_max[r]);
309 }
310 buckets.push_back(Bucket(vrp_node.id, interval_start, interval_end));
311 n_buckets++;
312 cum_sum++;
313
314 // Generate next combination.
315 int i = 0;
316 while (i < num_intervals) {
317 ++pos[i];
318 if (pos[i] < node_split_counts[i])
319 break;
320 else {
321 pos[i] = 0;
322 ++i;
323 }
324 }
325 if (i == num_intervals) break;
326 }
327 }
328 num_buckets[vrp_node.id] = n_buckets;
329 num_buckets_index[vrp_node.id] = cum_sum - n_buckets;
330 // Optionally assign node_interval_trees[vrp_node.id] = node_tree;
331 //
332 if constexpr (D == Direction::Forward) { n_segments = (n_buckets + 63) >> 6; }
333 }
334
335 // 5. Process fixed bucket ranges (if available).
336 if (!node_pair_fixed_ranges.empty()) {
337 size_t new_size = buckets.size();
338 size_t required_bitmap_words = std::max(size_t(1), ((new_size * new_size) + 63) / 64);
339 fixed_buckets_bitmap.clear();
340 fixed_buckets_bitmap.resize(required_bitmap_words, 0);
341 for (size_t from = 0; from < new_size; ++from) {
342 const auto &bucket_from = buckets[from];
343 int from_node = bucket_from.node_id;
344 if (node_pair_fixed_ranges.find(from_node) == node_pair_fixed_ranges.end()) continue;
345 for (size_t to = 0; to < new_size; ++to) {
346 const auto &bucket_to = buckets[to];
347 int to_node = bucket_to.node_id;
348 if (node_pair_fixed_ranges[from_node].find(to_node) == node_pair_fixed_ranges[from_node].end())
349 continue;
350 const auto &range = node_pair_fixed_ranges[from_node][to_node];
351 bool in_range = true;
352 for (int r = 0; r < options.main_resources.size(); ++r) {
353 if (bucket_from.lb[r] < range.first[r] || bucket_from.ub[r] > range.second[r]) {
354 in_range = false;
355 break;
356 }
357 }
358 if (in_range) {
359 size_t bit_pos = from * new_size + to;
360 size_t word_idx = bit_pos / 64;
361 if (word_idx < fixed_buckets_bitmap.size()) {
362 fixed_buckets_bitmap[word_idx] |= (1ULL << (bit_pos % 64));
363 }
364 }
365 }
366 }
367 }
368
369 // 6. Finalize: record the total bucket count.
370 buckets_size = buckets.size();
371}

380template <Direction D>
381void BucketGraph::generate_arcs() {
382 // Mutex for protecting global bucket updates.
383 std::mutex buckets_mutex;
384
385 // Clear the appropriate graph.
386 if constexpr (D == Direction::Forward) {
387 fw_bucket_graph.clear();
388 } else {
389 bw_bucket_graph.clear();
390 }
391
392 // Resolve references based on the direction.
393 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
394 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
395 auto &num_buckets_idx = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
396 // Note: we no longer rely solely on bucket_splits here.
397
398 // Pre-compute intervals (base widths) for each node and each resource,
399 // using the same logic as in define_buckets.
400 const int num_resources = options.resources.size();
401 std::vector<std::vector<double>> node_base_intervals(nodes.size());
402 // Also store the computed splits per resource for each node.
403 std::vector<std::vector<int>> node_split_counts(nodes.size());
404
405 for (size_t node_id = 0; node_id < nodes.size(); ++node_id) {
406 const auto &node = nodes[node_id];
407 node_base_intervals[node_id].resize(num_resources);
408 node_split_counts[node_id].resize(num_resources);
409 for (int r = 0; r < num_resources; ++r) {
410 double full_range = R_max[r] - R_min[r];
411 double node_range = node.ub[r] - node.lb[r];
412 int splits = 1;
413 if (std::fabs(full_range) > std::numeric_limits<double>::epsilon()) {
414 double splits_d = (node_range * intervals[r].interval) / full_range;
415 splits = std::max(1, static_cast<int>(std::round(splits_d)));
416 }
417 node_split_counts[node_id][r] = splits;
418 node_base_intervals[node_id][r] = (node.ub[r] - node.lb[r]) / static_cast<double>(splits);
419 }
420 }
421
422 // Clear all buckets and their arcs.
423 for (auto &bucket : buckets) {
424 bucket.clear();
425 bucket.clear_arcs(D == Direction::Forward);
426 }
427
428 // Helper lambda: validate an arc and, if valid, add it.
429 auto try_add_arc = [&](const VRPNode &curr_node, int from_bucket, const VRPNode &next_node, int to_bucket,
430 const std::vector<double> &curr_node_base, const std::vector<double> &next_node_base,
431 const std::vector<double> &res_inc, double cost_inc) -> bool {
432 bool valid = true;
433 if constexpr (D == Direction::Forward) {
434 for (int r = 0; r < res_inc.size() && valid; ++r) {
435 // For forward, check that adding the resource increment
436 // from the from-bucket does not push us past the next
437 // node's upper bound.
438 if (numericutils::gt(buckets[from_bucket].lb[r] + res_inc[r], next_node.ub[r])) {
439 valid = false;
440 } else {
441 double max_calc = std::max(buckets[from_bucket].lb[r] + res_inc[r], next_node.lb[r]);
442 if (numericutils::lt(max_calc, buckets[to_bucket].lb[r]) ||
443 numericutils::gte(max_calc, buckets[to_bucket].lb[r] + next_node_base[r])) {
444 valid = false;
445 }
446 }
447 }
448 } else {
449 for (int r = 0; r < res_inc.size() && valid; ++r) {
450 // For backward, check that subtracting the resource
451 // increment does not drop below next node's lb.
452 if (numericutils::lt(buckets[from_bucket].ub[r] - res_inc[r], next_node.lb[r])) {
453 valid = false;
454 } else {
455 double min_calc = std::min(buckets[from_bucket].ub[r] - res_inc[r], next_node.ub[r]);
456 if (numericutils::gt(min_calc, buckets[to_bucket].ub[r]) ||
457 numericutils::lte(min_calc, buckets[to_bucket].ub[r] - next_node_base[r])) {
458 valid = false;
459 }
460 }
461 }
462 }
463 if (valid) {
464 // Add the arc to the global structure.
465 {
466 std::lock_guard<std::mutex> lock(buckets_mutex);
467 add_arc<D>(from_bucket, to_bucket, res_inc, cost_inc);
468 buckets[from_bucket].template add_bucket_arc<D>(from_bucket, to_bucket, res_inc, cost_inc, false);
469 }
470 }
471 return valid;
472 };
473
474 // Lambda to process a single node.
475 auto process_node = [&](int node_id) {
476 const auto &node = nodes[node_id];
477 std::vector<double> res_inc(num_resources);
478 // Local container for arcs (if further processing is needed).
479 std::vector<std::pair<int, int>> local_arcs;
480 local_arcs.reserve(nodes.size());
481
482 const auto arcs = node.get_arcs<D>();
483 // Use the pre-computed base intervals for the current node.
484 const auto &curr_node_base = node_base_intervals[node.id];
485
486 // For every bucket in the current node.
487 for (int i = 0; i < num_buckets[node.id]; ++i) {
488 int from_bucket = i + num_buckets_idx[node.id];
489 // Process each outgoing arc.
490 for (const auto &arc : nodes) {
491 const auto &next_node = arc;
492 // Avoid self-loop.
493 if (node.id == next_node.id) continue;
494
495 const auto &next_node_base = node_base_intervals[next_node.id];
496 const double travel_cost = getcij(node.id, next_node.id);
497 double cost_inc = travel_cost - next_node.cost;
498 // Pre-calculate resource increments.
499 for (int r = 0; r < num_resources; r++) {
500 res_inc[r] = node.consumption[r];
501 if (options.resources[r] == "time") { res_inc[r] += travel_cost; }
502 }
503
504 // Iterate over buckets in the next node.
505 for (int j = 0; j < num_buckets[next_node.id]; ++j) {
506 int to_bucket = j + num_buckets_idx[next_node.id];
507 // if (from_bucket == to_bucket ||
508 // is_bucket_fixed<D>(from_bucket, to_bucket)) {
509 // continue;
510 // }
511 if (from_bucket == to_bucket) { continue; }
512 // // Validate and add arc if valid.
513 if (try_add_arc(node, from_bucket, next_node, to_bucket, curr_node_base, next_node_base, res_inc,
514 cost_inc)) {
515 local_arcs.emplace_back(from_bucket, to_bucket);
516 }
517 }
518 }
519 }
520 // If needed, local_arcs can be used for debugging or further
521 // processing.
522 };
523
524 // Create a list of tasks (each task is a node index).
525 std::vector<int> tasks(nodes.size());
526 std::iota(tasks.begin(), tasks.end(), 0);
527
528 // Choose a chunk size based on a heuristic.
529 constexpr int chunk_size = 10;
530
531 // Set up the thread pool.
532 const unsigned int thread_count = std::max(1u, std::thread::hardware_concurrency() / 2);
533 exec::static_thread_pool pool(thread_count);
534 auto sched = pool.get_scheduler();
535
536 // Launch tasks in parallel over chunks.
537 auto bulk_sender = stdexec::bulk(stdexec::just(), (tasks.size() + chunk_size - 1) / chunk_size,
538 [&, chunk_size](std::size_t chunk_idx) {
539 const size_t start_idx = chunk_idx * chunk_size;
540 const size_t end_idx = std::min(start_idx + chunk_size, tasks.size());
541 for (size_t idx = start_idx; idx < end_idx; ++idx) {
542 process_node(tasks[idx]);
543 }
544 });
545
546 auto work = stdexec::starts_on(sched, bulk_sender);
547 stdexec::sync_wait(std::move(work));
548}

561template <Direction D>
562Label *BucketGraph::get_best_label(const std::vector<int> &topological_order, const std::vector<double> &c_bar,

575 }

585template <Stage S, Symmetry SYM>
586void BucketGraph::ConcatenateLabel(const Label *L, int &b, std::atomic<double> &best_cost) {
587 // Determine the number of segments needed for the bucket visited
588 // bitmap.
589 const size_t n_segments = (fw_buckets_size + 63) / 64;
590 static thread_local std::vector<uint64_t> Bvisited;
591 if (Bvisited.size() != n_segments) { Bvisited.resize(n_segments); }
592 std::memset(Bvisited.data(), 0, n_segments * sizeof(uint64_t));
593
594 // Use thread_local storage for the bucket stack to avoid reallocations.
595 static thread_local std::vector<int> bucket_stack;
596 if (bucket_stack.capacity() < 64) bucket_stack.reserve(64);
597 bucket_stack.clear(); // Clear for reuse
598 bucket_stack.push_back(b);
599
600 static thread_local std::vector<std::pair<const Label *, double>> merge_candidates;
601
602 // Choose symmetric buckets and cost-bar arrays based on symmetry.
603 auto &other_buckets = assign_symmetry<SYM>(fw_buckets, bw_buckets);
604 auto &other_c_bar = assign_symmetry<SYM>(fw_c_bar, bw_c_bar);
605
606 // Cache frequently accessed values.
607 const int L_node_id = L->node_id;
608 const double L_cost = L->cost;
609
610 // Pre-compute if branching is active.
611 const bool has_branching = !branching_duals->empty();
612
613#if defined(SRC)
614 // SRC mode setup: Only relevant in Stage::Four.
615 decltype(cut_storage) cutter = nullptr;
616 if constexpr (S > Stage::Three) { cutter = cut_storage; }
617 const auto active_cuts = cutter->getActiveCuts();
618#endif
619
620// Lambda to apply SRC adjustments to total_cost (only for Stage::Four).
621#if defined(SRC)
622 auto applySRCAdjustments = [&](double total_cost, const Label *L, const Label *L_bw) -> double {
623 if constexpr (S == Stage::Four) {
624 for (const auto &active_cut : active_cuts) {
625 const auto &cut = *active_cut.cut_ptr;
626 const size_t idx = active_cut.index;
627 const double dual = active_cut.dual_value;
628 if (L->SRCmap[idx] + L_bw->SRCmap[idx] >= cut.p.den) { total_cost -= dual; }
629 }
630 }
631 return total_cost;
632 };
633#endif
634
635 // Process buckets until the stack is empty.
636 while (!bucket_stack.empty()) {
637 const int current_bucket = bucket_stack.back();
638 bucket_stack.pop_back();
639
640 // Mark current_bucket as visited in the bitmap.
641 const size_t segment = current_bucket >> 6;
642 const uint64_t bit_mask = bit_mask_lookup[current_bucket & 63];
643 Bvisited[segment] |= bit_mask;
644
645 // Get the node id associated with this bucket.
646 const int bucketLprimenode = other_buckets[current_bucket].node_id;
647 double travel_cost = getcij(L_node_id, bucketLprimenode);
648
649#if defined(RCC) || defined(EXACT_RCC)
650 // Adjust travel cost with arc duals in Stage Four.
651 if constexpr (S == Stage::Four) { travel_cost -= arc_duals.getDual(L_node_id, bucketLprimenode); }
652#endif
653
654 if (has_branching) { travel_cost -= branching_duals->getDual(L_node_id, bucketLprimenode); }
655
656 const double path_cost = L_cost + travel_cost;
657 const double bound = other_c_bar[current_bucket];
658
659 const double prune_limit = best_cost.load(std::memory_order_relaxed);
660
661 // Early bound check.
662 if ((S != Stage::Enumerate && numericutils::gte(path_cost + bound, prune_limit)) ||
663 (S == Stage::Enumerate && numericutils::gte(path_cost + bound, gap))) {
664 continue;
665 }
666
667 // Retrieve labels in the current bucket.
668 const auto &bucket = other_buckets[current_bucket];
669 const auto &labels = bucket.get_labels();
670 if (labels.empty()) continue;
671 merge_candidates.clear();
672 if (merge_candidates.capacity() < labels.size()) { merge_candidates.reserve(labels.size()); }
673
674 // Process each label in the current bucket.
675 for (const Label *L_bw : labels) {
676 // Skip invalid labels with combined check
677 if (L_bw == nullptr || L_bw->is_dominated || L_bw->node_id == L_node_id) { continue; }
678
679 // Check feasibility
680 if (!check_feasibility(L, L_bw)) { continue; }
681
682 double total_cost = path_cost + L_bw->cost;
683
684#if defined(SRC)
685 // Apply SRC adjustments if in Stage Four.
686 total_cost = applySRCAdjustments(total_cost, L, L_bw);
687#endif
688 double latest_best = best_cost.load(std::memory_order_relaxed);
689
690 // Filter based on total cost against best known cost or gap.
691 bool cost_acceptable;
692 if constexpr (S != Stage::Enumerate) {
693 cost_acceptable = numericutils::lt(total_cost, latest_best);
694 } else {
695 cost_acceptable = numericutils::lt(total_cost, gap);
696 }
697
698 if (!cost_acceptable) { continue; }
699 merge_candidates.emplace_back(L_bw, total_cost);
700 }
701
702 if (!merge_candidates.empty()) {
703 // Replay accepted candidates in order under one lock.
704 std::lock_guard<std::mutex> lock(merge_mutex);
705
706 double current_best = best_cost.load(std::memory_order_relaxed);
707 for (const auto &[candidate_label, candidate_cost] : merge_candidates) {
708 if (numericutils::lt(candidate_cost, current_best)) {
709 best_cost.store(candidate_cost, std::memory_order_relaxed);
710 current_best = candidate_cost;
711 }
712
713 if (numericutils::lte(candidate_cost, current_best)) {
714 auto pbest = compute_label<S>(L, candidate_label, candidate_cost);
715 merged_labels.push_back(pbest);
716 }
717 }
718 }
719 // Add neighbor buckets to the stack if they have not been visited.
720 const auto &neighbors = Phi_bw[current_bucket];
721 if (bucket_stack.capacity() < bucket_stack.size() + neighbors.size()) {
722 bucket_stack.reserve(bucket_stack.size() + neighbors.size());
723 }
724 for (int b_prime : neighbors) {
725 const size_t seg_prime = b_prime >> 6;
726 const uint64_t mask_prime = bit_mask_lookup[b_prime & 63];
727 if (!(Bvisited[seg_prime] & mask_prime)) { bucket_stack.push_back(b_prime); }
728 }
729 }
730}

744template <Direction D>
745void BucketGraph::SCC_handler() {
746 // Get necessary references.
747 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
748 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
749 auto &bucket_graph = assign_buckets<D>(fw_bucket_graph, bw_bucket_graph);
750
751 // Build an extended bucket graph from bucket_graph using Phi.
752 ankerl::unordered_dense::map<int, std::vector<int>> extended_bucket_graph = bucket_graph;
753 for (size_t i = 0; i < buckets.size(); ++i) {
754 extended_bucket_graph[static_cast<int>(i)];
755 const auto &phi_set = Phi[i];
756 if (phi_set.empty()) continue;
757 for (auto phi_bucket : phi_set) { extended_bucket_graph[phi_bucket].push_back(static_cast<int>(i)); }
758 }
759
760 // Compute SCCs using Tarjan's algorithm.
761 SCC scc_finder;
762 scc_finder.convertFromUnorderedMap(extended_bucket_graph);
763 auto sccs = scc_finder.tarjanSCC();
764 auto topological_order = scc_finder.topologicalOrderOfSCCs(sccs);
765
766#ifdef VERBOSE
767 // Optional: Print SCCs.
768 constexpr auto blue = "\033[34m";
769 constexpr auto reset = "\033[0m";
770 fmt::print((D == Direction::Forward) ? "FW SCCs:\n" : "BW SCCs:\n");
771 for (auto scc : topological_order) {
772 fmt::print("{}({}) -> {}", blue, scc, reset);
773 for (auto bucket : sccs[scc]) { fmt::print("{} ", bucket); }
774 fmt::print("\n");
775 }
776 fmt::print("\n");
777#endif
778
779 // Order SCCs according to the topological order.
780 std::vector<std::vector<int>> ordered_sccs;
781 ordered_sccs.reserve(sccs.size());
782 for (int scc : topological_order) { ordered_sccs.push_back(sccs[scc]); }
783
784 // Sort each SCC: For Forward, sort by bucket lower bound; for Backward,
785 // sort by bucket upper bound.
786 auto sorted_sccs = sccs; // Make a copy
787 for (auto &scc : sorted_sccs) {
788 if constexpr (D == Direction::Forward) {
789 pdqsort(scc.begin(), scc.end(), [&buckets](int a, int b) { return buckets[a].lb[0] < buckets[b].lb[0]; });
790 } else {
791 pdqsort(scc.begin(), scc.end(), [&buckets](int a, int b) { return buckets[a].ub[0] > buckets[b].ub[0]; });
792 }
793 }
794
795 // Ensure each VRP node has the appropriate SCC arc container allocated.
796 for (auto &node : nodes) {
797 if constexpr (D == Direction::Forward) {
798 node.fw_arcs_scc.resize(sccs.size());
799 } else {
800 node.bw_arcs_scc.resize(sccs.size());
801 }
802 }
803
804 // Lambda: Filter and collect arcs for a given bucket, node, and
805 // direction.
806 auto process_bucket_arcs = [&](int bucket, int scc_index) {
807 int from_node_id = buckets[bucket].node_id;
808 VRPNode &node = nodes[from_node_id];
809 // Select the proper arc container and node arcs based on the
810 // direction.
811 const auto &node_arcs = (D == Direction::Forward) ? node.fw_arcs : node.bw_arcs;
812 std::vector<Arc> &filtered_arcs =
813 (D == Direction::Forward) ? node.fw_arcs_scc[scc_index] : node.bw_arcs_scc[scc_index];
814
815 const auto &bucket_arcs = buckets[bucket].template get_bucket_arcs<D>();
816 for (const auto &arc : bucket_arcs) {
817 int to_node_id = buckets[arc.to_bucket].node_id;
818 // Look for an arc with destination equal to to_node_id.
819 auto it = std::find_if(node_arcs.begin(), node_arcs.end(),
820 [&to_node_id](const Arc &a) { return a.to == to_node_id; });
821 if (it != node_arcs.end()) { filtered_arcs.push_back(*it); }
822 }
823 };
824
825 // For each SCC, process arcs for each bucket in the SCC.
826 int scc_ctr = 0;
827 for (const auto &scc : sccs) {
828 for (int bucket : scc) { process_bucket_arcs(bucket, scc_ctr); }
829 ++scc_ctr;
830 }
831
832 // Lambda: Sort and deduplicate arcs within each node's SCC arc
833 // container.
834 auto sortAndDedup = [](std::vector<Arc> &arcs) {
835 pdqsort(arcs.begin(), arcs.end(),
836 [](const Arc &a, const Arc &b) { return std::tie(a.from, a.to) < std::tie(b.from, b.to); });
837 auto last = std::unique(arcs.begin(), arcs.end(),
838 [](const Arc &a, const Arc &b) { return a.from == b.from && a.to == b.to; });
839 arcs.erase(last, arcs.end());
840 };
841
842 // For each node, sort and deduplicate all arc lists.
843 for (auto &node : nodes) {
844 if constexpr (D == Direction::Forward) {
845 for (auto &arcs : node.fw_arcs_scc) { sortAndDedup(arcs); }
846 } else {
847 for (auto &arcs : node.bw_arcs_scc) { sortAndDedup(arcs); }
848 }
849 }
850
851 std::vector<int> bucket_scc_rank(buckets.size(), -1);
852 for (size_t rank = 0; rank < ordered_sccs.size(); ++rank) {
853 for (int bucket : ordered_sccs[rank]) { bucket_scc_rank[bucket] = static_cast<int>(rank); }
854 }
855
856 // Build union-find structure for SCCs using the ordered SCCs.
857 UnionFind uf(ordered_sccs);
858 if constexpr (D == Direction::Forward) {
859 fw_ordered_sccs = ordered_sccs;
860 fw_topological_order = topological_order;
861 fw_sccs = sccs;
862 fw_sccs_sorted = sorted_sccs;
863 fw_bucket_scc_rank = std::move(bucket_scc_rank);
864 fw_union_find = uf;
865 } else {
866 bw_ordered_sccs = ordered_sccs;
867 bw_topological_order = topological_order;
868 bw_sccs = sccs;
869 bw_sccs_sorted = sorted_sccs;
870 bw_bucket_scc_rank = std::move(bucket_scc_rank);
871 bw_union_find = uf;
872 }
873}

999template <Stage S>
1000void BucketGraph::heuristic_fixing() {
1001 if (fixed) { return; }
1002
1003 // Reset and initialize state.
1004 reset_pool();
1005 reset_fixed();
1006 common_initialization();
1007
1008 // Pre-allocate c-bar vectors with infinity.
1009 std::vector<double> forward_cbar(fw_buckets.size(), std::numeric_limits<double>::infinity());
1010 std::vector<double> backward_cbar(bw_buckets.size(), std::numeric_limits<double>::infinity());
1011
1012 gap = std::ceil(incumbent - (relaxation + std::min(0.0, min_red_cost)));
1013
1014 // Run the appropriate labeling algorithms based on stage
1015 if (S == Stage::Three) {
1016 // Run the labeling algorithms for Stage::Two with Full::Partial mode.
1017 run_labeling_algorithms<Stage::Two, Full::Partial>(forward_cbar, backward_cbar);
1018 } else if (S == Stage::Four) {
1019 fmt::print("\033[34m_STARTING ARC FIXING PROCEDURE\033[0m\n");
1020 fmt::print("gap: {}\n", gap);
1021 // Run the labeling algorithms for Stage::Four with Full::Full mode.
1022 run_labeling_algorithms<Stage::Four, Full::Full>(forward_cbar, backward_cbar);
1023 }
1024
1025 // Prepare containers for grouping labels by node.
1026 const size_t num_nodes = nodes.size();
1027
1028 // Preallocate vectors with nullptr to avoid resizing
1029 std::vector<const Label *> min_fw_labels(num_nodes, nullptr);
1030 std::vector<const Label *> min_bw_labels(num_nodes, nullptr);
1031
1032 // Find minimum cost label for each node directly from buckets
1033 // without creating intermediate label maps
1034 for (auto &bucket : fw_buckets) {
1035 for (const Label *label : bucket.get_labels()) {
1036 const size_t node_id = label->node_id;
1037 if (!min_fw_labels[node_id] || label->cost < min_fw_labels[node_id]->cost) {
1038 min_fw_labels[node_id] = label;
1039 }
1040 }
1041 }
1042
1043 for (auto &bucket : bw_buckets) {
1044 for (const Label *label : bucket.get_labels()) {
1045 const size_t node_id = label->node_id;
1046 if (!min_bw_labels[node_id] || label->cost < min_bw_labels[node_id]->cost) {
1047 min_bw_labels[node_id] = label;
1048 }
1049 }
1050 }
1051
1052 // Pre-compute the index of the "time" resource.
1053 const size_t time_index =
1054 std::distance(options.resources.begin(), std::find(options.resources.begin(), options.resources.end(), "time"));
1055
1056 // Precompute lookup tables for common values
1057 std::vector<double> fw_time_resources(num_nodes, 0.0);
1058 std::vector<double> bw_time_resources(num_nodes, 0.0);
1059 std::vector<double> fw_costs(num_nodes, 0.0);
1060 std::vector<double> bw_costs(num_nodes, 0.0);
1061
1062 for (size_t i = 0; i < num_nodes; ++i) {
1063 if (min_fw_labels[i]) {
1064 fw_time_resources[i] = min_fw_labels[i]->resources[time_index];
1065 fw_costs[i] = min_fw_labels[i]->cost;
1066 }
1067 if (min_bw_labels[i]) {
1068 bw_time_resources[i] = min_bw_labels[i]->resources[time_index];
1069 bw_costs[i] = min_bw_labels[i]->cost;
1070 }
1071 }
1072
1073 // Cache node data to avoid repeated lookups
1074 std::vector<double> node_durations(num_nodes);
1075 std::vector<std::vector<double>> node_consumptions(num_nodes, std::vector<double>(options.resources.size()));
1076
1077 for (size_t i = 0; i < num_nodes; ++i) {
1078 const VRPNode &node = nodes[i];
1079 node_durations[i] = node.duration;
1080 for (size_t r = 0; r < options.resources.size(); ++r) { node_consumptions[i][r] = node.consumption[r]; }
1081 }
1082
1083 size_t num_fixes = 0;
1084 const double gap_threshold = gap; // Cache this value
1085
1086 // Build a list of candidate node pairs to check
1087 struct NodePair {
1088 size_t i;
1089 size_t j;
1090 };
1091
1092 std::vector<NodePair> candidate_pairs;
1093 candidate_pairs.reserve(num_nodes * num_nodes / 4); // Estimate: 25% of pairs will be candidates
1094
1095 for (size_t i = 0; i < num_nodes; ++i) {
1096 if (!min_fw_labels[i]) continue;
1097
1098 for (size_t j = 0; j < num_nodes; ++j) {
1099 if (i == j || !min_bw_labels[j]) continue;
1100 candidate_pairs.push_back({i, j});
1101 }
1102 }
1103
1104 // Process candidate pairs in batches to improve cache locality
1105 const size_t BATCH_SIZE = 256;
1106 std::vector<std::pair<size_t, size_t>> arcs_to_fix;
1107 arcs_to_fix.reserve(BATCH_SIZE);
1108
1109 for (size_t start = 0; start < candidate_pairs.size(); start += BATCH_SIZE) {
1110 const size_t end = std::min(start + BATCH_SIZE, candidate_pairs.size());
1111 arcs_to_fix.clear();
1112
1113 for (size_t idx = start; idx < end; ++idx) {
1114 const auto &pair = candidate_pairs[idx];
1115 const size_t i = pair.i;
1116 const size_t j = pair.j;
1117
1118 const double cost = getcij(min_fw_labels[i]->node_id, min_bw_labels[j]->node_id);
1119
1120 // Early rejection using precomputed time resources (often most
1121 // restrictive)
1122 if (options.resources[time_index] == "time") {
1123 if (fw_time_resources[i] + cost + node_durations[min_fw_labels[i]->node_id] > bw_time_resources[j]) {
1124 continue;
1125 }
1126 }
1127
1128 // Check all other resources
1129 bool violated = false;
1130 const size_t fw_node_id = min_fw_labels[i]->node_id;
1131
1132 for (size_t r = 0; r < options.resources.size(); ++r) {
1133 if (r != time_index) {
1134 if (min_fw_labels[i]->resources[r] + node_consumptions[fw_node_id][r] >
1135 min_bw_labels[j]->resources[r]) {
1136 violated = true;
1137 break;
1138 }
1139 }
1140 }
1141
1142 // If all resource constraints are satisfied and the cost exceeds
1143 // the gap, fix the arc
1144 if (!violated && (fw_costs[i] + cost + bw_costs[j] > gap_threshold)) {
1145 arcs_to_fix.emplace_back(i, j);
1146 ++num_fixes;
1147 }
1148 }
1149
1150 // Fix arcs in batch
1151 for (const auto &[i, j] : arcs_to_fix) { fix_arc(i, j); }
1152 }
1153
1154 if (S == Stage::Four) {
1155 fmt::print("Applied {} arc fixes\n", num_fixes);
1156 fmt::print("\033[34m_ARC FIXING PROCEDURE FINISHED\033[0m\n");
1157 }
1158}

1160template <Symmetry SYM>
1161void BucketGraph::set_adjacency_list() {
1162 // === Step 1: Clear Existing Arcs from Each Node ===
1163 for (auto &node : nodes) { node.clear_arcs(); }
1164
1165 // === Step 2: MST-Based Clustering ===
1166 MST mst_solver(nodes, [this](int from, int to) { return getcij(from, to); });
1167 auto mst = mst_solver.compute_mst();
1168
1169 // Collect and sort MST edge weights.
1170 std::vector<double> edge_weights;
1171 edge_weights.reserve(mst.size());
1172 for (const auto &[weight, from, to] : mst) { edge_weights.push_back(weight); }
1173 pdqsort(edge_weights.begin(), edge_weights.end());
1174
1175 // Compute theta as the 90th percentile (scaled down by 100).
1176 size_t p90_index = static_cast<size_t>(0.9 * edge_weights.size());
1177 double theta = (edge_weights[p90_index]) / 100.0;
1178 print_info("Computed theta: {}\n", theta);
1179
1180 // Cluster nodes and build mapping: job_id -> cluster_id.
1181 auto clusters = mst_solver.cluster(theta);
1182 print_info("Number of clusters: {}\n", clusters.size());
1183 std::vector<int> job_to_cluster(nodes.size(), -1);
1184 for (int cluster_id = 0; cluster_id < clusters.size(); ++cluster_id) {
1185 for (int job : clusters[cluster_id]) { job_to_cluster[job] = cluster_id; }
1186 }
1187
1188 // === Step 3: Precompute Resource Information ===
1189 // Find the index for the "time" resource.
1190 auto time_it = std::find(options.resources.begin(), options.resources.end(), "time");
1191 const int time_resource_idx =
1192 (time_it != options.resources.end()) ? std::distance(options.resources.begin(), time_it) : -1;
1193
1194 // === Step 4: Process Arcs for Each Node ===
1195
1196 // Helper lambda to compute resource increments.
1197 // Depending on symmetry, consumption is either taken as-is or averaged.
1198 auto compute_resource_increments = [&](const VRPNode &from_node, const VRPNode &to_node, double travel_cost,
1199 std::vector<double> &inc) {
1200 const int nres = options.resources.size();
1201 if constexpr (SYM == Symmetry::Asymmetric) {
1202 for (int r = 0; r < nres; ++r) {
1203 inc[r] = (r == time_resource_idx) ? travel_cost + from_node.duration : from_node.consumption[r];
1204 }
1205 } else {
1206 for (int r = 0; r < nres; ++r) {
1207 inc[r] = (r == time_resource_idx) ? (from_node.duration / 2.0 + travel_cost + to_node.duration / 2.0)
1208 : ((from_node.consumption[r] + to_node.consumption[r]) / 2.0);
1209 }
1210 }
1211 };
1212
1213 // Lambda to add forward and reverse arcs from a given node.
1214 auto add_arcs_for_node = [&](const VRPNode &curr_node) {
1215 // Containers to accumulate arcs before insertion.
1216 using ArcTuple = std::tuple<double, int, std::vector<double>, double>;
1217 std::vector<ArcTuple> forward_arcs;
1218 std::vector<ArcTuple> reverse_arcs;
1219 forward_arcs.reserve(nodes.size());
1220 reverse_arcs.reserve(nodes.size());
1221
1222 // Iterate over candidate next nodes.
1223 for (const auto &next_node : nodes) {
1224 // Skip if the candidate is a depot, the same node, or a pstep
1225 // depot.
1226 if (next_node.id == options.depot || curr_node.id == next_node.id ||
1227 (options.pstep && (next_node.id == options.pstep_depot || next_node.id == options.pstep_end_depot))) {
1228 continue;
1229 }
1230
1231 double travel_cost = getcij(curr_node.id, next_node.id);
1232 double cost_inc = travel_cost - next_node.cost;
1233
1234 // Skip self-loops (or if bucket indexes match; here using node
1235 // id as bucket id).
1236 if (curr_node.id == next_node.id) continue;
1237
1238 // Compute resource increments.
1239 std::vector<double> res_inc(options.resources.size());
1240 compute_resource_increments(curr_node, next_node, travel_cost, res_inc);
1241
1242 // Check feasibility: ensure resource consumption does not
1243 // exceed next node's upper bound.
1244 bool feasible = true;
1245 for (int r = 0; r < options.resources.size(); ++r) {
1246 if (numericutils::gt(curr_node.lb[r] + res_inc[r], next_node.ub[r])) {
1247 feasible = false;
1248 break;
1249 }
1250 }
1251 if (!feasible) continue;
1252
1253 // Compute priorities based on cluster membership.
1254 bool same_cluster = (job_to_cluster[curr_node.id] == job_to_cluster[next_node.id]);
1255 double forward_priority = (same_cluster ? 5.0 : 1.0) + 1.E-5 * next_node.start_time;
1256 double reverse_priority = (same_cluster ? 1.0 : 5.0) + 1.E-5 * curr_node.start_time;
1257
1258 // Collect arcs for forward and reverse directions.
1259 forward_arcs.emplace_back(forward_priority, next_node.id, res_inc, cost_inc);
1260 reverse_arcs.emplace_back(reverse_priority, next_node.id, res_inc, cost_inc);
1261 }
1262
1263 // Insert the arcs into the corresponding nodes.
1264 for (const auto &[priority, to_node, res_inc, cost_inc] : forward_arcs) {
1265 nodes[curr_node.id].template add_arc<Direction::Forward>(curr_node.id, to_node, res_inc, cost_inc,
1266 priority);
1267 }
1268 for (const auto &[priority, from_node, res_inc, cost_inc] : reverse_arcs) {
1269 nodes[from_node].template add_arc<Direction::Backward>(from_node, curr_node.id, res_inc, cost_inc,
1270 priority);
1271 }
1272 };
1273
1274 // Process each node, skipping the end depot.
1275 for (const auto &node : nodes) {
1276 if (node.id != options.end_depot) { add_arcs_for_node(node); }
1277 }
1278}

1315 if constexpr (Direction::Forward == D) {
1316 dominance_checks_per_bucket.assign(buckets_size + 1, 0);
1317 non_dominated_labels_per_bucket = 0;
1318 }
1319
1320 // --- Clear all buckets ---
1321 for (size_t b = 0; b < buckets_size; b++) { buckets[b].clear(); }
1322
1323 // --- Warm start processing: sort warm_labels and compute reduced costs
1324 // ---
1325 if (options.warm_start && !just_fixed && !warm_labels.empty()) {
1326 pdqsort(warm_labels.begin(), warm_labels.end(),
1327 [](const WarmLabelState &a, const WarmLabelState &b) { return a.cost < b.cost; });
1328 const size_t process_size =
1329 std::min(warm_labels.size(), static_cast<size_t>(options.n_warm_start * warm_labels.size()));
1330 for (size_t i = 0; i < process_size; ++i) {
1331 auto new_label = compute_red_cost(warm_labels[i], D == Direction::Forward);
1332 if (new_label != nullptr) {
1333 if constexpr (D == Direction::Forward) {
1334 fw_buckets[new_label->vertex].add_sorted_label(new_label);
1335 } else {
1336 bw_buckets[new_label->vertex].add_sorted_label(new_label);
1337 }
1338 }
1339 }
1340 warm_labels.clear();
1341 }
1342
1343 // --- Prepare interval arrays and compute per-resource splits for depot
1344 // ---
1345 const size_t num_intervals = options.main_resources.size();
1346 std::vector<double> base_intervals(num_intervals);
1347 std::vector<double> interval_starts(num_intervals);
1348 std::vector<double> interval_ends(num_intervals);
1349 // Compute per-resource split counts for the depot
1350 const auto &VRPNode = nodes[0]; // assuming node 0 is the depot
1351 std::vector<int> depot_splits(num_intervals);
1352 for (size_t r = 0; r < num_intervals; ++r) {
1353 // Use the same ratio as earlier:
1354 double full_range = R_max[r] - R_min[r];
1355 double node_range = VRPNode.ub[r] - VRPNode.lb[r];
1356 depot_splits[r] = std::max(1, static_cast<int>(std::round((node_range / full_range) * intervals[r].interval)));
1357 // Compute the base interval for resource r.
1358 base_intervals[r] = (VRPNode.ub[r] - VRPNode.lb[r]) / depot_splits[r];
1359 interval_starts[r] = VRPNode.lb[r];
1360 interval_ends[r] = VRPNode.ub[r];
1361 }
1362
1363 // --- Interval-based bucket generation ---
1364 std::vector<int> current_pos(num_intervals, 0);
1365 int offset = 0;
1366 // Lambda for generating interval combinations and creating depots.
1367 auto generate_combinations = [&](bool is_forward, int &offset) {
1368 auto &label_pool = is_forward ? label_pool_fw : label_pool_bw;
1369 auto &buckets = is_forward ? fw_buckets : bw_buckets;
1370 const auto depot_id = is_forward ? options.depot : options.end_depot;
1371 const int calculated_index_base =
1372 is_forward ? num_buckets_index_fw[options.depot] : num_buckets_index_bw[options.end_depot];
1373
1374 std::vector<double> interval_bounds(num_intervals);
1375 // Recursive lambda to generate all interval combinations.
1376 std::function<void(int)> process_intervals = [&](int depth) {
1377 if (depth == num_intervals) {
1378 // Acquire a new depot label.
1379 auto depot = label_pool->acquire();
1380 int calculated_index = calculated_index_base + offset;
1381 for (int r = 0; r < static_cast<int>(num_intervals); ++r) {
1382 // For forward direction, start at the lower bound and
1383 // add multiples of the base interval; adjust similarly
1384 // for backward.
1385 interval_bounds[r] = is_forward ? interval_starts[r] + current_pos[r] * base_intervals[r]
1386 : interval_ends[r] - current_pos[r] * base_intervals[r];
1387 }
1388 depot->initialize(calculated_index, 0.0, interval_bounds, depot_id);
1389 depot->is_extended = false;
1390 depot->addNode(depot_id);
1391 set_node_visited(depot->visited_bitmap, depot_id);
1392 SRC_MODE_BLOCK(depot->SRCmap.assign(cut_storage->SRCDuals.size(), 0);)
1393 buckets[calculated_index].add_label(depot);
1394 buckets[calculated_index].node_id = depot_id;
1395
1396 offset++;
1397 return;
1398 }
1399
1400 // For the current resource dimension, loop using its specific
1401 // split count.
1402 for (int k = 0; k < depot_splits[depth]; ++k) {
1403 current_pos[depth] = k;
1404 process_intervals(depth + 1);
1405 }
1406 };
1407 process_intervals(0);
1408 };
1409
1410 // Generate combinations for the current direction.
1411 generate_combinations(D == Direction::Forward, offset);
1412}

85template <Direction D>
86inline int BucketGraph::get_bucket_number(
87 int node, std::vector<double> &resource_values_vec) noexcept {
88 const size_t num_resources = options.main_resources.size();
89
90 const auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
91
92 // Optionally adjust resource values (uncomment if you need epsilon
93 // adjustments)
94 for (size_t r = 0; r < num_resources; ++r) {
95 if constexpr (D == Direction::Forward) {
96 // resource_values_vec[r] += numericutils::eps;
97 } else { // Direction::Backward
98 // resource_values_vec[r] -= numericutils::eps;
99 }
100 }
101 auto &num_buckets_index =
102 assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
103 // Start with the offset for this node’s buckets.
104 int bucket_number = num_buckets_index[node];
105 int multiplier = 1;
106
107 for (size_t r = 0; r < num_resources; ++r) {
108 // Node-specific resource bounds.
109 double node_lb = nodes[node].lb[r];
110 double node_ub = nodes[node].ub[r];
111
112 // Global bounds (if needed for computing splits).
113 double global_lb = R_min[r];
114 double global_ub = R_max[r];
115 double full_range = global_ub - global_lb;
116 double node_range = node_ub - node_lb;
117
118 // Determine the number of splits (i.e. intervals) for resource r.
119 int splits =
120 std::max(1, static_cast<int>(std::round((node_range / full_range) *
121 intervals[r].interval)));
122 double interval_width = node_range / splits;
123
124 int pos = 0;
125 if constexpr (D == Direction::Forward) {
126 // Compute the bucket "digit" from the lower bound.
127 pos = std::min(
128 splits - 1,
129 static_cast<int>(std::floor((resource_values_vec[r] - node_lb) /
130 interval_width)));
131 } else { // Direction::Backward
132 // For backward, we consider the distance from the upper bound.
133 pos = std::min(
134 splits - 1,
135 static_cast<int>(std::floor((node_ub - resource_values_vec[r]) /
136 interval_width)));
137 }
138
139 bucket_number += pos * multiplier;
140 multiplier *= splits;
141 }
142
143 // check if buckets[bucket_number] contains the resource_vec
144 if (unlikely(!buckets[bucket_number].contains(resource_values_vec))) {
145 // print resource_values_vec
146 for (auto val : resource_values_vec) {
147 fmt::print("{}\n", val);
148 }
149 // print bucket bounds
150 fmt::print("{}\n", buckets[bucket_number].lb[0]);
151 fmt::print("{}\n", buckets[bucket_number].ub[0]);
152
153 std::throw_with_nested(
154 std::runtime_error("Resource values not contained in bucket"));
155 return -1;
156 }
157
158 return bucket_number;
159}

171// A constant representing the number of splits for a node that spans the full
172// range. For example, if a node covers the entire full range, it will get
173// FULL_RANGE_SPLITS buckets.
174void BucketGraph::define_buckets() {
175 // 1. Save the old buckets and fixed status.
176 std::vector<Bucket> old_buckets;
177 std::vector<bool> was_fixed;
178 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
179 size_t old_size = buckets.size();
180
181 // Map to store the fixed range (min_lb and max_ub) for each node pair.
182 ankerl::unordered_dense::map<
183 int, ankerl::unordered_dense::map<
184 int, std::pair<std::vector<double>, std::vector<double>>>>
185 node_pair_fixed_ranges;
186
187 if (old_size > 0) {
188 old_buckets = buckets;
189 was_fixed.resize(old_size * old_size, false);
190 auto &old_bitmap =
191 assign_buckets<D>(fw_fixed_buckets_bitmap, bw_fixed_buckets_bitmap);
192
193 // Mark fixed bucket pairs.
194 for (size_t i = 0; i < old_size; ++i) {
195 for (size_t j = 0; j < old_size; ++j) {
196 size_t pos = i * old_size + j;
197 if (pos / 64 < old_bitmap.size()) {
198 was_fixed[pos] =
199 ((old_bitmap[pos / 64] & (1ULL << (pos % 64))) != 0);
200 }
201 }
202 }
203
204 const int num_intervals = options.main_resources.size();
205 // For each fixed bucket pair, update the fixed range for the
206 // corresponding node pair.
207 for (size_t i = 0; i < old_size; ++i) {
208 for (size_t j = 0; j < old_size; ++j) {
209 if (!was_fixed[i * old_size + j]) continue;
210 const auto &bucket_from = old_buckets[i];
211 const auto &bucket_to = old_buckets[j];
212 int from_node = bucket_from.node_id;
213 int to_node = bucket_to.node_id;
214 // Initialize if necessary.
215 if (node_pair_fixed_ranges[from_node].find(to_node) ==
216 node_pair_fixed_ranges[from_node].end()) {
217 node_pair_fixed_ranges[from_node][to_node] = {
218 std::vector<double>(num_intervals,
219 std::numeric_limits<double>::max()),
220 std::vector<double>(
221 num_intervals,
222 std::numeric_limits<double>::lowest())};
223 }
224 auto &[min_lbs, max_ubs] =
225 node_pair_fixed_ranges[from_node][to_node];
226 for (int r = 0; r < num_intervals; ++r) {
227 min_lbs[r] = std::min(min_lbs[r], bucket_from.lb[r]);
228 max_ubs[r] = std::max(max_ubs[r], bucket_from.ub[r]);
229 }
230 }
231 }
232 } // End if old buckets exist.
233
234 // 2. Clear the current buckets and fixed bitmap.
235 buckets.clear();
236 auto &fixed_buckets_bitmap =
237 assign_buckets<D>(fw_fixed_buckets_bitmap, bw_fixed_buckets_bitmap);
238 fixed_buckets_bitmap.clear();
239
240 // 3. Prepare variables and containers for new bucket definition.
241 const int num_intervals = options.main_resources.size();
242 std::vector<double> total_ranges(num_intervals); // unused in current code
243 std::vector<double> base_intervals(
244 num_intervals); // unused in current code
245
246 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
247 auto &num_buckets_index =
248 assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
249 auto &node_interval_trees =
250 assign_buckets<D>(fw_node_interval_trees, bw_node_interval_trees);
251 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
252 auto &bucket_splits = assign_buckets<D>(fw_bucket_splits, bw_bucket_splits);
253
254 const size_t num_nodes = nodes.size();
255 num_buckets.resize(num_nodes);
256 num_buckets_index.resize(num_nodes);
257 node_interval_trees.assign(num_nodes, SplayTree());
258
259 // Lambda to calculate an interval for one dimension.
260 auto calculate_interval =
261 [&](double lb, double ub, double base_interval, int pos,
262 int max_intervals, bool is_forward) -> std::pair<double, double> {
263 double start, end;
264 if (is_forward) {
265 // For the first interval, start exactly at lb.
266 start = (pos == 0) ? lb : lb + pos * base_interval;
267 // For the last interval, end exactly at ub.
268 end = (pos == max_intervals - 1) ? ub
269 : lb + (pos + 1) * base_interval;
270 } else {
271 // For Backward: first interval gets ub exactly, last gets lb
272 // exactly.
273 start = (pos == max_intervals - 1) ? lb
274 : ub - (pos + 1) * base_interval;
275 end = (pos == 0) ? ub : ub - pos * base_interval;
276 }
277 start = roundToTwoDecimalPlaces(start);
278 end = roundToTwoDecimalPlaces(end);
279 return {start, end};
280 };
281
282 int cum_sum = 0;
283 int bucket_index = 0;
284 std::vector<double> interval_start(num_intervals);
285 std::vector<double> interval_end(num_intervals);
286 std::vector<int> pos(num_intervals,
287 0); // combination indices for multi-dimensions
288
289 // 4. Process each node to define new buckets.
290 for (const auto &vrp_node : nodes) {
291 std::vector<int> node_split_counts(num_intervals);
292 std::vector<double> node_base_interval(num_intervals);
293 // Determine the number of splits and base interval for each
294 // resource dimension.
295 for (int r = 0; r < num_intervals; ++r) {
296 double full_range = R_max[r] - R_min[r];
297 double node_range = vrp_node.ub[r] - vrp_node.lb[r];
298 int splits = 1;
299 if (std::fabs(full_range) >
300 std::numeric_limits<double>::epsilon()) {
301 double splits_d =
302 (node_range * intervals[r].interval) / full_range;
303 splits = std::max(1, static_cast<int>(std::round(splits_d)));
304 }
305 node_split_counts[r] = splits;
306 node_base_interval[r] =
307 (vrp_node.ub[r] - vrp_node.lb[r]) / static_cast<double>(splits);
308 }
309 // Optionally store node-specific split counts.
310 bucket_splits[vrp_node.id] = node_split_counts;
311
312 int n_buckets = 0;
313 if (num_intervals == 1) {
314 // Single-dimensional case.
315 for (int j = 0; j < node_split_counts[0]; ++j) {
316 auto [int_start, int_end] = calculate_interval(
317 vrp_node.lb[0], vrp_node.ub[0], node_base_interval[0], j,
318 node_split_counts[0], D == Direction::Forward);
319 if constexpr (D == Direction::Backward)
320 int_start = std::max(int_start, vrp_node.lb[0]);
321 else
322 int_end = std::min(int_end, vrp_node.ub[0]);
323 buckets.push_back(Bucket(vrp_node.id,
324 std::vector<double>{int_start},
325 std::vector<double>{int_end}));
326 n_buckets++;
327 cum_sum++;
328 }
329 } else {
330 // Multi-dimensional: iterate over all combinations.
331 fmt::print("Multiple intervals\n");
332 std::fill(pos.begin(), pos.end(), 0);
333 while (true) {
334 for (int r = 0; r < num_intervals; ++r) {
335 auto [int_start, int_end] = calculate_interval(
336 vrp_node.lb[r], vrp_node.ub[r], node_base_interval[r],
337 pos[r], node_split_counts[r], D == Direction::Forward);
338 interval_start[r] = int_start;
339 interval_end[r] = int_end;
340 if constexpr (D == Direction::Backward)
341 interval_start[r] =
342 std::max(interval_start[r], R_min[r]);
343 else
344 interval_end[r] = std::min(interval_end[r], R_max[r]);
345 }
346 buckets.push_back(
347 Bucket(vrp_node.id, interval_start, interval_end));
348 n_buckets++;
349 cum_sum++;
350
351 // Generate next combination.
352 int i = 0;
353 while (i < num_intervals) {
354 ++pos[i];
355 if (pos[i] < node_split_counts[i])
356 break;
357 else {
358 pos[i] = 0;
359 ++i;
360 }
361 }
362 if (i == num_intervals) break;
363 }
364 }
365 num_buckets[vrp_node.id] = n_buckets;
366 num_buckets_index[vrp_node.id] = cum_sum - n_buckets;
367 // Optionally assign node_interval_trees[vrp_node.id] = node_tree;
368 //
369 if constexpr (D == Direction::Forward) {
370 n_segments = (n_buckets + 63) >> 6;
371 }
372 }
373
374 // 5. Process fixed bucket ranges (if available).
375 if (!node_pair_fixed_ranges.empty()) {
376 size_t new_size = buckets.size();
377 size_t required_bitmap_words =
378 std::max(size_t(1), ((new_size * new_size) + 63) / 64);
379 fixed_buckets_bitmap.clear();
380 fixed_buckets_bitmap.resize(required_bitmap_words, 0);
381 for (size_t from = 0; from < new_size; ++from) {
382 const auto &bucket_from = buckets[from];
383 int from_node = bucket_from.node_id;
384 if (node_pair_fixed_ranges.find(from_node) ==
385 node_pair_fixed_ranges.end())
386 continue;
387 for (size_t to = 0; to < new_size; ++to) {
388 const auto &bucket_to = buckets[to];
389 int to_node = bucket_to.node_id;
390 if (node_pair_fixed_ranges[from_node].find(to_node) ==
391 node_pair_fixed_ranges[from_node].end())
392 continue;
393 const auto &range = node_pair_fixed_ranges[from_node][to_node];
394 bool in_range = true;
395 for (int r = 0; r < options.main_resources.size(); ++r) {
396 if (bucket_from.lb[r] < range.first[r] ||
397 bucket_from.ub[r] > range.second[r]) {
398 in_range = false;
399 break;
400 }
401 }
402 if (in_range) {
403 size_t bit_pos = from * new_size + to;
404 size_t word_idx = bit_pos / 64;
405 if (word_idx < fixed_buckets_bitmap.size()) {
406 fixed_buckets_bitmap[word_idx] |=
407 (1ULL << (bit_pos % 64));
408 }
409 }
410 }
411 }
412 }
413
414 // 6. Finalize: record the total bucket count.
415 buckets_size = buckets.size();
416}

425template <Direction D>
426void BucketGraph::generate_arcs() {
427 // Mutex for protecting global bucket updates.
428 std::mutex buckets_mutex;
429
430 // Clear the appropriate graph.
431 if constexpr (D == Direction::Forward) {
432 fw_bucket_graph.clear();
433 } else {
434 bw_bucket_graph.clear();
435 }
436
437 // Resolve references based on the direction.
438 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
439 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
440 auto &num_buckets_idx =
441 assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
442 // Note: we no longer rely solely on bucket_splits here.
443
444 // Pre-compute intervals (base widths) for each node and each resource,
445 // using the same logic as in define_buckets.
446 const int num_resources = options.resources.size();
447 std::vector<std::vector<double>> node_base_intervals(nodes.size());
448 // Also store the computed splits per resource for each node.
449 std::vector<std::vector<int>> node_split_counts(nodes.size());
450
451 for (size_t node_id = 0; node_id < nodes.size(); ++node_id) {
452 const auto &node = nodes[node_id];
453 node_base_intervals[node_id].resize(num_resources);
454 node_split_counts[node_id].resize(num_resources);
455 for (int r = 0; r < num_resources; ++r) {
456 double full_range = R_max[r] - R_min[r];
457 double node_range = node.ub[r] - node.lb[r];
458 int splits = 1;
459 if (std::fabs(full_range) >
460 std::numeric_limits<double>::epsilon()) {
461 double splits_d =
462 (node_range * intervals[r].interval) / full_range;
463 splits = std::max(1, static_cast<int>(std::round(splits_d)));
464 }
465 node_split_counts[node_id][r] = splits;
466 node_base_intervals[node_id][r] =
467 (node.ub[r] - node.lb[r]) / static_cast<double>(splits);
468 }
469 }
470
471 // Clear all buckets and their arcs.
472 for (auto &bucket : buckets) {
473 bucket.clear();
474 bucket.clear_arcs(D == Direction::Forward);
475 }
476
477 // Helper lambda: validate an arc and, if valid, add it.
478 auto try_add_arc =
479 [&](const VRPNode &curr_node, int from_bucket, const VRPNode &next_node,
480 int to_bucket, const std::vector<double> &curr_node_base,
481 const std::vector<double> &next_node_base,
482 const std::vector<double> &res_inc, double cost_inc) -> bool {
483 bool valid = true;
484 if constexpr (D == Direction::Forward) {
485 for (int r = 0; r < res_inc.size() && valid; ++r) {
486 // For forward, check that adding the resource increment
487 // from the from-bucket does not push us past the next
488 // node's upper bound.
489 if (numericutils::gt(buckets[from_bucket].lb[r] + res_inc[r],
490 next_node.ub[r])) {
491 valid = false;
492 } else {
493 double max_calc =
494 std::max(buckets[from_bucket].lb[r] + res_inc[r],
495 next_node.lb[r]);
496 if (numericutils::lt(max_calc, buckets[to_bucket].lb[r]) ||
497 numericutils::gte(max_calc, buckets[to_bucket].lb[r] +
498 next_node_base[r])) {
499 valid = false;
500 }
501 }
502 }
503 } else {
504 for (int r = 0; r < res_inc.size() && valid; ++r) {
505 // For backward, check that subtracting the resource
506 // increment does not drop below next node's lb.
507 if (numericutils::lt(buckets[from_bucket].ub[r] - res_inc[r],
508 next_node.lb[r])) {
509 valid = false;
510 } else {
511 double min_calc =
512 std::min(buckets[from_bucket].ub[r] - res_inc[r],
513 next_node.ub[r]);
514 if (numericutils::gt(min_calc, buckets[to_bucket].ub[r]) ||
515 numericutils::lte(min_calc, buckets[to_bucket].ub[r] -
516 next_node_base[r])) {
517 valid = false;
518 }
519 }
520 }
521 }
522 if (valid) {
523 // Add the arc to the global structure.
524 {
525 std::lock_guard<std::mutex> lock(buckets_mutex);
526 add_arc<D>(from_bucket, to_bucket, res_inc, cost_inc);
527 buckets[from_bucket].template add_bucket_arc<D>(
528 from_bucket, to_bucket, res_inc, cost_inc, false);
529 }
530 }
531 return valid;
532 };
533
534 // Lambda to process a single node.
535 auto process_node = [&](int node_id) {
536 const auto &node = nodes[node_id];
537 std::vector<double> res_inc(num_resources);
538 // Local container for arcs (if further processing is needed).
539 std::vector<std::pair<int, int>> local_arcs;
540 local_arcs.reserve(nodes.size());
541
542 const auto arcs = node.get_arcs<D>();
543 // Use the pre-computed base intervals for the current node.
544 const auto &curr_node_base = node_base_intervals[node.id];
545
546 // For every bucket in the current node.
547 for (int i = 0; i < num_buckets[node.id]; ++i) {
548 int from_bucket = i + num_buckets_idx[node.id];
549 // Process each outgoing arc.
550 for (const auto &arc : nodes) {
551 const auto &next_node = arc;
552 // Avoid self-loop.
553 if (node.id == next_node.id) continue;
554
555 const auto &next_node_base = node_base_intervals[next_node.id];
556 const double travel_cost = getcij(node.id, next_node.id);
557 double cost_inc = travel_cost - next_node.cost;
558 // Pre-calculate resource increments.
559 for (int r = 0; r < num_resources; r++) {
560 res_inc[r] = node.consumption[r];
561 if (options.resources[r] == "time") {
562 res_inc[r] += travel_cost;
563 }
564 }
565
566 // Iterate over buckets in the next node.
567 for (int j = 0; j < num_buckets[next_node.id]; ++j) {
568 int to_bucket = j + num_buckets_idx[next_node.id];
569 // if (from_bucket == to_bucket ||
570 // is_bucket_fixed<D>(from_bucket, to_bucket)) {
571 // continue;
572 // }
573 if (from_bucket == to_bucket) {
574 continue;
575 }
576 // // Validate and add arc if valid.
577 if (try_add_arc(node, from_bucket, next_node, to_bucket,
578 curr_node_base, next_node_base, res_inc,
579 cost_inc)) {
580 local_arcs.emplace_back(from_bucket, to_bucket);
581 }
582 }
583 }
584 }
585 // If needed, local_arcs can be used for debugging or further
586 // processing.
587 };
588
589 // Create a list of tasks (each task is a node index).
590 std::vector<int> tasks(nodes.size());
591 std::iota(tasks.begin(), tasks.end(), 0);
592
593 // Choose a chunk size based on a heuristic.
594 constexpr int chunk_size = 10;
595
596 // Set up the thread pool.
597 const unsigned int thread_count =
598 std::max(1u, std::thread::hardware_concurrency() / 2);
599 exec::static_thread_pool pool(thread_count);
600 auto sched = pool.get_scheduler();
601
602 // Launch tasks in parallel over chunks.
603 auto bulk_sender = stdexec::bulk(
604 stdexec::just(), (tasks.size() + chunk_size - 1) / chunk_size,
605 [&, chunk_size](std::size_t chunk_idx) {
606 const size_t start_idx = chunk_idx * chunk_size;
607 const size_t end_idx =
608 std::min(start_idx + chunk_size, tasks.size());
609 for (size_t idx = start_idx; idx < end_idx; ++idx) {
610 process_node(tasks[idx]);
611 }
612 });
613
614 auto work = stdexec::starts_on(sched, bulk_sender);
615 stdexec::sync_wait(std::move(work));
616}

629template <Direction D>
630Label *BucketGraph::get_best_label(const std::vector<int> &topological_order,
631 const std::vector<double> &c_bar,
632 const std::vector<std::vector<int>> &sccs) {
633 double best_cost = std::numeric_limits<double>::infinity();
634 Label *best_label = nullptr;
635 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
636
637 // Iterate over all buckets, and select the label with the lowest cost.
638 for (auto &bucket : buckets) {
639 Label *label = bucket.get_best_label();

642 best_cost = label->cost;
643 best_label = label;
644 }
645 }
646 return best_label;

654template <Stage S, Symmetry SYM>
655void BucketGraph::ConcatenateLabel(const Label *L, int &b,
656 std::atomic<double> &best_cost) {
657 // Determine the number of segments needed for the bucket visited
658 // bitmap.
659 const size_t n_segments = (fw_buckets_size + 63) / 64;
660 std::vector<uint64_t> Bvisited(n_segments, 0);
661
662 // Use thread_local storage for the bucket stack to avoid reallocations.
663 static thread_local std::vector<int> bucket_stack;
664 bucket_stack.reserve(R_SIZE); // Reserve once at initialization
665 bucket_stack.clear(); // Clear for reuse
666 bucket_stack.push_back(b);
667
668 // Choose symmetric buckets and cost-bar arrays based on symmetry.
669 auto &other_buckets = assign_symmetry<SYM>(fw_buckets, bw_buckets);
670 auto &other_c_bar = assign_symmetry<SYM>(fw_c_bar, bw_c_bar);
671
672 // Cache frequently accessed values.
673 const int L_node_id = L->node_id;
674 const auto &L_resources = L->resources;
675 const auto &L_last_node = nodes[L_node_id];
676 const double L_cost = L->cost;
677 const size_t bitmap_size = L->visited_bitmap.size();
678
679 // Pre-compute if branching is active.
680 const bool has_branching = !branching_duals->empty();
681
682#if defined(SRC)
683 // SRC mode setup: Only relevant in Stage::Four.
684 decltype(cut_storage) cutter = nullptr;
685 if constexpr (S > Stage::Three) {
686 cutter = cut_storage;
687 }
688 const auto active_cuts = cutter->getActiveCuts();
689#endif

691 // Lambda to check if the visited bitmaps of L and L_bw overlap.
692 auto overlapsVisited = [bitmap_size, this](const Label *L,
693 const Label *L_bw) -> bool {
694 // Make a local copy of the visited bitmap.
695 auto visited_bitmap = L->visited_bitmap;
696
697 // check if visited_bitmap overlaps
698 for (size_t i = 0; i < bitmap_size; ++i) {
699 if (visited_bitmap[i] & L_bw->visited_bitmap[i]) {
700 return true;
701 }
702 }
703 // Iterate in reverse over nodes_covered without allocating a new
704 // vector.
705 // for (auto it = L_bw->nodes_covered.rbegin();
706 // it != L_bw->nodes_covered.rend(); ++it) {
707 // uint16_t node_id = *it;
708 // if (node_id == L->node_id ||
709 // is_node_visited(visited_bitmap, node_id))
710 // return true;
711 // // Update the bitmap for each 64-bit word.
712 // for (size_t i = 0; i < visited_bitmap.size(); ++i) {
713 // uint64_t &current_visited = visited_bitmap[i];
714 // if (current_visited != 0)
715 // current_visited &= neighborhoods_bitmap[node_id][i];
716 // }
717 // set_node_visited(visited_bitmap, node_id);
718 // }
719 return false;
720 };

722// Lambda to apply SRC adjustments to total_cost (only for Stage::Four).
723#if defined(SRC)
724 auto applySRCAdjustments = [&](double total_cost, const Label *L,
725 const Label *L_bw) -> double {
726 if constexpr (S == Stage::Four) {
727 for (const auto &active_cut : active_cuts) {
728 const auto &cut = *active_cut.cut_ptr;
729 const size_t idx = active_cut.index;
730 const double dual = active_cut.dual_value;
731 if (L->SRCmap[idx] + L_bw->SRCmap[idx] >= cut.p.den) {
732 total_cost -= dual;
733 }
734 }
735 }
736 return total_cost;
737 };
738#endif
739
740 // Process buckets until the stack is empty.
741 while (!bucket_stack.empty()) {
742 const int current_bucket = bucket_stack.back();
743 bucket_stack.pop_back();
744
745 // Mark current_bucket as visited in the bitmap.
746 const size_t segment = current_bucket >> 6;
747 const uint64_t bit_mask = bit_mask_lookup[current_bucket & 63];
748 Bvisited[segment] |= bit_mask;
749
750 // Get the node id associated with this bucket.
751 const int bucketLprimenode = other_buckets[current_bucket].node_id;
752 double travel_cost = getcij(L_node_id, bucketLprimenode);
753
754#if defined(RCC) || defined(EXACT_RCC)
755 // Adjust travel cost with arc duals in Stage Four.
756 if constexpr (S == Stage::Four) {
757 travel_cost -= arc_duals.getDual(L_node_id, bucketLprimenode);
758 }
759#endif
760
761 if (has_branching) {
762 travel_cost -=
763 branching_duals->getDual(L_node_id, bucketLprimenode);
764 }
765
766 const double path_cost = L_cost + travel_cost;
767 const double bound = other_c_bar[current_bucket];
768
769 // Early bound check.
770 if ((S != Stage::Enumerate &&
771 numericutils::gte(path_cost + bound, best_cost.load())) ||
772 (S == Stage::Enumerate &&
773 numericutils::gte(path_cost + bound, gap))) {
774 continue;
775 }
776
777 // Retrieve labels in the current bucket.
778 auto &bucket = other_buckets[current_bucket];
779 auto labels = bucket.get_labels();
780 if (labels.empty()) continue;
781
782 // Process each label in the current bucket.
783 for (const Label *L_bw : labels) {
784 // Skip invalid labels with combined check
785 if (L_bw == nullptr || L_bw->is_dominated ||
786 L_bw->node_id == L_node_id) {
787 continue;
788 }
789
790 // Check feasibility
791 if (!check_feasibility(L, L_bw)) {
792 continue;
793 }
794
795 double total_cost = path_cost + L_bw->cost;
796
797#if defined(SRC)
798 // Apply SRC adjustments if in Stage Four.
799 total_cost = applySRCAdjustments(total_cost, L, L_bw);
800#endif
801 double latest_best = best_cost.load();
802
803 // Filter based on total cost against best known cost or gap.
804 bool cost_acceptable;
805 if constexpr (S != Stage::Enumerate) {
806 cost_acceptable = numericutils::lt(total_cost, latest_best);
807 } else {
808 cost_acceptable = numericutils::lt(total_cost, gap);
809 }
810
811 if (!cost_acceptable) {
812 continue;
813 }
814 // Use critical section only when necessary
815 {
816 std::lock_guard<std::mutex> lock(merge_mutex);
817
818 // Use atomic compare-exchange loop with exponential backoff
819 double expected = best_cost.load();
820 int backoff_counter = 0;
821
822 while (numericutils::lt(total_cost, expected) &&
823 !best_cost.compare_exchange_weak(expected, total_cost)) {
824 // Simple exponential backoff to reduce contention
825 if (++backoff_counter > 3) {
826 for (int i = 0; i < (1 << (backoff_counter - 3)); ++i) {
827 _mm_pause(); // CPU hint to reduce power in spin
828 // loops
829 }
830 }
831 }
832
833 // Add label if still an improvement
834 if (numericutils::lte(total_cost, best_cost.load())) {
835 auto pbest = compute_label<S>(L, L_bw, total_cost);
836 merged_labels.push_back(pbest);
837 }
838 }
839 }
840 // Add neighbor buckets to the stack if they have not been visited.
841 for (int b_prime : Phi_bw[current_bucket]) {
842 const size_t seg_prime = b_prime >> 6;
843 const uint64_t mask_prime = bit_mask_lookup[b_prime & 63];
844 if (!(Bvisited[seg_prime] & mask_prime)) {
845 bucket_stack.push_back(b_prime);
846 }
847 }
848 }

863template <Direction D>
864void BucketGraph::SCC_handler() {
865 // Get necessary references.
866 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
867 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
868 auto &bucket_graph = assign_buckets<D>(fw_bucket_graph, bw_bucket_graph);
869
870 // Build an extended bucket graph from bucket_graph using Phi.
871 ankerl::unordered_dense::map<int, std::vector<int>> extended_bucket_graph =
872 bucket_graph;
873 for (size_t i = 0; i < extended_bucket_graph.size(); ++i) {
874 const auto &phi_set = Phi[i];
875 if (phi_set.empty()) continue;
876 for (auto phi_bucket : phi_set) {
877 extended_bucket_graph[phi_bucket].push_back(i);
878 }
879 }
880
881 // Compute SCCs using Tarjan's algorithm.
882 SCC scc_finder;
883 scc_finder.convertFromUnorderedMap(extended_bucket_graph);
884 auto sccs = scc_finder.tarjanSCC();
885 auto topological_order = scc_finder.topologicalOrderOfSCCs(sccs);
886
887#ifdef VERBOSE
888 // Optional: Print SCCs.
889 constexpr auto blue = "\033[34m";
890 constexpr auto reset = "\033[0m";
891 fmt::print((D == Direction::Forward) ? "FW SCCs:\n" : "BW SCCs:\n");
892 for (auto scc : topological_order) {
893 fmt::print("{}({}) -> {}", blue, scc, reset);
894 for (auto bucket : sccs[scc]) {
895 fmt::print("{} ", bucket);
896 }
897 fmt::print("\n");
898 }
899 fmt::print("\n");
900#endif
901
902 // Order SCCs according to the topological order.
903 std::vector<std::vector<int>> ordered_sccs;
904 ordered_sccs.reserve(sccs.size());
905 for (int scc : topological_order) {
906 ordered_sccs.push_back(sccs[scc]);
907 }
908
909 // Sort each SCC: For Forward, sort by bucket lower bound; for Backward,
910 // sort by bucket upper bound.
911 auto sorted_sccs = sccs; // Make a copy
912 for (auto &scc : sorted_sccs) {
913 if constexpr (D == Direction::Forward) {
914 pdqsort(scc.begin(), scc.end(), [&buckets](int a, int b) {
915 return buckets[a].lb[0] < buckets[b].lb[0];
916 });
917 } else {
918 pdqsort(scc.begin(), scc.end(), [&buckets](int a, int b) {
919 return buckets[a].ub[0] > buckets[b].ub[0];
920 });
921 }
922 }
923
924 // Ensure each VRP node has the appropriate SCC arc container allocated.
925 for (auto &node : nodes) {
926 if constexpr (D == Direction::Forward) {
927 node.fw_arcs_scc.resize(sccs.size());
928 } else {
929 node.bw_arcs_scc.resize(sccs.size());
930 }
931 }
932
933 // Lambda: Filter and collect arcs for a given bucket, node, and
934 // direction.
935 auto process_bucket_arcs = [&](int bucket, int scc_index) {
936 int from_node_id = buckets[bucket].node_id;
937 VRPNode &node = nodes[from_node_id];
938 // Select the proper arc container and node arcs based on the
939 // direction.
940 const auto &node_arcs =
941 (D == Direction::Forward) ? node.fw_arcs : node.bw_arcs;
942 std::vector<Arc> &filtered_arcs = (D == Direction::Forward)
943 ? node.fw_arcs_scc[scc_index]
944 : node.bw_arcs_scc[scc_index];
945
946 const auto &bucket_arcs = buckets[bucket].template get_bucket_arcs<D>();
947 for (const auto &arc : bucket_arcs) {
948 int to_node_id = buckets[arc.to_bucket].node_id;
949 // Look for an arc with destination equal to to_node_id.
950 auto it = std::find_if(
951 node_arcs.begin(), node_arcs.end(),
952 [&to_node_id](const Arc &a) { return a.to == to_node_id; });
953 if (it != node_arcs.end()) {
954 filtered_arcs.push_back(*it);
955 }
956 }
957 };
958
959 // For each SCC, process arcs for each bucket in the SCC.
960 int scc_ctr = 0;
961 for (const auto &scc : sccs) {
962 for (int bucket : scc) {
963 process_bucket_arcs(bucket, scc_ctr);
964 }
965 ++scc_ctr;
966 }
967
968 // Lambda: Sort and deduplicate arcs within each node's SCC arc
969 // container.
970 auto sortAndDedup = [](std::vector<Arc> &arcs) {
971 pdqsort(arcs.begin(), arcs.end(), [](const Arc &a, const Arc &b) {
972 return std::tie(a.from, a.to) < std::tie(b.from, b.to);
973 });
974 auto last = std::unique(arcs.begin(), arcs.end(),
975 [](const Arc &a, const Arc &b) {
976 return a.from == b.from && a.to == b.to;
977 });
978 arcs.erase(last, arcs.end());
979 };
980
981 // For each node, sort and deduplicate all arc lists.
982 for (auto &node : nodes) {
983 if constexpr (D == Direction::Forward) {
984 for (auto &arcs : node.fw_arcs_scc) {
985 sortAndDedup(arcs);
986 }
987 } else {
988 for (auto &arcs : node.bw_arcs_scc) {
989 sortAndDedup(arcs);
990 }
991 }
992 }
993
994 // Build union-find structure for SCCs using the ordered SCCs.
995 UnionFind uf(ordered_sccs);
996 if constexpr (D == Direction::Forward) {
997 fw_ordered_sccs = ordered_sccs;
998 fw_topological_order = topological_order;
999 fw_sccs = sccs;
1000 fw_sccs_sorted = sorted_sccs;
1001 fw_union_find = uf;
1002 } else {
1003 bw_ordered_sccs = ordered_sccs;
1004 bw_topological_order = topological_order;
1005 bw_sccs = sccs;
1006 bw_sccs_sorted = sorted_sccs;
1007 bw_union_find = uf;
1008 }
1009}

1142template <Stage S>
1143void BucketGraph::heuristic_fixing() {
1144 if (fixed) {
1145 return;
1146 }
1147
1148 // Reset and initialize state.
1149 reset_pool();
1150 reset_fixed();
1151 common_initialization();
1152
1153 // Pre-allocate c-bar vectors with infinity.
1154 std::vector<double> forward_cbar(fw_buckets.size(),
1155 std::numeric_limits<double>::infinity());
1156 std::vector<double> backward_cbar(bw_buckets.size(),
1157 std::numeric_limits<double>::infinity());
1158
1159 gap = std::ceil(incumbent - (relaxation + std::min(0.0, min_red_cost)));
1160
1161 // Run the appropriate labeling algorithms based on stage
1162 if (S == Stage::Three) {
1163 // Run the labeling algorithms for Stage::Two with Full::Partial mode.
1164 run_labeling_algorithms<Stage::Two, Full::Partial>(forward_cbar,
1165 backward_cbar);
1166 } else if (S == Stage::Four) {
1167 fmt::print("\033[34m_STARTING ARC FIXING PROCEDURE\033[0m\n");
1168 fmt::print("gap: {}\n", gap);
1169 // Run the labeling algorithms for Stage::Four with Full::Full mode.
1170 run_labeling_algorithms<Stage::Four, Full::Full>(forward_cbar,
1171 backward_cbar);
1172 }
1173
1174 // Prepare containers for grouping labels by node.
1175 const size_t num_nodes = nodes.size();
1176
1177 // Preallocate vectors with nullptr to avoid resizing
1178 std::vector<const Label *> min_fw_labels(num_nodes, nullptr);
1179 std::vector<const Label *> min_bw_labels(num_nodes, nullptr);
1180
1181 // Find minimum cost label for each node directly from buckets
1182 // without creating intermediate label maps
1183 for (auto &bucket : fw_buckets) {
1184 for (const Label *label : bucket.get_labels()) {
1185 const size_t node_id = label->node_id;
1186 if (!min_fw_labels[node_id] ||
1187 label->cost < min_fw_labels[node_id]->cost) {
1188 min_fw_labels[node_id] = label;
1189 }
1190 }
1191 }
1192
1193 for (auto &bucket : bw_buckets) {
1194 for (const Label *label : bucket.get_labels()) {
1195 const size_t node_id = label->node_id;
1196 if (!min_bw_labels[node_id] ||
1197 label->cost < min_bw_labels[node_id]->cost) {
1198 min_bw_labels[node_id] = label;
1199 }
1200 }
1201 }
1202
1203 // Pre-compute the index of the "time" resource.
1204 const size_t time_index = std::distance(
1205 options.resources.begin(),
1206 std::find(options.resources.begin(), options.resources.end(), "time"));
1207
1208 // Precompute lookup tables for common values
1209 std::vector<double> fw_time_resources(num_nodes, 0.0);
1210 std::vector<double> bw_time_resources(num_nodes, 0.0);
1211 std::vector<double> fw_costs(num_nodes, 0.0);
1212 std::vector<double> bw_costs(num_nodes, 0.0);
1213
1214 for (size_t i = 0; i < num_nodes; ++i) {
1215 if (min_fw_labels[i]) {
1216 fw_time_resources[i] = min_fw_labels[i]->resources[time_index];
1217 fw_costs[i] = min_fw_labels[i]->cost;
1218 }
1219 if (min_bw_labels[i]) {
1220 bw_time_resources[i] = min_bw_labels[i]->resources[time_index];
1221 bw_costs[i] = min_bw_labels[i]->cost;
1222 }
1223 }
1224
1225 // Cache node data to avoid repeated lookups
1226 std::vector<double> node_durations(num_nodes);
1227 std::vector<std::vector<double>> node_consumptions(
1228 num_nodes, std::vector<double>(options.resources.size()));
1229
1230 for (size_t i = 0; i < num_nodes; ++i) {
1231 const VRPNode &node = nodes[i];
1232 node_durations[i] = node.duration;
1233 for (size_t r = 0; r < options.resources.size(); ++r) {
1234 node_consumptions[i][r] = node.consumption[r];
1235 }
1236 }
1237
1238 size_t num_fixes = 0;
1239 const double gap_threshold = gap; // Cache this value
1240
1241 // Build a list of candidate node pairs to check
1242 struct NodePair {
1243 size_t i;
1244 size_t j;
1245 };
1246
1247 std::vector<NodePair> candidate_pairs;
1248 candidate_pairs.reserve(num_nodes * num_nodes /
1249 4); // Estimate: 25% of pairs will be candidates
1250
1251 for (size_t i = 0; i < num_nodes; ++i) {
1252 if (!min_fw_labels[i]) continue;
1253
1254 for (size_t j = 0; j < num_nodes; ++j) {
1255 if (i == j || !min_bw_labels[j]) continue;
1256 candidate_pairs.push_back({i, j});
1257 }
1258 }
1259
1260 // Process candidate pairs in batches to improve cache locality
1261 const size_t BATCH_SIZE = 256;
1262 std::vector<std::pair<size_t, size_t>> arcs_to_fix;
1263 arcs_to_fix.reserve(BATCH_SIZE);
1264
1265 for (size_t start = 0; start < candidate_pairs.size();
1266 start += BATCH_SIZE) {
1267 const size_t end = std::min(start + BATCH_SIZE, candidate_pairs.size());
1268 arcs_to_fix.clear();
1269
1270 for (size_t idx = start; idx < end; ++idx) {
1271 const auto &pair = candidate_pairs[idx];
1272 const size_t i = pair.i;
1273 const size_t j = pair.j;
1274
1275 const double cost =
1276 getcij(min_fw_labels[i]->node_id, min_bw_labels[j]->node_id);
1277
1278 // Early rejection using precomputed time resources (often most
1279 // restrictive)
1280 if (options.resources[time_index] == "time") {
1281 if (fw_time_resources[i] + cost +
1282 node_durations[min_fw_labels[i]->node_id] >
1283 bw_time_resources[j]) {
1284 continue;
1285 }
1286 }
1287
1288 // Check all other resources
1289 bool violated = false;
1290 const size_t fw_node_id = min_fw_labels[i]->node_id;
1291
1292 for (size_t r = 0; r < options.resources.size(); ++r) {
1293 if (r != time_index) {
1294 if (min_fw_labels[i]->resources[r] +
1295 node_consumptions[fw_node_id][r] >
1296 min_bw_labels[j]->resources[r]) {
1297 violated = true;
1298 break;
1299 }
1300 }
1301 }
1302
1303 // If all resource constraints are satisfied and the cost exceeds
1304 // the gap, fix the arc
1305 if (!violated &&
1306 (fw_costs[i] + cost + bw_costs[j] > gap_threshold)) {
1307 arcs_to_fix.emplace_back(i, j);
1308 ++num_fixes;
1309 }
1310 }
1311
1312 // Fix arcs in batch
1313 for (const auto &[i, j] : arcs_to_fix) {
1314 fix_arc(i, j);
1315 }
1316 }
1317
1318 if (S == Stage::Four) {
1319 fmt::print("Applied {} arc fixes\n", num_fixes);
1320 fmt::print("\033[34m_ARC FIXING PROCEDURE FINISHED\033[0m\n");
1321 }

1324template <Symmetry SYM>
1325void BucketGraph::set_adjacency_list() {
1326 // === Step 1: Clear Existing Arcs from Each Node ===
1327 for (auto &node : nodes) {
1328 node.clear_arcs();
1329 }
1330
1331 // === Step 2: MST-Based Clustering ===
1332 MST mst_solver(nodes,
1333 [this](int from, int to) { return getcij(from, to); });
1334 auto mst = mst_solver.compute_mst();
1335
1336 // Collect and sort MST edge weights.
1337 std::vector<double> edge_weights;
1338 edge_weights.reserve(mst.size());
1339 for (const auto &[weight, from, to] : mst) {
1340 edge_weights.push_back(weight);
1341 }
1342 pdqsort(edge_weights.begin(), edge_weights.end());
1343
1344 // Compute theta as the 90th percentile (scaled down by 100).
1345 size_t p90_index = static_cast<size_t>(0.9 * edge_weights.size());
1346 double theta = (edge_weights[p90_index]) / 100.0;
1347 print_info("Computed theta: {}\n", theta);
1348
1349 // Cluster nodes and build mapping: job_id -> cluster_id.
1350 auto clusters = mst_solver.cluster(theta);
1351 print_info("Number of clusters: {}\n", clusters.size());
1352 std::vector<int> job_to_cluster(nodes.size(), -1);
1353 for (int cluster_id = 0; cluster_id < clusters.size(); ++cluster_id) {
1354 for (int job : clusters[cluster_id]) {
1355 job_to_cluster[job] = cluster_id;
1356 }
1357 }
1358
1359 // === Step 3: Precompute Resource Information ===
1360 // Find the index for the "time" resource.
1361 auto time_it =
1362 std::find(options.resources.begin(), options.resources.end(), "time");
1363 const int time_resource_idx =
1364 (time_it != options.resources.end())
1365 ? std::distance(options.resources.begin(), time_it)
1366 : -1;
1367
1368 // === Step 4: Process Arcs for Each Node ===
1369
1370 // Helper lambda to compute resource increments.
1371 // Depending on symmetry, consumption is either taken as-is or averaged.
1372 auto compute_resource_increments = [&](const VRPNode &from_node,
1373 const VRPNode &to_node,
1374 double travel_cost,
1375 std::vector<double> &inc) {
1376 const int nres = options.resources.size();
1377 if constexpr (SYM == Symmetry::Asymmetric) {
1378 for (int r = 0; r < nres; ++r) {
1379 inc[r] = (r == time_resource_idx)
1380 ? travel_cost + from_node.duration
1381 : from_node.consumption[r];
1382 }
1383 } else {
1384 for (int r = 0; r < nres; ++r) {
1385 inc[r] =
1386 (r == time_resource_idx)
1387 ? (from_node.duration / 2.0 + travel_cost +
1388 to_node.duration / 2.0)
1389 : ((from_node.consumption[r] + to_node.consumption[r]) /
1390 2.0);
1391 }
1392 }
1393 };
1394
1395 // Lambda to add forward and reverse arcs from a given node.
1396 auto add_arcs_for_node = [&](const VRPNode &curr_node) {
1397 // Containers to accumulate arcs before insertion.
1398 using ArcTuple = std::tuple<double, int, std::vector<double>, double>;
1399 std::vector<ArcTuple> forward_arcs;
1400 std::vector<ArcTuple> reverse_arcs;
1401 forward_arcs.reserve(nodes.size());
1402 reverse_arcs.reserve(nodes.size());
1403
1404 // Iterate over candidate next nodes.
1405 for (const auto &next_node : nodes) {
1406 // Skip if the candidate is a depot, the same node, or a pstep
1407 // depot.
1408 if (next_node.id == options.depot || curr_node.id == next_node.id ||
1409 (options.pstep && (next_node.id == options.pstep_depot ||
1410 next_node.id == options.pstep_end_depot))) {
1411 continue;
1412 }
1413
1414 double travel_cost = getcij(curr_node.id, next_node.id);
1415 double cost_inc = travel_cost - next_node.cost;
1416
1417 // Skip self-loops (or if bucket indexes match; here using node
1418 // id as bucket id).
1419 if (curr_node.id == next_node.id) continue;
1420
1421 // Compute resource increments.
1422 std::vector<double> res_inc(options.resources.size());
1423 compute_resource_increments(curr_node, next_node, travel_cost,
1424 res_inc);
1425
1426 // Check feasibility: ensure resource consumption does not
1427 // exceed next node's upper bound.
1428 bool feasible = true;
1429 for (int r = 0; r < options.resources.size(); ++r) {
1430 if (numericutils::gt(curr_node.lb[r] + res_inc[r],
1431 next_node.ub[r])) {
1432 feasible = false;
1433 break;
1434 }
1435 }
1436 if (!feasible) continue;
1437
1438 // Compute priorities based on cluster membership.
1439 bool same_cluster =
1440 (job_to_cluster[curr_node.id] == job_to_cluster[next_node.id]);
1441 double forward_priority =
1442 (same_cluster ? 5.0 : 1.0) + 1.E-5 * next_node.start_time;
1443 double reverse_priority =
1444 (same_cluster ? 1.0 : 5.0) + 1.E-5 * curr_node.start_time;
1445
1446 // Collect arcs for forward and reverse directions.
1447 forward_arcs.emplace_back(forward_priority, next_node.id, res_inc,
1448 cost_inc);
1449 reverse_arcs.emplace_back(reverse_priority, next_node.id, res_inc,
1450 cost_inc);
1451 }
1452
1453 // Insert the arcs into the corresponding nodes.
1454 for (const auto &[priority, to_node, res_inc, cost_inc] :
1455 forward_arcs) {
1456 nodes[curr_node.id].template add_arc<Direction::Forward>(
1457 curr_node.id, to_node, res_inc, cost_inc, priority);
1458 }
1459 for (const auto &[priority, from_node, res_inc, cost_inc] :
1460 reverse_arcs) {
1461 nodes[from_node].template add_arc<Direction::Backward>(
1462 from_node, curr_node.id, res_inc, cost_inc, priority);
1463 }
1464 };
1465
1466 // Process each node, skipping the end depot.
1467 for (const auto &node : nodes) {
1468 if (node.id != options.end_depot) {
1469 add_arcs_for_node(node);
1470 }
1471 }
1472}

1602 auto depot = label_pool->acquire();

1605 // For forward direction, start at the lower bound and
1606 // add multiples of the base interval; adjust similarly
1607 // for backward.
1608 interval_bounds[r] =
1609 is_forward ? interval_starts[r] +
1610 current_pos[r] * base_intervals[r]
1611 : interval_ends[r] -
1612 current_pos[r] * base_intervals[r];

1615 depot_id);
1616 depot->is_extended = false;
1617 depot->nodes_covered.push_back(depot_id);
1618 set_node_visited(depot->visited_bitmap, depot_id);

1622 buckets[calculated_index].node_id = depot_id;
1623
1624 offset++;
1625 return;
1626 }
1627
1628 // For the current resource dimension, loop using its specific
1629 // split count.
1630 for (int k = 0; k < depot_splits[depth]; ++k) {
1631 current_pos[depth] = k;
1632 process_intervals(depth + 1);
1633 }
1634 };
1635 process_intervals(0);
1636 };
1637
1638 // Generate combinations for the current direction.
1639 generate_combinations(D == Direction::Forward, offset);

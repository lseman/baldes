81template <Direction D>
82void BucketGraph::define_buckets() {
83 int num_intervals = options.main_resources.size();
84 std::vector<double> total_ranges(num_intervals);
85 std::vector<double> base_intervals(num_intervals);
86
87 // Ensure the base_intervals storage is resized for all nodes
88 if constexpr (D == Direction::Forward) {
89 fw_base_intervals.resize(num_intervals);
90 } else {
91 bw_base_intervals.resize(num_intervals);
92 }
93
94 // Determine the base interval and other relevant values for each resource
95 for (int r = 0; r < num_intervals; ++r) {
96 total_ranges[r] = R_max[r] - R_min[r];
97 base_intervals[r] = total_ranges[r] / intervals[r].interval;
98 }
99
100 if constexpr (D == Direction::Forward) {
101 fw_base_intervals = base_intervals;
102 } else {
103 bw_base_intervals = base_intervals;
104 }
105
106 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
107 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
108 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
109 auto &node_interval_trees = assign_buckets<D>(fw_node_interval_trees, bw_node_interval_trees);
110 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
111 num_buckets.resize(nodes.size());
112 num_buckets_index.resize(nodes.size());
113
114 int cum_sum = 0; // Tracks global bucket index
115 int bucket_index = 0;
116
117 node_interval_trees.assign(nodes.size(), SplayTree());
118
119 // Loop through each node to define its specific buckets
120 for (const auto &VRPNode : nodes) {
121 std::vector<double> node_base_interval(num_intervals);
122 for (int r = 0; r < num_intervals; ++r) {
123 node_base_interval[r] = (VRPNode.ub[r] - VRPNode.lb[r]) / intervals[r].interval;
124 }
125 SplayTree node_tree;
126
127 // Multiple buckets case
128 int n_buckets = 0;
129 std::vector<int> current_pos(num_intervals, 0);
130 std::vector<double> interval_start(num_intervals), interval_end(num_intervals);
131
132 // Calculate the start and end for each interval dimension
133 // Check if there is only one interval and proceed with the single interval logic
134 if (num_intervals == 1) {
135 for (auto j = 0; j < intervals[0].interval; ++j) {
136 // Perform the same interval logic for a single interval
137 if constexpr (D == Direction::Forward) {
138 interval_start[0] = VRPNode.lb[0] + current_pos[0] * node_base_interval[0];
139
140 if (j == intervals[0].interval - 1) {
141 interval_end[0] = VRPNode.ub[0];
142 } else {
143 interval_end[0] = VRPNode.lb[0] + (current_pos[0] + 1) * node_base_interval[0];
144 }
145 } else {
146 if (j == intervals[0].interval - 1) {
147 interval_start[0] = VRPNode.lb[0];
148 } else {
149 interval_start[0] = VRPNode.ub[0] - (current_pos[0] + 1) * node_base_interval[0];
150 }
151 interval_end[0] = VRPNode.ub[0] - current_pos[0] * node_base_interval[0];
152 }
153
154 interval_start[0] = roundToTwoDecimalPlaces(interval_start[0]);
155 interval_end[0] = roundToTwoDecimalPlaces(interval_end[0]);
156
157 if constexpr (D == Direction::Backward) {
158 interval_start[0] = std::max(interval_start[0], VRPNode.lb[0]);
159 } else {
160 interval_end[0] = std::min(interval_end[0], VRPNode.ub[0]);
161 }
162 buckets.push_back(Bucket(VRPNode.id, interval_start, interval_end));
163 node_tree.insert(interval_start, interval_end, bucket_index);
164
165 bucket_index++;
166 n_buckets++;
167 cum_sum++;
168 current_pos[0]++;
169 }
170 } else {
171 // Multiple intervals case, nested loop logic to generate all combinations
172 std::vector<int> pos(num_intervals, 0);
173
174 while (true) {
175 for (int r = 0; r < num_intervals; ++r) {
176 if constexpr (D == Direction::Forward) {
177 interval_start[r] = VRPNode.lb[r] + pos[r] * node_base_interval[r];
178
179 if (pos[r] == intervals[r].interval - 1) {
180 interval_end[r] = VRPNode.ub[r];
181 } else {
182 interval_end[r] = VRPNode.lb[r] + (pos[r] + 1) * node_base_interval[r];
183 }
184 } else {
185 if (pos[r] == intervals[r].interval - 1) {
186 interval_start[r] = VRPNode.lb[r];
187 } else {
188 interval_start[r] = VRPNode.ub[r] - (pos[r] + 1) * node_base_interval[r];
189 }
190 interval_end[r] = VRPNode.ub[r] - pos[r] * node_base_interval[r];
191 }
192
193 interval_start[r] = roundToTwoDecimalPlaces(interval_start[r]);
194 interval_end[r] = roundToTwoDecimalPlaces(interval_end[r]);
195
196 if constexpr (D == Direction::Backward) {
197 interval_start[r] = std::max(interval_start[r], R_min[r]);
198 } else {
199 interval_end[r] = std::min(interval_end[r], R_max[r]);
200 }
201 }
202
203 /*
204 // Print and store
205 fmt::print("Creating bucket with interval: [");
206 for (int i = 0; i < num_intervals; ++i) {
207 fmt::print("({}, {})", interval_start[i], interval_end[i]);
208 if (i < num_intervals - 1) fmt::print(", ");
209 }
210 fmt::print("]\n");
211 */
212
213 buckets.push_back(Bucket(VRPNode.id, interval_start, interval_end));
214 node_tree.insert(interval_start, interval_end, bucket_index);
215
216 bucket_index++;
217 n_buckets++;
218 cum_sum++;
219
220 // Increment the positions for combinations
221 int i = 0;
222 while (i < num_intervals && ++pos[i] >= intervals[i].interval) {
223 pos[i] = 0;
224 i++;
225 }
226 if (i == num_intervals) break; // All combinations generated
227 }
228 }
229
230 // Update node-specific bucket data
231 num_buckets[VRPNode.id] = n_buckets;
232 num_buckets_index[VRPNode.id] = cum_sum - n_buckets;
233 node_interval_trees[VRPNode.id] = node_tree;
234 }
235
236 // Update global bucket sizes based on direction
237 buckets_size = cum_sum;
238}

246template <Direction D>
247void BucketGraph::generate_arcs() {
248 // Mutex to synchronize access to shared resources
249 auto buckets_mutex = std::mutex();
250
251 // Clear the appropriate bucket graph (forward or backward)
252 if constexpr (D == Direction::Forward) {
253 fw_bucket_graph.clear(); // Clear the forward bucket graph
254 } else {
255 bw_bucket_graph.clear(); // Clear the backward bucket graph
256 }
257
258 // Assign the forward or backward fixed buckets and other bucket-related structures
259 auto &fixed_buckets = assign_buckets<D>(fw_fixed_buckets, bw_fixed_buckets);
260 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
261 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
262 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
263
264 // Clear all buckets in parallel, removing any existing arcs
265 std::for_each(buckets.begin(), buckets.end(), [&](auto &bucket) {
266 bucket.clear(); // Clear bucket data
267 bucket.clear_arcs(D == Direction::Forward); // Clear arcs in the bucket
268 });
269
270 auto add_arcs_for_node = [&](const VRPNode &node, int from_bucket, std::vector<double> &res_inc,
271 std::vector<std::pair<int, int>> &local_arcs) {
272 // Retrieve the arcs for the node in the given direction (Forward/Backward)
273 auto arcs = node.get_arcs<D>();
274 // Compute base intervals for each resource dimension based on R_max and R_min
275 std::vector<double> node_intervals(options.resources.size());
276 for (int r = 0; r < options.resources.size(); ++r) {
277 node_intervals[r] = (node.ub[r] - node.lb[r]) / intervals[r].interval;
278 }
279
280 // Iterate over all arcs of the node
281 for (const auto &arc : arcs) {
282 const auto &next_node = nodes[arc.to]; // Get the destination node of the arc
283
284 std::vector<double> next_node_intervals(options.resources.size());
285 for (int r = 0; r < options.resources.size(); ++r) {
286 next_node_intervals[r] = (next_node.ub[r] - next_node.lb[r]) / intervals[r].interval;
287 }
288 // Skip self-loops (no arc from a node to itself)
289 if (node.id == next_node.id) continue;
290
291 // Calculate travel cost and cost increment based on node's properties
292 const auto travel_cost = getcij(node.id, next_node.id);
293 double cost_inc = travel_cost - next_node.cost;
294 for (auto r = 0; r < options.resources.size(); r++) {
295 res_inc[r] = node.consumption[r];
296 if (options.resources[r] == "time") { res_inc[r] += travel_cost; }
297 }
298
299 for (int j = 0; j < num_buckets[next_node.id]; ++j) {
300 int to_bucket = j + num_buckets_index[next_node.id];
301 if (from_bucket == to_bucket) continue; // Skip arcs that loop back to the same bucket
302
303 if (fixed_buckets[from_bucket][to_bucket] == 1) continue; // Skip fixed arcs
304
305 bool valid_arc = true;
306 for (int r = 0; r < res_inc.size(); ++r) {
307 // Forward direction: Check that resource increment doesn't exceed upper bounds
308 if constexpr (D == Direction::Forward) {
309 if (buckets[from_bucket].lb[r] + res_inc[r] > next_node.ub[r]) {
310 valid_arc = false;
311 break;
312 }
313 }
314 // Backward direction: Check that resource decrement doesn't drop below lower bounds
315 else if constexpr (D == Direction::Backward) {
316 if (buckets[from_bucket].ub[r] - res_inc[r] < next_node.lb[r]) {
317 valid_arc = false;
318 break;
319 }
320 }
321 }
322 if (!valid_arc) continue; // Skip invalid arcs
323
324 // Further refine arc validity based on the base intervals and node bounds
325 if constexpr (D == Direction::Forward) {
326 for (int r = 0; r < res_inc.size(); ++r) {
327 double max_calc = std::max(buckets[from_bucket].lb[r] + res_inc[r], next_node.lb[r]);
328 if (max_calc < buckets[to_bucket].lb[r] ||
329 max_calc >= buckets[to_bucket].lb[r] + next_node_intervals[r] + numericutils::eps) {
330 valid_arc = false;
331 break;
332 }
333 }
334 } else if constexpr (D == Direction::Backward) {
335 for (int r = 0; r < res_inc.size(); ++r) {
336 double min_calc = std::min(buckets[from_bucket].ub[r] - res_inc[r], next_node.ub[r]);
337 if (min_calc > buckets[to_bucket].ub[r] ||
338 min_calc <= buckets[to_bucket].ub[r] - next_node_intervals[r] - numericutils::eps) {
339 valid_arc = false;
340 break;
341 }
342 }
343 }
344 if (!valid_arc) continue; // Skip invalid arcs
345
346 // Store the arc data locally before committing it to the global structure
347 local_arcs.emplace_back(from_bucket, to_bucket);
348 double local_cost_inc = cost_inc;
349 std::vector<double> local_res_inc = res_inc;
350
351 // Add the arc to the global structure and the bucket
352 {
353 std::lock_guard<std::mutex> lock(buckets_mutex); // Lock the mutex to ensure thread safety
354 add_arc<D>(from_bucket, to_bucket, local_res_inc, local_cost_inc); // Add the arc globally
355 buckets[from_bucket].template add_bucket_arc<D>(from_bucket, to_bucket, local_res_inc,
356 local_cost_inc, false); // Add the arc to the bucket
357 }
358 }
359 }
360 };
361
362 unsigned int total_threads = std::thread::hardware_concurrency();
363 // Divide by 2 to use only half of the available threads
364 unsigned int half_threads = total_threads / 2;
365
366 const int JOBS = half_threads;
367 exec::static_thread_pool pool(JOBS);
368 auto sched = pool.get_scheduler();
369
370 // Iterate over all nodes in parallel, generating arcs for each
371 std::vector<int> tasks; // Tasks will store node ids
372 for (int node_id = 0; node_id < nodes.size(); ++node_id) {
373 tasks.push_back(node_id); // Store the node id to be processed
374 }
375
376 // Define chunk size to reduce parallelization overhead
377 const int chunk_size = 10; // Adjust based on your performance needs
378
379 // Parallel execution in chunks
380 auto bulk_sender = stdexec::bulk(
381 stdexec::just(), (tasks.size() + chunk_size - 1) / chunk_size,
382 [this, &tasks, &num_buckets, &num_buckets_index, &add_arcs_for_node, chunk_size](std::size_t chunk_idx) {
383 size_t start_idx = chunk_idx * chunk_size;
384 size_t end_idx = std::min(start_idx + chunk_size, tasks.size());
385
386 // Process a chunk of tasks (i.e., a group of nodes)
387 for (size_t task_idx = start_idx; task_idx < end_idx; ++task_idx) {
388 int node_id = tasks[task_idx]; // Get the node id
389 const VRPNode &VRPNode = nodes[node_id];
390
391 std::vector<double> res_inc(options.resources.size()); // Resource increment vector
392 std::vector<std::pair<int, int>> local_arcs; // Local storage for arcs
393
394 // Generate arcs for all buckets associated with the current node
395 for (int i = 0; i < num_buckets[VRPNode.id]; ++i) {
396 int from_bucket = i + num_buckets_index[VRPNode.id]; // Determine the source bucket
397 add_arcs_for_node(VRPNode, from_bucket, res_inc,
398 local_arcs); // Add arcs for this node and bucket
399 }
400 }
401 });
402
403 // Submit work to the thread pool
404 auto work = stdexec::starts_on(sched, bulk_sender);
405 stdexec::sync_wait(std::move(work));
406}

417template <Direction D>
418Label *BucketGraph::get_best_label(const std::vector<int> &topological_order, const std::vector<double> &c_bar,
419 const std::vector<std::vector<int>> &sccs) {
420 double best_cost = std::numeric_limits<double>::infinity();
421 Label *best_label = nullptr; // Ensure this is initialized
422 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
423
424 for (const int component_index : topological_order) {
425 const auto &component_buckets = sccs[component_index];
426
427 for (const int bucket : component_buckets) {
428 const auto &label = buckets[bucket].get_best_label();
429 if (!label) continue;
430 // print label->cost
431 if (label->cost < best_cost) {
432 best_cost = label->cost;
433 best_label = label;
434 }
435 }
436 }
437
438 return best_label;
439}

445template <Stage S, Symmetry SYM>
446void BucketGraph::ConcatenateLabel(const Label *L, int &b, double &best_cost, std::vector<uint64_t> &Bvisited) {
447 static thread_local std::vector<int> bucket_stack;
448 bucket_stack.clear();
449 bucket_stack.reserve(50);
450 bucket_stack.push_back(b);
451
452 auto &other_buckets = assign_symmetry<SYM>(fw_buckets, bw_buckets);
453 auto &other_c_bar = assign_symmetry<SYM>(fw_c_bar, bw_c_bar);
454
455 // Cache frequently accessed values
456 const int L_node_id = L->node_id;
457 const auto &L_resources = L->resources;
458 const auto &L_last_node = nodes[L_node_id];
459 const double L_cost = L->cost;
460 const size_t bitmap_size = L->visited_bitmap.size();
461
462 // Pre-compute constants for bit operations
463 constexpr uint64_t one = 1ULL;
464 const bool has_branching = !branching_duals->empty();
465
466 // SRC mode setup
467#if defined(SRC)
468 decltype(cut_storage) cutter = nullptr;
469 decltype(cut_storage->SRCDuals) *SRCDuals = nullptr;
470 if constexpr (S > Stage::Three) {
471 cutter = cut_storage;
472 SRCDuals = &cutter->SRCDuals;
473 }
474#endif
475
476 while (!bucket_stack.empty()) {
477 const int current_bucket = bucket_stack.back();
478 bucket_stack.pop_back();
479
480 // Optimize bit operations for visited tracking
481 const size_t segment = current_bucket >> 6;
482 const uint64_t bit_mask = one << (current_bucket & 63);
483 Bvisited[segment] |= bit_mask;
484
485 const int bucketLprimenode = other_buckets[current_bucket].node_id;
486 double travel_cost = getcij(L_node_id, bucketLprimenode);
487
488 // Apply arc duals if needed
489#if defined(RCC) || defined(EXACT_RCC)
490 if constexpr (S == Stage::Four) { travel_cost -= arc_duals.getDual(L_node_id, bucketLprimenode); }
491#endif
492
493 if (has_branching) { travel_cost -= branching_duals->getDual(L_node_id, bucketLprimenode); }
494
495 const double path_cost = L_cost + travel_cost;
496 const double bound = other_c_bar[current_bucket];
497
498 // Early bound check
499 if ((S != Stage::Enumerate && path_cost + bound >= best_cost) ||
500 (S == Stage::Enumerate && path_cost + bound >= gap)) {
501 continue;
502 }
503
504 // Process labels in current bucket
505 const auto &bucket = other_buckets[current_bucket];
506 const auto &labels = bucket.get_labels();
507
508 for (const Label *L_bw : labels) {
509 // Early rejection tests
510 if (L_bw->node_id == L_node_id || !check_feasibility(L, L_bw)) { continue; }
511
512 // Visited nodes overlap check
513 if constexpr (S >= Stage::Three) {
514 bool has_overlap = false;
515 for (size_t i = 0; i < bitmap_size; ++i) {
516 if (L->visited_bitmap[i] & L_bw->visited_bitmap[i]) {
517 has_overlap = true;
518 break;
519 }
520 }
521 if (has_overlap) continue;
522 }
523
524 double total_cost = path_cost + L_bw->cost;
525
526 // SRC cost adjustments
527#if defined(SRC)
528 if constexpr (S == Stage::Four) {
529 for (auto it = cutter->begin(); it < cutter->end(); ++it) {
530 const auto &dual = (*SRCDuals)[it->id];
531 if (dual == 0) continue;
532
533 if (L->SRCmap[it->id] + L_bw->SRCmap[it->id] >= it->p.den) { total_cost -= dual; }
534 }
535 }
536#endif
537
538 // Cost-based filtering
539 if ((S != Stage::Enumerate && total_cost >= best_cost) || (S == Stage::Enumerate && total_cost >= gap)) {
540 continue;
541 }
542
543 // Create new merged label
544 auto pbest = compute_label<S>(L, L_bw);
545 best_cost = pbest->cost;
546 // fmt::print("Best cost: {}\n", best_cost);
547 merged_labels.push_back(pbest);
548 }
549
550 // Process neighbor buckets
551 for (int b_prime : Phi_bw[current_bucket]) {
552 const size_t seg_prime = b_prime >> 6;
553 const uint64_t mask_prime = one << (b_prime & 63);
554 if (!(Bvisited[seg_prime] & mask_prime)) { bucket_stack.push_back(b_prime); }
555 }
556 }
557}

568template <Direction D>
569void BucketGraph::SCC_handler() {
570 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
571 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
572 auto &bucket_graph = assign_buckets<D>(fw_bucket_graph, bw_bucket_graph);
573 ankerl::unordered_dense::map<int, std::vector<int>> extended_bucket_graph = bucket_graph;
574
575 // Extend the bucket graph with arcs defined by the Phi sets
576 for (auto i = 0; i < extended_bucket_graph.size(); ++i) {
577 auto phi_set = Phi[i];
578 if (phi_set.empty()) continue;
579 for (auto &phi_bucket : phi_set) { extended_bucket_graph[phi_bucket].push_back(i); }
580 }
581
582 SCC scc_finder;
583 scc_finder.convertFromUnorderedMap(extended_bucket_graph); // print extended bucket graph
584
585 auto sccs = scc_finder.tarjanSCC();
586 auto topological_order = scc_finder.topologicalOrderOfSCCs(sccs);
587
588#ifdef VERBOSE
589 // print SCCs and buckets in it
590 constexpr auto blue = "\033[34m";
591 constexpr auto reset = "\033[0m";
592 if constexpr (D == Direction::Forward) {
593 fmt::print("FW SCCs: \n");
594 } else {
595 fmt::print("BW SCCs: \n");
596 }
597 for (auto scc : topological_order) {
598 if constexpr (D == Direction::Forward) {
599 fmt::print("{}({}) -> {}", blue, scc, reset);
600 } else {
601 fmt::print("{}({}) -> {}", blue, scc, reset);
602 }
603 for (auto &bucket : sccs[scc]) { fmt::print("{} ", bucket); }
604 }
605 fmt::print("\n");
606#endif
607
608 std::vector<std::vector<int>> ordered_sccs;
609 ordered_sccs.reserve(sccs.size()); // Reserve space for all SCCs
610 for (int i : topological_order) { ordered_sccs.push_back(sccs[i]); }
611
612 auto sorted_sccs = sccs;
613 for (auto &scc : sorted_sccs) {
614 if constexpr (D == Direction::Forward) {
615 std::sort(scc.begin(), scc.end(), [&buckets](int a, int b) { return buckets[a].lb[0] < buckets[b].lb[0]; });
616 } else {
617 std::sort(scc.begin(), scc.end(), [&buckets](int a, int b) { return buckets[a].ub[0] > buckets[b].ub[0]; });
618 }
619 }
620
621 // iterate over nodes
622 for (auto &node : nodes) {
623 if constexpr (D == Direction::Forward) {
624 node.fw_arcs_scc.resize(sccs.size());
625 } else {
626 node.bw_arcs_scc.resize(sccs.size());
627 }
628 }
629
630 // Split arcs for each SCC
631 auto scc_ctr = 0;
632 for (const auto &scc : sccs) {
633 // Iterate over each bucket in the SCC
634 for (int bucket : scc) {
635 int from_node_id = buckets[bucket].node_id; // Get the source node ID
636 VRPNode &node = nodes[from_node_id]; // Access the corresponding node
637 // Define filtered arcs depending on the direction
638 if constexpr (D == Direction::Forward) {
639 std::vector<Arc> &filtered_fw_arcs = nodes[from_node_id].fw_arcs_scc[scc_ctr]; // For forward direction
640
641 // Iterate over the arcs from the current bucket
642 const auto &bucket_arcs = buckets[bucket].template get_bucket_arcs<D>();
643 for (const auto &arc : bucket_arcs) {
644 int to_node_id = buckets[arc.to_bucket].node_id; // Get the destination node ID
645
646 // Search for the arc from `from_node_id` to `to_node_id` in the node's arcs
647 auto it = std::find_if(node.fw_arcs.begin(), node.fw_arcs.end(),
648 [&to_node_id](const Arc &a) { return a.to == to_node_id; });
649
650 // If both nodes are within the current SCC, retain the arc
651 if (it != node.fw_arcs.end()) {
652 // Add the arc to the filtered arcs
653 filtered_fw_arcs.push_back(*it); // Forward arcs
654 }
655 }
656 } else {
657 std::vector<Arc> &filtered_bw_arcs = nodes[from_node_id].bw_arcs_scc[scc_ctr]; // For forward direction
658
659 // Iterate over the arcs from the current bucket
660 const auto &bucket_arcs = buckets[bucket].template get_bucket_arcs<D>();
661 for (const auto &arc : bucket_arcs) {

663
664 // Search for the arc from `from_node_id` to `to_node_id` in the node's arcs
665 auto it = std::find_if(node.bw_arcs.begin(), node.bw_arcs.end(),
666 [&to_node_id](const Arc &a) { return a.to == to_node_id; });
667
668 // If both nodes are within the current SCC, retain the arc
669 if (it != node.bw_arcs.end()) {
670 // Add the arc to the filtered arcs
671 filtered_bw_arcs.push_back(*it); // Forward arcs
672 }
673 }
674 }
675 }
676
677 // Increment SCC counter
678 ++scc_ctr;
679 }
680
681 for (auto &node : nodes) {
682 if constexpr (D == Direction::Forward) {
683 // Iterate over all SCCs for each node
684 for (auto &fw_arcs_scc : node.fw_arcs_scc) {
685 // Sort arcs by from_bucket and to_bucket
686 std::sort(fw_arcs_scc.begin(), fw_arcs_scc.end(),
687 [](const Arc &a, const Arc &b) { return std::tie(a.from, a.to) < std::tie(b.from, b.to); });
688
689 // Remove consecutive duplicates
690 auto last = std::unique(fw_arcs_scc.begin(), fw_arcs_scc.end(),
691 [](const Arc &a, const Arc &b) { return a.from == b.from && a.to == b.to; });
692
693 // Erase the duplicates from the vector
694 fw_arcs_scc.erase(last, fw_arcs_scc.end());
695 }
696 } else {
697 // Iterate over all SCCs for each node
698 for (auto &bw_arcs_scc : node.bw_arcs_scc) {
699 // Sort arcs by from_bucket and to_bucket
700 std::sort(bw_arcs_scc.begin(), bw_arcs_scc.end(),
701 [](const Arc &a, const Arc &b) { return std::tie(a.from, a.to) < std::tie(b.from, b.to); });
702
703 // Remove consecutive duplicates
704 auto last = std::unique(bw_arcs_scc.begin(), bw_arcs_scc.end(),
705 [](const Arc &a, const Arc &b) { return a.from == b.from && a.to == b.to; });
706
707 // Erase the duplicates from the vector
708 bw_arcs_scc.erase(last, bw_arcs_scc.end());
709 }
710 }
711 }
712
713 UnionFind uf(ordered_sccs);
714 if constexpr (D == Direction::Forward) {
715 fw_ordered_sccs = ordered_sccs;
716 fw_topological_order = topological_order;
717 fw_sccs = sccs;
718 fw_sccs_sorted = sorted_sccs;
719 fw_union_find = uf;
720 } else {
721 bw_ordered_sccs = ordered_sccs;
722 bw_topological_order = topological_order;
723 bw_sccs = sccs;
724 bw_sccs_sorted = sorted_sccs;
725 bw_union_find = uf;

832template <Stage S>
833void BucketGraph::heuristic_fixing() {
834 // Stage 3 fixing heuristic
835 reset_pool();
836 reset_fixed();
837 common_initialization();
838
839 std::vector<double> forward_cbar(fw_buckets.size(), std::numeric_limits<double>::infinity());
840 std::vector<double> backward_cbar(bw_buckets.size(), std::numeric_limits<double>::infinity());
841
842 run_labeling_algorithms<Stage::Two, Full::Partial>(forward_cbar, backward_cbar);
843
844 std::vector<std::vector<Label *>> fw_labels_map(nodes.size());
845 std::vector<std::vector<Label *>> bw_labels_map(nodes.size());
846
847 auto group_labels = [&](auto &buckets, auto &labels_map) {
848 for (auto &bucket : buckets) {
849 for (auto label : bucket.get_labels()) {
850 labels_map[label->node_id].push_back(label); // Directly index using node_id
851 }
852 }
853 };
854
855 // Create tasks for forward and backward labels grouping
856 auto forward_task = stdexec::schedule(bi_sched) | stdexec::then([&]() { group_labels(fw_buckets, fw_labels_map); });
857 auto backward_task =
858 stdexec::schedule(bi_sched) | stdexec::then([&]() { group_labels(bw_buckets, bw_labels_map); });
859
860 // Execute the tasks in parallel
861 auto work = stdexec::when_all(std::move(forward_task), std::move(backward_task));
862
863 stdexec::sync_wait(std::move(work));
864
865 auto num_fixes = 0;
866 // Function to find the minimum cost label in a vector of labels
867 auto find_min_cost_label = [](const std::vector<Label *> &labels) -> const Label * {
868 return *std::min_element(labels.begin(), labels.end(),
869 [](const Label *a, const Label *b) { return a->cost < b->cost; });
870 };
871 for (const auto &node_I : nodes) {
872 const auto &fw_labels = fw_labels_map[node_I.id];
873 if (fw_labels.empty()) continue; // Skip if no labels for this node_id
874
875 for (const auto &node_J : nodes) {
876 if (node_I.id == node_J.id) continue; // Compare based on id (or other key field)
877 const auto &bw_labels = bw_labels_map[node_J.id];
878 if (bw_labels.empty()) continue; // Skip if no labels for this node_id
879
880 const Label *min_fw_label = find_min_cost_label(fw_labels);
881 const Label *min_bw_label = find_min_cost_label(bw_labels);
882
883 if (!min_fw_label || !min_bw_label) continue;
884
885 const VRPNode &L_last_node = nodes[min_fw_label->node_id];
886 auto cost = getcij(min_fw_label->node_id, min_bw_label->node_id);
887
888 bool violated = false;
889 for (auto r = 0; r < options.resources.size(); ++r) {
890 if (options.resources[r] == "time") {
891 if (min_fw_label->resources[TIME_INDEX] + cost + L_last_node.duration >
892 min_bw_label->resources[TIME_INDEX]) {
893 violated = true;
894 break;
895 }
896 } else {
897 if (min_fw_label->resources[r] + L_last_node.consumption[r] > min_bw_label->resources[r]) {
898 violated = true;
899 break;
900 }
901 }
902 }
903
904 if (violated) continue;
905
906 if (min_fw_label->cost + cost + min_bw_label->cost > gap) {
907 fixed_arcs[node_I.id][node_J.id] = 1; // Index with node ids
908 num_fixes++;
909 }
910 }
911 }

915void BucketGraph::set_adjacency_list() {
916 // Clear existing arcs for each node
917 for (auto &node : nodes) {
918 node.clear_arcs(); // Remove any existing arcs associated with the node
919 }
920

937 // Create a job-to-cluster mapping (cluster ID for each job/node)
938 std::vector<int> job_to_cluster(nodes.size(), -1); // Mapping from job (node) to cluster ID
939 for (int cluster_id = 0; cluster_id < clusters.size(); ++cluster_id) {
940 for (int job : clusters[cluster_id]) { job_to_cluster[job] = cluster_id; }
941 }

943 // Step 2: Modify add_arcs_for_node to give priority based on cluster membership
944 auto add_arcs_for_node = [&](const VRPNode &node, int from_bucket, std::vector<double> &res_inc) {
945 using Arc = std::tuple<double, int, std::vector<double>,
946 double>; // Arc: priority, to_node, resource increments, cost increment

949 best_arcs.reserve(nodes.size()); // Reserve space for forward arcs
950
951 std::vector<Arc> best_arcs_rev;
952 best_arcs_rev.reserve(nodes.size()); // Reserve space for reverse arcs
953
954 for (const auto &next_node : nodes) {
955 if (next_node.id == options.depot || node.id == next_node.id) continue; // Skip depot and same node
956
957 if (options.pstep == true) {
958 if (next_node.id == options.pstep_depot || next_node.id == options.pstep_end_depot)
959 continue; // Skip depot and end depot
960 }

962 auto travel_cost = getcij(node.id, next_node.id); // Calculate travel cost
963 double cost_inc = travel_cost - next_node.cost; // Adjust cost increment by subtracting next node's cost
964
965 for (int r = 0; r < options.resources.size(); ++r) {
966 if (options.resources[r] == "time") {
967 if constexpr (SYM == Symmetry::Asymmetric) {
968 res_inc[r] = travel_cost + node.duration; // Update resource increment based on node duration
969 } else {
970 res_inc[r] = node.duration / 2 + travel_cost +
971 next_node.duration / 2; // Update resource increment based on node duration
972 }
973 } else {
974 if constexpr (SYM == Symmetry::Asymmetric) {
975 res_inc[r] = node.consumption[r];
976 } else {
977 res_inc[r] = node.consumption[r] / 2 + next_node.consumption[r] / 2;
978 }
979 }
980 }
981
982 int to_bucket = next_node.id;
983 if (from_bucket == to_bucket) continue; // Skip arcs that loop back to the same bucket
984
985 bool feasible = true; // Check feasibility based on resource constraints
986 for (int r = 0; r < options.resources.size(); ++r) {
987 if (node.lb[r] + res_inc[r] > next_node.ub[r]) {
988 feasible = false;
989 break;
990 }
991 }
992 if (!feasible) continue; // Skip infeasible arcs
993
994 // Step 3: Calculate priority based on cluster membership
995 double priority_value;
996 double reverse_priority_value;
997
998 // bool is_heuristic_arc = heur_arcs.has_arc(node.id, next_node.id);
999

1001 // Higher priority if both nodes are in the same cluster
1002 priority_value = 5.0 + 1.E-5 * next_node.start_time; // Adjust weight for same-cluster priority
1003 reverse_priority_value = 1.0 + 1.E-5 * node.start_time; // Adjust weight for same-cluster priority
1004 } else {
1005 // Lower priority for cross-cluster arcs
1006 priority_value = 1.0 + 1.E-5 * next_node.start_time; // Higher base value for cross-cluster arcs
1007 reverse_priority_value = 5.0 + 1.E-5 * node.start_time; // Higher base value for cross-cluster arcs
1008 }

1010 best_arcs_rev.emplace_back(reverse_priority_value, next_node.id, res_inc,
1011 cost_inc); // Store the reverse arc
1012 }
1013
1014 // Add forward arcs from the current node to its neighbors
1015 for (const auto &arc : best_arcs) {
1016 auto [priority_value, to_bucket, res_inc_local, cost_inc] = arc;
1017 nodes[node.id].template add_arc<Direction::Forward>(node.id, to_bucket, res_inc_local, cost_inc,
1018 priority_value); // Add forward arc
1019 // fmt::print("Node ID: {}, To Bucket: {}, Cost Inc: {}\n", node.id, to_bucket, cost_inc);
1020 }
1021
1022 // Add reverse arcs from neighboring nodes to the current node
1023 for (const auto &arc : best_arcs_rev) {
1024 auto [priority_value, to_bucket, res_inc_local, cost_inc] = arc;
1025 nodes[to_bucket].template add_arc<Direction::Backward>(to_bucket, node.id, res_inc_local, cost_inc,
1026 priority_value); // Add reverse arc
1027 }
1028 };
1029
1030 // Step 4: Iterate over all nodes to set the adjacency list
1031 // print depot and end depot
1032 for (const auto &VRPNode : nodes) {
1033 // fmt::print("Node ID: {}\n", VRPNode.id);
1034
1035 if (VRPNode.id == options.end_depot) {
1036 continue; // Skip the last node (depot)
1037 }
1038
1039 std::vector<double> res_inc(intervals.size()); // Resource increment vector
1040 add_arcs_for_node(VRPNode, VRPNode.id, res_inc); // Add arcs for the current node
1041 }

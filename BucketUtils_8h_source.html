20class ThreadLocalPool {
21 static thread_local std::vector<std::vector<T>> pool;
22 static thread_local size_t current_index;

24public:
25 std::vector<T> &acquire() {
26 if (current_index >= pool.size()) { pool.emplace_back(); }

47void BucketGraph::add_arc(int from_bucket, int to_bucket, const std::vector<double> &res_inc, double cost_inc) {
48 if constexpr (D == Direction::Forward) {
49 fw_arcs.emplace_back(from_bucket, to_bucket, res_inc, cost_inc);

58 bw_bucket_graph.resize(static_cast<size_t>(from_bucket) + 1);
59 }
60 bw_bucket_graph[from_bucket].push_back(to_bucket);
61 }

64template <Direction D>
65inline int BucketGraph::get_bucket_number(int node, std::vector<double> &resource_values_vec) noexcept {
66 const size_t num_resources = options.main_resources.size();
67
68 const auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
69
70 // Optionally adjust resource values (uncomment if you need epsilon
71 // adjustments)
72 for (size_t r = 0; r < num_resources; ++r) {
73 if constexpr (D == Direction::Forward) {
74 // resource_values_vec[r] += numericutils::eps;
75 } else { // Direction::Backward
76 // resource_values_vec[r] -= numericutils::eps;
77 }
78 }
79 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
80 // Start with the offset for this node’s buckets.
81 int bucket_number = num_buckets_index[node];
82 int multiplier = 1;
83
84 for (size_t r = 0; r < num_resources; ++r) {
85 // Node-specific resource bounds.
86 double node_lb = nodes[node].lb[r];
87 double node_ub = nodes[node].ub[r];
88
89 // Global bounds (if needed for computing splits).
90 double global_lb = R_min[r];
91 double global_ub = R_max[r];
92 double full_range = global_ub - global_lb;
93 double node_range = node_ub - node_lb;
94
95 // Determine the number of splits (i.e. intervals) for resource r.
96 int splits = std::max(1, static_cast<int>(std::round((node_range / full_range) * intervals[r].interval)));
97 double interval_width =
98 (splits == 0 || std::fabs(node_range) < std::numeric_limits<double>::epsilon()) ? 1.0 : node_range / splits;
99
100 int pos = 0;
101 if constexpr (D == Direction::Forward) {
102 if (std::fabs(interval_width) < std::numeric_limits<double>::epsilon()) {
103 pos = 0;
104 } else {
105 pos = std::clamp(static_cast<int>(std::floor((resource_values_vec[r] - node_lb) / interval_width)), 0,
106 splits - 1);
107 }
108 } else {
109 if (std::fabs(interval_width) < std::numeric_limits<double>::epsilon()) {
110 pos = 0;
111 } else {
112 pos = std::clamp(static_cast<int>(std::floor((node_ub - resource_values_vec[r]) / interval_width)), 0,
113 splits - 1);
114 }
115 }
116
117 bucket_number += pos * multiplier;
118 multiplier *= splits;
119 }
120
121 // check if buckets[bucket_number] contains the resource_vec
122 if (unlikely(!buckets[bucket_number].contains(resource_values_vec))) {
123 // print resource_values_vec
124 for (auto val : resource_values_vec) { fmt::print("{}\n", val); }
125 // print bucket bounds
126 fmt::print("{}\n", buckets[bucket_number].lb[0]);
127 fmt::print("{}\n", buckets[bucket_number].ub[0]);
128
129 std::throw_with_nested(std::runtime_error("Resource values not contained in bucket"));
130 return -1;
131 }
132
133 return bucket_number;
134}

145template <Direction D>
146// A constant representing the number of splits for a node that spans the full
147// range. For example, if a node covers the entire full range, it will get
148// FULL_RANGE_SPLITS buckets.
149void BucketGraph::define_buckets() {
150 // 1. Save the old buckets and fixed status.
151 std::vector<Bucket> old_buckets;
152 std::vector<bool> was_fixed;
153 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
154 size_t old_size = buckets.size();
155
156 // Map to store the fixed range (min_lb and max_ub) for each node pair.
157 ankerl::unordered_dense::map<int,
158 ankerl::unordered_dense::map<int, std::pair<std::vector<double>, std::vector<double>>>>
159 node_pair_fixed_ranges;
160
161 if (old_size > 0) {
162 old_buckets = buckets;
163 was_fixed.resize(old_size * old_size, false);
164 auto &old_bitmap = assign_buckets<D>(fw_fixed_buckets_bitmap, bw_fixed_buckets_bitmap);
165
166 // Mark fixed bucket pairs.
167 for (size_t i = 0; i < old_size; ++i) {
168 for (size_t j = 0; j < old_size; ++j) {
169 size_t pos = i * old_size + j;
170 if (pos / 64 < old_bitmap.size()) {
171 was_fixed[pos] = ((old_bitmap[pos / 64] & (1ULL << (pos % 64))) != 0);
172 }
173 }
174 }
175
176 const int num_intervals = options.main_resources.size();
177 // For each fixed bucket pair, update the fixed range for the
178 // corresponding node pair.
179 for (size_t i = 0; i < old_size; ++i) {
180 for (size_t j = 0; j < old_size; ++j) {
181 if (!was_fixed[i * old_size + j]) continue;
182 const auto &bucket_from = old_buckets[i];
183 const auto &bucket_to = old_buckets[j];
184 int from_node = bucket_from.node_id;
185 int to_node = bucket_to.node_id;
186 // Initialize if necessary.
187 if (node_pair_fixed_ranges[from_node].find(to_node) == node_pair_fixed_ranges[from_node].end()) {
188 node_pair_fixed_ranges[from_node][to_node] = {
189 std::vector<double>(num_intervals, std::numeric_limits<double>::max()),
190 std::vector<double>(num_intervals, std::numeric_limits<double>::lowest())};
191 }
192 auto &[min_lbs, max_ubs] = node_pair_fixed_ranges[from_node][to_node];
193 for (int r = 0; r < num_intervals; ++r) {
194 min_lbs[r] = std::min(min_lbs[r], bucket_from.lb[r]);
195 max_ubs[r] = std::max(max_ubs[r], bucket_from.ub[r]);
196 }
197 }
198 }
199 } // End if old buckets exist.
200
201 // 2. Clear the current buckets and fixed bitmap.
202 buckets.clear();
203 auto &fixed_buckets_bitmap = assign_buckets<D>(fw_fixed_buckets_bitmap, bw_fixed_buckets_bitmap);
204 fixed_buckets_bitmap.clear();
205
206 // 3. Prepare variables and containers for new bucket definition.
207 const int num_intervals = options.main_resources.size();
208 std::vector<double> total_ranges(num_intervals); // unused in current code
209 std::vector<double> base_intervals(num_intervals); // unused in current code
210
211 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
212 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
213 auto &node_interval_trees = assign_buckets<D>(fw_node_interval_trees, bw_node_interval_trees);
214 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
215 auto &bucket_splits = assign_buckets<D>(fw_bucket_splits, bw_bucket_splits);
216
217 const size_t num_nodes = nodes.size();
218 num_buckets.resize(num_nodes);
219 num_buckets_index.resize(num_nodes);
220 node_interval_trees.assign(num_nodes, SplayTree());
221
222 // Lambda to calculate an interval for one dimension. Backward buckets are
223 // deliberately emitted in the paper's kappa order:
224 // kappa = 0 -> interval ending at u_v,
225 // kappa = 1 -> next lower interval, ...
226 // This keeps component-wise smaller buckets equal to smaller local bucket
227 // positions in both forward and backward arrays.
228 auto calculate_interval = [&](double lb, double ub, double base_interval, int pos, int max_intervals,
229 bool is_forward) -> std::pair<double, double> {
230 double start, end;
231 if (is_forward) {
232 start = (pos == 0) ? lb : lb + pos * base_interval;
233 end = (pos == max_intervals - 1) ? ub : lb + (pos + 1) * base_interval;
234 } else {
235 start = (pos == max_intervals - 1) ? lb : ub - (pos + 1) * base_interval;
236 end = (pos == 0) ? ub : ub - pos * base_interval;
237 }
238 start = roundToTwoDecimalPlaces(start);
239 end = roundToTwoDecimalPlaces(end);
240 return {start, end};
241 };
242
243 int cum_sum = 0;
244 int bucket_index = 0;
245 std::vector<double> interval_start(num_intervals);
246 std::vector<double> interval_end(num_intervals);
247 std::vector<int> pos(num_intervals,
248 0); // combination indices for multi-dimensions
249
250 // 4. Process each node to define new buckets.
251 for (const auto &vrp_node : nodes) {
252 std::vector<int> node_split_counts(num_intervals);
253 std::vector<double> node_base_interval(num_intervals);
254 // Determine the number of splits and base interval for each
255 // resource dimension.
256 for (int r = 0; r < num_intervals; ++r) {
257 double full_range = R_max[r] - R_min[r];
258 double node_range = vrp_node.ub[r] - vrp_node.lb[r];
259 int splits = 1;
260 if (std::fabs(full_range) > std::numeric_limits<double>::epsilon()) {
261 double splits_d = (node_range * intervals[r].interval) / full_range;
262 splits = std::max(1, static_cast<int>(std::round(splits_d)));
263 }
264 node_split_counts[r] = splits;
265 node_base_interval[r] = (vrp_node.ub[r] - vrp_node.lb[r]) / static_cast<double>(splits);
266 }
267 // Optionally store node-specific split counts.
268 bucket_splits[vrp_node.id] = node_split_counts;
269
270 int n_buckets = 0;
271 if (num_intervals == 1) {
272 // Single-dimensional case.
273 for (int j = 0; j < node_split_counts[0]; ++j) {
274 auto [int_start, int_end] = calculate_interval(vrp_node.lb[0], vrp_node.ub[0], node_base_interval[0], j,
275 node_split_counts[0], D == Direction::Forward);
276 if constexpr (D == Direction::Backward)
277 int_start = std::max(int_start, vrp_node.lb[0]);
278 else
279 int_end = std::min(int_end, vrp_node.ub[0]);
280 buckets.push_back(Bucket(vrp_node.id, std::vector<double>{int_start}, std::vector<double>{int_end}));
281 n_buckets++;
282 cum_sum++;
283 }
284 } else {
285 // Multi-dimensional: iterate over all combinations.
286 fmt::print("Multiple intervals\n");
287 std::fill(pos.begin(), pos.end(), 0);
288 while (true) {
289 for (int r = 0; r < num_intervals; ++r) {
290 auto [int_start, int_end] =
291 calculate_interval(vrp_node.lb[r], vrp_node.ub[r], node_base_interval[r], pos[r],
292 node_split_counts[r], D == Direction::Forward);
293 interval_start[r] = int_start;
294 interval_end[r] = int_end;
295 if constexpr (D == Direction::Backward)
296 interval_start[r] = std::max(interval_start[r], R_min[r]);
297 else
298 interval_end[r] = std::min(interval_end[r], R_max[r]);
299 }
300 buckets.push_back(Bucket(vrp_node.id, interval_start, interval_end));
301 n_buckets++;
302 cum_sum++;
303
304 // Generate next combination.
305 int i = 0;
306 while (i < num_intervals) {
307 ++pos[i];
308 if (pos[i] < node_split_counts[i])
309 break;
310 else {
311 pos[i] = 0;
312 ++i;
313 }
314 }
315 if (i == num_intervals) break;
316 }
317 }
318 num_buckets[vrp_node.id] = n_buckets;
319 num_buckets_index[vrp_node.id] = cum_sum - n_buckets;
320 // Optionally assign node_interval_trees[vrp_node.id] = node_tree;
321 //
322 if constexpr (D == Direction::Forward) { n_segments = (n_buckets + 63) >> 6; }
323 }
324
325 // 5. Precompute RC2 bucket metadata for whole-bucket prune.
326 // Forward dominance needs candidate rc2 <= label rc2, so keep a prefix
327 // minimum of bucket lower bounds. Backward buckets are reversed, so
328 // candidate rc2 >= label rc2 also lives in the prefix.
329 auto &rc2_bin = assign_buckets<D>(fw_rc2_bin, bw_rc2_bin);
330 auto &rc2_till_this_bin = assign_buckets<D>(fw_rc2_till_this_bin, bw_rc2_till_this_bin);
331 rc2_bin.assign(num_nodes, {});
332 rc2_till_this_bin.assign(num_nodes, {});
333 for (int node_id = 0; node_id < static_cast<int>(num_nodes); ++node_id) {
334 int count = num_buckets[node_id];
335 rc2_bin[node_id].assign(count, 0.0);
336 rc2_till_this_bin[node_id].assign(count, 0.0);
337 if (count == 0) continue;
338
339 for (int local_b = 0; local_b < count; ++local_b) {
340 int global_b = num_buckets_index[node_id] + local_b;
341 const auto &bucket = buckets[global_b];
342 double rc2_value = 0.0;
343 if constexpr (D == Direction::Forward) {
344 if (bucket.lb.size() > 1) { rc2_value = bucket.lb[1]; }
345 } else {
346 if (bucket.ub.size() > 1) { rc2_value = bucket.ub[1]; }
347 }
348 rc2_bin[node_id][local_b] = rc2_value;
349 if (local_b == 0) {
350 rc2_till_this_bin[node_id][local_b] = rc2_value;
351 } else {
352 if constexpr (D == Direction::Forward) {
353 rc2_till_this_bin[node_id][local_b] = std::min(rc2_till_this_bin[node_id][local_b - 1], rc2_value);
354 } else {
355 rc2_till_this_bin[node_id][local_b] = std::max(rc2_till_this_bin[node_id][local_b - 1], rc2_value);
356 }
357 }
358 }
359 }
360
361 // 6. Process fixed bucket ranges (if available).
362 if (!node_pair_fixed_ranges.empty()) {
363 size_t new_size = buckets.size();
364 size_t required_bitmap_words = std::max(size_t(1), ((new_size * new_size) + 63) / 64);
365 fixed_buckets_bitmap.clear();
366 fixed_buckets_bitmap.resize(required_bitmap_words, 0);
367 for (size_t from = 0; from < new_size; ++from) {
368 const auto &bucket_from = buckets[from];
369 int from_node = bucket_from.node_id;
370 if (node_pair_fixed_ranges.find(from_node) == node_pair_fixed_ranges.end()) continue;
371 for (size_t to = 0; to < new_size; ++to) {
372 const auto &bucket_to = buckets[to];
373 int to_node = bucket_to.node_id;
374 if (node_pair_fixed_ranges[from_node].find(to_node) == node_pair_fixed_ranges[from_node].end())
375 continue;
376 const auto &range = node_pair_fixed_ranges[from_node][to_node];
377 bool in_range = true;
378 for (int r = 0; r < options.main_resources.size(); ++r) {
379 if (bucket_from.lb[r] < range.first[r] || bucket_from.ub[r] > range.second[r]) {
380 in_range = false;
381 break;
382 }
383 }
384 if (in_range) {
385 size_t bit_pos = from * new_size + to;
386 size_t word_idx = bit_pos / 64;
387 if (word_idx < fixed_buckets_bitmap.size()) {
388 fixed_buckets_bitmap[word_idx] |= (1ULL << (bit_pos % 64));
389 }
390 }
391 }
392 }
393 }
394
395 // 6. Finalize: record the total bucket count.
396 buckets_size = buckets.size();
397}

408template <Direction D>
409int BucketGraph::get_opposite_bucket_number(int current_bucket_index, std::vector<double> &inc) {
410 // Retrieve the current bucket based on the current direction.
411 const auto &current_bucket =
412 (D == Direction::Forward) ? fw_buckets[current_bucket_index] : bw_buckets[current_bucket_index];
413 const int node = current_bucket.node_id;
414 const auto &theNode = nodes[node];
415
416 // Pre-compute the reference point for the given node using the main
417 // resources.
418 const size_t num_resources = options.main_resources.size();
419 // Optionally, add an assertion or check that inc has the expected size.
420 assert(inc.size() == num_resources && theNode.lb.size() == num_resources && theNode.ub.size() == num_resources);
421
422 std::vector<double> reference_point(num_resources);
423
424 if constexpr (D == Direction::Forward) {
425 // For Forward direction:
426 // Use the maximum of inc and the node's lower bounds.
427 // This ensures we do not fall below the allowed lower bound.
428 for (size_t i = 0; i < num_resources; ++i) {
429 reference_point[i] = std::max(inc[i], theNode.lb[i]);
430 // Optionally, adjust with epsilon if needed:
431 // reference_point[i] += numericutils::eps;
432 }
433 // Get the opposite bucket using the Backward tree.
434 return get_bucket_number<Direction::Backward>(node, reference_point);
435 } else {
436 // For Backward direction:
437 // Use the minimum of inc and the node's upper bounds.
438 // This ensures we do not exceed the allowed upper bound.
439 for (size_t i = 0; i < num_resources; ++i) {
440 reference_point[i] = std::min(inc[i], theNode.ub[i]);
441 // Optionally, adjust with epsilon if needed:
442 // reference_point[i] -= numericutils::eps;
443 }
444 // Get the opposite bucket using the Forward tree.
445 return get_bucket_number<Direction::Forward>(node, reference_point);
446 }
447}

459template <Stage S>
460void BucketGraph::bucket_fixing() {
461 // Only execute if not already fixed.
462 if (!fixed) {
463 fmt::print("\033[34m_STARTING BUCKET FIXING PROCEDURE\033[0m\n");
464 fixed = true;
465
466 // Compute gap and check feasibility (if gap is negative, exit
467 // early).
468 gap = std::ceil(incumbent - (relaxation + std::min(0.0, min_red_cost)));
469 if (gap < 0) {
470 fmt::print("\033[34m_BUCKET FIXING PROCEDURE CAN'T BE EXECUTED DUE TO "
471 "GAP\033[0m\n");
472 fmt::print("gap: {}\n", gap);
473 return;
474 }
475
476 // Initialize common structures, arc scores, warm labels, etc.
477 common_initialization();
478
479 // Initialize forward and backward c_bar vectors.
480 std::vector<double> forward_cbar(fw_buckets.size(), std::numeric_limits<double>::infinity());
481 std::vector<double> backward_cbar(bw_buckets.size(), std::numeric_limits<double>::infinity());
482
483 // Run labeling algorithms for Stage Four with full mode.
484 run_labeling_algorithms<Stage::Four, Full::Full>(forward_cbar, backward_cbar);
485 // Save computed c_bar values.
486 fw_c_bar = forward_cbar;
487 bw_c_bar = backward_cbar;
488
489 // Print info about bucket arc elimination.
490 print_info("performing bucket arc elimination with theta = {}\n", gap);
491
492 // Run forward and backward sections in parallel.
493 PARALLEL_SECTIONS(
494 work, bi_sched,
495 SECTION {
496 // Section 1: Process forward direction.
497 BucketArcElimination<Direction::Forward>(gap);
498 ObtainJumpBucketArcs<Direction::Forward>();
499 },
500 SECTION {
501 // Section 2: Process backward direction.
502 BucketArcElimination<Direction::Backward>(gap);
503 ObtainJumpBucketArcs<Direction::Backward>();
504 });
505
506 // Generate arcs after elimination.
507 // generate_arcs();
508 fmt::print("\033[34m_BUCKET FIXING PROCEDURE FINISHED\033[0m\n");
509
510 just_fixed = true;
511 return;
512 }
513 just_fixed = false;
514}

523template <Stage S>
524void BucketGraph::heuristic_fixing() {
525 if (fixed) { return; }
526
527 // Reset and initialize state.
528 reset_pool();
529 reset_fixed();
530 common_initialization();
531
532 // Pre-allocate c-bar vectors with infinity.
533 std::vector<double> forward_cbar(fw_buckets.size(), std::numeric_limits<double>::infinity());
534 std::vector<double> backward_cbar(bw_buckets.size(), std::numeric_limits<double>::infinity());
535
536 gap = std::ceil(incumbent - (relaxation + std::min(0.0, min_red_cost)));
537
538 // Run the appropriate labeling algorithms based on stage
539 if (S == Stage::Three) {
540 // Run the labeling algorithms for Stage::Two with Full::Partial mode.
541 run_labeling_algorithms<Stage::Two, Full::Partial>(forward_cbar, backward_cbar);
542 } else if (S == Stage::Four) {
543 fmt::print("\033[34m_STARTING ARC FIXING PROCEDURE\033[0m\n");
544 fmt::print("gap: {}\n", gap);
545 // Run the labeling algorithms for Stage::Four with Full::Full mode.
546 run_labeling_algorithms<Stage::Four, Full::Full>(forward_cbar, backward_cbar);
547 }
548
549 // Prepare containers for grouping labels by node.
550 const size_t num_nodes = nodes.size();
551
552 // Preallocate vectors with nullptr to avoid resizing
553 std::vector<const Label *> min_fw_labels(num_nodes, nullptr);
554 std::vector<const Label *> min_bw_labels(num_nodes, nullptr);
555
556 // Find minimum cost label for each node directly from buckets
557 // without creating intermediate label maps
558 for (auto &bucket : fw_buckets) {
559 for (const Label *label : bucket.get_labels()) {
560 if (label == nullptr || label->is_dominated) continue;
561 const size_t node_id = label->node_id;
562 if (!min_fw_labels[node_id] || label->cost < min_fw_labels[node_id]->cost) {
563 min_fw_labels[node_id] = label;
564 }
565 }
566 }
567
568 for (auto &bucket : bw_buckets) {
569 for (const Label *label : bucket.get_labels()) {
570 if (label == nullptr || label->is_dominated) continue;
571 const size_t node_id = label->node_id;
572 if (!min_bw_labels[node_id] || label->cost < min_bw_labels[node_id]->cost) {
573 min_bw_labels[node_id] = label;
574 }
575 }
576 }
577
578 // Pre-compute the index of the "time" resource.
579 const size_t time_index =
580 std::distance(options.resources.begin(), std::find(options.resources.begin(), options.resources.end(), "time"));
581
582 // Precompute lookup tables for common values
583 std::vector<double> fw_time_resources(num_nodes, 0.0);
584 std::vector<double> bw_time_resources(num_nodes, 0.0);
585 std::vector<double> fw_costs(num_nodes, 0.0);
586 std::vector<double> bw_costs(num_nodes, 0.0);
587
588 for (size_t i = 0; i < num_nodes; ++i) {
589 if (min_fw_labels[i]) {
590 fw_time_resources[i] = min_fw_labels[i]->resources[time_index];
591 fw_costs[i] = min_fw_labels[i]->cost;
592 }
593 if (min_bw_labels[i]) {
594 bw_time_resources[i] = min_bw_labels[i]->resources[time_index];
595 bw_costs[i] = min_bw_labels[i]->cost;
596 }
597 }
598
599 // Cache node data to avoid repeated lookups
600 std::vector<double> node_durations(num_nodes);
601 std::vector<std::vector<double>> node_consumptions(num_nodes, std::vector<double>(options.resources.size()));
602
603 for (size_t i = 0; i < num_nodes; ++i) {
604 const VRPNode &node = nodes[i];
605 node_durations[i] = node.duration;
606 for (size_t r = 0; r < options.resources.size(); ++r) { node_consumptions[i][r] = node.consumption[r]; }
607 }
608
609 size_t num_fixes = 0;
610 const double gap_threshold = gap; // Cache this value
611
612 // Build a list of candidate node pairs to check
613 struct NodePair {
614 size_t i;
615 size_t j;
616 };
617
618 std::vector<NodePair> candidate_pairs;
619 candidate_pairs.reserve(num_nodes * num_nodes / 4); // Estimate: 25% of pairs will be candidates
620
621 for (size_t i = 0; i < num_nodes; ++i) {
622 if (!min_fw_labels[i]) continue;
623
624 for (size_t j = 0; j < num_nodes; ++j) {
625 if (i == j || !min_bw_labels[j]) continue;
626 candidate_pairs.push_back({i, j});
627 }
628 }
629
630 // Process candidate pairs in batches to improve cache locality
631 const size_t BATCH_SIZE = 256;
632 std::vector<std::pair<size_t, size_t>> arcs_to_fix;
633 arcs_to_fix.reserve(BATCH_SIZE);
634
635 for (size_t start = 0; start < candidate_pairs.size(); start += BATCH_SIZE) {
636 const size_t end = std::min(start + BATCH_SIZE, candidate_pairs.size());
637 arcs_to_fix.clear();
638
639 for (size_t idx = start; idx < end; ++idx) {
640 const auto &pair = candidate_pairs[idx];
641 const size_t i = pair.i;
642 const size_t j = pair.j;
643
644 const double cost = getcij(min_fw_labels[i]->node_id, min_bw_labels[j]->node_id);
645
646 // Early rejection using precomputed time resources (often most
647 // restrictive)
648 if (options.resources[time_index] == "time") {
649 if (fw_time_resources[i] + cost + node_durations[min_fw_labels[i]->node_id] > bw_time_resources[j]) {
650 continue;
651 }
652 }
653
654 // Check all other resources
655 bool violated = false;
656 const size_t fw_node_id = min_fw_labels[i]->node_id;
657
658 for (size_t r = 0; r < options.resources.size(); ++r) {
659 if (r != time_index) {
660 if (min_fw_labels[i]->resources[r] + node_consumptions[fw_node_id][r] >
661 min_bw_labels[j]->resources[r]) {
662 violated = true;
663 break;
664 }
665 }
666 }
667
668 // If all resource constraints are satisfied and the cost exceeds
669 // the gap, fix the arc
670 if (!violated && (fw_costs[i] + cost + bw_costs[j] > gap_threshold)) {
671 arcs_to_fix.emplace_back(i, j);
672 ++num_fixes;
673 }
674 }
675
676 // Fix arcs in batch
677 for (const auto &[i, j] : arcs_to_fix) { fix_arc(i, j); }
678 }
679
680 if (S == Stage::Four) {
681 fmt::print("Applied {} arc fixes\n", num_fixes);
682 fmt::print("\033[34m_ARC FIXING PROCEDURE FINISHED\033[0m\n");
683 }
684}

686template <Symmetry SYM>
687void BucketGraph::set_adjacency_list() {
688 // === Step 1: Clear Existing Arcs from Each Node ===
689 for (auto &node : nodes) { node.clear_arcs(); }
690
691 // === Step 2: MST-Based Clustering ===
692 MST mst_solver(nodes, [this](int from, int to) { return getcij(from, to); });
693 auto mst = mst_solver.compute_mst();
694
695 // Collect and sort MST edge weights.
696 std::vector<double> edge_weights;
697 edge_weights.reserve(mst.size());
698 for (const auto &[weight, from, to] : mst) { edge_weights.push_back(weight); }
699 pdqsort(edge_weights.begin(), edge_weights.end());
700
701 // Compute theta as the 90th percentile (scaled down by 100).
702 size_t p90_index = static_cast<size_t>(0.9 * edge_weights.size());
703 double theta = (edge_weights[p90_index]) / 100.0;
704 print_info("Computed theta: {}\n", theta);
705
706 // Cluster nodes and build mapping: job_id -> cluster_id.
707 auto clusters = mst_solver.cluster(theta);
708 print_info("Number of clusters: {}\n", clusters.size());
709 std::vector<int> job_to_cluster(nodes.size(), -1);
710 for (int cluster_id = 0; cluster_id < clusters.size(); ++cluster_id) {
711 for (int job : clusters[cluster_id]) { job_to_cluster[job] = cluster_id; }
712 }
713
714 // === Step 3: Precompute Resource Information ===
715 // Find the index for the "time" resource.
716 auto time_it = std::find(options.resources.begin(), options.resources.end(), "time");
717 const int time_resource_idx =
718 (time_it != options.resources.end()) ? std::distance(options.resources.begin(), time_it) : -1;
719
720 // === Step 4: Process Arcs for Each Node ===
721
722 // Helper lambda to compute resource increments.
723 // Depending on symmetry, consumption is either taken as-is or averaged.
724 auto compute_resource_increments = [&](const VRPNode &from_node, const VRPNode &to_node, double travel_cost,
725 std::vector<double> &inc) {
726 const int nres = options.resources.size();
727 if constexpr (SYM == Symmetry::Asymmetric) {
728 for (int r = 0; r < nres; ++r) {
729 inc[r] = (r == time_resource_idx) ? travel_cost + from_node.duration : from_node.consumption[r];
730 }
731 } else {
732 for (int r = 0; r < nres; ++r) {
733 inc[r] = (r == time_resource_idx) ? (from_node.duration / 2.0 + travel_cost + to_node.duration / 2.0)
734 : ((from_node.consumption[r] + to_node.consumption[r]) / 2.0);
735 }
736 }
737 };
738
739 // Lambda to add forward and reverse arcs from a given node.
740 auto add_arcs_for_node = [&](const VRPNode &curr_node) {
741 // Containers to accumulate arcs before insertion.
742 using ArcTuple = std::tuple<double, int, std::vector<double>, double>;
743 std::vector<ArcTuple> forward_arcs;
744 std::vector<ArcTuple> reverse_arcs;
745 forward_arcs.reserve(nodes.size());
746 reverse_arcs.reserve(nodes.size());
747
748 // Iterate over candidate next nodes.
749 for (const auto &next_node : nodes) {
750 // Skip if the candidate is a depot, the same node, or a pstep
751 // depot.
752 if (next_node.id == options.depot || curr_node.id == next_node.id ||
753 (options.pstep && (next_node.id == options.pstep_depot || next_node.id == options.pstep_end_depot))) {
754 continue;
755 }
756
757 double travel_cost = getcij(curr_node.id, next_node.id);
758 double cost_inc = travel_cost - next_node.cost;
759
760 // Skip self-loops (or if bucket indexes match; here using node
761 // id as bucket id).
762 if (curr_node.id == next_node.id) continue;
763
764 // Compute resource increments.
765 std::vector<double> res_inc(options.resources.size());
766 compute_resource_increments(curr_node, next_node, travel_cost, res_inc);
767
768 // Check feasibility: ensure resource consumption does not
769 // exceed next node's upper bound.
770 bool feasible = true;
771 for (int r = 0; r < options.resources.size(); ++r) {
772 if (numericutils::gt(curr_node.lb[r] + res_inc[r], next_node.ub[r])) {
773 feasible = false;
774 break;
775 }
776 }
777 if (!feasible) continue;
778
779 // Compute priorities based on cluster membership.
780 bool same_cluster = (job_to_cluster[curr_node.id] == job_to_cluster[next_node.id]);
781 double forward_priority = (same_cluster ? 5.0 : 1.0) + 1.E-5 * next_node.start_time;
782 double reverse_priority = (same_cluster ? 1.0 : 5.0) + 1.E-5 * curr_node.start_time;
783
784 // Collect arcs for forward and reverse directions.
785 forward_arcs.emplace_back(forward_priority, next_node.id, res_inc, cost_inc);
786 reverse_arcs.emplace_back(reverse_priority, next_node.id, res_inc, cost_inc);
787 }
788
789 // Insert the arcs into the corresponding nodes.
790 for (const auto &[priority, to_node, res_inc, cost_inc] : forward_arcs) {
791 nodes[curr_node.id].template add_arc<Direction::Forward>(curr_node.id, to_node, res_inc, cost_inc,
792 priority);
793 }
794 for (const auto &[priority, from_node, res_inc, cost_inc] : reverse_arcs) {
795 nodes[from_node].template add_arc<Direction::Backward>(from_node, curr_node.id, res_inc, cost_inc,
796 priority);
797 }
798 };
799
800 // Process each node, skipping the end depot.
801 for (const auto &node : nodes) {
802 if (node.id != options.end_depot) { add_arcs_for_node(node); }
803 }
804}

812void BucketGraph::common_initialization() {
813 // Retrieve active cuts and build a vector of ActiveCutInfo.
814 auto &cutter = cut_storage;
815 const auto active_cuts = cutter->getActiveCuts();
816 std::vector<ActiveCutInfo> active_cuts_info;
817 active_cuts_info.reserve(active_cuts.size());
818 for (const auto &cut : active_cuts) { active_cuts_info.push_back(cut); }
819
820 // --- Sort arcs by scores for each node ---
821 auto &arc_scores = assign_buckets<D>(fw_arc_scores, bw_arc_scores);
822 for (auto &node : nodes) {
823 if (!arc_scores[node.id].empty()) { node.sort_arcs_by_scores<D>(arc_scores[node.id], nodes, active_cuts_info); }
824 arc_scores[node.id].clear();
825 }
826
827 auto &warm_labels = assign_buckets<D>(fw_warm_labels, bw_warm_labels);
828 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
829 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
830 auto &c_bar = assign_buckets<D>(fw_c_bar, bw_c_bar);
831 auto &bucket_splits = assign_buckets<D>(fw_bucket_splits, bw_bucket_splits);
832
833 // --- Initialize c_bar vector ---
834 c_bar.resize(buckets_size, std::numeric_limits<double>::infinity());
835
836 // --- Initialize dominance check counters ---
837 if constexpr (Direction::Forward == D) {
838 dominance_checks_per_bucket.assign(buckets_size + 1, 0);
839 non_dominated_labels_per_bucket = 0;
840 } else {
841 non_dominated_labels_per_bucket_bw = 0;
842 }
843
844 // --- Clear all buckets ---
845 for (size_t b = 0; b < buckets_size; b++) { buckets[b].clear(); }
846
847 // --- Warm start processing: sort warm_labels and compute reduced costs
848 // ---
849 if (options.warm_start && !just_fixed && !warm_labels.empty()) {
850 pdqsort(warm_labels.begin(), warm_labels.end(),
851 [](const WarmLabelState &a, const WarmLabelState &b) { return a.cost < b.cost; });
852 const size_t process_size =
853 std::min(warm_labels.size(), static_cast<size_t>(options.n_warm_start * warm_labels.size()));
854 for (size_t i = 0; i < process_size; ++i) {
855 auto new_label = compute_red_cost(warm_labels[i], D == Direction::Forward);
856 if (new_label != nullptr) {
857 if constexpr (D == Direction::Forward) {
858 fw_buckets[new_label->vertex].add_sorted_label(new_label);
859 } else {
860 bw_buckets[new_label->vertex].add_sorted_label(new_label);
861 }
862 }
863 }
864 warm_labels.clear();
865 }
866
867 const int depot_id = (D == Direction::Forward) ? options.depot : options.end_depot;
868 auto &depot_node = nodes[depot_id];
869
870 std::vector<double> initial_resources(options.main_resources.size(), 0.0);
871 for (size_t r = 0; r < initial_resources.size(); ++r) {
872 initial_resources[r] = (D == Direction::Forward) ? depot_node.lb[r] : depot_node.ub[r];
873 }
874
875 const int initial_bucket = get_bucket_number<D>(depot_id, initial_resources);
876 auto &label_pool = assign_buckets<D>(label_pool_fw, label_pool_bw);
877 auto depot_label = label_pool->acquire(initial_bucket);
878 depot_label->initialize(initial_bucket, 0.0, initial_resources, depot_id);
879 depot_label->is_extended = false;
880 depot_label->addNode(depot_id);
881 set_node_visited(depot_label->visited_bitmap, depot_id);
882 SRC_MODE_BLOCK(depot_label->SRCmap.assign(cut_storage->activeSize(), 0);)
883 buckets[initial_bucket].add_label(depot_label);
884 buckets[initial_bucket].node_id = depot_id;
885}

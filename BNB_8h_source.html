26enum class BNBNodeSelectionStrategy {
27 DFS, // Depth-First Search
28 BFS, // Breadth-First Search
29 BestFirst // Best-First Search based on the boundValue

32struct BNBNodeCompare {
33 bool operator()(const BNBNode *lhs, const BNBNode *rhs) const {
34 return lhs > rhs; // Min heap for Best-First Search
35 }

53 void addBNBNode(BNBNode *&node) {
54 if (strategy == BNBNodeSelectionStrategy::BestFirst) {
55 std::lock_guard<std::mutex> lock(
56 bestFirstMutex); // Protect bestFirst queue
57 bestFirstBNBNodes.push(node);
58 } else if (strategy == BNBNodeSelectionStrategy::DFS) {
59 std::lock_guard<std::mutex> lock(bestFirstMutex);
60 dfsBNBNodes.push_front(node); // Push to front for LIFO order
61 } else {
62 otherBNBNodes.enqueue(node);
63 }
64 }
65
66 BNBNode *getNextBNBNode() {
67 BNBNode *node = nullptr;
68
69 if (strategy == BNBNodeSelectionStrategy::BestFirst) {
70 std::lock_guard<std::mutex> lock(
71 bestFirstMutex); // Protect bestFirst queue
72 if (!bestFirstBNBNodes.empty()) {
73 node = bestFirstBNBNodes.top();
74 bestFirstBNBNodes.pop();
75 return node;
76 }
77 } else if (strategy == BNBNodeSelectionStrategy::DFS) {
78 std::lock_guard<std::mutex> lock(bestFirstMutex);
79 if (!dfsBNBNodes.empty()) {
80 node = dfsBNBNodes.front(); // Pop from the front
81 dfsBNBNodes.pop_front();
82 return node;
83 }
84 } else {
85 otherBNBNodes.try_dequeue(node);
86 }
87
88 return nullptr; // No node available
89 }

118 void updateGlobalBest(double objectiveValue, BNBNode *currentBNBNode,
119 double boundValue) {
120 // Use atomic compare and swap for updating global best objective
121 double prevGlobalBest =
122 globalBestObjective.load(std::memory_order_acquire);
123 while (objectiveValue > prevGlobalBest &&
124 !globalBestObjective.compare_exchange_weak(
125 prevGlobalBest, objectiveValue, std::memory_order_release)) {
126 // Repeat until update is successful
127 }
128 print_info("Global best: {}\n",
129 globalBestObjective.load(std::memory_order_acquire) / 10);
130
131 if ((std::floor(boundValue) ==
132 globalBestObjective.load(std::memory_order_acquire)) &&
133 boundValue > 0) {
134 markSolutionFound(); // If bound equals the best objective, mark
135 // solution found
136 }
137 }

139 void processBNBNode(BNBNode *currentBNBNode) {
140 if (isSolutionFound()) return; // Early exit if solution already found
141
142 currentBNBNode->start();
143 currentBNBNode->enforceBranching();
144 problem->evaluate(currentBNBNode);
145 double boundValue = problem->bound(currentBNBNode);
146
147 if (isSolutionFound())
148 return; // Check again after a potentially long operation
149
150 if (currentBNBNode->getPrune() ||
151 boundValue < globalBestObjective.load()) {
152 return; // BNBNode can be pruned
153 }
154
155 double objectiveValue = problem->objective(currentBNBNode);
156 updateGlobalBest(objectiveValue, currentBNBNode, boundValue);
157
158 if (isSolutionFound())
159 return; // Exit if the solution was found during update
160
161 branch(currentBNBNode);
162 }

212 void solve() {
213 // init timer
214 fmt::print("\n");
215 fmt::print("\033[34m_STARTING BnB \033[0m");
216 fmt::print("\n");
217 auto start = std::chrono::high_resolution_clock::now();
218 while (BNBNode *currentBNBNode = getNextBNBNode()) {
219 currentBNBNode->start();
220 currentBNBNode->enforceBranching();
221 print_info("Solving for node {} with depth: {}\n",
222 currentBNBNode->getUUID(), currentBNBNode->getDepth());
223
224 problem->evaluate(currentBNBNode);
225 double boundValue = problem->bound(currentBNBNode);
226 print_info("Bound value: {}\n", boundValue / 10);
227
228 if (currentBNBNode->getPrune()) {
229 print_info("Pruned node (feasibility): {}\n",
230 currentBNBNode->getUUID());
231 continue;
232 }
233
234 if (boundValue > globalBestObjective.load()) {
235 print_branching("Pruned node (bound): {}\n",
236 currentBNBNode->getUUID());
237 // print globalBestObjective.load
238 print_branching("Global best: {}\n",
239 globalBestObjective.load() / 10);
240 continue;
241 }
242
243 auto objectiveValue = problem->objective(currentBNBNode);
244 print_info("Objective value: {}\n", objectiveValue / 10);
245
246 if (std::abs(boundValue - objectiveValue) < 1e-2) {
247 fmt::print("\n");
248 fmt::print("\033[34m_SOLUTION FOUND \033[0m: {}\n",
249 objectiveValue / 10);
250 fmt::print("\n");
251
252 problem->printSolution(currentBNBNode);
253
254 globalBestObjective.store(objectiveValue,
255 std::memory_order_release);
256 markSolutionFound();
257 break;
258 }
259
260 if (objectiveValue > globalBestObjective.load()) {
261 globalBestObjective.store(objectiveValue,
262 std::memory_order_release);
263 }
264
265 branch(currentBNBNode);
266 // print currentBNBNode->getDepth
267 }
268 auto end = std::chrono::high_resolution_clock::now();
269 std::chrono::duration<double> elapsed = end - start;
270 print_info("Elapsed time: {:.2f}\n", elapsed.count());
271 Logger::log("Elapsed time: {:.2f}\n", elapsed.count());
272 }

99 std::vector<int> topologicalOrderOfSCCs(const std::vector<std::vector<int>> &sccs) {
100 ankerl::unordered_dense::map<int, int> nodeToScc;
101 for (int i = 0; i < sccs.size(); ++i) {
102 for (int node : sccs[i]) { nodeToScc[node] = i; }
103 }
104
105 std::vector<ankerl::unordered_dense::set<int>> componentGraph(sccs.size());
106 for (int v = 0; v < adj.size(); ++v) {
107 for (int w : adj[v]) {
108 int sccV = nodeToScc[v];
109 int sccW = nodeToScc[w];
110 if (sccV != sccW) { componentGraph[sccV].insert(sccW); }
111 }
112 }
113
114 std::vector<int> inDegree(sccs.size(), 0);
115 for (const auto &neighbors : componentGraph) {
116 for (int neighbor : neighbors) { ++inDegree[neighbor]; }
117 }
118
119 std::stack<int> zeroInDegreeNodes;
120 for (int i = 0; i < inDegree.size(); ++i) {
121 if (inDegree[i] == 0) { zeroInDegreeNodes.push(i); }
122 }
123
124 std::vector<int> topologicalOrder;
125 while (!zeroInDegreeNodes.empty()) {
126 int node = zeroInDegreeNodes.top();
127 zeroInDegreeNodes.pop();
128 topologicalOrder.push_back(node);
129
130 for (int neighbor : componentGraph[node]) {
131 --inDegree[neighbor];
132 if (inDegree[neighbor] == 0) { zeroInDegreeNodes.push(neighbor); }
133 }
134 }

201 void strongConnect(int v, std::vector<std::vector<int>> &sccs) {
202 index[v] = currentIndex;
203 lowlink[v] = currentIndex;
204 currentIndex++;
205 stack.push(v);
206 onStack[v] = true;
207
208 for (int w : adj[v]) {
209 if (index[w] == -1) {
210 strongConnect(w, sccs);
211 lowlink[v] = std::min(lowlink[v], lowlink[w]);
212 } else if (onStack[w]) {
213 lowlink[v] = std::min(lowlink[v], index[w]);
214 }
215 }
216
217 if (lowlink[v] == index[v]) {
218 std::vector<int> scc;
219 int w;
220 do {
221 w = stack.top();
222 stack.pop();
223 onStack[w] = false;
224 scc.push_back(w);
225 } while (w != v);
226 sccs.push_back(scc);
227 }
228 }

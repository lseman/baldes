626 template <Direction D>
627 void split_buckets() {
628 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
629 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
630 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
631 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
632
633 int num_intervals = options.main_resources.size();
634 std::vector<double> total_ranges(num_intervals);
635 std::vector<double> base_intervals(num_intervals);
636
637 // Determine the base interval and other relevant values for each resource
638 for (int r = 0; r < num_intervals; ++r) {
639 total_ranges[r] = R_max[r] - R_min[r];
640 base_intervals[r] = total_ranges[r] / intervals[r].interval;
641 }
642
643 int original_num_buckets = buckets_size;
644
645 // To track splits per node_id
646 ankerl::unordered_dense::map<int, int> splits_per_node;
647
648 // Loop until the second-to-last bucket to avoid out-of-bounds access
649 for (int i = 0; i < original_num_buckets - 1; ++i) {
650 const auto &bucket = buckets[i];
651
652 // Calculate mid-point for each dimension
653 std::vector<double> mid_point(bucket.lb.size());
654 if constexpr (D == Direction::Forward) {
655 for (int r = 0; r < bucket.lb.size(); ++r) { mid_point[r] = (bucket.lb[r] + base_intervals[r]) / 2.0; }
656 } else {
657 for (int r = 0; r < bucket.lb.size(); ++r) { mid_point[r] = (bucket.ub[r] - base_intervals[r]) / 2.0; }
658 }
659
660 // Create two new buckets by splitting at the mid-point
661 Bucket bucket1, bucket2;
662 if constexpr (D == Direction::Forward) {
663 bucket1 = Bucket(bucket.node_id, bucket.lb, bucket.ub);
664 bucket2 = Bucket(bucket.node_id, mid_point, bucket.ub);
665 } else {
666 bucket1 = Bucket(bucket.node_id, bucket.lb, bucket.ub);
667 bucket2 = Bucket(bucket.node_id, bucket.lb, mid_point);
668 }
669
670 // Manually shift elements to the right to make space for the new bucket
671 for (int j = original_num_buckets; j > i + 1; --j) { buckets[j] = buckets[j - 1]; }
672
673 // Insert the new bucket at position i+1
674 buckets[i + 1] = bucket2;
675
676 // Replace the current bucket with the first half (bucket1)
677 buckets[i] = bucket1;
678
679 // Track the number of splits for this node_id
680 splits_per_node[bucket.node_id]++;
681
682 // Since we added a new bucket at i+1, increment i to skip over the new bucket
683 ++i;
684 ++original_num_buckets; // Adjust the count to account for the new bucket
685 }
686
687 // Update the global bucket size
688 buckets_size = original_num_buckets;
689
690 // Now update the num_buckets_index for each node_id based on the final positions
691 int current_index = 0;
692 for (int i = 0; i < original_num_buckets; ++i) {
693 const auto &bucket = buckets[i];
694 if (num_buckets_index[bucket.node_id] == -1) { num_buckets_index[bucket.node_id] = current_index; }
695 current_index++;
696 }
697 }

658 template <Direction D>
659 void split_buckets() {
660 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
661 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
662 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
663 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
664
665 int num_intervals = options.main_resources.size();
666 std::vector<double> total_ranges(num_intervals);
667 std::vector<double> base_intervals(num_intervals);
668
669 // Determine the base interval and other relevant values for
670 // each resource
671 for (int r = 0; r < num_intervals; ++r) {
672 total_ranges[r] = R_max[r] - R_min[r];
673 base_intervals[r] = total_ranges[r] / intervals[r].interval;
674 }
675
676 int original_num_buckets = buckets_size;
677
678 // To track splits per node_id
679 ankerl::unordered_dense::map<int, int> splits_per_node;
680
681 // Loop until the second-to-last bucket to avoid
682 // out-of-bounds access
683 for (int i = 0; i < original_num_buckets - 1; ++i) {
684 const auto &bucket = buckets[i];
685
686 // Calculate mid-point for each dimension
687 std::vector<double> mid_point(bucket.lb.size());
688 if constexpr (D == Direction::Forward) {
689 for (int r = 0; r < bucket.lb.size(); ++r) { mid_point[r] = (bucket.lb[r] + base_intervals[r]) / 2.0; }
690 } else {
691 for (int r = 0; r < bucket.lb.size(); ++r) { mid_point[r] = (bucket.ub[r] - base_intervals[r]) / 2.0; }
692 }
693
694 // Create two new buckets by splitting at the mid-point
695 Bucket bucket1, bucket2;
696 if constexpr (D == Direction::Forward) {
697 bucket1 = Bucket(bucket.node_id, bucket.lb, bucket.ub);
698 bucket2 = Bucket(bucket.node_id, mid_point, bucket.ub);
699 } else {
700 bucket1 = Bucket(bucket.node_id, bucket.lb, bucket.ub);
701 bucket2 = Bucket(bucket.node_id, bucket.lb, mid_point);
702 }
703
704 // Manually shift elements to the right to make space
705 // for the new bucket
706 for (int j = original_num_buckets; j > i + 1; --j) { buckets[j] = buckets[j - 1]; }
707
708 // Insert the new bucket at position i+1
709 buckets[i + 1] = bucket2;
710
711 // Replace the current bucket with the first half
712 // (bucket1)
713 buckets[i] = bucket1;
714
715 // Track the number of splits for this node_id
716 splits_per_node[bucket.node_id]++;
717
718 // Since we added a new bucket at i+1, increment i to
719 // skip over the new bucket
720 ++i;
721 ++original_num_buckets; // Adjust the count to account
722 // for the new bucket
723 }
724
725 // Update the global bucket size
726 buckets_size = original_num_buckets;
727
728 // Now update the num_buckets_index for each node_id based
729 // on the final positions
730 int current_index = 0;
731 for (int i = 0; i < original_num_buckets; ++i) {
732 const auto &bucket = buckets[i];
733 if (num_buckets_index[bucket.node_id] == -1) { num_buckets_index[bucket.node_id] = current_index; }
734 current_index++;
735 }
736 }

846 void updateSplit() {
847 // Base thresholds and parameters for adjustments.
848 constexpr double BASE_IMBALANCE_THRESHOLD = 0.15; // Slightly lower base threshold.
849 constexpr double MIN_ADJUSTMENT_FACTOR = 0.02;
850 constexpr double MAX_ADJUSTMENT_FACTOR = 0.10;
851 constexpr double LEARNING_RATE_DECAY = 0.99; // Decay factor for learning rate.
852 constexpr double EMA_ALPHA = 0.1; // Smoothing factor for EMA.
853
854 // Static variables to persist state across calls.
855 static double learning_rate = MAX_ADJUSTMENT_FACTOR;
856 static double ema_imbalance_ratio = 0.0;
857
858 // Convert label counts to double for ratio computations.
859 const double fw_labels = static_cast<double>(n_fw_labels);
860 const double bw_labels = static_cast<double>(n_bw_labels);
861 const double total_labels = fw_labels + bw_labels;
862
863 // Skip adjustment if too few labels exist.
864 if (total_labels < 10) return;
865
866 // Compute imbalance ratios for forward and backward labels.
867 const double fw_ratio = fw_labels / total_labels;
868 const double bw_ratio = bw_labels / total_labels;
869 const double imbalance_ratio = std::abs(fw_ratio - bw_ratio);
870
871 // Update exponential moving average (EMA) of the imbalance ratio.
872 ema_imbalance_ratio = EMA_ALPHA * imbalance_ratio + (1.0 - EMA_ALPHA) * ema_imbalance_ratio;
873
874 // Dynamic threshold decreases as total_labels increase.
875 const double dynamic_threshold = BASE_IMBALANCE_THRESHOLD * std::exp(-total_labels / 1000.0);
876
877 // Only adjust split if the imbalance exceeds the dynamic threshold.
878 if (ema_imbalance_ratio > dynamic_threshold) {
879 // Severity scales between 0 and 1.
880 const double severity = (ema_imbalance_ratio - dynamic_threshold) / (1.0 - dynamic_threshold);
881
882 // Adaptive adjustment factor increases with severity.
883 const double adjustment_factor =
884 MIN_ADJUSTMENT_FACTOR + (MAX_ADJUSTMENT_FACTOR - MIN_ADJUSTMENT_FACTOR) * severity;
885
886 // Decay the learning rate.
887 learning_rate *= LEARNING_RATE_DECAY;
888
889 // Compute resource range based on the main resource.
890 const double resource_range = R_max[options.main_resources[0]] - R_min[options.main_resources[0]];
891
892 // Base adjustment scales with the resource range and learning rate.
893 const double base_adjustment = adjustment_factor * resource_range * learning_rate;
894
895 // Adjust each split position in q_star.
896 for (size_t i = 0; i < q_star.size(); ++i) {
897 // Compute relative position (0: min, 1: max) within the
898 // resource range.
899 const double relative_pos = (q_star[i] - R_min[options.main_resources[0]]) / resource_range;
900
901 // Position factor gives smaller adjustments near boundaries.
902 const double position_factor = 4.0 * relative_pos * (1.0 - relative_pos);
903
904 // Determine direction: shift toward balancing label counts.
905 const double direction = (bw_labels > fw_labels) ? 1.0 : -1.0;
906
907 // Final adjustment for this split.
908 const double final_adjustment = base_adjustment * position_factor * direction;
909
910 // Apply the adjustment with boundary protection.
911 q_star[i] = std::clamp(q_star[i] + final_adjustment,
912 R_min[options.main_resources[0]] + resource_range * 0.1, // Lower bound.
913 R_max[options.main_resources[0]] - resource_range * 0.1 // Upper bound.
914 );
915 }
916
917 // Log adjustments if the base adjustment is significant.
918 if (std::abs(base_adjustment) > resource_range * 0.05) {
919 print_info("Split adjustment: imbalance={:.3f}, adjustment={:.3f}, "
920 "learning_rate={:.3f}\n",
921 ema_imbalance_ratio, base_adjustment, learning_rate);
922 }
923 }

1102 // Check main resource feasibility.
1103 const double main_resource = L.resources[options.main_resources[0]];
1104 const double q_star_value = q_star[options.main_resources[0]];
1105 if (fw) {

1108 if (main_resource <= q_star_value) return nullptr;

1111 // Ensure sufficient nodes in the route.
1112 if (L.path_len <= 3) return nullptr;
1113
1114 // Initialize an empty SRCmap vector with size equal to the current
1115 // number of cuts.
1116 std::vector<uint16_t> updated_SRCmap(cut_storage->size(), 0);

1120 auto new_label = pool->acquire();
1121 auto &route = new_label->nodes_covered;
1122 route.clear();
1123 route.reserve(L.nodes_covered.size());
1124 route.insert(route.end(), L.nodes_covered.begin(), L.nodes_covered.end());
1125 new_label->path_len = L.path_len;
1126
1127 // Variables to keep track of costs.
1128 int last_node = -1;
1129
1130 // Traverse the nodes in the route (from current label to root).
1131 for (auto node_id : L.nodes_covered) {
1132 // --- NG-route Feasibility Check ---
1133 // Clear visited bitmap in the neighborhood of node_id.
1134 for (size_t seg = 0, limit = new_label->visited_bitmap.size(); seg < limit; ++seg) {

1172 if (baseSet[segment] & bit_mask) {

1176 red_cost -= SRCDuals[idx];
1177 src_val -= multipliers.den;
1178 }
1179 }
1180 }
1181#endif
1182

1184#if defined(RCC) || defined(EXACT_RCC)

1186#endif
1187 if (branching_duals->size() > 0 && last_node != -1) {
1188 red_cost -= branching_duals->getDual(last_node, node_id);
1189 }
1190
1191 last_node = node_id; // Update last_node.
1192 }
1193
1194 // Set final label properties.
1195 new_label->cost = red_cost;
1196 new_label->real_cost = real_cost;
1197 new_label->parent = nullptr;
1198 new_label->node_id = L.node_id;
1199 new_label->is_extended = false;
1200 new_label->resources = L.resources;
1201 new_label->SRCmap = updated_SRCmap;
1202 new_label->fresh = false;
1203
1204 // Determine bucket number for the new label.
1205 std::vector<double> new_resources(options.resources.size());
1206 for (size_t i = 0; i < options.resources.size(); ++i) { new_resources[i] = new_label->resources[i]; }

1208 : get_bucket_number<Direction::Backward>(new_label->node_id, new_resources);

1210

1447 node_interval_trees[VRPNode.id] = node_tree;
1448 }

1451 buckets_size = buckets.size();

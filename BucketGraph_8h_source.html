648 bool shouldAttemptEnumeration(double lower_bound, bool cuts_stable, int exact_iterations,

651 if (s5 || stage >= 5) return false;
652 if (!cuts_stable) return false;

654 const double rel_gap = relative_gap_to_incumbent(lower_bound);

860 template <Direction D>
861 void split_buckets() {
862 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
863 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
864 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
865 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
866
867 int num_intervals = options.main_resources.size();
868 std::vector<double> total_ranges(num_intervals);
869 std::vector<double> base_intervals(num_intervals);
870
871 // Determine the base interval and other relevant values for
872 // each resource
873 for (int r = 0; r < num_intervals; ++r) {
874 total_ranges[r] = R_max[r] - R_min[r];
875 base_intervals[r] = total_ranges[r] / intervals[r].interval;
876 }
877
878 int original_num_buckets = buckets_size;
879
880 // To track splits per node_id
881 ankerl::unordered_dense::map<int, int> splits_per_node;
882
883 // Loop until the second-to-last bucket to avoid
884 // out-of-bounds access
885 for (int i = 0; i < original_num_buckets - 1; ++i) {
886 const auto &bucket = buckets[i];
887
888 // Calculate mid-point for each dimension
889 std::vector<double> mid_point(bucket.lb.size());
890 if constexpr (D == Direction::Forward) {
891 for (int r = 0; r < bucket.lb.size(); ++r) { mid_point[r] = (bucket.lb[r] + base_intervals[r]) / 2.0; }
892 } else {
893 for (int r = 0; r < bucket.lb.size(); ++r) { mid_point[r] = (bucket.ub[r] - base_intervals[r]) / 2.0; }
894 }
895
896 // Create two new buckets by splitting at the mid-point
897 Bucket bucket1, bucket2;
898 if constexpr (D == Direction::Forward) {
899 bucket1 = Bucket(bucket.node_id, bucket.lb, bucket.ub);
900 bucket2 = Bucket(bucket.node_id, mid_point, bucket.ub);
901 } else {
902 bucket1 = Bucket(bucket.node_id, bucket.lb, bucket.ub);
903 bucket2 = Bucket(bucket.node_id, bucket.lb, mid_point);
904 }
905
906 // Manually shift elements to the right to make space
907 // for the new bucket
908 for (int j = original_num_buckets; j > i + 1; --j) { buckets[j] = buckets[j - 1]; }
909
910 // Insert the new bucket at position i+1
911 buckets[i + 1] = bucket2;
912
913 // Replace the current bucket with the first half
914 // (bucket1)
915 buckets[i] = bucket1;
916
917 // Track the number of splits for this node_id
918 splits_per_node[bucket.node_id]++;
919
920 // Since we added a new bucket at i+1, increment i to
921 // skip over the new bucket
922 ++i;
923 ++original_num_buckets; // Adjust the count to account
924 // for the new bucket
925 }
926
927 // Update the global bucket size
928 buckets_size = original_num_buckets;
929
930 // Now update the num_buckets_index for each node_id based
931 // on the final positions
932 int current_index = 0;
933 for (int i = 0; i < original_num_buckets; ++i) {
934 const auto &bucket = buckets[i];
935 if (num_buckets_index[bucket.node_id] == -1) { num_buckets_index[bucket.node_id] = current_index; }
936 current_index++;
937 }
938 }

1010 inline bool check_feasibility(const Label *fw_label, const Label *bw_label,
1011 const SpliceState *splice_state = nullptr) {
1012 if (!fw_label || !bw_label) return false;
1013
1014 // Cache resources and node data.
1015 const auto &fw_resources = splice_state ? splice_state->resources : fw_label->resources;
1016 const auto &bw_resources = bw_label->resources;
1017 const auto &node = nodes[fw_label->node_id];
1018
1019 const size_t n_resources = options.resources.size();
1020
1021 // Compute travel time only once.
1022 const auto travel_time = getcij(fw_label->node_id, bw_label->node_id);
1023
1024 // Assuming the "time" resource is at a known index (e.g., index 0).
1025 // If not, you can either iterate to find it once or precompute a
1026 // boolean array.
1027 if (n_resources > 0 && options.resources[0] == "time") {
1028 // Check time feasibility.
1029 const double fw_time = splice_state ? fw_resources[0] : fw_resources[0] + travel_time + node.duration;
1030 if (numericutils::gt(fw_time, bw_resources[0])) return false;
1031 } else {
1032 // Fallback: iterate to check any resource named "time".
1033 for (size_t r = 0; r < n_resources; ++r) {
1034 if (options.resources[r] == "time") {
1035 const double fw_time =
1036 splice_state ? fw_resources[r] : fw_resources[r] + travel_time + node.duration;
1037 if (numericutils::gt(fw_time, bw_resources[r])) return false;
1038 }
1039 }
1040 }
1041
1042 // Check additional resource feasibility (assuming resources[0] is
1043 // "time") and additional resources (from index 1 onward) represent
1044 // capacity constraints.
1045 for (size_t i = 1; i < n_resources; ++i) {
1046 const double fw_resource = splice_state ? fw_resources[i] : fw_resources[i] + node.demand;
1047 if (fw_resource > bw_resources[i]) return false;
1048 }
1049
1050 return true;
1051 }

1053 void updateSplit() {
1054 // Base thresholds and parameters for adjustments.
1055 constexpr double BASE_IMBALANCE_THRESHOLD = 0.10; // Lower threshold —
1056 // work-weighted signal
1057 // is more sensitive
1058 // than raw counts.
1059 constexpr double MIN_ADJUSTMENT_FACTOR = 0.002;
1060 constexpr double MAX_ADJUSTMENT_FACTOR = 0.015;
1061 constexpr double EMA_ALPHA = 0.1; // Smoothing factor for EMA.
1062 constexpr int SPLIT_UPDATE_COOLDOWN = 5;
1063
1064 // Prefer a work-weighted signal: total labels scanned during dominance
1065 // walks captures both check count and per-bin depth, which is what the
1066 // labeler actually pays for. Fall back to label counts when the
1067 // profiler hasn't recorded data yet (first iteration / profiling off).
1068 // Sum across stages because Stage 1-3 may be active early and Stage 4
1069 // dominates later iterations.
1070 uint64_t fw_work = 0, bw_work = 0;
1071 for (size_t i = 0; i < PROFILE_STAGE_COUNT; ++i) {
1072 fw_work += profile_inner_bin_scan_labels_fw[i];
1073 bw_work += profile_inner_bin_scan_labels_bw[i];
1074 }
1075 const bool have_work_signal = (fw_work + bw_work) > 0;
1076 const double fw_labels = have_work_signal ? static_cast<double>(fw_work) : static_cast<double>(n_fw_labels);
1077 const double bw_labels = have_work_signal ? static_cast<double>(bw_work) : static_cast<double>(n_bw_labels);
1078 const double total_labels = fw_labels + bw_labels;
1079
1080 // Skip adjustment if too few labels exist.
1081 if (total_labels < 10) return;
1082
1083 // Compute imbalance ratios for forward and backward labels.
1084 const double fw_ratio = fw_labels / total_labels;
1085 const double bw_ratio = bw_labels / total_labels;
1086 const double imbalance_ratio = std::abs(fw_ratio - bw_ratio);
1087
1088 // Update exponential moving average (EMA) of the imbalance ratio.
1089 split_imbalance_ema = EMA_ALPHA * imbalance_ratio + (1.0 - EMA_ALPHA) * split_imbalance_ema;
1090
1091 // Keep a real dead-band even on large passes. Letting the threshold
1092 // decay to zero makes q_star chase noise and creates concat churn.
1093 const double dynamic_threshold = std::max(0.03, BASE_IMBALANCE_THRESHOLD * std::exp(-total_labels / 50000.0));
1094 if (iter - last_split_update_iter < SPLIT_UPDATE_COOLDOWN) return;
1095
1096 // Only adjust split if the imbalance exceeds the dynamic threshold.
1097 if (split_imbalance_ema > dynamic_threshold) {
1098 // Severity scales between 0 and 1.
1099 const double severity = (split_imbalance_ema - dynamic_threshold) / (1.0 - dynamic_threshold);
1100
1101 // Adaptive adjustment factor increases with severity.
1102 const double adjustment_factor =
1103 MIN_ADJUSTMENT_FACTOR + (MAX_ADJUSTMENT_FACTOR - MIN_ADJUSTMENT_FACTOR) * severity;
1104
1105 // Compute resource range based on the main resource.
1106 const double resource_range = R_max[options.main_resources[0]] - R_min[options.main_resources[0]];
1107
1108 // Base adjustment scales with the resource range; no decay so
1109 // adjustments stay aggressive when imbalance persists late.
1110 const double base_adjustment = adjustment_factor * resource_range;
1111 last_split_update_iter = iter;
1112
1113 // Adjust each split position in q_star.
1114 for (size_t i = 0; i < q_star.size(); ++i) {
1115 // Compute relative position (0: min, 1: max) within the
1116 // resource range.
1117 const double relative_pos = (q_star[i] - R_min[options.main_resources[0]]) / resource_range;
1118
1119 // Position factor gives smaller adjustments near boundaries.
1120 const double position_factor = 4.0 * relative_pos * (1.0 - relative_pos);
1121
1122 // Determine direction: shift toward balancing label counts.
1123 const double direction = (bw_labels > fw_labels) ? 1.0 : -1.0;
1124
1125 // Final adjustment for this split.
1126 const double final_adjustment = base_adjustment * position_factor * direction;
1127
1128 // Apply the adjustment with boundary protection.
1129 q_star[i] = std::clamp(q_star[i] + final_adjustment,
1130 R_min[options.main_resources[0]] + resource_range * 0.1, // Lower bound.
1131 R_max[options.main_resources[0]] - resource_range * 0.1 // Upper bound.
1132 );
1133 }
1134
1135 // Log adjustments if the base adjustment is significant.
1136 if (std::abs(base_adjustment) > resource_range * 0.01) {
1137 print_info("Split adjustment: imbalance={:.3f}, adjustment={:.3f}\n", split_imbalance_ema,
1138 base_adjustment);
1139 }
1140 }
1141 }

1145 // Mirrors the paper's A+ bucket refinement. After each labeling pass we
1146 // sample dominance_checks / non_dominant_labels and fold it into a running
1147 // average. When the average exceeds REGEN_RATIO_THRESHOLD we double
1148 // bucket_interval. In BALDES this value is xi-like bucket density, not a
1149 // physical step size, so doubling it gives finer per-bin partitioning.
1150 //
1151 // Guards mirror RouteOpt:
1152 // - need total bucket count below REGEN_BUCKETS_CAP (don't explode)
1153 bool considerRegenerate() {
1154 constexpr double REGEN_RATIO_THRESHOLD = 800.0; // RouteOpt's value.
1155 constexpr int REGEN_BUCKETS_CAP = 200000; // ~10k arcs/vertex
1156 // analog for our scale.
1157
1158 // Sample the most recent labeling pass: total dominance scans across
1159 // all forward buckets, divided by non-dominant label count summed over
1160 // both directions. Avoids div-by-zero when the pass produced nothing.
1161 double total_checks = 0.0;
1162 for (int v : dominance_checks_per_bucket) total_checks += v;
1163 const int non_dom_total = non_dominated_labels_per_bucket + non_dominated_labels_per_bucket_bw;
1164 if (non_dom_total <= 0) return false;
1165 const double pass_ratio = total_checks / static_cast<double>(non_dom_total);
1166
1167 // Update running average.
1168 regen_ratio_sum += pass_ratio;
1169 ++regen_ratio_count;
1170 const double avg_ratio = regen_ratio_sum / regen_ratio_count;
1171
1172 if (avg_ratio <= REGEN_RATIO_THRESHOLD) return false;
1173
1174 // Guards.
1175 const int total_buckets = static_cast<int>(fw_buckets.size() + bw_buckets.size());
1176 if (total_buckets >= REGEN_BUCKETS_CAP) return false;
1177
1178 if (bucket_interval > std::numeric_limits<int>::max() / 2) return false;
1179 const int new_interval = bucket_interval * 2;

1181 avg_ratio);

1183 redefine(bucket_interval);
1184
1185 // Reset the running average so we judge the new partition on its own.
1186 regen_ratio_sum = 0.0;
1187 regen_ratio_count = 0;
1188 fixed = false;
1189 return true;

1593 buckets.clear();

1608 const size_t num_nodes = nodes.size();

1610 num_buckets_index.resize(num_nodes);
1611 node_interval_trees.assign(num_nodes, SplayTree());
1612
1613 // Lambda to calculate an interval for a given resource dimension.
1614 // Rounding is applied after computing boundaries.
1615 auto calculate_interval = [&](double lb, double ub, double base_interval, int pos, int max_interval,
1616 bool is_forward) -> std::pair<double, double> {
1617 double start, end;
1618 if (is_forward) {
1619 start = (pos == 0) ? lb : lb + pos * base_interval;
1620 end = (pos == max_interval - 1) ? ub : lb + (pos + 1) * base_interval;
1621 } else {
1622 start = (pos == max_interval - 1) ? lb : ub - (pos + 1) * base_interval;
1623 end = (pos == 0) ? ub : ub - pos * base_interval;
1624 }
1625 return {roundToTwoDecimalPlaces(start), roundToTwoDecimalPlaces(end)};
1626 };
1627
1628 int bucket_index = 0;
1629 int cum_sum = 0;
1630 std::vector<double> interval_start(num_intervals);
1631 std::vector<double> interval_end(num_intervals);
1632 // "pos" holds the current combination indices when
1633 // multi-dimensional.
1634 std::vector<int> pos(num_intervals, 0);
1635
1636 // Process each node (job) to define its buckets.
1637 for (const auto &VRPNode : nodes) {
1638 // Compute the number of splits for each resource dimension.
1639 // For a single dimension, we use the fixed number directly.
1640 // For multiple dimensions we choose an equal split in each
1641 // dimension.
1642 std::vector<int> node_split_counts(num_intervals, 1);
1643 if (num_intervals == 1) {
1644 node_split_counts[0] = fixed_num_buckets;
1645 } else {
1646 // Compute an equal split count per dimension (rounded).
1647 int equal_split =
1648 std::max(1, static_cast<int>(std::round(std::pow(fixed_num_buckets, 1.0 / num_intervals))));
1649 for (int r = 0; r < num_intervals; ++r) { node_split_counts[r] = equal_split; }
1650 // Adjust the first dimension if the product is less than
1651 // fixed_num_buckets.
1652 int product = 1;
1653 for (int r = 0; r < num_intervals; ++r) { product *= node_split_counts[r]; }
1654 if (product < fixed_num_buckets) {
1655 int factor = fixed_num_buckets / product;
1656 node_split_counts[0] *= factor;
1657 }
1658 }
1659
1660 // Compute the width (base interval) for each resource
1661 // dimension.
1662 std::vector<double> node_base_interval(num_intervals);
1663 for (int r = 0; r < num_intervals; ++r) {
1664 node_base_interval[r] = (VRPNode.ub[r] - VRPNode.lb[r]) / static_cast<double>(node_split_counts[r]);
1665 }
1666
1667 // Optionally store the per-node split information.
1668 // (Here we store just the first dimension's split count; you
1669 // could store the full vector.)
1670 bucket_splits[VRPNode.id] = (num_intervals == 1) ? node_split_counts[0] : node_split_counts[0];
1671
1672 SplayTree node_tree;
1673 int n_buckets = 0;
1674
1675 if (num_intervals == 1) {
1676 // Single-dimensional splitting.
1677 for (int j = 0; j < node_split_counts[0]; ++j) {
1678 auto [start, end] = calculate_interval(VRPNode.lb[0], VRPNode.ub[0], node_base_interval[0], j,
1679 node_split_counts[0], D == Direction::Forward);
1680 // Clamp to the node's bounds.
1681 if constexpr (D == Direction::Backward) {
1682 start = std::max(start, VRPNode.lb[0]);
1683 } else {
1684 end = std::min(end, VRPNode.ub[0]);
1685 }
1686 buckets.push_back(Bucket(VRPNode.id, std::vector<double>{start}, std::vector<double>{end}));
1687 node_tree.insert(std::vector<double>{start}, std::vector<double>{end}, bucket_index++);
1688 n_buckets++;
1689 cum_sum++;
1690 }
1691 } else {
1692 // Multi-dimensional splitting.
1693 std::fill(pos.begin(), pos.end(), 0);
1694 while (true) {
1695 for (int r = 0; r < num_intervals; ++r) {
1696 auto [start, end] = calculate_interval(VRPNode.lb[r], VRPNode.ub[r], node_base_interval[r],
1697 pos[r], node_split_counts[r], D == Direction::Forward);
1698 interval_start[r] = start;
1699 interval_end[r] = end;
1700 // Optionally clamp to global resource bounds.
1701 if constexpr (D == Direction::Backward) {
1702 interval_start[r] = std::max(interval_start[r], R_min[r]);
1703 } else {
1704 interval_end[r] = std::min(interval_end[r], R_max[r]);
1705 }
1706 }
1707 buckets.push_back(Bucket(VRPNode.id, interval_start, interval_end));
1708 node_tree.insert(interval_start, interval_end, bucket_index++);
1709 n_buckets++;
1710 cum_sum++;
1711
1712 // Generate next combination of splits.
1713 int i = 0;
1714 while (i < num_intervals) {
1715 ++pos[i];
1716 if (pos[i] < node_split_counts[i]) {
1717 break;
1718 } else {
1719 pos[i] = 0;

1729 num_buckets_index[VRPNode.id] = cum_sum - n_buckets;
1730 node_interval_trees[VRPNode.id] = node_tree;
1731 }
1732
1733 // Update the overall bucket count.
1734 buckets_size = buckets.size();

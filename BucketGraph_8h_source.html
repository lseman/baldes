682 template <Direction D>
683 void split_buckets() {
684 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
685 auto &buckets_size =
686 assign_buckets<D>(fw_buckets_size, bw_buckets_size);
687 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
688 auto &num_buckets_index =
689 assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
690
691 int num_intervals = options.main_resources.size();
692 std::vector<double> total_ranges(num_intervals);
693 std::vector<double> base_intervals(num_intervals);
694
695 // Determine the base interval and other relevant values for
696 // each resource
697 for (int r = 0; r < num_intervals; ++r) {
698 total_ranges[r] = R_max[r] - R_min[r];
699 base_intervals[r] = total_ranges[r] / intervals[r].interval;
700 }
701
702 int original_num_buckets = buckets_size;
703
704 // To track splits per node_id
705 ankerl::unordered_dense::map<int, int> splits_per_node;
706
707 // Loop until the second-to-last bucket to avoid
708 // out-of-bounds access
709 for (int i = 0; i < original_num_buckets - 1; ++i) {
710 const auto &bucket = buckets[i];
711
712 // Calculate mid-point for each dimension
713 std::vector<double> mid_point(bucket.lb.size());
714 if constexpr (D == Direction::Forward) {
715 for (int r = 0; r < bucket.lb.size(); ++r) {
716 mid_point[r] = (bucket.lb[r] + base_intervals[r]) / 2.0;
717 }
718 } else {
719 for (int r = 0; r < bucket.lb.size(); ++r) {
720 mid_point[r] = (bucket.ub[r] - base_intervals[r]) / 2.0;
721 }
722 }
723
724 // Create two new buckets by splitting at the mid-point
725 Bucket bucket1, bucket2;
726 if constexpr (D == Direction::Forward) {
727 bucket1 = Bucket(bucket.node_id, bucket.lb, bucket.ub);
728 bucket2 = Bucket(bucket.node_id, mid_point, bucket.ub);
729 } else {
730 bucket1 = Bucket(bucket.node_id, bucket.lb, bucket.ub);
731 bucket2 = Bucket(bucket.node_id, bucket.lb, mid_point);
732 }
733
734 // Manually shift elements to the right to make space
735 // for the new bucket
736 for (int j = original_num_buckets; j > i + 1; --j) {
737 buckets[j] = buckets[j - 1];
738 }
739
740 // Insert the new bucket at position i+1
741 buckets[i + 1] = bucket2;
742
743 // Replace the current bucket with the first half
744 // (bucket1)
745 buckets[i] = bucket1;
746
747 // Track the number of splits for this node_id
748 splits_per_node[bucket.node_id]++;
749
750 // Since we added a new bucket at i+1, increment i to
751 // skip over the new bucket
752 ++i;
753 ++original_num_buckets; // Adjust the count to account
754 // for the new bucket
755 }
756
757 // Update the global bucket size
758 buckets_size = original_num_buckets;
759
760 // Now update the num_buckets_index for each node_id based
761 // on the final positions
762 int current_index = 0;
763 for (int i = 0; i < original_num_buckets; ++i) {
764 const auto &bucket = buckets[i];
765 if (num_buckets_index[bucket.node_id] == -1) {
766 num_buckets_index[bucket.node_id] = current_index;
767 }
768 current_index++;
769 }
770 }

891 void updateSplit() {
892 // Base thresholds and parameters for adjustments.
893 constexpr double BASE_IMBALANCE_THRESHOLD =
894 0.15; // Slightly lower base threshold.
895 constexpr double MIN_ADJUSTMENT_FACTOR = 0.02;
896 constexpr double MAX_ADJUSTMENT_FACTOR = 0.10;
897 constexpr double LEARNING_RATE_DECAY =
898 0.99; // Decay factor for learning rate.
899 constexpr double EMA_ALPHA = 0.1; // Smoothing factor for EMA.
900
901 // Static variables to persist state across calls.
902 static double learning_rate = MAX_ADJUSTMENT_FACTOR;
903 static double ema_imbalance_ratio = 0.0;
904
905 // Convert label counts to double for ratio computations.
906 const double fw_labels = static_cast<double>(n_fw_labels);
907 const double bw_labels = static_cast<double>(n_bw_labels);
908 const double total_labels = fw_labels + bw_labels;
909
910 // Skip adjustment if too few labels exist.
911 if (total_labels < 10) return;
912
913 // Compute imbalance ratios for forward and backward labels.
914 const double fw_ratio = fw_labels / total_labels;
915 const double bw_ratio = bw_labels / total_labels;
916 const double imbalance_ratio = std::abs(fw_ratio - bw_ratio);
917
918 // Update exponential moving average (EMA) of the imbalance ratio.
919 ema_imbalance_ratio = EMA_ALPHA * imbalance_ratio +
920 (1.0 - EMA_ALPHA) * ema_imbalance_ratio;
921
922 // Dynamic threshold decreases as total_labels increase.
923 const double dynamic_threshold =
924 BASE_IMBALANCE_THRESHOLD * std::exp(-total_labels / 1000.0);
925
926 // Only adjust split if the imbalance exceeds the dynamic threshold.
927 if (ema_imbalance_ratio > dynamic_threshold) {
928 // Severity scales between 0 and 1.
929 const double severity = (ema_imbalance_ratio - dynamic_threshold) /
930 (1.0 - dynamic_threshold);
931
932 // Adaptive adjustment factor increases with severity.
933 const double adjustment_factor =
934 MIN_ADJUSTMENT_FACTOR +
935 (MAX_ADJUSTMENT_FACTOR - MIN_ADJUSTMENT_FACTOR) * severity;
936
937 // Decay the learning rate.
938 learning_rate *= LEARNING_RATE_DECAY;
939
940 // Compute resource range based on the main resource.
941 const double resource_range = R_max[options.main_resources[0]] -
942 R_min[options.main_resources[0]];
943
944 // Base adjustment scales with the resource range and learning rate.
945 const double base_adjustment =
946 adjustment_factor * resource_range * learning_rate;
947
948 // Adjust each split position in q_star.
949 for (size_t i = 0; i < q_star.size(); ++i) {
950 // Compute relative position (0: min, 1: max) within the
951 // resource range.
952 const double relative_pos =
953 (q_star[i] - R_min[options.main_resources[0]]) /
954 resource_range;
955
956 // Position factor gives smaller adjustments near boundaries.
957 const double position_factor =
958 4.0 * relative_pos * (1.0 - relative_pos);
959
960 // Determine direction: shift toward balancing label counts.
961 const double direction = (bw_labels > fw_labels) ? 1.0 : -1.0;
962
963 // Final adjustment for this split.
964 const double final_adjustment =
965 base_adjustment * position_factor * direction;
966
967 // Apply the adjustment with boundary protection.
968 q_star[i] =
969 std::clamp(q_star[i] + final_adjustment,
970 R_min[options.main_resources[0]] +
971 resource_range * 0.1, // Lower bound.
972 R_max[options.main_resources[0]] -
973 resource_range * 0.1 // Upper bound.
974 );
975 }
976
977 // Log adjustments if the base adjustment is significant.
978 if (std::abs(base_adjustment) > resource_range * 0.05) {
979 print_info(
980 "Split adjustment: imbalance={:.3f}, adjustment={:.3f}, "
981 "learning_rate={:.3f}\n",
982 ema_imbalance_ratio, base_adjustment, learning_rate);
983 }
984 }

1180 double real_cost = 0.0;
1181 double red_cost = 0.0;
1182
1183 // Check main resource feasibility.
1184 const double main_resource = L->resources[options.main_resources[0]];
1185 const double q_star_value = q_star[options.main_resources[0]];
1186 if (fw) {
1187 if (main_resource > q_star_value) return nullptr;
1188 } else {
1189 if (main_resource <= q_star_value) return nullptr;
1190 }
1191
1192 // Ensure sufficient nodes in the route.
1193 if (L->nodes_covered.size() <= 3) return nullptr;
1194
1195 // Initialize an empty SRCmap vector with size equal to the current
1196 // number of cuts.
1197 std::vector<uint16_t> updated_SRCmap(cut_storage->size(), 0);
1198
1199 // Prepare new label.
1200 auto new_label = new Label();
1201 new_label->nodes_covered.clear();
1202 new_label->nodes_covered = L->nodes_covered; // Copy the route
1203
1204 // Variables to keep track of costs.
1205 int last_node = -1;
1206
1207 // Traverse the nodes in the route (from current label to root).
1208 for (auto node_id : L->nodes_covered) {
1209 // --- NG-route Feasibility Check ---
1210 // Clear visited bitmap in the neighborhood of node_id.
1211 for (size_t seg = 0, limit = new_label->visited_bitmap.size();
1212 seg < limit; ++seg) {
1213 uint64_t current_visited = new_label->visited_bitmap[seg];
1214 if (!current_visited) continue;
1215 uint64_t neighborhood_mask = neighborhoods_bitmap[node_id][seg];
1216 new_label->visited_bitmap[seg] =
1217 current_visited & neighborhood_mask;
1218 }
1219 set_node_visited(new_label->visited_bitmap, node_id);
1220
1221 // --- Cost Accumulation ---
1222 if (last_node != -1) {
1223 double cij_cost = getcij(last_node, node_id);
1224 real_cost += cij_cost;
1225 red_cost += cij_cost;
1226 }
1227 red_cost -= nodes[node_id].cost;
1228
1229 // --- SRC Logic ---
1230 size_t segment = node_id >> 6;
1231 const uint64_t bit_mask = bit_mask_lookup[node_id & 63];
1232 auto &cutter = cut_storage;
1233 auto &SRCDuals = cutter->SRCDuals;
1234
1235#if defined(SRC)
1236 for (std::size_t idx = 0; idx < cutter->size(); ++idx) {
1237 const auto &cut = cutter->getCut(idx);
1238 // Retrieve base set and order information.
1239 const auto &baseSet = cut.baseSet;

1241 const auto &neighbors = cut.neighbors;
1242 const auto &multipliers = cut.p;
1243
1244 // If the current node is not in the neighbor set, reset SRCmap
1245 // value.
1246 if (!(neighbors[segment] & bit_mask)) {
1247 updated_SRCmap[idx] = 0;
1248 continue;
1249 }
1250 // If the current node is in the base set, update SRC value.
1251 if (baseSet[segment] & bit_mask) {
1252 auto &src_val = updated_SRCmap[idx];
1253 src_val += multipliers.num[baseSetOrder[node_id]];
1254 if (src_val >= multipliers.den) {
1255 red_cost -= SRCDuals[idx];
1256 src_val -= multipliers.den;
1257 }
1258 }
1259 }
1260#endif
1261
1262 // --- Arc and Branching Dual Adjustments ---
1263#if defined(RCC) || defined(EXACT_RCC)
1264 if (last_node != -1) {
1265 red_cost -= arc_duals.getDual(last_node, node_id);
1266 }
1267#endif
1268 if (branching_duals->size() > 0 && last_node != -1) {
1269 red_cost -= branching_duals->getDual(last_node, node_id);
1270 }
1271
1272 last_node = node_id; // Update last_node.
1273 }
1274
1275 // Set final label properties.
1276 new_label->cost = red_cost;
1277 new_label->real_cost = real_cost;
1278 new_label->parent = nullptr;
1279 new_label->node_id = L->node_id;
1280 new_label->is_extended = false;
1281 new_label->resources = L->resources;
1282 new_label->SRCmap = updated_SRCmap;
1283 new_label->fresh = false;
1284
1285 // Determine bucket number for the new label.
1286 std::vector<double> new_resources(options.resources.size());
1287 for (size_t i = 0; i < options.resources.size(); ++i) {
1288 new_resources[i] = new_label->resources[i];
1289 }
1290 int bucket = fw ? get_bucket_number<Direction::Forward>(
1291 new_label->node_id, new_resources)
1292 : get_bucket_number<Direction::Backward>(
1293 new_label->node_id, new_resources);
1294 new_label->vertex = bucket;
1295
1296 return new_label;
1297 }

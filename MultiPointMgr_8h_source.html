94 void updatePool(const DualSolution &new_point, double obj_value) {
95 if (!active) return;
96
97 assert(new_point.size() == solution_size);
98
99 updateObjectiveBounds(obj_value);
100 double relative_improvement = calculateImprovement(obj_value);
101 updateMetrics(relative_improvement);
102
103 if (shouldDeactivate()) {
104 active = false;
105 return;
106 }
107
108 adjustPoolSize(relative_improvement);
109
110 if (shouldAddPoint(new_point, obj_value)) {
111 addPoint(new_point, obj_value);
112 updatePointQualities();
113 maintainPoolSize();
114 }

137 double computeAdaptiveWeight(const DualSolution &dir_sol, const DualSolution &mp_sol,
138 const DualSolution &subgradient, double subgradient_norm) const {
139 if (stability_points.empty() || subgradient_norm < EPSILON) { return 0.5; }
140
141 if (metrics.stagnant_iterations > 0) {
142 double stagnation_factor = std::min(0.8, 0.1 * metrics.stagnant_iterations);
143 return 0.1 + stagnation_factor;
144 }
145
146 std::vector<double> diff_dir(solution_size);
147 double diff_norm = computeDirection(diff_dir, dir_sol, mp_sol);
148
149 if (diff_norm < EPSILON) return 0.4;
150
151 double alignment = computeAlignment(diff_dir, diff_norm, subgradient, subgradient_norm);
152 double improvement_factor = std::exp(-metrics.average_improvement / FAST_CONV_THRESHOLD);
153
154 double base_weight = 0.4 - alignment * 0.6;
155 double adjusted_weight = base_weight * improvement_factor;
156
157 return std::clamp(adjusted_weight, 0.1, 0.8);
158 }

206 double calculateImprovement(double obj_value) {
207 double best_known = getBestObjectiveValue();
208 return std::abs((obj_value - best_known) / (std::abs(best_known) + EPSILON));
209 }

214 void updateMetrics(double relative_improvement) {
215 metrics.last_improvement = relative_improvement;
216 improvement_history.push_back(relative_improvement);
217
218 if (improvement_history.size() > HISTORY_SIZE) { improvement_history.pop_front(); }
219
220 // Update statistical metrics
221 metrics.average_improvement =
222 std::accumulate(improvement_history.begin(), improvement_history.end(), 0.0) / improvement_history.size();
223
224 metrics.variance_improvement = std::accumulate(improvement_history.begin(), improvement_history.end(), 0.0,
225 [&](double acc, double imp) {
226 double diff = imp - metrics.average_improvement;
227 return acc + diff * diff;
228 }) /
229 improvement_history.size();
230
231 // Adapt convergence threshold
232 conv_threshold = FAST_CONV_THRESHOLD * (1.0 + std::sqrt(metrics.variance_improvement));
233 }

238 bool shouldDeactivate() const {
239 return metrics.stagnant_iterations >= Metrics::MAX_STAGNANT ||
240 (improvement_history.size() == HISTORY_SIZE && allSmallImprovements() &&
241 metrics.variance_improvement < EPSILON);
242 }

247 void adjustPoolSize(double relative_improvement) {
248 if (relative_improvement < conv_threshold) {
249 metrics.stagnant_iterations++;
250 current_pool_size = std::max(MIN_POOL_SIZE, current_pool_size - 1);
251 } else {
252 metrics.stagnant_iterations = 0;
253 current_pool_size = std::min(MAX_POOL_SIZE, current_pool_size + 1);
254 }
255 }

260 void updatePointQualities() {
261 double obj_range = best_objective - worst_objective + EPSILON;
262
263 for (auto &point : stability_points) {
264 double obj_quality = (point.objective_value - worst_objective) / obj_range;
265 double age_quality = std::exp(-AGE_DECAY_RATE * point.age);
266 point.quality_score = QUALITY_OBJ_WEIGHT * obj_quality + (1.0 - QUALITY_OBJ_WEIGHT) * age_quality;
267 }
268 }

273 DualSolution computeWeightedSolution() const {
274 DualSolution result(solution_size, 0.0);
275 auto [weights, total_weight] = computeWeights();
276
277 if (total_weight > EPSILON) {
278 for (size_t i = 0; i < stability_points.size(); i++) {
279 double normalized_weight = weights[i] / total_weight;
280 for (size_t j = 0; j < solution_size; j++) {
281 result[j] += normalized_weight * stability_points[i].duals[j];
282 }
283 }
284 }
285
286 return result;
287 }

357 std::pair<std::vector<double>, double> computeWeights() const {

359 double total_weight = 0.0;
360
361 for (size_t i = 0; i < stability_points.size(); i++) {

372 double computeDirection(std::vector<double> &diff_dir, const DualSolution &dir_sol,
373 const DualSolution &mp_sol) const {
374 double diff_norm = 0.0;
375 for (size_t i = 0; i < solution_size; i++) {
376 diff_dir[i] = mp_sol[i] - dir_sol[i];
377 diff_norm += diff_dir[i] * diff_dir[i];
378 }
379 return std::sqrt(diff_norm + EPSILON);
380 }

385 double computeAlignment(const std::vector<double> &diff_dir, double diff_norm, const DualSolution &subgradient,
386 double subgradient_norm) const {
387 double dot_product = 0.0;
388 for (size_t i = 0; i < solution_size; i++) {
389 dot_product += (diff_dir[i] / diff_norm) * (subgradient[i] / subgradient_norm);
390 }
391 return std::clamp(dot_product, -1.0, 1.0);
392 }

397 double vectorNorm(const DualSolution &v1, const DualSolution &v2) const {
398 assert(v1.size() == v2.size());
399 double sum = 0.0;
400 for (size_t i = 0; i < v1.size(); i++) {
401 double diff = v2[i] - v1[i];
402 sum += diff * diff;
403 }
404 return std::sqrt(sum + EPSILON);
405 }
406
410 double vectorNorm(const DualSolution &v) const {
411 double sum = 0.0;
412 for (double val : v) { sum += val * val; }
413 return std::sqrt(sum + EPSILON);
414 }

419 void normalizeVector(DualSolution &v) const {
420 double norm = vectorNorm(v);

422 for (double &val : v) { val /= norm; }

447inline double computeRelativeGap(const DualSolution &sol1, const DualSolution &sol2) {
448 assert(sol1.size() == sol2.size());
449 double num = 0.0;
450 double den = 0.0;
451 for (size_t i = 0; i < sol1.size(); i++) {
452 num += std::abs(sol1[i] - sol2[i]);
453 den += std::abs(sol1[i]) + std::abs(sol2[i]);
454 }
455 return num / (den + MultiPointManager::EPSILON);

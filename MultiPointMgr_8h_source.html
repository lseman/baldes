69 void updatePointQualities() {
70 double obj_range = worst_objective - best_objective + EPSILON;
71
72 double obj_factor = QUALITY_OBJ_WEIGHT / obj_range;
73 double age_factor = 1.0 - QUALITY_OBJ_WEIGHT;
74
75 for (auto& point : stability_points) {
76 point.quality_score =
77 obj_factor * (worst_objective - point.objective_value) +

148 void updatePool(const DualSolution& new_point, double obj_value) {
149 if (!active || new_point.empty() || new_point.size() != solution_size)
150 return;
151
152 // Update any bounds related to the objective value.
153 updateObjectiveBounds(obj_value);
154
155 // For minimization, a lower objective is better.
156 // Use positive infinity as the default best value if no best point
157 // exists.
158 double prev_best = best_point ? best_point->objective_value
159 : std::numeric_limits<double>::infinity();
160
161 // Compute the relative improvement: (prev_best - new_value) /
162 // (|prev_best| + EPSILON)
163 double improvement =
164 std::abs((prev_best - obj_value) / (std::abs(prev_best) + EPSILON));
165 updateMetrics(improvement);
166
167 // Check convergence: if improvement is below threshold over a number of
168 // iterations, deactivate.
169 if (improvement < conv_threshold && improvement_history.back() != 0.0 &&
170 std::all_of(
171 improvement_history.begin(), improvement_history.end(),
172 [this](double imp) { return imp < conv_threshold; })) {
173 active = false;
174 return;
175 }
176
177 adjustPoolSize(improvement);
178
179 // If the new point is good enough to be added to the pool, add it.
180 if (shouldAddPoint(new_point, obj_value)) {
181 addPoint(new_point, obj_value);
182 updatePointQualities();
183 maintainPoolSize();
184 updateBestPoint();
185 }
186 }

188 DualSolution getWeightedSolution() const {
189 if (stability_points.empty()) {
190 return DualSolution(solution_size, 0.0);
191 }
192
193 if ((improvement_history.back() < conv_threshold) && best_point) {
194 return best_point->duals;
195 }
196
197 std::vector<double> weights;
198 double total_weight = 0.0;
199 for (const auto& point : stability_points) {
200 weights.push_back(point.quality_score);
201 total_weight += point.quality_score;
202 }
203
204 if (total_weight <= EPSILON) {
205 return DualSolution(solution_size, 0.0);
206 }
207
208 DualSolution result(solution_size, 0.0);
209 for (size_t i = 0; i < stability_points.size(); i++) {
210 double normalized_weight = weights[i] / total_weight;
211 std::transform(stability_points[i].duals.begin(),
212 stability_points[i].duals.end(), result.begin(),
213 result.begin(),
214 [normalized_weight](double x, double y) {
215 return y + normalized_weight * x;
216 });
217 }
218 return result;
219 }

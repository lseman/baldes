58 void updatePool(const DualSolution &new_point, double obj_value) {
59 if (!active) return;
60
61 // Update objective bounds
62 best_objective = std::max(best_objective, obj_value);
63 worst_objective = std::min(worst_objective, obj_value);
64
65 // Calculate improvement
66 double best_known = getBestObjectiveValue();
67 double relative_improvement = std::abs((obj_value - best_known) / (std::abs(best_known) + EPSILON));
68
69 // Track improvement history
70 improvement_history.push_back(relative_improvement);
71 if (improvement_history.size() > HISTORY_SIZE) {
72 improvement_history.erase(improvement_history.begin());
73 }
74
75 // Update metrics and check convergence
76 metrics.last_improvement = relative_improvement;
77 if (relative_improvement < FAST_CONV_THRESHOLD) {
78 metrics.stagnant_iterations++;
79 if (metrics.stagnant_iterations >= Metrics::MAX_STAGNANT ||
80 (improvement_history.size() == HISTORY_SIZE &&
81 allSmallImprovements())) {
82 active = false;
83 return;
84 }
85 current_pool_size = MIN_POOL_SIZE;
86 } else {
87 metrics.stagnant_iterations = 0;
88 current_pool_size = std::min(MAX_POOL_SIZE, current_pool_size + 1);
89 }
90
91 if (shouldAddPoint(new_point, obj_value)) {
92 addPoint(new_point, obj_value);
93 maintainPoolSize();
94 }
95 }

99
100 if (stability_points.empty()) return result;

116 }

214 std::pair<std::vector<double>, double> computeWeights() const {
215 std::vector<double> weights(stability_points.size());
216 double total_weight = 0.0;
217
218 double obj_range = best_objective - worst_objective;
219
220 for (size_t i = 0; i < stability_points.size(); i++) {
221 const auto &point = stability_points[i];
222 double age_factor = 1.0 / (1.0 + 0.7 * point.age); // Faster decay
223 double obj_factor = obj_range > EPSILON ?
224 (point.objective_value - worst_objective) / obj_range : 1.0;
225
226 weights[i] = age_factor * (0.2 + 0.8 * obj_factor); // More emphasis on quality
227 total_weight += weights[i];
228 }
229
230 return {weights, total_weight};
231 }
